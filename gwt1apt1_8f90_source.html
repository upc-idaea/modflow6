224  subroutine apt_mc(this, moffset, iasln, jasln)
225 ! ******************************************************************************
226 ! apt_mc -- map package connection to matrix
227 ! ******************************************************************************
228 !
229 ! SPECIFICATIONS:
230 ! ------------------------------------------------------------------------------
231  use sparsemodule, only: sparsematrix
232  ! -- dummy
233  class(gwtapttype),intent(inout) :: this
234  integer(I4B), intent(in) :: moffset
235  integer(I4B), dimension(:), intent(in) :: iasln
236  integer(I4B), dimension(:), intent(in) :: jasln
237  ! -- local
238  integer(I4B) :: n, j, jj, iglo, jglo
239  integer(I4B) :: ipos
240  ! -- format
241 ! ------------------------------------------------------------------------------
242  !
243  !
244  if (this%imatrows /= 0) then
245  !
246  ! -- allocate pointers to global matrix
247  allocate(this%idxlocnode(this%ncv))
248  allocate(this%idxpakdiag(this%ncv))
249  allocate(this%idxdglo(this%maxbound))
250  allocate(this%idxoffdglo(this%maxbound))
251  allocate(this%idxsymdglo(this%maxbound))
252  allocate(this%idxsymoffdglo(this%maxbound))
253  n = 0
254  if (this%idxbudfjf /= 0) then
255  n = this%flowbudptr%budterm(this%idxbudfjf)%maxlist
256  end if
257  allocate(this%idxfjfdglo(n))
258  allocate(this%idxfjfoffdglo(n))
259  !
260  ! -- Find the position of each connection in the global ia, ja structure
261  ! and store them in idxglo. idxglo allows this model to insert or
262  ! retrieve values into or from the global A matrix
263  ! -- apt rows
264  do n = 1, this%ncv
265  this%idxlocnode(n) = this%dis%nodes + this%ioffset + n
266  iglo = moffset + this%dis%nodes + this%ioffset + n
267  this%idxpakdiag(n) = iasln(iglo)
268  end do
269  do ipos = 1, this%flowbudptr%budterm(this%idxbudgwf)%nlist
270  n = this%flowbudptr%budterm(this%idxbudgwf)%id1(ipos)
271  j = this%flowbudptr%budterm(this%idxbudgwf)%id2(ipos)
272  iglo = moffset + this%dis%nodes + this%ioffset + n
273  jglo = j + moffset
274  searchloop: do jj = iasln(iglo), iasln(iglo + 1) - 1
275  if(jglo == jasln(jj)) then
276  this%idxdglo(ipos) = iasln(iglo)
277  this%idxoffdglo(ipos) = jj
278  exit searchloop
279  endif
280  enddo searchloop
281  end do
282  !
283  ! -- apt contributions to gwf portion of global matrix
284  do ipos = 1, this%flowbudptr%budterm(this%idxbudgwf)%nlist
285  n = this%flowbudptr%budterm(this%idxbudgwf)%id1(ipos)
286  j = this%flowbudptr%budterm(this%idxbudgwf)%id2(ipos)
287  iglo = j + moffset
288  jglo = moffset + this%dis%nodes + this%ioffset + n
289  symsearchloop: do jj = iasln(iglo), iasln(iglo + 1) - 1
290  if(jglo == jasln(jj)) then
291  this%idxsymdglo(ipos) = iasln(iglo)
292  this%idxsymoffdglo(ipos) = jj
293  exit symsearchloop
294  endif
295  enddo symsearchloop
296  end do
297  !
298  ! -- apt-apt contributions to gwf portion of global matrix
299  if (this%idxbudfjf /= 0) then
300  do ipos = 1, this%flowbudptr%budterm(this%idxbudfjf)%nlist
301  n = this%flowbudptr%budterm(this%idxbudfjf)%id1(ipos)
302  j = this%flowbudptr%budterm(this%idxbudfjf)%id2(ipos)
303  iglo = moffset + this%dis%nodes + this%ioffset + n
304  jglo = moffset + this%dis%nodes + this%ioffset + j
305  fjfsearchloop: do jj = iasln(iglo), iasln(iglo + 1) - 1
306  if(jglo == jasln(jj)) then
307  this%idxfjfdglo(ipos) = iasln(iglo)
308  this%idxfjfoffdglo(ipos) = jj
309  exit fjfsearchloop
310  endif
311  enddo fjfsearchloop
312  end do
313  end if
314  else
315  allocate(this%idxlocnode(0))
316  allocate(this%idxpakdiag(0))
317  allocate(this%idxdglo(0))
318  allocate(this%idxoffdglo(0))
319  allocate(this%idxsymdglo(0))
320  allocate(this%idxsymoffdglo(0))
321  allocate(this%idxfjfdglo(0))
322  allocate(this%idxfjfoffdglo(0))
323  endif
324  !
325  ! -- return
326  return
327  end subroutine apt_mc

807  subroutine apt_fc_expanded(this, rhs, ia, idxglo, amatsln)
808 ! ******************************************************************************
809 ! apt_fc_expanded -- formulate for the expanded matrix case in which new
810 ! rows are added to the system of equations for each feature
811 ! ****************************************************************************
812 !
813 ! SPECIFICATIONS:
814 ! ------------------------------------------------------------------------------
815  ! -- modules
816  ! -- dummy
817  class(gwtapttype) :: this
818  real(DP), dimension(:), intent(inout) :: rhs
819  integer(I4B), dimension(:), intent(in) :: ia
820  integer(I4B), dimension(:), intent(in) :: idxglo
821  real(DP), dimension(:), intent(inout) :: amatsln
822  ! -- local
823  integer(I4B) :: j, n, n1, n2
824  integer(I4B) :: iloc
825  integer(I4B) :: iposd, iposoffd
826  integer(I4B) :: ipossymd, ipossymoffd
827  real(DP) :: cold
828  real(DP) :: qbnd
829  real(DP) :: omega
830  real(DP) :: rrate
831  real(DP) :: rhsval
832  real(DP) :: hcofval
833 ! ------------------------------------------------------------------------------
834  !
835  ! -- call the specific method for the advanced transport package, such as
836  ! what would be overridden by
837  ! GwtLktType, GwtSftType, GwtMwtType, GwtUztType
838  ! This routine will add terms for rainfall, runoff, or other terms
839  ! specific to the package
840  call this%pak_fc_expanded(rhs, ia, idxglo, amatsln)
841  !
842  ! -- mass storage in features
843  do n = 1, this%ncv
844  cold = this%xoldpak(n)
845  iloc = this%idxlocnode(n)
846  iposd = this%idxpakdiag(n)
847  call this%apt_stor_term(n, n1, n2, rrate, rhsval, hcofval)
848  amatsln(iposd) = amatsln(iposd) + hcofval
849  rhs(iloc) = rhs(iloc) + rhsval
850  end do
851  !
852  ! -- add to mover contribution
853  if (this%idxbudtmvr /= 0) then
854  do j = 1, this%flowbudptr%budterm(this%idxbudtmvr)%nlist
855  call this%apt_tmvr_term(j, n1, n2, rrate, rhsval, hcofval)
856  iloc = this%idxlocnode(n1)
857  iposd = this%idxpakdiag(n1)
858  amatsln(iposd) = amatsln(iposd) + hcofval
859  rhs(iloc) = rhs(iloc) + rhsval
860  end do
861  end if
862  !
863  ! -- add from mover contribution
864  if (this%idxbudfmvr /= 0) then
865  do n = 1, this%ncv
866  rhsval = this%qmfrommvr(n)
867  iloc = this%idxlocnode(n)
868  rhs(iloc) = rhs(iloc) - rhsval
869  end do
870  end if
871  !
872  ! -- go through each apt-gwf connection
873  do j = 1, this%flowbudptr%budterm(this%idxbudgwf)%nlist
874  !
875  ! -- set n to feature number and process if active feature
876  n = this%flowbudptr%budterm(this%idxbudgwf)%id1(j)
877  if (this%iboundpak(n) /= 0) then
878  !
879  ! -- set acoef and rhs to negative so they are relative to apt and not gwt
880  qbnd = this%flowbudptr%budterm(this%idxbudgwf)%flow(j)
881  omega = dzero
882  if (qbnd < dzero) omega = done
883  !
884  ! -- add to apt row
885  iposd = this%idxdglo(j)
886  iposoffd = this%idxoffdglo(j)
887  amatsln(iposd) = amatsln(iposd) + omega * qbnd
888  amatsln(iposoffd) = amatsln(iposoffd) + (done - omega) * qbnd
889  !
890  ! -- add to gwf row for apt connection
891  ipossymd = this%idxsymdglo(j)
892  ipossymoffd = this%idxsymoffdglo(j)
893  amatsln(ipossymd) = amatsln(ipossymd) - (done - omega) * qbnd
894  amatsln(ipossymoffd) = amatsln(ipossymoffd) - omega * qbnd
895  end if
896  end do
897  !
898  ! -- go through each apt-apt connection
899  if (this%idxbudfjf /= 0) then
900  do j = 1, this%flowbudptr%budterm(this%idxbudfjf)%nlist
901  n1 = this%flowbudptr%budterm(this%idxbudfjf)%id1(j)
902  n2 = this%flowbudptr%budterm(this%idxbudfjf)%id2(j)
903  qbnd = this%flowbudptr%budterm(this%idxbudfjf)%flow(j)
904  if (qbnd <= dzero) then
905  omega = done
906  else
907  omega = dzero
908  end if
909  iposd = this%idxfjfdglo(j)
910  iposoffd = this%idxfjfoffdglo(j)
911  amatsln(iposd) = amatsln(iposd) + omega * qbnd
912  amatsln(iposoffd) = amatsln(iposoffd) + (done - omega) * qbnd
913  end do
914  end if
915  !
916  ! -- Return
917  return
918  end subroutine apt_fc_expanded
