535  subroutine hashbndtimeseries(this)
536 ! ******************************************************************************
537 ! HashBndTimeSeries --
538 ! Store all boundary (stress) time series links in
539 ! TsContainers and construct hash table BndTsHashTable.
540 ! ******************************************************************************
541 !
542 ! SPECIFICATIONS:
543 ! ------------------------------------------------------------------------------
544  ! -- dummy
545  class(timeseriesmanagertype), intent(inout) :: this
546  ! -- local
547  integer(I4B) :: i, j, k, numtsfiles, numts
548  character(len=LENTIMESERIESNAME) :: name
549  type(timeseriesfiletype), pointer :: tsfile => null()
550 ! ------------------------------------------------------------------------------
551  !
552  ! Initialize the hash table
553  call hash_table_cr(this%BndTsHashTable)
554  !
555  ! Allocate the TsContainers array to accommodate all time-series links.
556  numts = this%tsfileList%CountTimeSeries()
557  allocate(this%TsContainers(numts))
558  !
559  ! Store a pointer to each time series in the TsContainers array
560  ! and put its key (time-series name) and index in the hash table.
561  numtsfiles = this%tsfileList%Counttsfiles()
562  k = 0
563  do i = 1, numtsfiles
564  tsfile => this%tsfileList%Gettsfile(i)
565  numts = tsfile%Count()
566  do j=1,numts
567  k = k + 1
568  this%TsContainers(k)%timeSeries => tsfile%GetTimeSeries(j)
569  if (associated(this%TsContainers(k)%timeSeries)) then
570  name = this%TsContainers(k)%timeSeries%Name
571  call this%BndTsHashTable%add_entry(name, k)
572  endif
573  enddo
574  enddo
575  !
576  return
577  end subroutine hashbndtimeseries

581  subroutine read_value_or_time_series(textInput, ii, jj, bndElem, &
582  pkgName, auxOrBnd, tsManager, iprpak, tsLink)
583 ! ******************************************************************************
584 ! read_value_or_time_series --
585 ! Call this subroutine if the time-series link is available or needed.
586 ! ******************************************************************************
587 !
588 ! SPECIFICATIONS:
589 ! ------------------------------------------------------------------------------
590  ! -- dummy
591  character(len=*), intent(in) :: textinput
592  integer(I4B), intent(in) :: ii
593  integer(I4B), intent(in) :: jj
594  real(dp), pointer, intent(inout) :: bndelem
595  character(len=*), intent(in) :: pkgname
596  character(len=3), intent(in) :: auxorbnd
597  type(timeseriesmanagertype), intent(inout) :: tsmanager
598  integer(I4B), intent(in) :: iprpak
599  type(timeserieslinktype), pointer, intent(inout) :: tslink
600  ! -- local
601  type(timeseriestype), pointer :: timeseries => null()
602  type(timeserieslinktype), pointer :: tsltemp => null()
603  integer(I4B) :: i, istat, nlinks
604  real(dp) :: r
605  character(len=LINELENGTH) :: errmsg
606  character(len=LENTIMESERIESNAME) :: tsnametemp
607  logical :: found
608 ! ------------------------------------------------------------------------------
609  !
610  read (textinput,*,iostat=istat) r
611  if (istat == 0) then
612  bndelem = r
613  else
614  tsnametemp = textinput
615  call upcase(tsnametemp)
616  ! -- If text is a time-series name, get value
617  ! from time series.
618  timeseries => tsmanager%get_time_series(tsnametemp)
619  ! -- Create a time series link and add it to the package
620  ! list of time series links used by the array.
621  if (associated(timeseries)) then
622  ! -- Assign value from time series to current
623  ! array element
624  r = timeseries%GetValue(totimsav, totim, tsmanager%extendTsToEndOfSimulation)
625  bndelem = r
626  ! Look to see if this array element already has a time series
627  ! linked to it. If not, make a link to it.
628  nlinks = tsmanager%CountLinks(auxorbnd)
629  found = .false.
630  searchlinks: do i=1,nlinks
631  tsltemp => tsmanager%GetLink(auxorbnd, i)
632  if (tsltemp%PackageName == pkgname) then
633  ! -- Check ii, jj against iRow, jCol stored in link
634  if (tsltemp%IRow==ii .and. tsltemp%JCol==jj) then
635  ! -- This array element is already linked to a time series.
636  tslink => tsltemp
637  found = .true.
638  exit searchlinks
639  endif
640  endif
641  enddo searchlinks
642  if (.not. found) then
643  ! -- Link was not found. Make one and add it to the list.
644  call tsmanager%make_link(timeseries, pkgname, auxorbnd, bndelem, &
645  ii, jj, iprpak, tslink, '', '')
646  endif

763 !
764 ! SPECIFICATIONS:
765 ! ------------------------------------------------------------------------------
766  ! -- return variable
767  logical :: found
768  ! -- dummy
769  type(timeseriesmanagertype), intent(inout) :: tsmanager
770  integer(I4B), intent(in) :: ii
771  integer(I4B), intent(in) :: jj

781  !
782  ! -- determine if link exists
783  nlinks = tsmanager%CountLinks(auxorbnd)
784  found = .false.
785  removelink = -1
786  csearchlinks: do i = 1, nlinks
787  tsltemp => tsmanager%GetLink(auxorbnd, i)

799  end if
800  end if

804  if (removelink > 0) then

806  call tsmanager%boundTsLinks%RemoveNode(removelink, .true.)
807  else if (auxorbnd == 'AUX') then
