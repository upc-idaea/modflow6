184  !> @brief Check that a given property value is valid
185  !!
186  !! Deferred procedure implementation called by the TvBaseType code after a
187  !! property value change occurs. Check if the property value of the given
188  !! variable at the given node is invalid, and log an error if so. Update
189  !! K22 and K33 values appropriately when specified as anisotropy.
190  !!
191  !<
192  subroutine tvk_validate_change(this, n, varName)
193  ! -- dummy variables
194  class(tvktype) :: this
195  integer(I4B), intent(in) :: n
196  character(len=*), intent(in) :: varName
197  ! -- local variables
198  character(len=LINELENGTH) :: cellstr
199  ! -- formats
200  character(len=*), parameter :: fmtkerr = &
201  "(1x, a, ' changed hydraulic property ', a, ' is <= 0 for cell ', a, &
202  &' ', 1pg15.6)"
203  !
204  ! -- Mark the node as being changed this time step
205  this%nodekchange(n) = 1
206  !
207  ! -- Check the changed value is ok
208  if(varname == 'K') then
209  if(this%k11(n) <= dzero) then
210  call this%dis%noder_to_string(n, cellstr)
211  write(errmsg, fmtkerr) trim(adjustl(this%packName)), 'K', trim(cellstr), this%k11(n)
212  call store_error(errmsg)
213  end if
214  elseif(varname == 'K22') then
215  if(this%ik22overk == 1) then
216  this%k22(n) = this%k22(n) * this%k11(n)
217  end if
218  if(this%k22(n) <= dzero) then
219  call this%dis%noder_to_string(n, cellstr)
220  write(errmsg, fmtkerr) trim(adjustl(this%packName)), 'K22', trim(cellstr), this%k22(n)
221  call store_error(errmsg)
222  end if
223  elseif(varname == 'K33') then
224  if(this%ik33overk == 1) then
225  this%k33(n) = this%k33(n) * this%k33(n)
226  end if
227  if(this%k33(n) <= dzero) then
228  call this%dis%noder_to_string(n, cellstr)
229  write(errmsg, fmtkerr) trim(adjustl(this%packName)), 'K33', trim(cellstr), this%k33(n)
230  call store_error(errmsg)
231  end if
232  end if
233  !
234  return
235  end subroutine tvk_validate_change
