185 ! flowtable_df -- Define the new flow table object
186 ! ******************************************************************************

188 ! SPECIFICATIONS:
189 ! ------------------------------------------------------------------------------
190  ! -- modules
191  ! -- dummy
192  class(budgetobjecttype) :: this
193  integer(I4B), intent(in) :: iout
194  character(len=*), intent(in), optional :: cellids

196  character(len=LINELENGTH) :: title
197  character(len=LINELENGTH) :: text
198  character(len=LENBUDTXT) :: flowtype
199  character(len=LENBUDTXT) :: tag
200  character(len=LENBUDTXT) :: coupletype
201  logical :: lfound
202  logical :: add_cellids
203  integer(I4B) :: maxcol
204  integer(I4B) :: idx
205  integer(I4B) :: ipos
206  integer(I4B) :: i
207 ! ------------------------------------------------------------------------------
208  !
209  ! -- process optional variables
210  if (present(cellids)) then
211  add_cellids = .true.
212  coupletype = cellids
213  else
214  add_cellids = .false.
215  end if
216  !
217  ! -- allocate scalars
218  allocate(this%add_cellids)
219  allocate(this%icellid)
220  allocate(this%nflowterms)
221  !
222  ! -- initialize scalars
223  this%add_cellids = add_cellids
224  this%nflowterms = 0
225  this%icellid = 0
226  !
227  ! -- determine the number of columns in the table
228  maxcol = 3
229  if (add_cellids) then
230  maxcol = maxcol + 1
231  end if
232  do i = 1, this%nbudterm
233  lfound = .false.
234  flowtype = this%budterm(i)%get_flowtype()
235  if (trim(adjustl(flowtype)) == 'FLOW-JA-FACE') then
236  lfound = .true.
237  maxcol = maxcol + 2

239  lfound = .true.
240  maxcol = maxcol + 1
241  end if
242  if (lfound) then
243  this%nflowterms = this%nflowterms + 1
244  if (add_cellids) then
245  if (trim(adjustl(flowtype)) == trim(adjustl(coupletype))) then
246  this%icellid = i
247  end if
248  end if
249  end if
250  end do
251  !
252  ! -- allocate arrays
253  allocate(this%istart(this%nflowterms))
254  allocate(this%iflowterms(this%nflowterms))
255  !
256  ! -- set up flow tableobj
257  title = trim(this%name) // ' PACKAGE - SUMMARY OF FLOWS FOR ' // &
258  'EACH CONTROL VOLUME'
259  call table_cr(this%flowtab, this%name, title)
260  call this%flowtab%table_df(this%ncv, maxcol, iout, transient=.true.)
261  !
262  ! -- Go through and set up flow table budget terms
263  text = 'NUMBER'
264  call this%flowtab%initialize_column(text, 10, alignment=tabcenter)
265  if (add_cellids) then
266  text = 'CELLID'
267  call this%flowtab%initialize_column(text, 20, alignment=tableft)
268  end if
269  idx = 1
270  do i = 1, this%nbudterm
271  lfound = .false.
272  flowtype = this%budterm(i)%get_flowtype()
273  tag = trim(adjustl(flowtype))
274  ipos = index(tag, '-')
275  if (ipos > 0) then
276  tag(ipos:ipos) = ' '
277  end if
278  if (trim(adjustl(flowtype)) == 'FLOW-JA-FACE') then
279  lfound = .true.
280  text = 'INFLOW'
281  call this%flowtab%initialize_column(text, 12, alignment=tabcenter)
282  text = 'OUTFLOW'
283  call this%flowtab%initialize_column(text, 12, alignment=tabcenter)
284  else if (trim(adjustl(flowtype)) /= 'AUXILIARY') then
285  lfound = .true.
286  call this%flowtab%initialize_column(tag, 12, alignment=tabcenter)
287  end if
288  if (lfound) then
289  this%iflowterms(idx) = i
290  idx = idx + 1
291  end if
292  end do
293  text = 'IN - OUT'
294  call this%flowtab%initialize_column(text, 12, alignment=tabcenter)
295  text = 'PERCENT DIFFERENCE'
296  call this%flowtab%initialize_column(text, 12, alignment=tabcenter)
297  !
298  ! -- Return
299  return
300  end subroutine flowtable_df

302  subroutine accumulate_terms(this)
303 ! ******************************************************************************
304 ! accumulate_terms -- add up accumulators and submit to budget table
305 ! ******************************************************************************
306 !
307 ! SPECIFICATIONS:
308 ! ------------------------------------------------------------------------------
309  ! -- modules
310  use tdismodule, only: delt
311  ! -- dummy
312  class(budgetobjecttype) :: this

314  character(len=LENBUDTXT) :: flowtype

316  real(DP) :: ratin, ratout

344  subroutine write_flowtable(this, dis, kstp, kper, cellidstr)
345 ! ******************************************************************************
346 ! write_flowtable -- Write the flow table for each advanced package control
347 ! volume
348 ! ******************************************************************************
349 !
350 ! SPECIFICATIONS:
351 ! ------------------------------------------------------------------------------
352  ! -- modules
353  ! -- dummy
354  class(budgetobjecttype) :: this
355  class(disbasetype), intent(in) :: dis
356  integer(I4B), intent(in) :: kstp
357  integer(I4B), intent(in) :: kper
358  character(len=20), dimension(:), optional :: cellidstr
359  ! -- dummy
360  character(len=LENBUDTXT) :: flowtype
361  character(len=20) :: cellid
362  integer(I4B) :: nlist
363  integer(I4B) :: id1
364  integer(I4B) :: id2
365  integer(I4B) :: icv
366  integer(I4B) :: idx
367  integer(I4B) :: i
368  integer(I4B) :: j
369  real(DP) :: v
370  real(DP) :: qin
371  real(DP) :: qout
372  real(DP) :: q
373  real(DP) :: qinflow
374  real(DP) :: qoutflow
375  real(DP) :: qerr
376  real(DP) :: qavg
377  real(DP) :: qpd
378 ! ------------------------------------------------------------------------------
379  !
380  ! -- reset starting position
381  do j = 1, this%nflowterms
382  this%istart(j) = 1
383  end do
384  !
385  ! -- set table kstp and kper
386  call this%flowtab%set_kstpkper(kstp, kper)
387  !
388  ! -- write the table
389  do icv = 1, this%ncv
390  call this%flowtab%add_term(icv)
391  !
392  ! -- initialize flow terms for the control volume
393  qin = dzero
394  qout = dzero
395  !
396  ! -- add cellid if required
397  if (this%add_cellids) then
398  if (present(cellidstr)) then
399  !
400  ! -- If there are not maxbound entries for this%budterm(idx),
401  ! which can happen for sfr, for example, if 'none' connections
402  ! are specified, then cellidstr should be passed in if the flow
403  ! table needs a cellid label.
404  cellid = cellidstr(icv)
405  else
406  !
407  ! -- Determine the cellid for this entry. The cellid, such as
408  ! (1, 10, 10), is assumed to be in the id2 column of this budterm.
409  j = this%icellid
410  idx = this%iflowterms(j)
411  i = this%istart(j)
412  id2 = this%budterm(idx)%get_id2(i)
413  if (id2 > 0) then
414  call dis%noder_to_string(id2, cellid)
415  else
416  cellid = 'NONE'
417  end if
418  end if
419  call this%flowtab%add_term(cellid)
420  end if
421  !
422  ! -- iterate over the flow terms
423  do j = 1, this%nflowterms
424  !
425  ! -- initialize flow terms for the row
426  q = dzero
427  qinflow = dzero
428  qoutflow = dzero
429  !
430  ! -- determine the index, flowtype and length of
431  ! the flowterm
432  idx = this%iflowterms(j)
433  flowtype = this%budterm(idx)%get_flowtype()
434  nlist = this%budterm(idx)%get_nlist()
435  !
436  ! -- iterate over the entries in the flowtype. If id1 is not ordered
437  ! then need to look through the entire list each time
438  colterm: do i = this%istart(j), nlist
439  id1 = this%budterm(idx)%get_id1(i)
440  if (this%budterm(idx)%ordered_id1) then
441  if(id1 > icv) then
442  this%istart(j) = i
443  exit colterm
444  end if
445  else
446  if (id1 /= icv) cycle colterm
447  end if
448  v = this%budterm(idx)%get_flow(i)
449  if (trim(adjustl(flowtype)) == 'FLOW-JA-FACE') then
450  if (v < dzero) then
451  qoutflow = qoutflow + v
452  else
453  qinflow = qinflow + v
454  end if
455  end if
456  !
457  ! -- accumulators
458  q = q + v
459  if (v < dzero) then
460  qout = qout + v
461  else
462  qin = qin + v
463  end if
464  end do colterm
465  !
466  ! -- add entry to table
467  if (trim(adjustl(flowtype)) == 'FLOW-JA-FACE') then
468  call this%flowtab%add_term(qinflow)
469  call this%flowtab%add_term(qoutflow)
470  else
471  call this%flowtab%add_term(q)
472  end if
473  end do
474  !
475  ! -- calculate in-out and percent difference
476  qerr = qin + qout
477  qavg = dhalf * (qin - qout)
478  qpd = dzero
479  if (qavg > dzero) then
480  qpd = dhundred * qerr / qavg
481  end if
482  call this%flowtab%add_term(qerr)
483  call this%flowtab%add_term(qpd)
484  end do
485  !
486  ! -- return
487  return
488  end subroutine write_flowtable

713 ! bfr_advance -- copy the information from the binary file into budterms
714 ! ******************************************************************************
715 !
716 ! SPECIFICATIONS:
717 ! ------------------------------------------------------------------------------
718  ! -- modules
719  use tdismodule, only: kstp, kper
720  ! -- dummy
721  class(budgetobjecttype) :: this
722  class(disbasetype), intent(in) :: dis
723  integer(I4B), intent(in) :: iout
724  ! -- dummy
725  logical :: readnext
726  character(len=*), parameter :: fmtkstpkper = &
727  "(1x,/1x, a, ' READING BUDGET TERMS FOR KSTP ', i0, ' KPER ', i0)"
728  character(len=*), parameter :: fmtbudkstpkper = &
729  "(1x,/1x, a, ' SETTING BUDGET TERMS FOR KSTP ', i0, ' AND KPER ', &
730  &i0, ' TO BUDGET FILE TERMS FROM KSTP ', i0, ' AND KPER ', i0)"
731 ! ------------------------------------------------------------------------------
732  !
733  ! -- Do not read the budget if the budget is at end of file or if the next
734  ! record in the budget file is the first timestep of the next stress
735  ! period. Also do not read if it is the very first time step because
736  ! the first chunk of data is read as part of the initialization
737  readnext = .true.
738  if (kstp * kper == 1) then
739  readnext = .false.
740  else if (kstp * kper > 1) then
741  if (this%bfr%endoffile) then
742  readnext = .false.
743  else
744  if (this%bfr%kpernext == kper + 1 .and. this%bfr%kstpnext == 1) &
745  readnext = .false.
746  endif
747  endif
748  !
749  ! -- Read the next record
750  if (readnext) then
751  !
752  ! -- Write the current time step and stress period
753  if (iout > 0) &
754  write(iout, fmtkstpkper) this%name, kstp, kper
755  !
756  ! -- read flows from the binary file and copy them into this%budterm(:)
757  call this%fill_from_bfr(dis, iout)
758  else
759  if (iout > 0) &
760  write(iout, fmtbudkstpkper) trim(this%name), kstp, kper, &
761  this%bfr%kstp, this%bfr%kper
762  endif
763  !
764  ! -- Return
765  return
766  end subroutine bfr_advance
