264  subroutine dsp_fc(this, kiter, nodes, nja, njasln, amatsln, idxglo, rhs, cnew)
265 ! ******************************************************************************
266 ! dsp_fc -- Calculate coefficients and fill amat and rhs
267 ! ******************************************************************************
268 !
269 ! SPECIFICATIONS:
270 ! ------------------------------------------------------------------------------
271  ! -- modules
272  ! -- dummy
273  class(gwtdsptype) :: this
274  integer(I4B) :: kiter
275  integer(I4B),intent(in) :: nodes
276  integer(I4B),intent(in) :: nja
277  integer(I4B),intent(in) :: njasln
278  real(DP),dimension(njasln),intent(inout) :: amatsln
279  integer(I4B),intent(in),dimension(nja) :: idxglo
280  real(DP),intent(inout),dimension(nodes) :: rhs
281  real(DP),intent(inout),dimension(nodes) :: cnew
282  ! -- local
283  integer(I4B) :: n, m, idiag, idiagm, ipos, isympos, isymcon
284  real(DP) :: dnm
285 ! ------------------------------------------------------------------------------
286  !
287  if(this%ixt3d > 0) then
288  call this%xt3d%xt3d_fc(kiter, njasln, amatsln, idxglo, rhs, cnew)
289  else
290  do n = 1, nodes
291  if(this%fmi%ibdgwfsat0(n) == 0) cycle
292  idiag = this%dis%con%ia(n)
293  do ipos = this%dis%con%ia(n) + 1, this%dis%con%ia(n + 1) - 1
294  m = this%dis%con%ja(ipos)
295  if (m < n) cycle
296  if(this%fmi%ibdgwfsat0(m) == 0) cycle
297  isympos = this%dis%con%jas(ipos)
298  dnm = this%dispcoef(isympos)
299  !
300  ! -- Contribution to row n
301  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + dnm
302  amatsln(idxglo(idiag)) = amatsln(idxglo(idiag)) - dnm
303  !
304  ! -- Contribution to row m
305  idiagm = this%dis%con%ia(m)
306  isymcon = this%dis%con%isym(ipos)
307  amatsln(idxglo(isymcon)) = amatsln(idxglo(isymcon)) + dnm
308  amatsln(idxglo(idiagm)) = amatsln(idxglo(idiagm)) - dnm
309  enddo
310  enddo
311  endif
312  !
313  ! -- Return
314  return
315  end subroutine dsp_fc

690  subroutine calcdispellipse(this)
691 ! ******************************************************************************
692 ! calcdispellipse -- Calculate dispersion coefficients
693 ! ******************************************************************************
694 !
695 ! SPECIFICATIONS:
696 ! ------------------------------------------------------------------------------
697  ! -- modules
698  ! -- dummy
699  class(gwtdsptype) :: this
700  ! -- local
701  integer(I4B) :: nodes, n
702  real(DP) :: q, qx, qy, qz
703  real(DP) :: alh, alv, ath1, ath2, atv, a
704  real(DP) :: al, at1, at2
705  real(DP) :: qzoqsquared
706  real(DP) :: dstar
707 ! ------------------------------------------------------------------------------
708  !
709  ! -- loop through and calculate dispersion coefficients and angles
710  nodes = size(this%d11)
711  do n = 1, nodes
712  !
713  ! -- initialize
714  this%d11(n) = dzero
715  this%d22(n) = dzero
716  this%d33(n) = dzero
717  this%angle1(n) = dzero
718  this%angle2(n) = dzero
719  this%angle3(n) = dzero
720  if(this%fmi%ibdgwfsat0(n) == 0) cycle
721  !
722  ! -- specific discharge
723  qx = dzero
724  qy = dzero
725  qz = dzero
726  q = dzero
727  qx = this%fmi%gwfspdis(1, n)
728  qy = this%fmi%gwfspdis(2, n)
729  qz = this%fmi%gwfspdis(3, n)
730  q = qx ** 2 + qy ** 2 + qz ** 2
731  if (q > dzero) q = sqrt(q)
732  !
733  ! -- dispersion coefficients
734  alh = dzero
735  alv = dzero
736  ath1 = dzero
737  ath2 = dzero
738  atv = dzero
739  if (this%idisp > 0) then
740  alh = this%alh(n)
741  alv = this%alv(n)
742  ath1 = this%ath1(n)
743  ath2 = this%ath2(n)
744  atv = this%atv(n)
745  endif
746  dstar = dzero
747  if (this%idiffc > 0) then
748  dstar = this%diffc(n) * this%porosity(n)
749  endif
750  !
751  ! -- Calculate the longitudal and transverse dispersivities
752  al = dzero
753  at1 = dzero
754  at2 = dzero
755  if (q > dzero) then
756  qzoqsquared = (qz / q) ** 2
757  al = alh * (done - qzoqsquared) + alv * qzoqsquared
758  at1 = ath1 * (done - qzoqsquared) + atv * qzoqsquared
759  at2 = ath2 * (done - qzoqsquared) + atv * qzoqsquared
760  endif
761  !
762  ! -- Calculate and save the diagonal components of the dispersion tensor
763  this%d11(n) = al * q + dstar
764  this%d22(n) = at1 * q + dstar
765  this%d33(n) = at2 * q + dstar
766  !
767  ! -- Angles of rotation if velocity based dispersion tensor
768  if (this%idisp > 0) then
769  !
770  ! -- angles of rotation from model coordinates to direction of velocity
771  ! qx / q = cos(a1) * cos(a2)
772  ! qy / q = sin(a1) * cos(a2)
773  ! qz / q = sin(a2)
774  !
775  ! -- angle3 is zero
776  this%angle3(n) = dzero
777  !
778  ! -- angle2
779  a = dzero
780  if (q > dzero) a = qz / q
781  this%angle2(n) = asin(a)
782  !
783  ! -- angle1
784  a = q * cos(this%angle2(n))
785  if (a /= dzero) then
786  a = qx / a
787  else
788  a = dzero
789  endif
790  !
791  ! -- acos(1) not defined, so set to zero if necessary
792  if (a <= -done) then
793  this%angle1(n) = dpi
794  elseif (a >= done) then
795  this%angle1(n) = dzero
796  else
797  this%angle1(n) = acos(a)
798  endif
799  !
800  endif
801  enddo
802  !
803  ! -- Return
804  return
805  end subroutine calcdispellipse

807  subroutine calcdispcoef(this)
808 ! ******************************************************************************
809 ! calcdispcoef -- Calculate dispersion coefficients
810 ! ******************************************************************************
811 !
812 ! SPECIFICATIONS:
813 ! ------------------------------------------------------------------------------
814  ! -- modules
815  use gwfnpfmodule, only: hyeff_calc
816  ! -- dummy
817  class(gwtdsptype) :: this
818  ! -- local
819  integer(I4B) :: nodes, n, m, idiag, ipos
820  real(DP) :: clnm, clmn, dn, dm
821  real(DP) :: vg1, vg2, vg3
822  integer(I4B) :: ihc, isympos
823  integer(I4B) :: iavgmeth
824  real(DP) :: satn, satm, topn, topm, botn, botm
825  real(DP) :: hwva, cond, cn, cm, denom
826  real(DP) :: anm, amn, thksatn, thksatm, sill_top, sill_bot, tpn, tpm
827 ! ------------------------------------------------------------------------------
828  !
829  ! -- set iavgmeth = 1 to use arithmetic averaging for effective dispersion
830  iavgmeth = 1
831  !
832  ! -- Proces connections
833  nodes = size(this%d11)
834  do n = 1, nodes
835  if(this%fmi%ibdgwfsat0(n) == 0) cycle
836  idiag = this%dis%con%ia(n)
837  do ipos = this%dis%con%ia(n) + 1, this%dis%con%ia(n + 1) - 1
838  !
839  ! -- Set m to connected cell
840  m = this%dis%con%ja(ipos)
841  !
842  ! -- skip for lower triangle
843  if (m < n) cycle
844  isympos = this%dis%con%jas(ipos)
845  this%dispcoef(isympos) = dzero
846  if(this%fmi%ibdgwfsat0(m) == 0) cycle
847  !
848  ! -- cell dimensions
849  hwva = this%dis%con%hwva(isympos)
850  clnm = this%dis%con%cl1(isympos)
851  clmn = this%dis%con%cl2(isympos)
852  ihc = this%dis%con%ihc(isympos)
853  topn = this%dis%top(n)
854  topm = this%dis%top(m)
855  botn = this%dis%bot(n)
856  botm = this%dis%bot(m)
857  !
858  ! -- flow model information
859  satn = this%fmi%gwfsat(n)
860  satm = this%fmi%gwfsat(m)
861  !
862  ! -- Calculate dispersion coefficient for cell n in the direction
863  ! normal to the shared n-m face and for cell m in the direction
864  ! normal to the shared n-m face.
865  call this%dis%connection_normal(n, m, ihc, vg1, vg2, vg3, ipos)
866  dn = hyeff_calc(this%d11(n), this%d22(n), this%d33(n), &
867  this%angle1(n), this%angle2(n), this%angle3(n), &
868  vg1, vg2, vg3, iavgmeth)
869  dm = hyeff_calc(this%d11(m), this%d22(m), this%d33(m), &
870  this%angle1(m), this%angle2(m), this%angle3(m), &
871  vg1, vg2, vg3, iavgmeth)
872  !
873  ! -- Calculate dispersion conductance based on NPF subroutines and the
874  ! effective dispersion coefficients dn and dm.
875  if(ihc == 0) then
876  clnm = satn * (topn - botn) * dhalf
877  clmn = satm * (topm - botm) * dhalf
878  anm = hwva
879  !
880  ! -- n is convertible and unsaturated
881  if (satn == dzero) then
882  anm = dzero
883  else if (n > m .and. satn < done) then
884  anm = dzero
885  endif
886  !
887  ! -- m is convertible and unsaturated
888  if (satm == dzero) then
889  anm = dzero
890  else if (m > n .and. satm < done) then
891  anm = dzero
892  endif
893  !
894  ! -- amn is the same as anm for vertical flow
895  amn = anm
896  !
897  else
898  !
899  ! -- horizontal conductance
900  thksatn = (topn - botn) * satn
901  thksatm = (topm - botm) * satm
902  !
903  ! -- handle vertically staggered case
904  if (ihc == 2) then
905  sill_top = min(topn, topm)
906  sill_bot = max(botn, botm)
907  tpn = botn + thksatn
908  tpm = botm + thksatm
909  thksatn = max(min(tpn, sill_top) - sill_bot, dzero)
910  thksatm = max(min(tpm, sill_top) - sill_bot, dzero)
911  end if
912  !
913  ! -- calculate the saturated area term
914  anm = thksatn * hwva
915  amn = thksatm * hwva
916  !
917  ! -- n or m is unsaturated, so no dispersion
918  if (satn == dzero .or. satm == dzero) then
919  anm = dzero
920  amn = dzero
921  endif
922  !
923  end if
924  !
925  ! -- calculate conductance using the two half cell conductances
926  cn = dzero
927  if (clnm > dzero) cn = dn * anm / clnm
928  cm = dzero
929  if (clmn > dzero) cm = dm * amn / clmn
930  denom = cn + cm
931  if (denom > dzero) then
932  cond = cn * cm / denom
933  else
934  cond = dzero
935  end if
936  !
937  ! -- Assign the calculated dispersion conductance
938  this%dispcoef(isympos) = cond
939  !
940  enddo
941  enddo
942  !
943  ! -- Return
944  return
945  end subroutine calcdispcoef
