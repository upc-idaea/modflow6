443  subroutine solution_groups_create()
444 ! ******************************************************************************
445 ! Set the solution_groups to be used for the simulation
446 ! ******************************************************************************
447 !
448 ! SPECIFICATIONS:
449 ! ------------------------------------------------------------------------------
450  use solutiongroupmodule, only: solutiongrouptype, &
451  solutiongroup_create
452  use basesolutionmodule, only: basesolutiontype
453  use basemodelmodule, only: basemodeltype
454  use baseexchangemodule, only: baseexchangetype
455  use numericalsolutionmodule, only: solution_create
456  ! -- dummy
457  ! -- local
458  type(solutiongrouptype), pointer :: sgp
459  class(basesolutiontype), pointer :: sp
460  class(basemodeltype), pointer :: mp
461  integer(I4B) :: ierr
462  logical :: isfound, endOfBlock
463  integer(I4B) :: isoln
464  integer(I4B) :: isgp
465  integer(I4B) :: isgpsoln
466  integer(I4B) :: sgid
467  integer(I4B) :: mid
468  character(len=LINELENGTH) :: errmsg
469  character(len=LENBIGLINE) :: keyword
470  character(len=LINELENGTH) :: fname, mname
471  ! -- formats
472  character(len=*), parameter :: fmterrmxiter = &
473  "('ERROR. MXITER IS SET TO ', i0, ' BUT THERE IS ONLY ONE SOLUTION', &
474  &' IN SOLUTION GROUP ', i0, '. SET MXITER TO 1 IN SIMULATION CONTROL', &
475  &' FILE.')"
476 ! ------------------------------------------------------------------------------
477  !
478  ! -- isoln is the cumulative solution number, isgp is the cumulative
479  ! solution group number.
480  isoln = 0
481  isgp = 0
482  !
483  !Read through the simulation name file and process each SOLUTION_GROUP
484  sgploop: do
485  !
486  call parser%GetBlock('SOLUTIONGROUP', isfound, ierr, &
487  supportopenclose=.true.)
488  if(ierr /= 0) exit sgploop
489  if (.not. isfound) exit sgploop
490  isgp = isgp + 1
491  !
492  ! -- Get the solutiongroup id and check that it is listed consecutively.
493  sgid = parser%GetInteger()
494  if(isgp /= sgid) then
495  write(errmsg, '(a)') 'Solution groups are not listed consecutively.'
496  call store_error(errmsg)
497  write(errmsg, '(a,i0,a,i0)' ) 'Found ', sgid, ' when looking for ',isgp
498  call store_error(errmsg)
499  call parser%StoreErrorUnit()
500  call ustop()
501  endif
502  !
503  ! -- Create the solutiongroup and add it to the solutiongrouplist
504  call solutiongroup_create(sgp, sgid)
505  call addsolutiongrouptolist(solutiongrouplist, sgp)
506  !
507  ! -- Begin processing the solution group
508  write(iout,'(/1x,a)')'READING SOLUTIONGROUP'
509  !
510  ! -- Initialize isgpsoln to 0. isgpsoln is the solution counter for this
511  ! particular solution group. It goes from 1 to the number of solutions
512  ! in this group.
513  isgpsoln = 0
514  do
515  call parser%GetNextLine(endofblock)
516  if (endofblock) exit
517  call parser%GetStringCaps(keyword)
518  select case (keyword)
519  !
520  case ('MXITER')
521  sgp%mxiter = parser%GetInteger()
522  !
523  case ('IMS6')
524  !
525  ! -- Initialize and increment counters
526  isoln = isoln + 1
527  isgpsoln = isgpsoln + 1
528  !
529  ! -- Create the solution, retrieve from the list, and add to sgp
530  call parser%GetString(fname)
531  call solution_create(fname, isoln)
532  sp => getbasesolutionfromlist(basesolutionlist, isoln)
533  call sgp%add_solution(isoln, sp)
534  !
535  ! -- Add all of the models that are listed on this line to
536  ! the current solution (sp)
537  do
538  !
539  ! -- Set istart and istop to encompass model name. Exit this
540  ! loop if there are no more models.
541  call parser%GetStringCaps(mname)
542  if (mname == '') exit
543  !
544  ! -- Find the model id, and then get model
545  mid = ifind(modelname, mname)
546  if(mid <= 0) then
547  write(errmsg, '(a,a)') 'Error. Invalid modelname: ', &
548  trim(mname)
549  call store_error(errmsg)
550  call parser%StoreErrorUnit()
551  call ustop()
552  endif
553  mp => getbasemodelfromlist(basemodellist, mid)
554  !
555  ! -- Add the model to the solution
556  call sp%add_model(mp)
557  mp%idsoln = isoln
558  !
559  enddo
560  !
561  case default
562  write(errmsg, '(4x,a,a)') &
563  '****ERROR. UNKNOWN SOLUTIONGROUP ENTRY: ', &
564  trim(keyword)
565  call store_error(errmsg)
566  call parser%StoreErrorUnit()
567  call ustop()
568  end select
569  end do
570  !
571  ! -- Make sure there is a solution in this solution group
572  if(isgpsoln == 0) then
573  write(errmsg, '(4x,a,i0)') &
574  'ERROR. THERE ARE NO SOLUTIONS FOR SOLUTION GROUP ', isgp
575  call store_error(errmsg)
576  call parser%StoreErrorUnit()
577  call ustop()
578  endif
579  !
580  ! -- If there is only one solution then mxiter should be 1.
581  if(isgpsoln == 1 .and. sgp%mxiter > 1) then
582  write(errmsg, fmterrmxiter) sgp%mxiter, isgpsoln
583  call store_error(errmsg)
584  call parser%StoreErrorUnit()
585  call ustop()
586  endif
587  !
588  ! -- todo: more error checking?
589  !
590  write(iout,'(1x,a)')'END OF SIMULATION SOLUTIONGROUP'
591  !
592  enddo sgploop
593  !
594  ! -- Check and make sure at least one solution group was found
595  if(solutiongrouplist%Count() == 0) then
596  call store_error('ERROR. THERE ARE NO SOLUTION GROUPS.')
597  call parser%StoreErrorUnit()
598  call ustop()
599  endif
600  !
601  ! -- return
602  return
603  end subroutine solution_groups_create

628  !> @brief Assign exchanges to solutions
629  !!
630  !! This assigns NumericalExchanges to NumericalSolutions,
631  !! based on the link between the models in the solution and
632  !! those exchanges. The BaseExchange%connects_model() function
633  !! should be overridden to indicate if such a link exists.
634  !<
635  subroutine assign_exchanges()
636  ! -- local
637  class(basesolutiontype), pointer :: sp
638  class(baseexchangetype), pointer :: ep
639  class(basemodeltype), pointer :: mp
640  type(listtype), pointer :: models_in_solution
641  integer(I4B) :: is, ie, im
642 
643  do is = 1, basesolutionlist%Count()
644  sp => getbasesolutionfromlist(basesolutionlist, is)
645  !
646  ! -- now loop over exchanges
647  do ie = 1, baseexchangelist%Count()
648  ep => getbaseexchangefromlist(baseexchangelist, ie)
649  !

651  models_in_solution => sp%get_models()
652  do im = 1, models_in_solution%Count()
653  mp => getbasemodelfromlist(models_in_solution, im)
654  if (ep%connects_model(mp)) then
655  !
656  ! -- add to solution (and only once)
657  call sp%add_exchange(ep)

659  end if
660  end do
