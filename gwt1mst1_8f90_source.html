374  subroutine mst_srb_term(isrb, thetamfrac, rhob, vcell, tled, cnew, cold, &
375  swnew, swold, const1, const2, rate, hcofval, rhsval)
376 ! ******************************************************************************
377 ! mst_srb_term -- Calculate sorption terms
378 ! ******************************************************************************
379 !
380 ! SPECIFICATIONS:
381 ! ------------------------------------------------------------------------------
382  ! -- modules
383  ! -- dummy
384  integer(I4B), intent(in) :: isrb
385  real(DP), intent(in) :: thetamfrac
386  real(DP), intent(in) :: rhob
387  real(DP), intent(in) :: vcell
388  real(DP), intent(in) :: tled
389  real(DP), intent(in) :: cnew
390  real(DP), intent(in) :: cold
391  real(DP), intent(in) :: swnew
392  real(DP), intent(in) :: swold
393  real(DP), intent(in) :: const1
394  real(DP), intent(in) :: const2
395  real(DP), intent(out), optional :: rate
396  real(DP), intent(out), optional :: hcofval
397  real(DP), intent(out), optional :: rhsval
398  ! -- local
399  real(DP) :: term
400  real(DP) :: derv
401  real(DP) :: cbarnew
402  real(DP) :: cbarold
403  real(DP) :: cavg
404  real(DP) :: cbaravg
405  real(DP) :: swavg
406 ! ------------------------------------------------------------------------------
407  if (isrb == 1) then
408  ! -- linear
409  term = - thetamfrac * rhob * vcell * tled * const1
410  if (present(hcofval)) hcofval = term * swnew
411  if (present(rhsval)) rhsval = term * swold * cold
412  if (present(rate)) rate = term * swnew * cnew - term * swold * cold
413  else
414  !
415  ! -- calculate average aqueous concentration
416  cavg = dhalf * (cold + cnew)
417  !
418  ! -- set values based on isotherm
419  if (isrb == 2) then
420  ! -- freundlich
421  cbarnew = get_freundlich_conc(cnew, const1, const2)
422  cbarold = get_freundlich_conc(cold, const1, const2)
423  derv = get_freundlich_derivative(cavg, const1, const2)
424  else if (isrb == 3) then
425  ! -- langmuir
426  cbarnew = get_langmuir_conc(cnew, const1, const2)
427  cbarold = get_langmuir_conc(cold, const1, const2)
428  derv = get_langmuir_derivative(cavg, const1, const2)
429  end if
430  !
431  ! -- calculate hcof, rhs, and rate for freundlich and langmuir
432  term = - thetamfrac * rhob * vcell * tled
433  cbaravg = (cbarold + cbarnew) * dhalf
434  swavg = (swnew + swold) * dhalf
435  if (present(hcofval)) then
436  hcofval = term * derv * swavg
437  end if
438  if (present(rhsval)) then
439  rhsval = term * derv * swavg * cold - term * cbaravg * (swnew - swold)
440  end if
441  if (present(rate)) then
442  rate = term * derv * swavg * (cnew - cold) &
443  + term * cbaravg * (swnew - swold)
444  end if
445  end if
446  return
447  end subroutine mst_srb_term

529  subroutine mst_fc_dcy_srb(this, nodes, cold, nja, njasln, amatsln, idxglo, &

532 ! mst_fc_dcy_srb -- Calculate coefficients and fill amat and rhs
533 ! ******************************************************************************
534 !
535 ! SPECIFICATIONS:
536 ! ------------------------------------------------------------------------------
537  ! -- modules
538  ! -- dummy
539  class(gwtmsttype) :: this
540  integer, intent(in) :: nodes
541  real(DP), intent(in), dimension(nodes) :: cold
542  integer(I4B), intent(in) :: nja
543  integer(I4B), intent(in) :: njasln
544  real(DP), dimension(njasln), intent(inout) :: amatsln
545  integer(I4B), intent(in), dimension(nja) :: idxglo
546  real(DP), intent(inout), dimension(nodes) :: rhs
547  real(DP), intent(in), dimension(nodes) :: cnew
548  ! -- local
549  integer(I4B) :: n, idiag
550  real(DP) :: hhcof, rrhs
551  real(DP) :: vcell
552  real(DP) :: swnew
553  real(DP) :: distcoef
554  real(DP) :: thetamfrac
555  real(DP) :: term
556  real(DP) :: csrb
557 ! ------------------------------------------------------------------------------
558  !
559  ! -- loop through and calculate sorption contribution to hcof and rhs
560  do n = 1, this%dis%nodes
561  !
562  ! -- skip if transport inactive
563  if(this%ibound(n) <= 0) cycle
564  !
565  ! -- set variables
566  hhcof = dzero
567  rrhs = dzero
568  vcell = this%dis%area(n) * (this%dis%top(n) - this%dis%bot(n))

570  distcoef = this%distcoef(n)
571  idiag = this%dis%con%ia(n)
572  thetamfrac = this%get_thetamfrac(n)
573  term = this%decay_sorbed(n) * thetamfrac * this%bulk_density(n) * &
574  swnew * vcell
575  !
576  ! -- add sorbed mass decay rate terms to accumulators
577  if (this%idcy == 1) then
578  !
579  if (this%isrb == 1) then
580  !
581  ! -- first order decay rate is a function of concentration, so add
582  ! to left hand side
583  hhcof = - term * distcoef
584  else if (this%isrb == 2) then
585  !
586  ! -- nonlinear Freundlich sorption, so add to RHS
587  csrb = get_freundlich_conc(cnew(n), distcoef, this%sp2(n))
588  rrhs = term * csrb
589  else if (this%isrb == 3) then
590  !
591  ! -- nonlinear Lanmuir sorption, so add to RHS
592  csrb = get_freundlich_conc(cnew(n), distcoef, this%sp2(n))
593  rrhs = term * csrb
594  end if
595  elseif (this%idcy == 2) then
596  !
597  ! -- zero-order decay rate is not a function of concentration, so add
598  ! to right hand side
599  if (distcoef > dzero) then
600  ! -- Add zero order sorption term only if distribution coefficient > 0
601  rrhs = term
602  end if
603  endif
604  !
605  ! -- Add hhcof to diagonal and rrhs to right-hand side
606  amatsln(idxglo(idiag)) = amatsln(idxglo(idiag)) + hhcof
607  rhs(n) = rhs(n) + rrhs
608  !
609  enddo
610  !
611  ! -- Return

655 ! ******************************************************************************
656 ! mst_cq_sto -- Calculate budget terms
657 ! ******************************************************************************
658 !
659 ! SPECIFICATIONS:
660 ! ------------------------------------------------------------------------------
661  ! -- modules
662  use tdismodule, only: delt
663  ! -- dummy
664  class(gwtmsttype) :: this
665  integer(I4B), intent(in) :: nodes

670  integer(I4B) :: n
671  integer(I4B) :: idiag

681  ! -- Calculate storage change

826  subroutine mst_cq_dcy_srb(this, nodes, cnew, cold, flowja)
827 ! ******************************************************************************
828 ! mst_cq_dcy_srb -- Calculate budget terms
829 ! ******************************************************************************
830 !
831 ! SPECIFICATIONS:
832 ! ------------------------------------------------------------------------------
833  ! -- modules
834  ! -- dummy
835  class(gwtmsttype) :: this
836  integer(I4B), intent(in) :: nodes
837  real(DP), intent(in), dimension(nodes) :: cnew
838  real(DP), intent(in), dimension(nodes) :: cold
839  real(DP), dimension(:), contiguous, intent(inout) :: flowja
840  ! -- local
841  integer(I4B) :: n
842  integer(I4B) :: idiag
843  real(DP) :: rate
844  real(DP) :: hhcof, rrhs
845  real(DP) :: vcell
846  real(DP) :: swnew
847  real(DP) :: distcoef
848  real(DP) :: thetamfrac
849  real(DP) :: term
850  real(DP) :: csrb
851 ! ------------------------------------------------------------------------------
852  !
853  ! -- Calculate sorbed decay change
854  ! This routine will only be called if sorption and decay are active
855  do n = 1, nodes
856  !
857  ! -- initialize rates
858  this%ratedcys(n) = dzero
859  !
860  ! -- skip if transport inactive
861  if(this%ibound(n) <= 0) cycle
862  !
863  ! -- set variables
864  hhcof = dzero
865  rrhs = dzero
866  vcell = this%dis%area(n) * (this%dis%top(n) - this%dis%bot(n))
867  swnew = this%fmi%gwfsat(n)
868  distcoef = this%distcoef(n)
869  thetamfrac = this%get_thetamfrac(n)
870  term = this%decay_sorbed(n) * thetamfrac * this%bulk_density(n) * &
871  swnew * vcell
872  !
873  ! -- add sorbed mass decay rate terms to accumulators
874  if (this%idcy == 1) then
875  !
876  if (this%isrb == 1) then
877  !
878  ! -- first order decay rate is a function of concentration, so add
879  ! to left hand side
880  hhcof = - term * distcoef
881  else if (this%isrb == 2) then
882  !
883  ! -- nonlinear Freundlich sorption, so add to RHS
884  csrb = get_freundlich_conc(cnew(n), distcoef, this%sp2(n))
885  rrhs = term * csrb
886  else if (this%isrb == 3) then
887  !
888  ! -- nonlinear Lanmuir sorption, so add to RHS
889  csrb = get_freundlich_conc(cnew(n), distcoef, this%sp2(n))
890  rrhs = term * csrb
891  end if
892  elseif (this%idcy == 2) then
893  !
894  ! -- zero-order decay rate is not a function of concentration, so add
895  ! to right hand side
896  if (distcoef > dzero) then
897  ! -- Add zero order sorption term only if distribution coefficient > 0
898  rrhs = term
899  end if
900  endif
901  !
902  ! -- calculate rate
903  rate = hhcof * cnew(n) - rrhs
904  this%ratedcys(n) = rate
905  idiag = this%dis%con%ia(n)
906  flowja(idiag) = flowja(idiag) + rate
907  !
908  enddo
909  !
910  ! -- Return
911  return
912  end subroutine mst_cq_dcy_srb
