309  &stress period ', i0, ' = ', g12.5)"
310  !
311  if(this%inunit == 0) return
312  !
313  ! -- Get stress period data
314  if(this%ionper < kper) then
315  !
316  ! -- Get PERIOD block
317  call this%parser%GetBlock('PERIOD', isfound, ierr, &
318  supportopenclose=.true.)
319  if(isfound) then
320  !
321  ! -- Read ionper and check for increasing period numbers
322  call this%read_check_ionper()

326  if(ierr < 0) then
327  ! -- End of file found; data applies for remainder of simulation.
328  this%ionper = nper + 1
329  else

332  write(errmsg, fmtblkerr) adjustl(trim(line))
333  call store_error(errmsg)
334  end if
335  end if
336  end if

338  ! -- Read data if ionper == kper
339  if(this%ionper == kper) then
340  !

342  call this%reset_change_flags()
343  !

345  do

347  if(endofblock) then
348  exit
349  end if

411  !> @brief Advance the package
412  !!
413  !! Advance data for a new time step.
414  !!
415  !<
416  subroutine ad(this)
417  ! -- dummy variables
418  class(tvbasetype) :: this
419  ! -- local variables
420  integer(I4B) :: i, n, numlinks
421  type(timeserieslinktype), pointer :: tsLink
422  !
423  ! -- Advance the time series manager;
424  ! -- this will apply any time series changes to property values
425  call this%tsmanager%ad()
426  !
427  ! -- If there are no time series property changes,
428  ! -- there is nothing else to be done
429  numlinks = this%tsmanager%CountLinks('BND')
430  if(numlinks <= 0) then
431  return
432  end if
433  !
434  ! -- Record that changes were made at the current time step
435  ! -- (as we have at least one active time series link)
436  call this%set_changed_at(kper, kstp)
437  !
438  ! -- Reset node K change flags at all time steps except the first of each
439  ! -- period (the first is done in rp(), to allow non-time series changes)
440  if(kstp /= 1) then
441  call this%reset_change_flags()
442  end if
443  !
444  ! -- Validate all new property values
445  do i = 1, numlinks
446  tslink => this%tsmanager%GetLink('BND', i)
447  n = tslink%iRow
448  call this%validate_change(n, tslink%Text)
449  end do
450  !
451  ! -- Terminate if there were errors
452  if(count_errors() > 0) then
453  call this%parser%StoreErrorUnit()
454  call ustop()
455  end if
456  !
457  return
458  end subroutine ad
