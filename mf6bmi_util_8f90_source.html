44  !! is inside to avoid lots of duplication

49  ! -- dummy variables

52  character(len=LENVARNAME), intent(out) :: var_name !< name of the variable
53  logical(LGP), intent(out) :: success !< false when invalid
54  ! -- local variables
55  character(len=LENMEMPATH) :: var_address
56  logical(LGP) :: valid, found
57 
58  success = .false.
59 
60  ! try and split the address string:
61  var_address = char_array_to_string(c_var_address, strlen(c_var_address))
62  call split_mem_address(var_address, mem_path, var_name, valid)
63  if (.not. valid) then
64  write (bmi_last_error, fmt_invalid_var) trim(var_address)
65  call report_bmi_error(bmi_last_error)
66  return
67  end if
68 
69  ! check if the variable even exists:
70  call check_mem_address(mem_path, var_name, found)
71  if (.not. found) then
72  write (bmi_last_error, fmt_unknown_var) trim(var_name), trim(mem_path)
73  call report_bmi_error(bmi_last_error)
74  return
75  end if

77  success = .true.
