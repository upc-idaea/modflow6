220  subroutine constructobservation(newObservation, defLine, numunit, &
221  formatted, indx, obsData, inunit)
222 ! **************************************************************************
223 ! ConstructObservation -- Construct and return an ObserveType object based
224 ! on the contents of defLine.
225 ! **************************************************************************
226 !
227 ! SPECIFICATIONS:
228 ! --------------------------------------------------------------------------
229  ! -- dummy variables
230  type(observetype), pointer :: newobservation
231  character(len=*), intent(in) :: defline
232  integer(I4B), intent(in) :: numunit ! Output unit number
233  logical, intent(in) :: formatted ! Formatted output?
234  integer(I4B), intent(in) :: indx ! Index in ObsOutput array
235  type(obsdatatype), dimension(:), pointer, intent(in) :: obsdata
236  integer(I4B), intent(in) :: inunit
237  ! -- local
238  real(dp) :: r
239  integer(I4B) :: i, icol, iout, istart, istop, ltrim, n
240  ! --------------------------------------------------------------------------
241  !
242  ! -- initialize
243  iout = 0
244  icol = 1
245  !
246  ! -- Allocate an ObserveType object.
247  allocate(newobservation)
248  allocate(newobservation%indxbnds(0))
249  !
250  ! -- Set indxbnds_count to 0
251  newobservation%indxbnds_count = 0
252  !
253  ! -- Define the contents of the ObservationSingleType object based on the
254  ! contents of defLine.
255  !
256  ! -- Get observation name and store it
257  call urword(defline,icol,istart,istop,1,n,r,iout,inunit)
258  newobservation%Name = defline(istart:istop)
259  !
260  ! -- Get observation type, convert it to uppercase, and store it.
261  call urword(defline,icol,istart,istop,1,n,r,iout,inunit)
262  newobservation%ObsTypeId = defline(istart:istop)
263  !
264  ! -- Look up package ID for this observation type and store it
265  do i=1,maxobstypes
266  if (obsdata(i)%ObsTypeID == newobservation%ObsTypeId) then
267  newobservation%obsDatum => obsdata(i)
268  exit
269  elseif (obsdata(i)%ObsTypeID == '') then
270  exit
271  endif
272  enddo
273  !
274  ! -- Remaining text is ID [and ID2]; store the remainder of the string
275  istart = istop + 1
276  istop = len_trim(defline)
277  if (istart > istop) then
278  istart = istop
279  end if
280  newobservation%IDstring = defline(istart:istop)
281  !
282  ! Store UnitNumber, FormattedOutput, and IndxObsOutput
283  newobservation%UnitNumber = numunit
284  newobservation%FormattedOutput = formatted
285  newobservation%IndxObsOutput = indx
286  !
287  return
288  end subroutine constructobservation
