114  subroutine riv_ck(this)
115 ! ******************************************************************************
116 ! riv_ck -- Check river boundary condition data
117 ! ******************************************************************************
118 !
119 ! SPECIFICATIONS:
120 ! ------------------------------------------------------------------------------
121  ! -- modules
122  use constantsmodule, only: linelength
123  use simmodule, only: store_error, count_errors, store_error_unit
124  ! -- dummy
125  class(rivtype),intent(inout) :: this
126  ! -- local
127  character(len=LINELENGTH) :: errmsg
128  integer(I4B) :: i
129  integer(I4B) :: node
130  real(DP) :: bt
131  real(DP) :: stage
132  real(DP) :: rbot
133  ! -- formats
134  character(len=*), parameter :: fmtriverr = &
135  "('RIV BOUNDARY (',i0,') RIVER BOTTOM (',f10.4,') IS LESS " // &
136  "THAN CELL BOTTOM (',f10.4,')')"
137  character(len=*), parameter :: fmtriverr2 = &
138  "('RIV BOUNDARY (',i0,') RIVER STAGE (',f10.4,') IS LESS " // &
139  "THAN RIVER BOTTOM (',f10.4,')')"
140  character(len=*), parameter :: fmtriverr3 = &
141  "('RIV BOUNDARY (',i0,') RIVER STAGE (',f10.4,') IS LESS " // &
142  "THAN CELL BOTTOM (',f10.4,')')"
143 ! ------------------------------------------------------------------------------
144  !
145  ! -- check stress period data
146  do i = 1, this%nbound
147  node = this%nodelist(i)
148  bt = this%dis%bot(node)
149  stage = this%bound(1,i)
150  rbot = this%bound(3,i)
151  ! -- accumulate errors
152  if (rbot < bt .and. this%icelltype(node) /= 0) then
153  write(errmsg, fmt=fmtriverr) i, rbot, bt
154  call store_error(errmsg)
155  end if
156  if (stage < rbot) then
157  write(errmsg, fmt=fmtriverr2) i, stage, rbot
158  call store_error(errmsg)
159  end if
160  if (stage < bt .and. this%icelltype(node) /= 0) then
161  write(errmsg, fmt=fmtriverr3) i, stage, bt
162  call store_error(errmsg)
163  end if
164  end do
165  !
166  ! -- write summary of river package error messages
167  if (count_errors() > 0) then
168  call store_error_unit(this%inunit)
169  end if
170  !
171  ! -- return
172  return
173  end subroutine riv_ck

175 subroutine riv_cf(this, reset_mover)
176 ! ******************************************************************************
177 ! riv_cf -- Formulate the HCOF and RHS terms
178 ! Subroutine: (1) skip in no rivs
179 ! (2) calculate hcof and rhs
180 ! ******************************************************************************
181 !
182 ! SPECIFICATIONS:
183 ! ------------------------------------------------------------------------------
184  ! -- dummy
185  class(rivtype) :: this
186  logical, intent(in), optional :: reset_mover
187  ! -- local
188  integer(I4B) :: i, node
189  real(DP) :: hriv, criv, rbot
190  logical :: lrm
191 ! ------------------------------------------------------------------------------
192  !
193  ! -- Return if no rivs
194  if(this%nbound.eq.0) return
195  !
196  ! -- pakmvrobj cf
197  lrm = .true.
198  if (present(reset_mover)) lrm = reset_mover
199  if(this%imover == 1 .and. lrm) then
200  call this%pakmvrobj%cf()
201  endif
202  !
203  ! -- Calculate hcof and rhs for each riv entry
204  do i=1,this%nbound
205  node=this%nodelist(i)
206  if(this%ibound(node)<=0) then
207  this%hcof(i)=dzero
208  this%rhs(i)=dzero
209  cycle
210  endif
211  hriv=this%bound(1,i)
212  criv=this%bound(2,i)
213  rbot=this%bound(3,i)
214  if(this%xnew(node)<=rbot) then
215  this%rhs(i)=-criv*(hriv-rbot)
216  this%hcof(i) = dzero
217  else
218  this%rhs(i) = -criv*hriv
219  this%hcof(i) = -criv
220  endif
221  enddo
222  !
223  ! -- return
224  return
225 end subroutine riv_cf

227  subroutine riv_fc(this, rhs, ia, idxglo, amatsln)
228 ! **************************************************************************
229 ! riv_fc -- Copy rhs and hcof into solution rhs and amat
230 ! **************************************************************************
231 !
232 ! SPECIFICATIONS:
233 ! --------------------------------------------------------------------------
234  ! -- dummy
235  class(rivtype) :: this
236  real(DP), dimension(:), intent(inout) :: rhs
237  integer(I4B), dimension(:), intent(in) :: ia
238  integer(I4B), dimension(:), intent(in) :: idxglo
239  real(DP), dimension(:), intent(inout) :: amatsln
240  ! -- local
241  integer(I4B) :: i, n, ipos
242  real(DP) :: cond, stage, qriv !, rbot
243 ! --------------------------------------------------------------------------
244  !
245  ! -- pakmvrobj fc
246  if(this%imover == 1) then
247  call this%pakmvrobj%fc()
248  endif
249  !
250  ! -- Copy package rhs and hcof into solution rhs and amat
251  do i = 1, this%nbound
252  n = this%nodelist(i)
253  rhs(n) = rhs(n) + this%rhs(i)
254  ipos = ia(n)
255  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + this%hcof(i)
256  !
257  ! -- If mover is active and this river cell is discharging,
258  ! store available water (as positive value).
259  stage = this%bound(1,i)
260  if(this%imover == 1 .and. this%xnew(n) > stage) then
261  cond = this%bound(2,i)
262  qriv = cond * (this%xnew(n) - stage)
263  call this%pakmvrobj%accumulate_qformvr(i, qriv)
264  endif
265  enddo
266  !
267  ! -- return
268  return
269  end subroutine riv_fc
