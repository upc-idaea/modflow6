214  subroutine ist_fc(this, rhs, ia, idxglo, amatsln)
215 ! ******************************************************************************
216 ! ist_fc -- Calculate coefficients and fill amat and rhs
217 ! ******************************************************************************
218 !
219 ! SPECIFICATIONS:
220 ! ------------------------------------------------------------------------------
221  ! -- modules
222  use tdismodule, only: delt
223  ! -- dummy
224  class(gwtisttype) :: this
225  real(DP), dimension(:), intent(inout) :: rhs
226  integer(I4B), dimension(:), intent(in) :: ia
227  integer(I4B), dimension(:), intent(in) :: idxglo
228  real(DP), dimension(:), intent(inout) :: amatsln
229  ! -- local
230  integer(I4B) :: n, idiag
231  real(DP) :: tled
232  real(DP) :: hhcof, rrhs
233  real(DP) :: swt, swtpdt
234  real(DP) :: vcell
235  real(DP) :: thetamfrac
236  real(DP) :: thetaimfrac
237  real(DP) :: kd
238  real(DP) :: rhob
239  real(DP) :: lambda1im
240  real(DP) :: lambda2im
241  real(DP) :: gamma1im
242  real(DP) :: gamma2im
243 ! ------------------------------------------------------------------------------
244  !
245  ! -- set variables
246  tled = done / delt
247  !
248  ! -- loop through and calculate immobile domain contribution to hcof and rhs
249  do n = 1, this%dis%nodes
250  !
251  ! -- skip if transport inactive
252  if(this%ibound(n) <= 0) cycle
253  !
254  ! -- calculate new and old water volumes
255  vcell = this%dis%area(n) * (this%dis%top(n) - this%dis%bot(n))
256  swtpdt = this%fmi%gwfsat(n)
257  swt = this%fmi%gwfsatold(n, delt)
258  idiag = ia(n)
259  !
260  ! -- Set thetamfrac and thetaimfrac
261  thetamfrac = this%mst%get_thetamfrac(n)
262  thetaimfrac = this%mst%get_thetaimfrac(n, this%thetaim(n))
263  !
264  ! -- Add dual domain mass transfer contributions to rhs and hcof
265  kd = dzero
266  rhob = dzero
267  lambda1im = dzero
268  lambda2im = dzero
269  gamma1im = dzero
270  gamma2im = dzero
271  if (this%idcy == 1) lambda1im = this%decay(n)
272  if (this%idcy == 2) gamma1im = this%decay(n)
273  if (this%isrb > 0) then
274  kd = this%distcoef(n)
275  rhob = this%bulk_density(n)
276  if (this%idcy == 1) lambda2im = this%decay_sorbed(n)
277  if (this%idcy == 2) gamma2im = this%decay_sorbed(n)
278  end if
279  call calcddhcofrhs(this%thetaim(n), vcell, delt, swtpdt, swt, &
280  thetamfrac, thetaimfrac, rhob, kd, &
281  lambda1im, lambda2im, gamma1im, gamma2im, &
282  this%zetaim(n), this%cim(n), hhcof, rrhs)
283  amatsln(idxglo(idiag)) = amatsln(idxglo(idiag)) + hhcof
284  rhs(n) = rhs(n) + rrhs
285  !
286  enddo
287  !
288  ! -- Return
289  return
290  end subroutine ist_fc

292  subroutine ist_cq(this, x, flowja, iadv)
293 ! ******************************************************************************
294 ! ist_cq -- Calculate flows
295 ! ******************************************************************************
296 !
297 ! SPECIFICATIONS:
298 ! ------------------------------------------------------------------------------
299  ! -- modules
300  use tdismodule, only: delt
301  use constantsmodule, only: dzero
302  ! -- dummy
303  class(gwtisttype), intent(inout) :: this
304  real(DP), dimension(:), intent(in) :: x
305  real(DP), dimension(:), contiguous, intent(inout) :: flowja
306  integer(I4B), optional, intent(in) :: iadv
307  ! -- local
308  integer(I4B) :: idiag
309  integer(I4B) :: n
310  real(DP) :: rate
311  real(DP) :: swt, swtpdt
312  real(DP) :: hhcof, rrhs
313  real(DP) :: vcell
314  real(DP) :: thetamfrac
315  real(DP) :: thetaimfrac
316  real(DP) :: kd
317  real(DP) :: rhob
318  real(DP) :: lambda1im
319  real(DP) :: lambda2im
320  real(DP) :: gamma1im
321  real(DP) :: gamma2im
322  ! -- formats
323 ! ------------------------------------------------------------------------------
324  !
325  ! -- Calculate immobile domain rhs and hcof
326  do n = 1, this%dis%nodes
327  !
328  ! -- skip if transport inactive
329  rate = dzero
330  if(this%ibound(n) > 0) then
331  !
332  ! -- calculate new and old water volumes
333  vcell = this%dis%area(n) * (this%dis%top(n) - this%dis%bot(n))
334  swtpdt = this%fmi%gwfsat(n)
335  swt = this%fmi%gwfsatold(n, delt)
336  !
337  ! -- Set thetamfrac and thetaimfrac
338  thetamfrac = this%mst%get_thetamfrac(n)
339  thetaimfrac = this%mst%get_thetaimfrac(n, this%thetaim(n))
340  !
341  ! -- Calculate exchange with immobile domain
342  rate = dzero
343  hhcof = dzero
344  rrhs = dzero
345  kd = dzero
346  rhob = dzero
347  lambda1im = dzero
348  lambda2im = dzero
349  gamma1im = dzero
350  gamma2im = dzero
351  if (this%idcy == 1) lambda1im = this%decay(n)
352  if (this%idcy == 2) gamma1im = this%decay(n)
353  if (this%isrb > 0) then
354  kd = this%distcoef(n)
355  rhob = this%bulk_density(n)
356  if (this%idcy == 1) lambda2im = this%decay_sorbed(n)
357  if (this%idcy == 2) gamma2im = this%decay_sorbed(n)
358  end if
359  call calcddhcofrhs(this%thetaim(n), vcell, delt, swtpdt, swt, &
360  thetamfrac, thetaimfrac, rhob, kd, &
361  lambda1im, lambda2im, gamma1im, gamma2im, &
362  this%zetaim(n), this%cim(n), hhcof, rrhs)
363  rate = hhcof * x(n) - rrhs
364  end if
365 
366  this%strg(n) = rate
367  idiag = this%dis%con%ia(n)
368  flowja(idiag) = flowja(idiag) + rate
369 
370  !
371  enddo
372  !
373  ! -- update cim
374  call this%calccim(this%cim, x)
375  !
376  ! -- Calculate and store the rates for the immobile domain
377  this%budterm(:, :) = dzero
378  call this%calcddbud(this%budterm, x)
379  !
380  ! -- return
381  return
382  end subroutine ist_cq

923  subroutine calcddhcofrhs(thetaim, vcell, delt, swtpdt, swt, thetamfrac, &
924  thetaimfrac, rhob, kd, lambda1im, lambda2im, &
925  gamma1im, gamma2im, zetaim, cimt, hcof, rhs)
926 ! ******************************************************************************
927 ! calcddhcofrhs -- calculate the hcof and rhs contributions for dual domain
928 ! mass transfer
929 ! ******************************************************************************
930 !
931 ! SPECIFICATIONS:
932 ! ------------------------------------------------------------------------------
933  ! -- dummy
934  real(DP), intent(in) :: thetaim
935  real(DP), intent(in) :: vcell
936  real(DP), intent(in) :: delt
937  real(DP), intent(in) :: swtpdt
938  real(DP), intent(in) :: swt
939  real(DP), intent(in) :: thetamfrac
940  real(DP), intent(in) :: thetaimfrac
941  real(DP), intent(in) :: rhob
942  real(DP), intent(in) :: kd
943  real(DP), intent(in) :: lambda1im
944  real(DP), intent(in) :: lambda2im
945  real(DP), intent(in) :: gamma1im
946  real(DP), intent(in) :: gamma2im
947  real(DP), intent(in) :: zetaim
948  real(DP), intent(in) :: cimt
949  real(DP), intent(inout) :: hcof
950  real(DP), intent(inout) :: rhs
951  ! -- local
952  real(DP), dimension(9) :: ddterm
953  real(DP) :: f
954 ! ------------------------------------------------------------------------------
955  !
956  ! -- calculate the ddterms
957  call calcddterms(thetaim, vcell, delt, swtpdt, swt, thetamfrac, &
958  thetaimfrac, rhob, kd, lambda1im, lambda2im, &
959  gamma1im, gamma2im, zetaim, cimt, ddterm, f)
960  !
961  ! -- calculate hcof
962  hcof = ddterm(9) ** 2 / f - ddterm(9)
963  !
964  ! -- calculate rhs, and switch the sign because this term needs to
965  ! be moved to the left hand side
966  rhs = (ddterm(2) + ddterm(4)) * cimt - ddterm(7) - ddterm(8)
967  rhs = rhs * ddterm(9) / f
968  rhs = -rhs
969  !
970  ! -- Return
971  return
972  end subroutine calcddhcofrhs

974  subroutine calcddbud(this, budterm, cnew)
975 ! ******************************************************************************
976 ! calcddbud -- calculate the individual budget terms for the immobile domain
977 ! ******************************************************************************
978 !
979 ! SPECIFICATIONS:
980 ! ------------------------------------------------------------------------------
981  ! -- modules
982  use tdismodule, only: delt
983  ! -- dummy
984  class(gwtisttype) :: this
985  real(DP), dimension(:, :), intent(inout) :: budterm
986  real(DP), dimension(:), intent(in) :: cnew
987  ! -- local
988  integer(I4B) :: n, i
989  real(DP) :: vcell
990  real(DP) :: swt
991  real(DP) :: swtpdt
992  real(DP) :: thetamfrac
993  real(DP) :: thetaimfrac
994  real(DP) :: kd
995  real(DP) :: lambda1im
996  real(DP) :: lambda2im
997  real(DP) :: gamma1im
998  real(DP) :: gamma2im
999  real(DP) :: ddterm(9)
1000  real(DP) :: f
1001  real(DP) :: cimt
1002  real(DP) :: cimtpdt
1003  real(DP) :: rate
1004  real(DP) :: rhob
1005 ! ------------------------------------------------------------------------------
1006  !
1007  ! -- Calculate cim
1008  do n = 1, this%dis%nodes
1009  if (this%ibound(n) <= 0) cycle
1010  vcell = this%dis%area(n) * (this%dis%top(n) - this%dis%bot(n))
1011  swt = this%fmi%gwfsatold(n, delt)
1012  swtpdt = this%fmi%gwfsat(n)
1013  thetamfrac = this%mst%get_thetamfrac(n)
1014  thetaimfrac = this%mst%get_thetaimfrac(n, this%thetaim(n))
1015  kd = dzero
1016  rhob = dzero
1017  lambda1im = dzero
1018  lambda2im = dzero
1019  gamma1im = dzero
1020  gamma2im = dzero
1021  if (this%idcy == 1) lambda1im = this%decay(n)
1022  if (this%idcy == 2) gamma1im = this%decay(n)
1023  if (this%isrb > 0) then
1024  kd = this%distcoef(n)
1025  rhob = this%bulk_density(n)
1026  if (this%idcy == 1) lambda2im = this%decay_sorbed(n)
1027  if (this%idcy == 2) gamma2im = this%decay_sorbed(n)
1028  end if
1029  !
1030  ! -- calculate the ddterms
1031  cimt = this%cim(n)
1032  call calcddterms(this%thetaim(n), vcell, delt, swtpdt, swt, thetamfrac, &
1033  thetaimfrac, rhob, kd, lambda1im, &
1034  lambda2im, gamma1im, gamma2im, this%zetaim(n), cimt, &
1035  ddterm, f)
1036  cimtpdt = calcddconc(this%thetaim(n), vcell, delt, swtpdt, swt, &
1037  thetamfrac, thetaimfrac, rhob, kd, &
1038  lambda1im, lambda2im, gamma1im, gamma2im, &
1039  this%zetaim(n), this%cim(n), cnew(n))
1040  !
1041  ! -- calculate STORAGE-AQUEOUS
1042  i = 1
1043  rate = - ddterm(1) * cimtpdt + ddterm(2) * cimt
1044  if (rate > dzero) then
1045  budterm(1, i) = budterm(1, i) + rate
1046  else
1047  budterm(2, i) = budterm(2, i) - rate
1048  endif
1049  !
1050  ! -- calculate STORAGE-SORBED
1051  i = 2
1052  rate = - ddterm(3) * cimtpdt + ddterm(4) * cimt
1053  if (rate > dzero) then
1054  budterm(1, i) = budterm(1, i) + rate
1055  else
1056  budterm(2, i) = budterm(2, i) - rate
1057  endif
1058  !
1059  ! -- calculate DECAY-AQUEOUS
1060  i = 3
1061  rate = dzero
1062  if (this%idcy == 1) then
1063  rate = - ddterm(5) * cimtpdt
1064  else if (this%idcy == 2) then
1065  rate = - ddterm(7)
1066  else
1067  rate = dzero
1068  endif
1069  if (rate > dzero) then
1070  budterm(1, i) = budterm(1, i) + rate
1071  else
1072  budterm(2, i) = budterm(2, i) - rate
1073  endif
1074  !
1075  ! -- calculate DECAY-SORBED
1076  i = 4
1077  if (this%idcy == 1) then
1078  rate = - ddterm(6) * cimtpdt
1079  else if (this%idcy == 2) then
1080  rate = - ddterm(8)
1081  else
1082  rate = dzero
1083  endif
1084  if (rate > dzero) then
1085  budterm(1, i) = budterm(1, i) + rate
1086  else
1087  budterm(2, i) = budterm(2, i) - rate
1088  endif
1089  !
1090  ! -- calculate MOBILE-DOMAIN
1091  i = 5
1092  rate = ddterm(9) * cnew(n) - ddterm(9) * cimtpdt
1093  if (rate > dzero) then
1094  budterm(1, i) = budterm(1, i) + rate
1095  else
1096  budterm(2, i) = budterm(2, i) - rate
1097  endif
1098  !
1099  enddo
1100  !
1101  ! -- Return
1102  return
1103  end subroutine calcddbud
