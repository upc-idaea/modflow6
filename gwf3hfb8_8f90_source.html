201  subroutine hfb_fc(this, kiter, njasln, amat, idxglo, rhs, hnew)
202 ! ******************************************************************************
203 ! hfb_fc -- Fill amatsln for the following conditions:
204 ! 1. Not Newton, and
205 ! 2. Cell type n is convertible or cell type m is convertible
206 ! OR
207 ! 3. XT3D
208 ! ******************************************************************************
209 !
210 ! SPECIFICATIONS:
211 ! ------------------------------------------------------------------------------
212  ! -- modules
213  use constantsmodule, only: dhalf, dzero
214  ! -- dummy
215  class(gwfhfbtype) :: this
216  integer(I4B) :: kiter
217  integer(I4B),intent(in) :: njasln
218  real(DP),dimension(njasln),intent(inout) :: amat
219  integer(I4B),intent(in),dimension(:) :: idxglo
220  real(DP),intent(inout),dimension(:) :: rhs
221  real(DP),intent(inout),dimension(:) :: hnew
222  ! -- local
223  integer(I4B) :: nodes, nja
224  integer(I4B) :: ihfb, n, m
225  integer(I4B) :: ipos
226  integer(I4B) :: idiag, isymcon
227  integer(I4B) :: ixt3d
228  real(DP) :: cond, condhfb, aterm
229  real(DP) :: fawidth, faheight
230  real(DP) :: topn, topm, botn, botm
231 ! ------------------------------------------------------------------------------
232  !
233  nodes = this%dis%nodes
234  nja = this%dis%con%nja
235  if (associated(this%xt3d%ixt3d)) then
236  ixt3d = this%xt3d%ixt3d
237  else
238  ixt3d = 0
239  end if
240  !
241  if(ixt3d > 0) then
242  !
243  do ihfb = 1, this%nhfb
244  n = min(this%noden(ihfb), this%nodem(ihfb))
245  m = max(this%noden(ihfb), this%nodem(ihfb))
246  ! -- Skip if either cell is inactive.
247  if(this%ibound(n) == 0 .or. this%ibound(m) == 0) cycle
248  !!! if(this%icelltype(n) == 1 .or. this%icelltype(m) == 1) then
249  ! -- Compute scale factor for hfb correction
250  if(this%hydchr(ihfb) > dzero) then
251  if(this%inewton == 0) then
252  ipos = this%idxloc(ihfb)
253  topn = this%top(n)
254  topm = this%top(m)
255  botn = this%bot(n)
256  botm = this%bot(m)
257  if(this%icelltype(n) == 1) then
258  if(hnew(n) < topn) topn = hnew(n)
259  endif
260  if(this%icelltype(m) == 1) then
261  if(hnew(m) < topm) topm = hnew(m)
262  endif
263  if(this%ihc(this%jas(ipos)) == 2) then
264  faheight = min(topn, topm) - max(botn, botm)
265  else
266  faheight = dhalf * ( (topn - botn) + (topm - botm) )
267  endif
268  fawidth = this%hwva(this%jas(ipos))
269  condhfb = this%hydchr(ihfb) * fawidth * faheight
270  else
271  condhfb = this%hydchr(ihfb)
272  end if
273  else
274  condhfb = this%hydchr(ihfb)
275  endif
276  ! -- Make hfb corrections for xt3d
277  call this%xt3d%xt3d_fhfb(kiter, nodes, nja, njasln, amat, idxglo, &
278  rhs, hnew, n, m, condhfb)
279  end do
280  !
281  else
282  !
283  ! -- For Newton, the effect of the barrier is included in condsat.
284  if(this%inewton == 0) then
285  do ihfb = 1, this%nhfb
286  ipos = this%idxloc(ihfb)
287  aterm = amat(idxglo(ipos))
288  n = this%noden(ihfb)
289  m = this%nodem(ihfb)
290  if(this%ibound(n) == 0 .or. this%ibound(m) == 0) cycle
291  if(this%icelltype(n) == 1 .or. this%icelltype(m) == 1) then
292  !
293  ! -- Calculate hfb conductance
294  topn = this%top(n)
295  topm = this%top(m)
296  botn = this%bot(n)
297  botm = this%bot(m)
298  if(this%icelltype(n) == 1) then
299  if(hnew(n) < topn) topn = hnew(n)
300  endif
301  if(this%icelltype(m) == 1) then
302  if(hnew(m) < topm) topm = hnew(m)
303  endif
304  if(this%ihc(this%jas(ipos)) == 2) then
305  faheight = min(topn, topm) - max(botn, botm)
306  else
307  faheight = dhalf * ( (topn - botn) + (topm - botm) )
308  endif
309  if(this%hydchr(ihfb) > dzero) then
310  fawidth = this%hwva(this%jas(ipos))
311  condhfb = this%hydchr(ihfb) * fawidth * faheight
312  cond = aterm * condhfb / (aterm + condhfb)
313  else
314  cond = - aterm * this%hydchr(ihfb)
315  endif
316  !
317  ! -- Save cond for budget calculation
318  this%condsav(ihfb) = cond
319  !
320  ! -- Fill row n diag and off diag
321  idiag = this%ia(n)
322  amat(idxglo(idiag)) = amat(idxglo(idiag)) + aterm - cond
323  amat(idxglo(ipos)) = cond
324  !
325  ! -- Fill row m diag and off diag
326  isymcon = this%isym(ipos)
327  idiag = this%ia(m)
328  amat(idxglo(idiag)) = amat(idxglo(idiag)) + aterm - cond
329  amat(idxglo(isymcon)) = cond
330  !
331  endif
332  enddo
333  endif
334  !
335  endif
336  !
337  ! -- return
338  return
339  end subroutine hfb_fc

341  subroutine hfb_cq(this, hnew, flowja)
342 ! ******************************************************************************
343 ! hfb_cq -- flowja will automatically include the effects of the hfb
344 ! for confined and newton cases when xt3d is not used. This method
345 ! recalculates flowja for the other cases.
346 ! ******************************************************************************
347 !
348 ! SPECIFICATIONS:
349 ! ------------------------------------------------------------------------------
350  ! -- modules
351  use constantsmodule, only: dhalf, dzero
352  ! -- dummy
353  class(gwfhfbtype) :: this
354  real(DP),intent(inout),dimension(:) :: hnew
355  real(DP),intent(inout),dimension(:) :: flowja
356  ! -- local
357  integer(I4B) :: ihfb, n, m
358  integer(I4B) :: ipos
359  real(DP) :: qnm
360  real(DP) :: cond
361  integer(I4B) :: ixt3d
362  real(DP) :: condhfb
363  real(DP) :: fawidth, faheight
364  real(DP) :: topn, topm, botn, botm
365 ! ------------------------------------------------------------------------------
366 !
367  if (associated(this%xt3d%ixt3d)) then
368  ixt3d = this%xt3d%ixt3d
369  else
370  ixt3d = 0
371  end if
372  !
373  if(ixt3d > 0) then
374  !
375  do ihfb = 1, this%nhfb
376  n = min(this%noden(ihfb), this%nodem(ihfb))
377  m = max(this%noden(ihfb), this%nodem(ihfb))
378  ! -- Skip if either cell is inactive.
379  if(this%ibound(n) == 0 .or. this%ibound(m) == 0) cycle
380  !!! if(this%icelltype(n) == 1 .or. this%icelltype(m) == 1) then
381  ! -- Compute scale factor for hfb correction
382  if(this%hydchr(ihfb) > dzero) then
383  if(this%inewton == 0) then
384  ipos = this%idxloc(ihfb)
385  topn = this%top(n)
386  topm = this%top(m)
387  botn = this%bot(n)
388  botm = this%bot(m)
389  if(this%icelltype(n) == 1) then
390  if(hnew(n) < topn) topn = hnew(n)
391  endif
392  if(this%icelltype(m) == 1) then
393  if(hnew(m) < topm) topm = hnew(m)
394  endif
395  if(this%ihc(this%jas(ipos)) == 2) then
396  faheight = min(topn, topm) - max(botn, botm)
397  else
398  faheight = dhalf * ( (topn - botn) + (topm - botm) )
399  endif
400  fawidth = this%hwva(this%jas(ipos))
401  condhfb = this%hydchr(ihfb) * fawidth * faheight
402  else
403  condhfb = this%hydchr(ihfb)
404  end if
405  else
406  condhfb = this%hydchr(ihfb)
407  endif
408  ! -- Make hfb corrections for xt3d
409  call this%xt3d%xt3d_flowjahfb(n, m, hnew, flowja, condhfb)
410  end do
411  !
412  else
413  !
414  ! -- Recalculate flowja for non-newton unconfined.
415  if(this%inewton == 0) then
416  do ihfb = 1, this%nhfb
417  n = this%noden(ihfb)
418  m = this%nodem(ihfb)
419  if(this%ibound(n) == 0 .or. this%ibound(m) == 0) cycle
420  if(this%icelltype(n) == 1 .or. this%icelltype(m) == 1) then
421  ipos = this%dis%con%getjaindex(n, m)
422  cond = this%condsav(ihfb)
423  qnm = cond * (hnew(m) - hnew(n))
424  flowja(ipos) = qnm
425  ipos = this%dis%con%getjaindex(m, n)
426  flowja(ipos) = -qnm
427  !
428  endif
429  enddo
430  endif
431  !
432  end if
433  !
434  ! -- return
435  return
436  end subroutine hfb_cq

735 ! ******************************************************************************
736 ! check_data -- Check for hfb's between two unconnected cells and write a
737 ! warning. Store ipos in idxloc.

826  subroutine condsat_modify(this)
827 ! ******************************************************************************
828 ! condsat_modify -- Modify condsat for the following conditions:
829 ! 1. If Newton is active
830 ! 2. If icelltype for n and icelltype for m is 0
831 ! ******************************************************************************
832 !
833 ! SPECIFICATIONS:
834 ! ------------------------------------------------------------------------------
835  ! -- modules
836  use constantsmodule, only: dhalf, dzero
837  ! -- dummy
838  class(gwfhfbtype) :: this
839  ! -- local
840  integer(I4B) :: ihfb, n, m
841  integer(I4B) :: ipos
842  real(DP) :: cond, condhfb
843  real(DP) :: fawidth, faheight
844  real(DP) :: topn, topm, botn, botm
845 ! ------------------------------------------------------------------------------
846  !
847  do ihfb = 1, this%nhfb
848  ipos = this%idxloc(ihfb)
849  cond = this%condsat(this%jas(ipos))
850  this%csatsav(ihfb) = cond
851  n = this%noden(ihfb)
852  m = this%nodem(ihfb)
853  if(this%inewton == 1 .or. &
854  (this%icelltype(n) == 0 .and. this%icelltype(m) == 0) ) then
855  !
856  ! -- Calculate hfb conductance
857  topn = this%top(n)
858  topm = this%top(m)
859  botn = this%bot(n)
860  botm = this%bot(m)
861  if(this%ihc(this%jas(ipos)) == 2) then
862  faheight = min(topn, topm) - max(botn, botm)
863  else
864  faheight = dhalf * ( (topn - botn) + (topm - botm) )
865  endif
866  if(this%hydchr(ihfb) > dzero) then
867  fawidth = this%hwva(this%jas(ipos))
868  condhfb = this%hydchr(ihfb) * fawidth * faheight
869  cond = cond * condhfb / (cond + condhfb)
870  else
871  cond = - cond * this%hydchr(ihfb)
872  endif
873  this%condsat(this%jas(ipos)) = cond
874  endif
875  enddo
876  !
877  ! -- return
878  return
879  end subroutine condsat_modify
