312  subroutine gwf_gwf_ar(this)
313 ! ******************************************************************************
314 ! gwf_gwf_ar -- Calculate the saturated conductance. Must be called after
315 ! npf_ar for both GWF models.
316 ! ******************************************************************************
317 !
318 ! SPECIFICATIONS:
319 ! ------------------------------------------------------------------------------
320  ! -- modules
321  use constantsmodule, only: linelength, dzero, dhalf, done, dpio180
322  use simmodule, only: store_error, ustop
323  use gwfnpfmodule, only: condmean, vcond, hcond
324  ! -- dummy
325  class(gwfexchangetype) :: this
326  ! -- local
327  integer(I4B) :: iexg
328  integer(I4B) :: n, m, ihc
329  real(DP) :: topn, topm
330  real(DP) :: botn, botm
331  real(DP) :: satn, satm
332  real(DP) :: thickn, thickm
333  real(DP) :: angle, hyn, hym
334  real(DP) :: csat
335  real(DP) :: fawidth
336  real(DP), dimension(3) :: vg
337 ! ------------------------------------------------------------------------------
338  !
339  ! -- If mover is active, then call ar routine
340  if(this%inmvr > 0) call this%mvr%mvr_ar()
341  !
342  ! -- Check to see if horizontal anisotropy is in either model1 or model2.
343  ! If so, then ANGLDEGX must be provided as an auxiliary variable for this
344  ! GWF-GWF exchange (this%ianglex > 0).
345  if(this%gwfmodel1%npf%ik22 /= 0 .or. this%gwfmodel2%npf%ik22 /= 0) then
346  if(this%ianglex == 0) then
347  write(errmsg, '(a)') 'GWF-GWF requires that ANGLDEGX be ' // &
348  'specified as an auxiliary variable because ' // &
349  'K22 was specified in one or both ' // &
350  'groundwater models.'
351  call store_error(errmsg)
352  call ustop()
353  endif
354  endif
355  !
356  ! -- Check to see if specific discharge is needed for model1 or model2.
357  ! If so, then ANGLDEGX must be provided as an auxiliary variable for this
358  ! GWF-GWF exchange (this%ianglex > 0).
359  if(this%gwfmodel1%npf%icalcspdis /= 0 .or. &
360  this%gwfmodel2%npf%icalcspdis /= 0) then
361  if(this%ianglex == 0) then
362  write(errmsg, '(a)') 'GWF-GWF requires that ANGLDEGX be ' // &
363  'specified as an auxiliary variable because ' // &
364  'specific discharge is being calculated in' // &
365  ' one or both groundwater models.'
366  call store_error(errmsg)
367  call ustop()
368  endif
369  if(this%icdist == 0) then
370  write(errmsg, '(a)') 'GWF-GWF requires that CDIST be ' // &
371  'specified as an auxiliary variable because ' // &
372  'specific discharge is being calculated in' // &
373  ' one or both groundwater models.'
374  call store_error(errmsg)
375  call ustop()
376  endif
377  endif
378  !
379  ! -- Go through each connection and calculate the saturated conductance
380  do iexg = 1, this%nexg
381  !
382  ihc = this%ihc(iexg)
383  n = this%nodem1(iexg)
384  m = this%nodem2(iexg)
385  topn = this%gwfmodel1%dis%top(n)
386  topm = this%gwfmodel2%dis%top(m)
387  botn = this%gwfmodel1%dis%bot(n)
388  botm = this%gwfmodel2%dis%bot(m)
389  satn = this%gwfmodel1%npf%sat(n)
390  satm = this%gwfmodel2%npf%sat(m)
391  thickn = (topn - botn) * satn
392  thickm = (topm - botm) * satm
393  !
394  ! -- Calculate conductance depending on connection orientation
395  if(ihc == 0) then
396  !
397  ! -- Vertical conductance for fully saturated conditions
398  vg(1) = dzero
399  vg(2) = dzero
400  vg(3) = done
401  hyn = this%gwfmodel1%npf%hy_eff(n, 0, ihc, vg=vg)
402  hym = this%gwfmodel2%npf%hy_eff(m, 0, ihc, vg=vg)
403  csat = vcond(1, 1, 1, 1, 0, 1, 1, done, &
404  botn, botm, &
405  hyn, hym, &
406  satn, satm, &
407  topn, topm, &
408  botn, botm, &
409  this%hwva(iexg))
410  else
411  !
412  ! -- Calculate horizontal conductance
413  hyn = this%gwfmodel1%npf%k11(n)
414  hym = this%gwfmodel2%npf%k11(m)
415  !
416  ! -- Check for anisotropy in models, and recalculate hyn and hym
417  if(this%ianglex > 0) then
418  angle = this%auxvar(this%ianglex, iexg) * dpio180
419  vg(1) = abs(cos(angle))
420  vg(2) = abs(sin(angle))
421  vg(3) = dzero
422  !
423  ! -- anisotropy in model 1
424  if(this%gwfmodel1%npf%ik22 /= 0) then
425  hyn = this%gwfmodel1%npf%hy_eff(n, 0, ihc, vg=vg)
426  endif
427  !
428  ! -- anisotropy in model 2
429  if(this%gwfmodel2%npf%ik22 /= 0) then
430  hym = this%gwfmodel2%npf%hy_eff(m, 0, ihc, vg=vg)
431  endif
432  endif
433  !
434  fawidth = this%hwva(iexg)
435  csat = hcond(1, 1, 1, 1, this%inewton, 0, ihc, &
436  this%icellavg, 0, 0, done, &
437  topn, topm, satn, satm, hyn, hym, &
438  topn, topm, &
439  botn, botm, &
440  this%cl1(iexg), this%cl2(iexg), &
441  fawidth, this%satomega)
442  endif
443  !
444  ! -- store csat in condsat
445  this%condsat(iexg) = csat
446  enddo
447  !
448  ! -- Observation AR
449  call this%obs%obs_ar()
450  !
451  ! -- Return
452  return
453  end subroutine gwf_gwf_ar

525  subroutine gwf_gwf_fc(this, kiter, iasln, amatsln, rhssln, inwtflag)
526 ! ******************************************************************************
527 ! gwf_gwf_fc -- Fill the matrix
528 ! ******************************************************************************
529 !
530 ! SPECIFICATIONS:
531 ! ------------------------------------------------------------------------------
532  ! -- modules
533  use constantsmodule, only: dhalf
534  use gwfnpfmodule, only: hcond, vcond
535  ! -- dummy
536  class(gwfexchangetype) :: this
537  integer(I4B), intent(in) :: kiter
538  integer(I4B), dimension(:), intent(in) :: iasln
539  real(DP), dimension(:), intent(inout) :: amatsln
540  real(DP), dimension(:), intent(inout) ::rhssln
541  integer(I4B), optional, intent(in) :: inwtflag
542  ! -- local
543  integer(I4B) :: inwt, iexg
544  integer(I4B) :: i, nodem1sln, nodem2sln, idiagsln
545  integer(I4B) :: njasln
546 ! ------------------------------------------------------------------------------
547  !
548  ! -- calculate the conductance for each exchange connection
549  call this%condcalc()
550  !
551  ! -- if gnc is active, then copy cond into gnc cond (might consider a
552  ! pointer here in the future)
553  if(this%ingnc > 0) then
554  do iexg = 1, this%nexg
555  this%gnc%cond(iexg) = this%cond(iexg)
556  enddo
557  endif
558  !
559  ! -- Put this%cond into amatsln
560  do i = 1, this%nexg
561  amatsln(this%idxglo(i)) = this%cond(i)
562  amatsln(this%idxsymglo(i)) = this%cond(i)
563  nodem1sln = this%nodem1(i) + this%gwfmodel1%moffset
564  nodem2sln = this%nodem2(i) + this%gwfmodel2%moffset
565  idiagsln = iasln(nodem1sln)
566  amatsln(idiagsln) = amatsln(idiagsln) - this%cond(i)
567  idiagsln = iasln(nodem2sln)
568  amatsln(idiagsln) = amatsln(idiagsln) - this%cond(i)
569  enddo
570  !
571  ! -- Fill the gnc terms in the solution matrix
572  if(this%ingnc > 0) then
573  call this%gnc%gnc_fc(kiter, amatsln)
574  endif
575  !
576  ! -- Call mvr fc routine
577  if(this%inmvr > 0) call this%mvr%mvr_fc()
578  !
579  ! -- Set inwt to exchange newton, but shut off if requested by caller
580  inwt = this%inewton
581  if(present(inwtflag)) then
582  if (inwtflag == 0) inwt = 0
583  endif
584  if (inwt /= 0) then
585  call this%exg_fn(kiter, iasln, amatsln)
586  endif
587  !
588  ! -- Ghost node Newton-Raphson
589  if (this%ingnc > 0) then
590  if (inwt /= 0) then
591  njasln = size(amatsln)
592  call this%gnc%gnc_fn(kiter, njasln, amatsln, this%condsat, &
593  ihc_opt=this%ihc, ivarcv_opt=this%ivarcv, &
594  ictm1_opt=this%gwfmodel1%npf%icelltype, &
595  ictm2_opt=this%gwfmodel2%npf%icelltype)
596  endif
597  endif
598  !
599  ! -- Return
600  return
601  end subroutine gwf_gwf_fc

631  real(DP) :: consterm

634  !
635  do iexg = 1, this%nexg

637  m = this%nodem2(iexg)
638  nodensln = this%nodem1(iexg) + this%gwfmodel1%moffset
639  nodemsln = this%nodem2(iexg) + this%gwfmodel2%moffset
640  ibdn = this%gwfmodel1%ibound(n)

646  hn = this%gwfmodel1%x(n)
647  hm = this%gwfmodel2%x(m)
648  if(this%ihc(iexg) == 0) then
649  ! -- vertical connection, newton not supported
650  else
651  ! -- determine upstream node
652  nisup = .false.
653  if(hm < hn) nisup = .true.
654  !
655  ! -- set upstream top and bot
656  if(nisup) then
657  topup = topn
658  botup = botn
659  hup = hn
660  hdn = hm
661  else
662  topup = topm
663  botup = botm
664  hup = hm
665  hdn = hn
666  endif
667  !
668  ! -- no newton terms if upstream cell is confined
669  if (nisup) then
670  if (this%gwfmodel1%npf%icelltype(n) == 0) cycle
671  else
672  if (this%gwfmodel2%npf%icelltype(m) == 0) cycle
673  end if

675  ! -- set topup and botup
676  if(this%ihc(iexg) == 2) then
677  topup = min(topn, topm)
678  botup = max(botn, botm)

684  ! -- TO DO deal with MODFLOW-NWT upstream weighting option

686  ! -- compute terms
687  consterm = -cond * (hup - hdn)
688  derv = squadraticsaturationderivative(topup, botup, hup)

690  idiagmsln = iasln(nodemsln)
691  if(nisup) then
692  !
693  ! -- fill jacobian with n being upstream
694  term = consterm * derv
695  this%gwfmodel1%rhs(n) = this%gwfmodel1%rhs(n) + term * hn
696  this%gwfmodel2%rhs(m) = this%gwfmodel2%rhs(m) - term * hn

700  endif
701  else
702  !
703  ! -- fill jacobian with m being upstream
704  term = -consterm * derv
705  this%gwfmodel1%rhs(n) = this%gwfmodel1%rhs(n) + term * hm
706  this%gwfmodel2%rhs(m) = this%gwfmodel2%rhs(m) - term * hm
707  amatsln(idiagmsln) = amatsln(idiagmsln) - term
708  if(ibdn > 0) then
709  amatsln(this%idxglo(iexg)) = amatsln(this%idxglo(iexg)) + term
710  endif
711  endif
712  endif
713  enddo
714  !
715  ! -- Return
716  return

719  subroutine gwf_gwf_cq(this, icnvg, isuppress_output, isolnid)
720 ! ******************************************************************************
721 ! gwf_gwf_cq -- Calculate flow between two cells
722 ! ******************************************************************************
723 !
724 ! SPECIFICATIONS:
725 ! ------------------------------------------------------------------------------
726  ! -- modules
727  use constantsmodule, only: dzero, dpio180
728  use gwfnpfmodule, only: thksatnm
729  ! -- dummy
730  class(gwfexchangetype) :: this
731  integer(I4B), intent(inout) :: icnvg
732  integer(I4B), intent(in) :: isuppress_output
733  integer(I4B), intent(in) :: isolnid
734  ! -- local
735  integer(I4B) :: i
736  integer(I4B) :: n1
737  integer(I4B) :: n2
738  integer(I4B) :: ihc
739  integer(I4B) :: ibdn1
740  integer(I4B) :: ibdn2
741  integer(I4B) :: ictn1
742  integer(I4B) :: ictn2
743  integer(I4B) :: iusg
744  real(DP) :: topn1
745  real(DP) :: topn2
746  real(DP) :: botn1
747  real(DP) :: botn2
748  real(DP) :: satn1
749  real(DP) :: satn2
750  real(DP) :: hn1
751  real(DP) :: hn2
752  real(DP) :: rrate
753  real(DP) :: thksat
754  real(DP) :: angle
755  real(DP) :: nx
756  real(DP) :: ny
757  real(DP) :: distance
758  real(DP) :: dltot
759  real(DP) :: hwva
760  real(DP) :: area
761 ! ------------------------------------------------------------------------------
762  !
763  ! -- calculate flow and store in simvals
764  do i = 1, this%nexg
765  rrate = dzero
766  n1 = this%nodem1(i)
767  n2 = this%nodem2(i)
768  ibdn1 = this%gwfmodel1%ibound(n1)
769  ibdn2 = this%gwfmodel2%ibound(n2)
770  if(ibdn1 /= 0 .and. ibdn2 /= 0) then
771  rrate = this%qcalc(i, n1, n2)
772  if(this%ingnc > 0) then
773  rrate = rrate + this%gnc%deltaqgnc(i)
774  endif
775  endif
776  this%simvals(i) = rrate
777  end do
778  !
779  ! -- Return if there neither model needs to calculate specific discharge
780  if (this%gwfmodel1%npf%icalcspdis == 0 .and. &
781  this%gwfmodel2%npf%icalcspdis == 0) return
782  !
783  ! -- initialize
784  iusg = 0
785  !
786  ! -- Loop through all exchanges using the flow rate
787  ! stored in simvals
788  do i = 1, this%nexg
789  rrate = this%simvals(i)
790  n1 = this%nodem1(i)
791  n2 = this%nodem2(i)
792  ihc = this%ihc(i)
793  hwva = this%hwva(i)
794  ibdn1 = this%gwfmodel1%ibound(n1)
795  ibdn2 = this%gwfmodel2%ibound(n2)
796  ictn1 = this%gwfmodel1%npf%icelltype(n1)
797  ictn2 = this%gwfmodel2%npf%icelltype(n2)
798  topn1 = this%gwfmodel1%dis%top(n1)
799  topn2 = this%gwfmodel2%dis%top(n2)
800  botn1 = this%gwfmodel1%dis%bot(n1)
801  botn2 = this%gwfmodel2%dis%bot(n2)
802  satn1 = this%gwfmodel1%npf%sat(n1)
803  satn2 = this%gwfmodel2%npf%sat(n2)
804  hn1 = this%gwfmodel1%x(n1)
805  hn2 = this%gwfmodel2%x(n2)
806  !
807  ! -- Calculate face normal components
808  if(ihc == 0) then
809  nx = dzero
810  ny = dzero
811  area = hwva
812  if (botn1 < botn2) then
813  ! -- n1 is beneath n2, so rate is positive downward. Flip rate
814  ! upward so that points in positive z direction
815  rrate = - rrate
816  endif
817  else
818  if(this%ianglex > 0) then
819  angle = this%auxvar(this%ianglex, i) * dpio180
820  nx = cos(angle)
821  ny = sin(angle)
822  else
823  ! error?
824  call ustop('error in gwf_gwf_cq')
825  endif
826  !
827  ! -- Calculate the saturated thickness at interface between n1 and n2
828  thksat = thksatnm(ibdn1, ibdn2, ictn1, ictn2, this%inewton, ihc, &
829  iusg, hn1, hn2, satn1, satn2, &
830  topn1, topn2, botn1, botn2, this%satomega)
831  area = hwva * thksat
832  endif
833  !
834  ! -- Submit this connection and flow information to the npf
835  ! package of gwfmodel1
836  if(this%icdist > 0) then
837  dltot = this%auxvar(this%icdist, i)
838  else
839  call ustop('error in gwf_gwf_cq')
840  endif
841  distance = dltot * this%cl1(i) / (this%cl1(i) + this%cl2(i))
842  if (this%gwfmodel1%npf%icalcspdis == 1) then
843  call this%gwfmodel1%npf%set_edge_properties(n1, ihc, rrate, area, &
844  nx, ny, distance)
845  endif
846  !
847  ! -- Submit this connection and flow information to the npf
848  ! package of gwfmodel2
849  if(this%icdist > 0) then
850  dltot = this%auxvar(this%icdist, i)
851  else
852  call ustop('error in gwf_gwf_cq')
853  endif
854  if (this%gwfmodel2%npf%icalcspdis == 1) then
855  distance = dltot * this%cl2(i) / (this%cl1(i) + this%cl2(i))
856  if (ihc /= 0) rrate = -rrate
857  call this%gwfmodel2%npf%set_edge_properties(n2, ihc, rrate, area, &
858  -nx, -ny, distance)
859  endif
860  !
861  enddo
862  !
863  ! -- return
864  return
865  end subroutine gwf_gwf_cq
