184  subroutine exg_ar(this)
185 ! ******************************************************************************

189 ! SPECIFICATIONS:
190 ! ------------------------------------------------------------------------------

193  ! -- dummy
194  class(gwfgwtexchangetype) :: this
195  ! -- local
196  class(basemodeltype), pointer :: mb => null()
197  type(gwfmodeltype), pointer :: gwfmodel => null()
198  type(gwtmodeltype), pointer :: gwtmodel => null()
199  ! -- formats
200  character(len=*),parameter :: fmtdiserr = &
201  "('GWF and GWT Models do not have the same discretization for exchange&
202  & ',a,'.&
203  & GWF Model has ', i0, ' user nodes and ', i0, ' reduced nodes.&
204  & GWT Model has ', i0, ' user nodes and ', i0, ' reduced nodes.&
205  & Ensure discretization packages, including IDOMAIN, are identical.')"
206 ! ------------------------------------------------------------------------------
207  !
208  ! -- set gwfmodel
209  mb => getbasemodelfromlist(basemodellist, this%m1id)
210  select type (mb)
211  type is (gwfmodeltype)
212  gwfmodel => mb
213  end select
214  !
215  ! -- set gwtmodel
216  mb => getbasemodelfromlist(basemodellist, this%m2id)
217  select type (mb)
218  type is (gwtmodeltype)
219  gwtmodel => mb
220  end select
221  !
222  ! -- Check to make sure sizes are identical
223  if (gwtmodel%dis%nodes /= gwfmodel%dis%nodes .or.&
224  gwtmodel%dis%nodesuser /= gwfmodel%dis%nodesuser) then
225  write(errmsg, fmtdiserr) trim(this%name), &
226  gwfmodel%dis%nodesuser, &
227  gwfmodel%dis%nodes, &
228  gwtmodel%dis%nodesuser, &
229  gwtmodel%dis%nodes
230  call store_error(errmsg, terminate=.true.)
231  end if
232  !
233  ! -- setup pointers to gwf variables allocated in gwf_ar
234  gwtmodel%fmi%gwfhead => gwfmodel%x
235  gwtmodel%fmi%gwfsat => gwfmodel%npf%sat
236  gwtmodel%fmi%gwfspdis => gwfmodel%npf%spdis
237  !
238  ! -- setup pointers to the flow storage rates. GWF strg arrays are
239  ! available after the gwf_ar routine is called.
240  if(gwtmodel%inmst > 0) then
241  if (gwfmodel%insto > 0) then
242  gwtmodel%fmi%gwfstrgss => gwfmodel%sto%strgss
243  gwtmodel%fmi%igwfstrgss = 1
244  if (gwfmodel%sto%iusesy == 1) then

246  gwtmodel%fmi%igwfstrgsy = 1
247  endif
248  endif
249  endif

253  call gwfmodel%buy%set_concentration_pointer(gwtmodel%name, gwtmodel%x, &
254  gwtmodel%ibound)
255  endif
256  !

258  call this%gwfbnd2gwtfmi()
259  !

262  gwtmodel%fmi%mvrbudobj => gwfmodel%mvr%budobj
