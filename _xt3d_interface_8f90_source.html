173  subroutine xt3d_mc(this, moffset, iasln, jasln)
174 ! ******************************************************************************
175 ! xt3d_mc -- Map connections and construct iax, jax, and idxglox
176 ! ******************************************************************************
177 !
178 ! SPECIFICATIONS:
179 ! ------------------------------------------------------------------------------
180  ! -- modules
181  use memorymanagermodule, only: mem_allocate
182  ! -- dummy
183  class(xt3dtype) :: this
184  integer(I4B), intent(in) :: moffset

186  integer(I4B), dimension(:), intent(in) :: jasln

188  integer(I4B) :: i, j, jj, iglo, jglo, jjg, niax, njax, ipos
189  integer(I4B) :: igfirstnod, iglastnod
190  logical :: isextnbr
191 ! ------------------------------------------------------------------------------
192  !
193  ! -- If not rhs, map connections for extended neighbors and construct iax,
194  ! -- jax, and idxglox
195  if (this%ixt3d == 1) then
196  !
197  ! -- calculate the first node for the model and the last node in global
198  ! numbers
199  igfirstnod = moffset + 1
200  iglastnod = moffset + this%dis%nodes
201  !
202  ! -- allocate iax, jax, and idxglox
203  niax = this%dis%nodes + 1
204  njax = this%numextnbrs ! + 1
205  call mem_allocate(this%iax, niax, 'IAX', trim(this%memoryPath))
206  call mem_allocate(this%jax, njax, 'JAX', trim(this%memoryPath))
207  call mem_allocate(this%idxglox, njax, 'IDXGLOX', trim(this%memoryPath))
208  !
209  ! -- load first iax entry
210  ipos = 1
211  this%iax(1) = ipos
212  !
213  ! -- loop over nodes
214  do i = 1, this%dis%nodes
215  !
216  ! -- calculate global node number
217  iglo = i + moffset
218  !
219  ! -- loop over neighbors in global matrix
220  do jjg = iasln(iglo), iasln(iglo + 1) - 1
221  !
222  ! -- if jglo is in a different model, then it cannot be an extended
223  ! neighbor, so skip over it
224  jglo = jasln(jjg)
225  if (jglo < igfirstnod .or. jglo > iglastnod) then
226  cycle
227  endif
228  !
229  ! -- determine whether this neighbor is an extended neighbor
230  ! by searching the original neighbors
231  isextnbr = .true.
232  searchloop: do jj = this%dis%con%ia(i), this%dis%con%ia(i+1) - 1
233  j = this%dis%con%ja(jj)
234  jglo = j + moffset
235  !
236  ! -- if an original neighbor, note that and end the search
237  if(jglo == jasln(jjg)) then
238  isextnbr = .false.
239  exit searchloop
240  endif
241  enddo searchloop
242  !
243  ! -- if an extended neighbor, add it to jax and idxglox
244  if (isextnbr) then
245  this%jax(ipos) = jasln(jjg) - moffset
246  this%idxglox(ipos) = jjg
247  ipos = ipos + 1
248  endif
249  enddo
250  ! -- load next iax entry
251  this%iax(i+1) = ipos
252  enddo
253  !
254  else
255  !
256  call mem_allocate(this%iax, 0, 'IAX', trim(this%memoryPath))
257  call mem_allocate(this%jax, 0, 'JAX', trim(this%memoryPath))
258  call mem_allocate(this%idxglox, 0, 'IDXGLOX', trim(this%memoryPath))
259  !
260  endif
261  !
262  ! -- Return
263  return
264  end subroutine xt3d_mc

369  subroutine xt3d_fc(this, kiter, njasln, amat, idxglo, rhs, hnew)
370 ! ******************************************************************************
371 ! xt3d_fc -- Formulate
372 ! ******************************************************************************
373 !
374 ! SPECIFICATIONS:
375 ! ------------------------------------------------------------------------------
376  use constantsmodule, only: done
377  use xt3dalgorithmmodule, only: qconds
378  ! -- dummy
379  class(xt3dtype) :: this
380  integer(I4B) :: kiter
381  integer(I4B),intent(in) :: njasln
382  real(DP),dimension(njasln),intent(inout) :: amat
383  integer(I4B),intent(in),dimension(:) :: idxglo
384  real(DP),intent(inout),dimension(:) :: rhs
385  real(DP),intent(inout),dimension(:) :: hnew
386  ! -- local
387  integer(I4B) :: nodes, nja
388  integer(I4B) :: n, m, ipos
389  !
390  logical :: allhc0, allhc1
391  integer(I4B) :: nnbr0, nnbr1
392  integer(I4B) :: il0, ii01, jjs01, il01, il10, ii00, ii11, ii10
393  integer(I4B) :: i
394  integer(I4B),dimension(this%nbrmax) :: inbr0, inbr1
395  real(DP) :: ar01, ar10
396  real(DP),dimension(this%nbrmax,3) :: vc0, vn0, vc1, vn1
397  real(DP),dimension(this%nbrmax) :: dl0, dl0n, dl1, dl1n
398  real(DP),dimension(3,3) :: ck0, ck1
399  real(DP) :: chat01
400  real(DP),dimension(this%nbrmax) :: chati0, chat1j
401  real(DP) :: qnm, qnbrs
402 ! ------------------------------------------------------------------------------
403  !
404  ! -- Calculate xt3d conductance-like coefficients and put into amat and rhs
405  ! -- as appropriate
406  !
407  nodes = this%dis%nodes
408  nja = this%dis%con%nja
409  if (this%lamatsaved) then
410  do i = 1, this%dis%con%nja
411  amat(idxglo(i)) = amat(idxglo(i)) + this%amatpc(i)
412  end do
413  do i = 1, this%numextnbrs
414  amat(this%idxglox(i)) = amat(this%idxglox(i)) + this%amatpcx(i)
415  end do
416  end if
417  !
418  do n = 1, nodes
419  ! -- Skip if inactive.
420  if (this%ibound(n).eq.0) cycle
421  ! -- Skip if all connections are permanently confined
422  if (this%lamatsaved) then
423  if (this%iallpc(n) == 1) cycle
424  end if
425  nnbr0 = this%dis%con%ia(n+1) - this%dis%con%ia(n) - 1
426  ! -- Load conductivity and connection info for cell 0.
427  call this%xt3d_load(nodes, n, nnbr0, inbr0, vc0, vn0, dl0, dl0n, &
428  ck0, allhc0)
429  ! -- Loop over active neighbors of cell 0 that have a higher
430  ! -- cell number (taking advantage of reciprocity).
431  do il0 = 1,nnbr0
432  ipos = this%dis%con%ia(n) + il0
433  if (this%dis%con%mask(ipos) == 0) cycle
434 
435  m = inbr0(il0)
436  ! -- Skip if neighbor is inactive or has lower cell number.
437  if ((m.eq.0).or.(m.lt.n)) cycle
438  nnbr1 = this%dis%con%ia(m+1) - this%dis%con%ia(m) - 1
439  ! -- Load conductivity and connection info for cell 1.
440  call this%xt3d_load(nodes, m, nnbr1, inbr1, vc1, vn1, dl1, dl1n, &
441  ck1, allhc1)
442  ! -- Set various indices.
443  call this%xt3d_indices(n, m, il0, ii01, jjs01, il01, il10, &
444  ii00, ii11, ii10)
445  ! -- Compute areas.
446  if (this%inewton /= 0) then
447  ar01 = done
448  ar10 = done
449  else
450  call this%xt3d_areas(nodes, n, m, jjs01, .false., ar01, ar10, hnew)
451  end if
452  ! -- Compute "conductances" for interface between
453  ! -- cells 0 and 1.
454  call qconds(this%nbrmax, nnbr0, inbr0, il01, vc0, vn0, dl0, dl0n, &
455  ck0, nnbr1, inbr1, il10, vc1, vn1, dl1, dl1n, ck1, ar01, ar10, &
456  this%vcthresh, allhc0, allhc1, chat01, chati0, chat1j)
457  ! -- If Newton, compute and save saturated flow, then scale
458  ! -- conductance-like coefficients by the actual area for
459  ! -- subsequent amat and rhs assembly.
460  if (this%inewton /= 0) then
461  ! -- Contribution to flow from primary connection.
462  qnm = chat01*(hnew(m) - hnew(n))
463  ! -- Contribution from immediate neighbors of node 0.
464  call this%xt3d_qnbrs(nodes, n, m, nnbr0, inbr0, chati0, hnew, qnbrs)
465  qnm = qnm + qnbrs
466  ! -- Contribution from immediate neighbors of node 1.
467  call this%xt3d_qnbrs(nodes, m, n, nnbr1, inbr1, chat1j, hnew, qnbrs)
468  qnm = qnm - qnbrs
469  ! -- Multiply by saturated area and save in qsat.
470  call this%xt3d_areas(nodes, n, m, jjs01, .true., ar01, ar10, hnew)
471  this%qsat(ii01) = qnm*ar01
472  ! -- Scale coefficients by actual area.
473  call this%xt3d_areas(nodes, n, m, jjs01, .false., ar01, ar10, hnew)
474  chat01 = chat01*ar01
475  chati0 = chati0*ar01
476  chat1j = chat1j*ar01
477  end if
478  ! -- Contribute to rows for cells 0 and 1.
479  amat(idxglo(ii00)) = amat(idxglo(ii00)) - chat01
480  amat(idxglo(ii01)) = amat(idxglo(ii01)) + chat01
481  amat(idxglo(ii11)) = amat(idxglo(ii11)) - chat01
482  amat(idxglo(ii10)) = amat(idxglo(ii10)) + chat01
483  if (this%ixt3d == 1) then
484  call this%xt3d_amat_nbrs(nodes, n, ii00, nnbr0, nja, njasln, &
485  inbr0, amat, idxglo, chati0)
486  call this%xt3d_amat_nbrnbrs(nodes, n, m, ii01, nnbr1, nja, njasln, &
487  inbr1, amat, idxglo, chat1j)
488  call this%xt3d_amat_nbrs(nodes, m, ii11, nnbr1, nja, njasln, &
489  inbr1, amat, idxglo, chat1j)
490  call this%xt3d_amat_nbrnbrs(nodes, m, n, ii10, nnbr0, nja, njasln, &
491  inbr0, amat, idxglo, chati0)
492  else
493  call this%xt3d_rhs(nodes, n, m, nnbr0, inbr0, chati0, hnew, rhs)
494  call this%xt3d_rhs(nodes, m, n, nnbr1, inbr1, chat1j, hnew, rhs)
495  endif
496  !
497  enddo
498  enddo
499  !
500  ! -- Return
501  return
502  end subroutine xt3d_fc

590  subroutine xt3d_fhfb(this, kiter, nodes, nja, njasln, amat, idxglo, rhs, hnew, &
591  n, m, condhfb)
592 ! ******************************************************************************
593 ! xt3d_fhfb -- Formulate HFB correction
594 ! ******************************************************************************
595 !
596 ! SPECIFICATIONS:
597 ! ------------------------------------------------------------------------------
598  use constantsmodule, only: done
599  use xt3dalgorithmmodule, only: qconds
600  ! -- dummy
601  class(xt3dtype) :: this
602  integer(I4B) :: kiter
603  integer(I4B),intent(in) :: nodes
604  integer(I4B),intent(in) :: nja
605  integer(I4B),intent(in) :: njasln
606  integer(I4B) :: n, m
607  real(DP),dimension(njasln),intent(inout) :: amat
608  integer(I4B),intent(in),dimension(nja) :: idxglo
609  real(DP),intent(inout),dimension(nodes) :: rhs
610  real(DP),intent(inout),dimension(nodes) :: hnew
611  real(DP) :: condhfb
612  ! -- local
613  !
614  logical :: allhc0, allhc1
615  integer(I4B) :: nnbr0, nnbr1
616  integer(I4B) :: il0, ii01, jjs01, il01, il10, ii00, ii11, ii10, il
617  integer(I4B),dimension(this%nbrmax) :: inbr0, inbr1
618  real(DP) :: ar01, ar10
619  real(DP),dimension(this%nbrmax,3) :: vc0, vn0, vc1, vn1
620  real(DP),dimension(this%nbrmax) :: dl0, dl0n, dl1, dl1n
621  real(DP),dimension(3,3) :: ck0, ck1
622  real(DP) :: chat01
623  real(DP),dimension(this%nbrmax) :: chati0, chat1j
624  real(DP) :: qnm, qnbrs
625  real(DP) :: term
626 ! ------------------------------------------------------------------------------
627  !
628  ! -- Calculate hfb corrections to xt3d conductance-like coefficients and
629  ! -- put into amat and rhs as appropriate
630  !
631  nnbr0 = this%dis%con%ia(n+1) - this%dis%con%ia(n) - 1
632  ! -- Load conductivity and connection info for cell 0.
633  call this%xt3d_load(nodes, n, nnbr0, inbr0, vc0, vn0, dl0, dl0n, &
634  ck0, allhc0)
635  ! -- Find local neighbor number of cell 1.
636  do il = 1,nnbr0
637  if (inbr0(il).eq.m) then
638  il0 = il
639  exit
640  end if
641  end do
642  nnbr1 = this%dis%con%ia(m+1) - this%dis%con%ia(m) - 1
643  ! -- Load conductivity and connection info for cell 1.
644  call this%xt3d_load(nodes, m, nnbr1, inbr1, vc1, vn1, dl1, dl1n, &
645  ck1, allhc1)
646  ! -- Set various indices.
647  call this%xt3d_indices(n, m, il0, ii01, jjs01, il01, il10, &
648  ii00, ii11, ii10)
649  ! -- Compute areas.
650  if (this%inewton /= 0) then
651  ar01 = done
652  ar10 = done
653  else
654  call this%xt3d_areas(nodes, n, m, jjs01, .false., ar01, ar10, hnew)
655  end if
656  ! -- Compute "conductances" for interface between
657  ! -- cells 0 and 1.
658  call qconds(this%nbrmax, nnbr0, inbr0, il01, vc0, vn0, dl0, dl0n, &
659  ck0, nnbr1, inbr1, il10, vc1, vn1, dl1, dl1n, ck1, ar01, ar10, &
660  this%vcthresh, allhc0, allhc1, chat01, chati0, chat1j)
661  ! -- Apply scale factor to compute "conductances" for hfb correction
662  if(condhfb > dzero) then
663  term = chat01/(chat01 + condhfb)
664  else
665  term = -condhfb
666  endif
667  chat01 = -chat01*term
668  chati0 = -chati0*term
669  chat1j = -chat1j*term
670  ! -- If Newton, compute and save saturated flow, then scale
671  ! -- conductance-like coefficients by the actual area for
672  ! -- subsequent amat and rhs assembly.
673  if (this%inewton /= 0) then
674  ! -- Contribution to flow from primary connection.
675  qnm = chat01*(hnew(m) - hnew(n))
676  ! -- Contribution from immediate neighbors of node 0.
677  call this%xt3d_qnbrs(nodes, n, m, nnbr0, inbr0, chati0, hnew, qnbrs)
678  qnm = qnm + qnbrs
679  ! -- Contribution from immediate neighbors of node 1.
680  call this%xt3d_qnbrs(nodes, m, n, nnbr1, inbr1, chat1j, hnew, qnbrs)
681  qnm = qnm - qnbrs
682  ! -- Multiply by saturated area and add correction to qsat.
683  call this%xt3d_areas(nodes, n, m, jjs01, .true., ar01, ar10, hnew)
684  this%qsat(ii01) = this%qsat(ii01) + qnm*ar01
685  ! -- Scale coefficients by actual area.
686  call this%xt3d_areas(nodes, n, m, jjs01, .false., ar01, ar10, hnew)
687  chat01 = chat01*ar01
688  chati0 = chati0*ar01
689  chat1j = chat1j*ar01
690  end if
691  ! -- Contribute to rows for cells 0 and 1.
692  amat(idxglo(ii00)) = amat(idxglo(ii00)) - chat01
693  amat(idxglo(ii01)) = amat(idxglo(ii01)) + chat01
694  amat(idxglo(ii11)) = amat(idxglo(ii11)) - chat01
695  amat(idxglo(ii10)) = amat(idxglo(ii10)) + chat01
696  if (this%ixt3d == 1) then
697  call this%xt3d_amat_nbrs(nodes, n, ii00, nnbr0, nja, njasln, &
698  inbr0, amat, idxglo, chati0)
699  call this%xt3d_amat_nbrnbrs(nodes, n, m, ii01, nnbr1, nja, njasln, &
700  inbr1, amat, idxglo, chat1j)
701  call this%xt3d_amat_nbrs(nodes, m, ii11, nnbr1, nja, njasln, &
702  inbr1, amat, idxglo, chat1j)
703  call this%xt3d_amat_nbrnbrs(nodes, m, n, ii10, nnbr0, nja, njasln, &
704  inbr0, amat, idxglo, chati0)
705  else
706  call this%xt3d_rhs(nodes, n, m, nnbr0, inbr0, chati0, hnew, rhs)
707  call this%xt3d_rhs(nodes, m, n, nnbr1, inbr1, chat1j, hnew, rhs)
708  endif
709  !
710  ! -- Return
711  return
712  end subroutine xt3d_fhfb

714  subroutine xt3d_fn(this, kiter, nodes, nja, njasln, amat, idxglo, rhs, hnew)
715 ! ******************************************************************************
716 ! xt3d_fn -- Fill Newton terms for xt3d
717 ! ******************************************************************************
718 !
719 ! SPECIFICATIONS:
720 ! ------------------------------------------------------------------------------
721  use constantsmodule, only: done
722  use smoothingmodule, only: squadraticsaturationderivative
723  ! -- dummy
724  class(xt3dtype) :: this
725  integer(I4B) :: kiter
726  integer(I4B),intent(in) :: nodes
727  integer(I4B),intent(in) :: nja
728  integer(I4B),intent(in) :: njasln
729  real(DP),dimension(njasln),intent(inout) :: amat
730  integer(I4B),intent(in),dimension(nja) :: idxglo
731  real(DP),intent(inout),dimension(nodes) :: rhs
732  real(DP),intent(inout),dimension(nodes) :: hnew
733  ! -- local
734  integer(I4B) :: n, m, ipos
735  !
736  integer(I4B) :: nnbr0
737  integer(I4B) :: il0, ii01, jjs01, il01, il10, ii00, ii11, ii10
738  integer(I4B),dimension(this%nbrmax) :: inbr0
739  integer(I4B) :: iups, idn
740  real(DP) :: topup, botup, derv, term
741 ! ------------------------------------------------------------------------------
742  !
743  ! -- Update amat and rhs with Newton terms
744  do n = 1, nodes
745  ! -- Skip if inactive.
746  if (this%ibound(n).eq.0) cycle
747  ! -- No Newton correction if amat saved (which implies no rhs option)
748  ! -- and all connections for the cell are permanently confined.
749  if (this%lamatsaved) then
750  if (this%iallpc(n) == 1) cycle
751  end if
752  nnbr0 = this%dis%con%ia(n+1) - this%dis%con%ia(n) - 1
753  ! -- Load neighbors of cell. Set cell numbers for inactive
754  ! -- neighbors to zero.
755  call this%xt3d_load_inbr(n, nnbr0, inbr0)
756  ! -- Loop over active neighbors of cell 0 that have a higher
757  ! -- cell number (taking advantage of reciprocity).
758  do il0 = 1,nnbr0
759  ipos = this%dis%con%ia(n) + il0
760  if (this%dis%con%mask(ipos) == 0) cycle
761 
762  m = inbr0(il0)
763  ! -- Skip if neighbor is inactive or has lower cell number.
764  if ((inbr0(il0).eq.0).or.(m.lt.n)) cycle
765  ! -- Set various indices.
766  call this%xt3d_indices(n, m, il0, ii01, jjs01, il01, il10, &
767  ii00, ii11, ii10)
768  ! determine upstream node
769  iups = m
770  if (hnew(m) < hnew(n)) iups = n
771  idn = n
772  if (iups == n) idn = m
773  ! -- no Newton terms if upstream cell is confined
774  ! -- and no rhs option
775  if ((this%icelltype(iups) == 0).and.(this%ixt3d.eq.1)) cycle
776  ! -- Set the upstream top and bot, and then recalculate for a
777  ! vertically staggered horizontal connection
778  topup = this%dis%top(iups)
779  botup = this%dis%bot(iups)
780  if(this%dis%con%ihc(jjs01) == 2) then
781  topup = min(this%dis%top(n), this%dis%top(m))
782  botup = max(this%dis%bot(n), this%dis%bot(m))
783  endif
784  ! derivative term
785  derv = squadraticsaturationderivative(topup, botup, hnew(iups))
786  term = this%qsat(ii01) * derv
787  ! fill Jacobian for n being the upstream node
788  if (iups == n) then
789  ! fill in row of n
790  amat(idxglo(ii00)) = amat(idxglo(ii00)) + term
791  rhs(n) = rhs(n) + term * hnew(n)
792  ! fill in row of m
793  amat(idxglo(ii10)) = amat(idxglo(ii10)) - term
794  rhs(m) = rhs(m) - term * hnew(n)
795  ! fill Jacobian for m being the upstream node
796  else
797  ! fill in row of n
798  amat(idxglo(ii01)) = amat(idxglo(ii01)) + term
799  rhs(n) = rhs(n) + term * hnew(m)
800  ! fill in row of m
801  amat(idxglo(ii11)) = amat(idxglo(ii11)) - term
802  rhs(m) = rhs(m) - term * hnew(m)
803  end if
804  enddo
805  enddo
806  !
807  ! -- Return
808  return
809  end subroutine xt3d_fn

1138  subroutine xt3d_iallpc(this)
1139 ! ******************************************************************************
1140 ! xt3d_iallpc -- Allocate and populate iallpc array. Set lamatsaved.
1141 ! ******************************************************************************
1142 !
1143 ! SPECIFICATIONS:
1144 ! ------------------------------------------------------------------------------
1145  ! -- modules
1146  use memorymanagermodule, only: mem_allocate, mem_deallocate
1147  ! -- dummy
1148  class(xt3dtype) :: this
1149  ! -- local
1150  integer(I4B) :: n, m, mm, il0, il1
1151  integer(I4B) :: nnbr0, nnbr1
1152  integer(I4B),dimension(this%nbrmax) :: inbr0, inbr1
1153 ! ------------------------------------------------------------------------------
1154  !
1155  if(this%ixt3d == 2) then
1156  this%lamatsaved = .false.
1157  call mem_allocate(this%iallpc, 0, 'IALLPC', this%memoryPath)
1158  else
1159  !
1160  ! -- allocate memory for iallpc and initialize to 1
1161  call mem_allocate(this%iallpc, this%dis%nodes, 'IALLPC', this%memoryPath)
1162  do n = 1, this%dis%nodes
1163  this%iallpc(n) = 1
1164  end do
1165  !
1166  ! -- Go through cells and connections and set iallpc to 0 if any
1167  ! connected cell is not confined
1168  do n = 1, this%dis%nodes
1169  if (this%icelltype(n) /= 0) then
1170  this%iallpc(n) = 0
1171  cycle
1172  end if
1173  nnbr0 = this%dis%con%ia(n+1) - this%dis%con%ia(n) - 1
1174  call this%xt3d_load_inbr(n, nnbr0, inbr0)
1175  do il0 = 1,nnbr0
1176  m = inbr0(il0)
1177  if (m.lt.n) cycle
1178  if (this%icelltype(m) /= 0) then
1179  this%iallpc(n) = 0
1180  this%iallpc(m) = 0
1181  cycle
1182  end if
1183  nnbr1 = this%dis%con%ia(m+1) - this%dis%con%ia(m) - 1
1184  call this%xt3d_load_inbr(m, nnbr1, inbr1)
1185  do il1 = 1,nnbr1
1186  mm = inbr1(il1)
1187  if (this%icelltype(mm) /= 0) then
1188  this%iallpc(n) = 0
1189  this%iallpc(m) = 0
1190  this%iallpc(mm) = 0
1191  end if
1192  enddo
1193  enddo
1194  enddo
1195  !
1196  ! -- If any cells have all permanently confined connections then
1197  ! performance can be improved by precalculating coefficients
1198  ! so set lamatsaved to true.
1199  this%lamatsaved = .false.
1200  do n = 1, this%dis%nodes
1201  if (this%iallpc(n) == 1) then
1202  this%lamatsaved = .true.
1203  exit
1204  end if
1205  enddo
1206  end if
1207  !
1208  if (.not.this%lamatsaved) then
1209  call mem_deallocate(this%iallpc)
1210  call mem_allocate(this%iallpc, 0, 'IALLPC', this%memoryPath)
1211  end if

1213  ! -- Return
1214  return
1215  end subroutine xt3d_iallpc

1254  subroutine xt3d_load(this, nodes, n, nnbr, inbr, vc, vn, dl, dln, ck, allhc)
1255 ! ******************************************************************************
1256 ! xt3d_load -- Load conductivity and connection info for a cell into arrays
1257 ! used by XT3D.
1258 ! ******************************************************************************
1259 !
1260 ! SPECIFICATIONS:
1261 ! ------------------------------------------------------------------------------
1262  ! -- module
1263  use constantsmodule, only: dzero, dhalf, done
1264  ! -- dummy
1265  class(xt3dtype) :: this
1266  logical :: allhc
1267  integer(I4B),intent(in) :: nodes
1268  integer(I4B) :: n, nnbr
1269  integer(I4B),dimension(this%nbrmax) :: inbr
1270  real(DP),dimension(this%nbrmax,3) :: vc, vn
1271  real(DP),dimension(this%nbrmax) :: dl, dln
1272  real(DP),dimension(3,3) :: ck
1273  ! -- local
1274  integer(I4B) :: il, ii, jj, jjs
1275  integer(I4B) :: ihcnjj
1276  real(DP) :: satn, satjj
1277  real(DP) :: cl1njj, cl2njj, dltot, ooclsum
1278 ! ------------------------------------------------------------------------------
1279  !
1280  ! -- Set conductivity tensor for cell.
1281  ck = dzero
1282  ck(1,1) = this%k11(n)
1283  if(this%ik22 == 0) then
1284  ck(2,2) = ck(1,1)
1285  else
1286  ck(2,2) = this%k22(n)
1287  end if
1288  if(this%ik33 == 0) then
1289  ck(3,3) = ck(1,1)
1290  else
1291  ck(3,3) = this%k33(n)
1292  endif
1293  call this%xt3d_fillrmatck(n)
1294  ck = matmul(this%rmatck, ck)
1295  ck = matmul(ck, transpose(this%rmatck))
1296  !
1297  ! -- Load neighbors of cell. Set cell numbers for inactive
1298  ! -- neighbors to zero so xt3d knows to ignore them. Compute
1299  ! -- direct connection lengths from perpendicular connection
1300  ! -- lengths. Also determine if all active connections are
1301  ! -- horizontal.
1302  allhc = .true.
1303  do il = 1,nnbr
1304  ii = il + this%dis%con%ia(n)
1305  jj = this%dis%con%ja(ii)
1306  jjs = this%dis%con%jas(ii)
1307  if (this%ibound(jj).ne.0) then
1308  inbr(il) = jj
1309  satn = this%sat(n)
1310  satjj = this%sat(jj)
1311  ! -- DISV and DIS
1312  ihcnjj = this%dis%con%ihc(jjs)
1313  call this%dis%connection_normal(n, jj, ihcnjj, &
1314  vn(il, 1), vn(il, 2), vn(il, 3), ii)
1315  call this%dis%connection_vector(n, jj, this%nozee, satn, satjj, &
1316  ihcnjj, vc(il, 1), vc(il, 2), vc(il, 3), dltot)
1317  if(jj > n) then
1318  cl1njj = this%dis%con%cl1(jjs)
1319  cl2njj = this%dis%con%cl2(jjs)
1320  else
1321  cl1njj = this%dis%con%cl2(jjs)
1322  cl2njj = this%dis%con%cl1(jjs)
1323  endif
1324  ooclsum = 1d0/(cl1njj + cl2njj)
1325  dl(il) = dltot*cl1njj*ooclsum
1326  dln(il) = dltot*cl2njj*ooclsum
1327  if (this%dis%con%ihc(jjs).eq.0) allhc = .false.
1328  else
1329  inbr(il) = 0
1330  end if
1331  end do
1332  !
1333  return
1334  end subroutine xt3d_load

1367  subroutine xt3d_areas(this, nodes, n, m, jjs01, lsat, ar01, ar10, hnew)
1368 ! ******************************************************************************
1369 ! xt3d_areas -- Compute interfacial areas.
1370 ! ******************************************************************************
1371 !
1372 ! SPECIFICATIONS:
1373 ! ------------------------------------------------------------------------------
1374  ! -- module
1375  ! -- dummy
1376  class(xt3dtype) :: this
1377  logical :: lsat
1378  integer(I4B) :: nodes, n, m, jjs01
1379  real(DP) :: ar01, ar10
1380  real(DP), intent(inout), dimension(:), optional :: hnew
1381  ! -- local
1382  real(DP) :: topn, botn, topm, botm, thksatn, thksatm
1383  real(DP) :: sill_top, sill_bot, tpn, tpm
1384  real(DP) :: satups
1385 ! ------------------------------------------------------------------------------
1386  !
1387  ! -- Compute area depending on connection type
1388  if (this%dis%con%ihc(jjs01) == 0) then
1389  !
1390  ! -- vertical connection
1391  ar01 = this%dis%con%hwva(jjs01)
1392  ar10 = ar01
1393  else if (this%inewton /= 0) then
1394  !
1395  ! -- If Newton horizontal connection
1396  if (lsat) then
1397  !
1398  ! -- lsat true means use full saturation
1399  topn = this%dis%top(n)
1400  botn = this%dis%bot(n)
1401  topm = this%dis%top(m)
1402  botm = this%dis%bot(m)
1403  thksatn = topn - botn
1404  thksatm = topm - botm
1405  if (this%dis%con%ihc(jjs01).eq.2) then
1406  ! -- vertically staggered
1407  sill_top = min(topn, topm)
1408  sill_bot = max(botn, botm)
1409  tpn = botn + thksatn
1410  tpm = botm + thksatm
1411  thksatn = max(min(tpn, sill_top) - sill_bot, dzero)
1412  thksatm = max(min(tpm, sill_top) - sill_bot, dzero)
1413  end if
1414  ar01 = this%dis%con%hwva(jjs01) * dhalf * (thksatn + thksatm)
1415  else
1416  ! -- If Newton and lsat=.false., it is assumed that the fully saturated
1417  ! -- areas have already been calculated and are being passed in through
1418  ! -- ar01 and ar10. The actual areas are obtained simply by scaling by
1419  ! -- the upstream saturation.
1420  if (hnew(m) < hnew(n)) then
1421  satups = this%sat(n)
1422  else
1423  satups = this%sat(m)
1424  end if
1425  ar01 = ar01*satups
1426  end if
1427  ar10 = ar01
1428  else
1429  !
1430  ! -- Non-Newton horizontal connection
1431  topn = this%dis%top(n)
1432  botn = this%dis%bot(n)
1433  topm = this%dis%top(m)
1434  botm = this%dis%bot(m)
1435  thksatn = topn - botn
1436  thksatm = topm - botm
1437  if (.not. lsat) then
1438  thksatn = this%sat(n) * thksatn
1439  thksatm = this%sat(m) * thksatm
1440  end if
1441  if (this%dis%con%ihc(jjs01) == 2) then
1442  ! -- vertically staggered
1443  sill_top = min(topn, topm)
1444  sill_bot = max(botn, botm)
1445  tpn = botn + thksatn
1446  tpm = botm + thksatm
1447  thksatn = max(min(tpn, sill_top) - sill_bot, dzero)
1448  thksatm = max(min(tpm, sill_top) - sill_bot, dzero)
1449  end if
1450  ar01 = this%dis%con%hwva(jjs01) * thksatn
1451  ar10 = this%dis%con%hwva(jjs01) * thksatm
1452  endif
1453  !
1454  return
1455  end subroutine xt3d_areas
