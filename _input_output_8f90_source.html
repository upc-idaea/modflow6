733  SUBROUTINE urword(LINE,ICOL,ISTART,ISTOP,NCODE,N,R,IOUT,IN)
734 !C ******************************************************************
735 !C ROUTINE TO EXTRACT A WORD FROM A LINE OF TEXT, AND OPTIONALLY
736 !C CONVERT THE WORD TO A NUMBER.
737 !C ISTART AND ISTOP WILL BE RETURNED WITH THE STARTING AND
738 !C ENDING CHARACTER POSITIONS OF THE WORD.
739 !C THE LAST CHARACTER IN THE LINE IS SET TO BLANK SO THAT IF ANY
740 !C PROBLEMS OCCUR WITH FINDING A WORD, ISTART AND ISTOP WILL
741 !C POINT TO THIS BLANK CHARACTER. THUS, A WORD WILL ALWAYS BE
742 !C RETURNED UNLESS THERE IS A NUMERIC CONVERSION ERROR. BE SURE
743 !C THAT THE LAST CHARACTER IN LINE IS NOT AN IMPORTANT CHARACTER
744 !C BECAUSE IT WILL ALWAYS BE SET TO BLANK.
745 !C A WORD STARTS WITH THE FIRST CHARACTER THAT IS NOT A SPACE OR
746 !C COMMA, AND ENDS WHEN A SUBSEQUENT CHARACTER THAT IS A SPACE
747 !C OR COMMA. NOTE THAT THESE PARSING RULES DO NOT TREAT TWO
748 !C COMMAS SEPARATED BY ONE OR MORE SPACES AS A NULL WORD.
749 !C FOR A WORD THAT BEGINS WITH "'" OR '"', THE WORD STARTS WITH
750 !C THE CHARACTER AFTER THE QUOTE AND ENDS WITH THE CHARACTER
751 !C PRECEDING A SUBSEQUENT QUOTE. THUS, A QUOTED WORD CAN
752 !C INCLUDE SPACES AND COMMAS. THE QUOTED WORD CANNOT CONTAIN
753 !C A QUOTE CHARACTER OF THE SAME TYPE WITHIN THE WORD BUT
754 !C CAN CONTAIN A DIFFERENT QUOTE CHARACTER. FOR EXAMPLE,
755 !C "WORD'S" OR 'WORD"S'.
756 !C IF NCODE IS 1, THE WORD IS CONVERTED TO UPPER CASE.
757 !C IF NCODE IS 2, THE WORD IS CONVERTED TO AN INTEGER.
758 !C IF NCODE IS 3, THE WORD IS CONVERTED TO A REAL NUMBER.
759 !C NUMBER CONVERSION ERROR IS WRITTEN TO UNIT IOUT IF IOUT IS
760 !C POSITIVE; ERROR IS WRITTEN TO DEFAULT OUTPUT IF IOUT IS 0;
761 !C NO ERROR MESSAGE IS WRITTEN IF IOUT IS NEGATIVE.
762 !C ******************************************************************
763 !C
764 !C SPECIFICATIONS:
765 !C ------------------------------------------------------------------
766  integer(I4B), intent(inout) :: n
767  real(dp),intent(inout) :: r
768  CHARACTER(len=*) line
769  CHARACTER(len=20) string
770  CHARACTER(len=30) rw
771  CHARACTER(len=1) tab
772  CHARACTER(len=1) charend
773  character(len=200) :: msg
774  character(len=LINELENGTH) :: msg_line

776  tab=char(9)
777 !C
778 !C1------Set last char in LINE to blank and set ISTART and ISTOP to point
779 !C1------to this blank as a default situation when no word is found. If
780 !C1------starting location in LINE is out of bounds, do not look for a
781 !C1------word.
782  linlen=len(line)
783  line(linlen:linlen)=' '
784  istart=linlen
785  istop=linlen
786  linlen=linlen-1
787  IF(icol.LT.1 .OR. icol.GT.linlen) GO TO 100
788 !C
789 !C2------Find start of word, which is indicated by first character that
790 !C2------is not a blank, a comma, or a tab.
791  DO 10 i=icol,linlen
792  IF(line(i:i).NE.' ' .AND. line(i:i).NE.',' &
793  & .AND. line(i:i).NE.tab) GO TO 20
794 10 CONTINUE
795  icol=linlen+1
796  GO TO 100
797 !C
798 !C3------Found start of word. Look for end.
799 !C3A-----When word is quoted, only a quote can terminate it.
800 !C-------SEARCH FOR A SINGLE (CHAR(39)) OR DOUBLE (CHAR(34)) QUOTE
801 20 IF(line(i:i).EQ.char(34) .OR. line(i:i).EQ.char(39)) THEN
802  IF (line(i:i).EQ.char(34)) THEN
803  charend = char(34)
804  ELSE
805  charend = char(39)
806  END IF
807  i=i+1
808  IF(i.LE.linlen) THEN
809  DO 25 j=i,linlen
810  IF(line(j:j).EQ.charend) GO TO 40
811 25 CONTINUE
812  END IF
813 !C
814 !C3B-----When word is not quoted, space, comma, or tab will terminate.
815  ELSE
816  DO 30 j=i,linlen
817  IF(line(j:j).EQ.' ' .OR. line(j:j).EQ.',' &
818  & .OR. line(j:j).EQ.tab) GO TO 40
819 30 CONTINUE
820  END IF
821 !C
822 !C3C-----End of line without finding end of word; set end of word to
823 !C3C-----end of line.
824  j=linlen+1
825 !C
826 !C4------Found end of word; set J to point to last character in WORD and
827 !C-------set ICOL to point to location for scanning for another word.
828 40 icol=j+1
829  j=j-1
830  IF(j.LT.i) GO TO 100
831  istart=i
832  istop=j
833 !C
834 !C5------Convert word to upper case and RETURN if NCODE is 1.
835  IF(ncode.EQ.1) THEN
836  idiff=ichar('a')-ichar('A')
837  DO 50 k=istart,istop
838  IF(line(k:k).GE.'a' .AND. line(k:k).LE.'z') &
839  & line(k:k)=char(ichar(line(k:k))-idiff)
840 50 CONTINUE
841  RETURN
842  END IF
843 !C
844 !C6------Convert word to a number if requested.
845 100 IF(ncode.EQ.2 .OR. ncode.EQ.3) THEN
846  rw=' '
847  l=30-istop+istart
848  IF(l.LT.1) GO TO 200
849  rw(l:30)=line(istart:istop)
850  IF(ncode.EQ.2) READ(rw,'(I30)',err=200) n
851  IF(ncode.EQ.3) READ(rw,'(F30.0)',err=200) r
852  END IF
853  RETURN
854 !C
855 !C7------Number conversion error.
856 200 IF(ncode.EQ.3) THEN
857  string= 'A REAL NUMBER'
858  l=13
859  ELSE
860  string= 'AN INTEGER'
861  l=10
862  END IF
863 !C
864 !C7A-----If output unit is negative, set last character of string to 'E'.
865  IF(iout.LT.0) THEN
866  n=0
867  r=0.
868  line(linlen+1:linlen+1)='E'
869  RETURN
870 !C
871 !C7B-----If output unit is positive; write a message to output unit.
872  ELSE IF(iout.GT.0) THEN
873  IF(in.GT.0) THEN
874  write(msg_line,201) in,line(istart:istop),string(1:l)
875  ELSE
876  WRITE(msg_line,202) line(istart:istop),string(1:l)
877  END IF
878  call sim_message(msg_line, iunit=iout, skipbefore=1)
879  call sim_message(line, iunit=iout, fmt='(1x,a)')
880 201 FORMAT(1x,'FILE UNIT ',i4,' : ERROR CONVERTING "',a, &
881  & '" TO ',a,' IN LINE:')
882 202 FORMAT(1x,'KEYBOARD INPUT : ERROR CONVERTING "',a, &
883  '" TO ',a,' IN LINE:')
884 !C
885 !C7C-----If output unit is 0; write a message to default output.
886  ELSE
887  IF(in.GT.0) THEN
888  write(msg_line,201) in,line(istart:istop),string(1:l)
889  ELSE
890  WRITE(msg_line,202) line(istart:istop),string(1:l)
891  END IF
892  call sim_message(msg_line, iunit=iout, skipbefore=1)
893  call sim_message(line, iunit=iout, fmt='(1x,a)')
894  END IF
895 !C
896 !C7D-----STOP after storing error message.
897  call lowcase(string)
898  if (in > 0) then
899  write(msg,205) in,line(istart:istop),trim(string)
900  else
901  write(msg,207) line(istart:istop),trim(string)
902  endif
903 205 format('File unit ',i0,': Error converting "',a, &
904  & '" to ',a,' in following line:')
905 207 format('Keyboard input: Error converting "',a, &
906  & '" to ',a,' in following line:')
907  call store_error(msg)
908  call store_error(trim(line))
909  call store_error_unit(in)
910  !
911  END SUBROUTINE urword
