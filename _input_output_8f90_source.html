746  SUBROUTINE urword(LINE,ICOL,ISTART,ISTOP,NCODE,N,R,IOUT,IN)
747 !C ******************************************************************
748 !C ROUTINE TO EXTRACT A WORD FROM A LINE OF TEXT, AND OPTIONALLY
749 !C CONVERT THE WORD TO A NUMBER.
750 !C ISTART AND ISTOP WILL BE RETURNED WITH THE STARTING AND
751 !C ENDING CHARACTER POSITIONS OF THE WORD.
752 !C THE LAST CHARACTER IN THE LINE IS SET TO BLANK SO THAT IF ANY
753 !C PROBLEMS OCCUR WITH FINDING A WORD, ISTART AND ISTOP WILL
754 !C POINT TO THIS BLANK CHARACTER. THUS, A WORD WILL ALWAYS BE
755 !C RETURNED UNLESS THERE IS A NUMERIC CONVERSION ERROR. BE SURE
756 !C THAT THE LAST CHARACTER IN LINE IS NOT AN IMPORTANT CHARACTER
757 !C BECAUSE IT WILL ALWAYS BE SET TO BLANK.
758 !C A WORD STARTS WITH THE FIRST CHARACTER THAT IS NOT A SPACE OR
759 !C COMMA, AND ENDS WHEN A SUBSEQUENT CHARACTER THAT IS A SPACE
760 !C OR COMMA. NOTE THAT THESE PARSING RULES DO NOT TREAT TWO
761 !C COMMAS SEPARATED BY ONE OR MORE SPACES AS A NULL WORD.
762 !C FOR A WORD THAT BEGINS WITH "'", THE WORD STARTS WITH THE
763 !C CHARACTER AFTER THE QUOTE AND ENDS WITH THE CHARACTER
764 !C PRECEDING A SUBSEQUENT QUOTE. THUS, A QUOTED WORD CAN
765 !C INCLUDE SPACES AND COMMAS. THE QUOTED WORD CANNOT CONTAIN
766 !C A QUOTE CHARACTER.
767 !C IF NCODE IS 1, THE WORD IS CONVERTED TO UPPER CASE.
768 !C IF NCODE IS 2, THE WORD IS CONVERTED TO AN INTEGER.
769 !C IF NCODE IS 3, THE WORD IS CONVERTED TO A REAL NUMBER.
770 !C NUMBER CONVERSION ERROR IS WRITTEN TO UNIT IOUT IF IOUT IS
771 !C POSITIVE; ERROR IS WRITTEN TO DEFAULT OUTPUT IF IOUT IS 0;
772 !C NO ERROR MESSAGE IS WRITTEN IF IOUT IS NEGATIVE.
773 !C ******************************************************************
774 !C
775 !C SPECIFICATIONS:
776 !C ------------------------------------------------------------------
777  integer(I4B), intent(inout) :: n
778  real(dp),intent(inout) :: r
779  CHARACTER(len=*) line
780  CHARACTER(len=20) string
781  CHARACTER(len=30) rw
782  CHARACTER(len=1) tab
783  character(len=200) :: msg
784  character(len=LINELENGTH) :: msg_line
785 !C ------------------------------------------------------------------
786  tab=char(9)
787 !C
788 !C1------Set last char in LINE to blank and set ISTART and ISTOP to point
789 !C1------to this blank as a default situation when no word is found. If
790 !C1------starting location in LINE is out of bounds, do not look for a
791 !C1------word.
792  linlen=len(line)
793  line(linlen:linlen)=' '
794  istart=linlen
795  istop=linlen
796  linlen=linlen-1
797  IF(icol.LT.1 .OR. icol.GT.linlen) GO TO 100
798 !C
799 !C2------Find start of word, which is indicated by first character that
800 !C2------is not a blank, a comma, or a tab.
801  DO 10 i=icol,linlen
802  IF(line(i:i).NE.' ' .AND. line(i:i).NE.',' &
803  & .AND. line(i:i).NE.tab) GO TO 20
804 10 CONTINUE
805  icol=linlen+1
806  GO TO 100
807 !C
808 !C3------Found start of word. Look for end.
809 !C3A-----When word is quoted, only a quote can terminate it.
810 20 IF(line(i:i).EQ.'''') THEN
811  i=i+1
812  IF(i.LE.linlen) THEN
813  DO 25 j=i,linlen
814  IF(line(j:j).EQ.'''') GO TO 40
815 25 CONTINUE
816  END IF
817 !C
818 !C3B-----When word is not quoted, space, comma, or tab will terminate.
819  ELSE
820  DO 30 j=i,linlen
821  IF(line(j:j).EQ.' ' .OR. line(j:j).EQ.',' &
822  & .OR. line(j:j).EQ.tab) GO TO 40
823 30 CONTINUE
824  END IF
825 !C
826 !C3C-----End of line without finding end of word; set end of word to
827 !C3C-----end of line.
828  j=linlen+1
829 !C
830 !C4------Found end of word; set J to point to last character in WORD and
831 !C-------set ICOL to point to location for scanning for another word.
832 40 icol=j+1
833  j=j-1
834  IF(j.LT.i) GO TO 100
835  istart=i
836  istop=j
837 !C
838 !C5------Convert word to upper case and RETURN if NCODE is 1.
839  IF(ncode.EQ.1) THEN
840  idiff=ichar('a')-ichar('A')
841  DO 50 k=istart,istop
842  IF(line(k:k).GE.'a' .AND. line(k:k).LE.'z') &
843  & line(k:k)=char(ichar(line(k:k))-idiff)
844 50 CONTINUE
845  RETURN
846  END IF
847 !C
848 !C6------Convert word to a number if requested.
849 100 IF(ncode.EQ.2 .OR. ncode.EQ.3) THEN
850  rw=' '
851  l=30-istop+istart
852  IF(l.LT.1) GO TO 200
853  rw(l:30)=line(istart:istop)
854  IF(ncode.EQ.2) READ(rw,'(I30)',err=200) n
855  IF(ncode.EQ.3) READ(rw,'(F30.0)',err=200) r
856  END IF
857  RETURN
858 !C
859 !C7------Number conversion error.
860 200 IF(ncode.EQ.3) THEN
861  string= 'A REAL NUMBER'
862  l=13
863  ELSE
864  string= 'AN INTEGER'
865  l=10
866  END IF
867 !C
868 !C7A-----If output unit is negative, set last character of string to 'E'.
869  IF(iout.LT.0) THEN
870  n=0
871  r=0.
872  line(linlen+1:linlen+1)='E'
873  RETURN
874 !C
875 !C7B-----If output unit is positive; write a message to output unit.
876  ELSE IF(iout.GT.0) THEN
877  IF(in.GT.0) THEN
878  write(msg_line,201) in,line(istart:istop),string(1:l)
879  ELSE
880  WRITE(msg_line,202) line(istart:istop),string(1:l)
881  END IF
882  call sim_message(msg_line, iunit=iout, skipbefore=1)
883  call sim_message(line, iunit=iout, fmt='(1x,a)')
884 201 FORMAT(1x,'FILE UNIT ',i4,' : ERROR CONVERTING "',a, &
885  & '" TO ',a,' IN LINE:')
886 202 FORMAT(1x,'KEYBOARD INPUT : ERROR CONVERTING "',a, &
887  '" TO ',a,' IN LINE:')
888 !C
889 !C7C-----If output unit is 0; write a message to default output.
890  ELSE
891  IF(in.GT.0) THEN
892  write(msg_line,201) in,line(istart:istop),string(1:l)
893  ELSE
894  WRITE(msg_line,202) line(istart:istop),string(1:l)
895  END IF
896  call sim_message(msg_line, iunit=iout, skipbefore=1)
897  call sim_message(line, iunit=iout, fmt='(1x,a)')
898  END IF
899 !C
900 !C7D-----STOP after storing error message.
901  call lowcase(string)
902  if (in > 0) then
903  write(msg,205) in,line(istart:istop),trim(string)
904  else
905  write(msg,207) line(istart:istop),trim(string)
906  endif
907 205 format('File unit ',i0,': Error converting "',a, &
908  & '" to ',a,' in following line:')
909 207 format('Keyboard input: Error converting "',a, &
910  & '" to ',a,' in following line:')
911  call store_error(msg)
912  call store_error(trim(line))
913  call store_error_unit(in)
914  call ustop()
915  !
916  END SUBROUTINE urword
