1 ! HashTableType implements a hash table for storing integers,
2 ! for use as an index for an array that could contain
3 ! any data type. This HashTableModule was designed using the
4 ! dictionary implementation by Arjen Markus of the Flibs
5 ! collection of Fortran utilities. This hash table works
6 ! like a dictionary. There can be n number of character
7 ! strings and each string will be assigned a unique number
8 ! between 1 and n, allowing an efficient way to store a
9 ! unique integer index with a character string.

107 ! ******************************************************************************

109 ! ******************************************************************************

112 ! ------------------------------------------------------------------------------

120 ! ------------------------------------------------------------------------------

123  ! get an unassociated elem that corresponds to this key

266  function hashfunc(key) result(ihash)
267 ! ******************************************************************************
268 ! hashfunc -- function to convert key into an integer hash number
269 ! ******************************************************************************
270 !
271 ! SPECIFICATIONS:
272 ! ------------------------------------------------------------------------------
273  ! -- dummy
274  character(len=*), intent(in) :: key
275  ! -- local
276  integer(I4B) :: ihash
277  integer(I4B) :: i
278 ! ------------------------------------------------------------------------------
279  ihash = 0
280  do i = 1,len(key)
281  ihash = modulo( multiplier * ihash + ichar(key(i:i)), hash_size)
282  enddo
283  ihash = 1 + modulo(ihash - 1, hash_size)
284  !
285  ! -- return
286  return
287  end function hashfunc
