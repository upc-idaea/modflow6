1 ! This module defines type ObsType, which is the highest-level
2 ! derived type for implementing observations. All objects derived from
3 ! NumericalModelType or BndType already contain an ObsType member.
4 !
5 ! Examples:
6 ! NumericalModelType.obs
7 ! BndType.obs
8 !
9 ! Similarly, an ObsType member could be added to, say,
10 ! NumericalExchangeType or any other type that has DF, AR, RP, AD, BD, and OT
11 ! routines.
12 !
13 ! ------------------------------------------------------------------------------
14 ! IMPLEMENTATION OF OBSERVATIONS IN A MODEL OR PACKAGE
15 !
16 ! For simple boundary packages like RIV and DRN, only steps 1-6 are
17 ! needed. For models and advanced packages like MAW and SFR, additional
18 ! steps are needed.
19 !
20 ! 1. (package only) Override BndType.bnd_obs_supported to return true.
21 ! bnd_obs_supported is called from various places in code.
22 !
23 ! 2. (optional) Write a subroutine that implements abstract interface
24 ! ObserveModule.ProcessIdSub. (Not needed if IDstring, which identifies
25 ! location in model to be observed, is either a single node number or
26 ! a single {lay, row, col} set of indices).
27 !
28 ! Examples:
29 ! gwf_process_head_drawdown_obs_id, gwf_process_intercell_obs_id
30 !
31 ! A package can allow IDstring to be a boundary name.
32 ! Example: ObsModule.DefaultObsIdProcessor
33 !
34 ! 3. Override BndType.bnd_df_obs() to define string(s) to be
35 ! recognized as observation type(s) and (optional) assign ProcessIdPtr
36 ! (not needed if IDstring is either a node number or a {lay, row, col}
37 ! set of indices).
38 !
39 ! Examples: gwf_df_obs, drn_df_obs
40 !
41 ! When boundary names are allowed and developer wants simulated value
42 ! to be cumulative (flow, for example) if user specifies multiple
43 ! boundaries with the same BOUNDNAME, in bnd_df_obs call to
44 ! ObsPackage.StoreObsType, provide cumulative argument as true.
45 ! Otherwise, simulated values are not cumulative.
46 !
47 ! 4. In DF routine: Call bnd_df_obs
48 !
49 ! 5. In AR routine: Call ObsType.obs_ar. This reads the OBS input
50 ! file.
51 ! Example (gwf_ar): call this%obs%obs_ar()
52 ! Example (lak_ar): call this%obs%obs_ar()
53 !
54 ! 6. Override BndType.bnd_rp_obs for any package that needs to
55 ! check user input or process observation input in any special way.
56 ! If no special processing is needed, BndType.bnd_rp_obs can
57 ! be used. This routine also expands the ObserveType%indxbnds array for
58 ! each observation in a package. ObserveType%indxbnds is used to sum
59 ! simulated values from multiple boundaries when BOUNDNAMES is used.
60 ! Equivalent routine may or may not be needed for model observations.
61 ! If needed, call it from bottom of RP routine.
62 !
63 ! Examples:
64 ! BndType.bnd_rp_obs, which is called from gwf_rp
65 !
66 ! 7. In AD routine: Call ObsType.obs_ad
67 ! Example: gwf_ad
68 !
69 ! 8. Write a *_bd_obs routine. This is the routine that actually
70 ! calculates the simulated value for each observation type supported
71 ! by the model/package. Call *_bd_obs from the bottom of the
72 ! _bd routine.
73 ! *_bd_obs needs to:
74 ! Call ObsType.obs_bd_clear
75 ! For each observation:
76 ! Calculate the simulated value
77 ! Call ObsType.SaveOneSimval
78 ! Examples: gwf_bd_obs, maw_bd_obs, lak_bd_obs
79 !
80 ! 9. In BD routine:
81 ! Call BndType.bnd_bd_obs
82 ! Examples: BndType.bnd_bd calls bnd_bd_obs
83 ! GwfModelType.gwf_bd calls gwf_bd_obs
84 ! MawType.maw_bd calls maw_bd_obs
85 ! LakType.lak_bd calls lak_bd_obs
86 !
87 ! 10. Ensure that ObsType.obs_ot is called. For packages, obs_ot is called
88 ! from the model _ot procedure. The model _ot procedure should also call
89 ! obs_ot for its own observations. Do not call obs_ot from a package _ot
90 ! procedure because the package _ot procedure may not be called, depending
91 ! on Output Control settings (ibudfl).
92 !
93 ! Note: BndType.bnd_ot_obs calls:
94 ! ObsType.obs_ot
95 !
96 ! Note: ObsType.obs_ot calls:
97 ! store_all_simvals
98 ! write_continuous_simvals
99 ! obsOutputList.WriteOutputLines
100 !
101 ! BINARY OUTPUT:
102 !
103 ! When observation-output files are written, the user has the option to have
104 ! output written to a binary file. Binary obs output files start with a
105 ! 100-byte header structured as follows:
106 !
107 ! bytes 1-4 (ascii): Observation type contained in file; options are:
108 ! "sngl" -- Single observations
109 ! "cont" -- Continuous observations
110 ! byte 5: blank
111 ! bytes 6-11 (ascii): Precision of all floating-point values; options are:
112 ! "single" -- Single precision
113 ! "double" -- Double precision
114 ! bytes 12-15 (ascii): LENOBSNAME (integer; length of observation names,
115 ! in bytes)
116 ! bytes 16-100: blank
117 !
118 ! IN A FILE OF CONTINUOUS OBSERVATIONS:
119 !
120 ! The 100-byte header is followed by:
121 ! NOBS (4-byte integer) -- Number of observations.
122 ! NOBS repetitions of OBSNAME (ascii, LENOBSNAME bytes each).
123 ! Any number of repetitions of:
124 ! TIME SIMVAL-1 SIMVAL-2 ... SIMVAL-NOBS (floating point)
125 !
126 !-------------------------------------------------------------------------------

238 ! ******************************************************************************
239 ! DefaultObsIdProcessor -- Process IDstring provided for each observation. The
240 ! IDstring identifies the location in the model of the node(s) or feature(s)
241 ! where the simulated value is to be extracted and recorded.
242 ! Subroutine: (1) interprets the IDstring
243 ! (2) stores the location of interest in the ObserveType object that
244 ! contains information about the observation
245 ! ******************************************************************************
246 !
247 ! SPECIFICATIONS:
248 ! ------------------------------------------------------------------------------
249  ! -- dummy
250  type(observetype), intent(inout) :: obsrv

252  integer(I4B), intent(in) :: inunitobs
253  integer(I4B), intent(in) :: iout
254  ! -- local
255  integer(I4B) :: n
256  integer(I4B) :: icol, istart, istop
257  character(len=LINELENGTH) :: ermsg, strng
258  logical :: flag_string
259 ! ------------------------------------------------------------------------------
260  !

262  strng = obsrv%IDstring

264  flag_string = .true. ! Allow strng to contain a boundary name
265  !

267  iout, strng, flag_string)
268  !

273  ! name (already converted to uppercase)
274  obsrv%FeatureName = strng(istart:istop)

276  ! so assign NodeNumber as a value that indicates observation
277  ! is for a named boundary or group of boundaries.

279  else
280  ermsg = 'Error reading data from ID string'
281  call store_error(ermsg)
282  call store_error_unit(inunitobs)
283  call ustop()
284  endif
285  !
286  return
287  end subroutine defaultobsidprocessor

606  subroutine obs_ar2(this, dis)
607 ! ******************************************************************************
608 ! obs_ar2
609 ! -- Call procedure provided by package to interpret IDstring and
610 ! store required data.
611 ! ******************************************************************************
612 !
613 ! SPECIFICATIONS:
614 ! ------------------------------------------------------------------------------
615  ! -- dummy
616  class(obstype), intent(inout) :: this
617  class(disbasetype) :: dis
618  ! -- local
619  integer(I4B) :: i
620  type(obsdatatype), pointer :: obsDat => null()
621  character(len=LENOBSTYPE) :: obsTypeID
622  class(observetype), pointer :: obsrv => null()

624  !

627  call this%get_obs_array(this%npakobs, this%pakobs)
628  !

630  obsrv => this%pakobs(i)%obsrv
631  ! -- Call IDstring processor procedure provided by package
632  obstypeid = obsrv%ObsTypeId
633  obsdat => this%get_obs_datum(obstypeid)

636  this%inUnitObs, this%iout)
637  else
638  call defaultobsidprocessor(obsrv, dis, &
639  this%inUnitObs, this%iout)
640  endif
641  enddo
642  !

644  call store_error_unit(this%inunitobs)
645  call ustop()
646  end if
647  !

817 ! ******************************************************************************
818 ! build_headers
819 ! -- Build headers for CSV-formatted and unformatted continuous-observation
820 ! output files and write them to those files.
821 ! Each formatted header will have the form: "time,obsname-1,obsname-2, ..."
822 ! ******************************************************************************
823 !
824 ! SPECIFICATIONS:
825 ! ------------------------------------------------------------------------------
826  ! -- module
827  use iso_fortran_env, only: int32
828  ! -- dummy
829  class(obstype), target :: this
830  ! -- local
831  integer(I4B) :: i
832  integer(I4B) :: ii
833  integer(I4B) :: idx
834  integer(I4B) :: iu
835  integer(I4B) :: num
836  integer(int32) :: nobs
837  character(len=4) :: clenobsname
838  type(observetype), pointer :: obsrv => null()
839  type(obsoutputtype), pointer :: obsOutput => null()

842  ! -- Cycle through ObsOutputList to write headers
843  ! to formatted and unformatted file(s).
844  idx = 1
845  num = this%obsOutputList%Count()
846  all_obsfiles: do i = 1, num
847  obsoutput => this%obsOutputList%Get(i)
848  nobs = obsoutput%nobs
849  iu = obsoutput%nunit
850  !
851  ! -- write header information to the formatted file
852  if (obsoutput%FormattedOutput) then
853  write(iu, '(a)', advance='NO') 'time'

862  write(iu) 'cont double'
863  end if

865  write(clenobsname,'(i4)') lenobsname
866  write(iu) clenobsname
867  ! -- write blanks to complete 100-byte header
868  do ii = 16, 100
869  write(iu) ' '
870  end do
871  ! -- write NOBS
872  write(iu) nobs
873  end if
874  !
875  ! -- write observation name
876  obsfile: do ii = 1, nobs
877  obsrv => this%get_obs(idx)
878  if (obsoutput%FormattedOutput) then
879  write(iu, '(a,a)', advance='NO') ',', trim(obsrv%Name)

881  ! -- terminate the line on the last observation in file
882  if (ii == nobs) then
883  write(iu, '(a)', advance='YES') ''
884  end if
885  else
886  write(iu) obsrv%Name
887  end if
888  idx = idx + 1
889  end do obsfile
890  end do all_obsfiles
891  !
892  ! -- return
893  return
894  end subroutine build_headers
