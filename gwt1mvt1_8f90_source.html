180  subroutine mvt_fc(this, nodes, cold, nja, njasln, amatsln, idxglo, cnew, rhs)
181 ! ******************************************************************************
182 ! mvt_fc -- Calculate coefficients and fill amat and rhs
183 !
184 ! The mvt package adds the mass flow rate to the provider qmfrommvr
185 ! array. The advanced packages know enough to subract any mass that is
186 ! leaving, so the mvt just adds mass coming in from elsewhere. Because the
187 ! movers change change by stress period, their solute effects must be
188 ! added to the right-hand side of the gwt matrix equations.
189 ! ******************************************************************************
190 !
191 ! SPECIFICATIONS:
192 ! ------------------------------------------------------------------------------
193  ! -- modules
194  ! -- dummy
195  class(gwtmvttype) :: this
196  integer, intent(in) :: nodes
197  real(DP), intent(in), dimension(nodes) :: cold
198  integer(I4B), intent(in) :: nja
199  integer(I4B), intent(in) :: njasln
200  real(DP), dimension(njasln), intent(inout) :: amatsln
201  integer(I4B), intent(in), dimension(nja) :: idxglo
202  real(DP), intent(in), dimension(nodes) :: cnew
203  real(DP), intent(inout), dimension(nodes) :: rhs
204  ! -- local
205  integer(I4B) :: i, n
206  integer(I4B) :: id1, id2, nlist
207  integer(I4B) :: ipr, irc
208  integer(I4B) :: igwtnode
209  integer(I4B) :: nbudterm
210  real(DP) :: q, cp
211  real(DP), dimension(:), pointer :: concpak
212 ! ------------------------------------------------------------------------------
213  !
214  ! -- initialize the mass flow into advanced package from the mover
215  do i = 1, this%fmi%nflowpack
216  if (this%fmi%iatp(i) == 0) cycle
217  do n = 1, size(this%fmi%datp(i)%qmfrommvr)
218  this%fmi%datp(i)%qmfrommvr(n) = dzero
219  end do
220  end do
221  !
222  ! -- Add mover QC terms to the receiver packages
223  nbudterm = this%fmi%mvrbudobj%nbudterm
224  do i = 1, nbudterm
225  nlist = this%fmi%mvrbudobj%budterm(i)%nlist
226  if (nlist > 0) then
227  call this%fmi%get_package_index(this%fmi%mvrbudobj%budterm(i)%text2id1, ipr)
228  call this%fmi%get_package_index(this%fmi%mvrbudobj%budterm(i)%text2id2, irc)
229  if (this%fmi%iatp(ipr) /= 0) concpak => this%fmi%datp(ipr)%concpack
230  do n = 1, nlist
231  !
232  ! -- lak/sfr/maw/uzf id1 (provider) and id2 (receiver)
233  id1 = this%fmi%mvrbudobj%budterm(i)%id1(n)
234  id2 = this%fmi%mvrbudobj%budterm(i)%id2(n)
235  !
236  ! -- Obtain mover flow rate from the mover flow budget object
237  q = this%fmi%mvrbudobj%budterm(i)%flow(n)
238  !
239  ! -- Assign concentration of the provider
240  cp = dzero
241  if (this%fmi%iatp(ipr) /= 0) then
242  !
243  ! -- Provider package is being represented by an APT (SFT, LKT, MWT, UZT)
244  ! so set the concentration to the simulated concentation of APT
245  cp = concpak(id1)
246  else
247  !
248  ! -- Provider is a regular stress package (WEL, DRN, RIV, etc.) or the
249  ! provider is an advanced stress package but is not represented with
250  ! SFT, LKT, MWT, or UZT, so use the GWT cell concentration
251  igwtnode = this%fmi%gwfpackages(ipr)%nodelist(id1)
252  cp = cnew(igwtnode)
253  end if
254  !
255  ! -- add the mover rate times the provider concentration into the receiver
256  ! make sure these are accumulated since multiple providers can move
257  ! water into the same receiver
258  if (this%fmi%iatp(irc) /= 0) then
259  this%fmi%datp(irc)%qmfrommvr(id2) = this%fmi%datp(irc)%qmfrommvr(id2) - q * cp
260  end if
261  end do
262  end if
263  end do
264  !
265  ! -- Return
266  return
267  end subroutine mvt_fc

380  subroutine mvt_ot_bdsummary(this, ibudfl)
381 ! ******************************************************************************
382 ! mvt_ot_bdsummary -- Write mover budget to listing file
383 ! ******************************************************************************
384 !
385 ! SPECIFICATIONS:
386 ! ------------------------------------------------------------------------------
387  ! -- modules
388  use tdismodule, only: kstp, kper, delt, totim
389  use arrayhandlersmodule, only: ifind, expandarray
390  ! -- dummy
391  class(gwtmvttype) :: this
392  integer(I4B), intent(in) :: ibudfl
393  ! -- locals
394  integer(I4B) :: i, j, n
395  real(DP), allocatable, dimension(:) :: ratin, ratout
396 ! ------------------------------------------------------------------------------
397  !
398  ! -- Allocate and initialize ratin/ratout
399  allocate(ratin(this%maxpackages), ratout(this%maxpackages))
400  do j = 1, this%maxpackages
401  ratin(j) = dzero
402  ratout(j) = dzero
403  enddo
404  !
405  ! -- Accumulate the rates
406  do i = 1, this%maxpackages
407 
408  do j = 1, this%budobj%nbudterm
409 
410  do n = 1, this%budobj%budterm(j)%nlist
411 
412  !
413  ! -- provider is inflow to mover
414  if(this%paknames(i) == this%budobj%budterm(j)%text2id1) then
415  ratin(i) = ratin(i) + this%budobj%budterm(j)%flow(n)
416  endif
417  !
418  ! -- receiver is outflow from mover
419  if(this%paknames(i) == this%budobj%budterm(j)%text2id2) then
420  ratout(i) = ratout(i) + this%budobj%budterm(j)%flow(n)
421  endif
422 
423  end do
424 
425  end do
426 
427  end do
428 
429  !
430  ! -- Send rates to budget object
431  call this%budget%reset()
432  do j = 1, this%maxpackages
433  call this%budget%addentry(ratin(j), ratout(j), delt, this%paknames(j))
434  enddo
435  !
436  ! -- Write the budget
437  if (ibudfl /= 0) then
438  call this%budget%budget_ot(kstp, kper, this%iout)
439  end if
440  !
441  ! -- Write budget csv
442  call this%budget%writecsv(totim)
443  !
444  ! -- Deallocate
445  deallocate(ratin, ratout)
446  !
447  ! -- Output mvr budget
448  ! Not using budobj write_table here because it would result
449  ! in a table that has one entry. A custom table looks
450  ! better here with a row for each package.
451  !call this%budobj%write_budtable(kstp, kper, this%iout)
452  !
453  ! -- Return
454  return
455  end subroutine mvt_ot_bdsummary

538  subroutine read_options(this)
539 ! ******************************************************************************
540 ! read_options -- Read Options
541 ! ******************************************************************************
542 !
543 ! SPECIFICATIONS:
544 ! ------------------------------------------------------------------------------
545  ! -- modules
546  use openspecmodule, only: access, form
547  use inputoutputmodule, only: getunit, openfile
548  ! -- dummy
549  class(gwtmvttype) :: this
550  ! -- local
551  character(len=LINELENGTH) :: errmsg, keyword
552  character(len=MAXCHARLEN) :: fname
553  integer(I4B) :: ierr
554  logical :: isfound, endOfBlock
555  character(len=*),parameter :: fmtflow = &
556  "(4x, a, 1x, a, 1x, ' WILL BE SAVED TO FILE: ', a, /4x, 'OPENED ON UNIT: ', I0)"
557  character(len=*),parameter :: fmtflow2 = &
558  "(4x, 'FLOWS WILL BE SAVED TO BUDGET FILE')"
559 ! ------------------------------------------------------------------------------
560  !
561  ! -- get options block
562  call this%parser%GetBlock('OPTIONS', isfound, ierr, blockrequired=.false., &
563  supportopenclose=.true.)
564  !
565  ! -- parse options block if detected
566  if (isfound) then
567  write(this%iout,'(1x,a)')'PROCESSING MVT OPTIONS'
568  do
569  call this%parser%GetNextLine(endofblock)
570  if (endofblock) exit
571  call this%parser%GetStringCaps(keyword)
572  select case (keyword)
573  case ('SAVE_FLOWS')
574  this%ipakcb = -1
575  write(this%iout, fmtflow2)
576  case ('PRINT_INPUT')
577  this%iprpak = 1
578  write(this%iout,'(4x,a)') 'MVT INPUT WILL BE PRINTED.'
579  case ('PRINT_FLOWS')
580  this%iprflow = 1
581  write(this%iout,'(4x,a)') &
582  'MVT FLOWS WILL BE PRINTED TO LISTING FILE.'
583  case('BUDGET')
584  call this%parser%GetStringCaps(keyword)
585  if (keyword == 'FILEOUT') then
586  call this%parser%GetString(fname)
587  this%ibudgetout = getunit()
588  call openfile(this%ibudgetout, this%iout, fname, 'DATA(BINARY)', &
589  form, access, 'REPLACE')
590  write(this%iout,fmtflow) 'MVT', 'BUDGET', fname, this%ibudgetout
591  else
592  call store_error('OPTIONAL BUDGET KEYWORD MUST BE FOLLOWED BY FILEOUT')
593  end if
594  case('BUDGETCSV')
595  call this%parser%GetStringCaps(keyword)
596  if (keyword == 'FILEOUT') then
597  call this%parser%GetString(fname)
598  this%ibudcsv = getunit()
599  call openfile(this%ibudcsv, this%iout, fname, 'CSV', &

603  call store_error('OPTIONAL BUDGETCSV KEYWORD MUST BE FOLLOWED BY &
604  &FILEOUT')
605  end if
606  case default
607  write(errmsg,'(4x,a,a)')'***ERROR. UNKNOWN MVT OPTION: ', &
608  trim(keyword)
609  call store_error(errmsg)
610  call this%parser%StoreErrorUnit()
611  end select
612  end do
613  write(this%iout,'(1x,a)') 'END OF MVT OPTIONS'
614  end if
615  !

674  subroutine mvt_fill_budobj(this, cnew)
675 ! ******************************************************************************
676 ! mvt_fill_budobj -- copy flow terms into this%budobj
677 ! ******************************************************************************
678 !
679 ! SPECIFICATIONS:
680 ! ------------------------------------------------------------------------------
681  ! -- modules
682  ! -- dummy
683  class(gwtmvttype) :: this
684  real(DP), intent(in), dimension(:) :: cnew
685  ! -- local
686  integer(I4B) :: nbudterm
687  integer(I4B) :: nlist
688  integer(I4B) :: ipr
689  integer(I4B) :: irc
690  integer(I4B) :: i
691  integer(I4B) :: j
692  integer(I4B) :: n1, n2
693  integer(I4B) :: igwtnode
694  real(DP) :: cp
695  real(DP) :: q
696  real(DP) :: rate
697  ! -- formats
698 ! -----------------------------------------------------------------------------
699  !
700  ! -- Go through the water mover budget terms and set up the transport
701  ! mover budget terms
702  nbudterm = this%fmi%mvrbudobj%nbudterm
703  do i = 1, nbudterm
704  nlist = this%fmi%mvrbudobj%budterm(i)%nlist
705  call this%fmi%get_package_index(this%fmi%mvrbudobj%budterm(i)%text2id1, ipr)
706  call this%fmi%get_package_index(this%fmi%mvrbudobj%budterm(i)%text2id2, irc)
707  call this%budobj%budterm(i)%reset(nlist)
708  do j = 1, nlist
709  n1 = this%fmi%mvrbudobj%budterm(i)%id1(j)
710  n2 = this%fmi%mvrbudobj%budterm(i)%id2(j)
711  q = this%fmi%mvrbudobj%budterm(i)%flow(j)
712  cp = dzero
713  if (this%fmi%iatp(ipr) /= 0) then
714  cp = this%fmi%datp(ipr)%concpack(n1)
715  else
716  ! -- Must be a regular stress package
717  igwtnode = this%fmi%gwfpackages(ipr)%nodelist(n1)
718  cp = cnew(igwtnode)
719  end if
720  !
721  ! -- Calculate solute mover rate
722  rate = dzero
723  if (this%fmi%iatp(irc) /= 0) then
724  rate = -q * cp
725  end if
726  !
727  ! -- add the rate to the budterm
728  call this%budobj%budterm(i)%update_term(n1, n2, rate)
729  end do
730  end do
731  !
732  ! --Terms are filled, now accumulate them for this time step
733  call this%budobj%accumulate_terms()

735  ! -- return
736  return
