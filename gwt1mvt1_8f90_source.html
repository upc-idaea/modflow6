178  subroutine mvt_fc(this, nodes, cold, nja, njasln, amatsln, idxglo, cnew, rhs)
179 ! ******************************************************************************
180 ! mvt_fc -- Calculate coefficients and fill amat and rhs
181 !
182 ! The mvt package adds the mass flow rate to the provider qmfrommvr
183 ! array. The advanced packages know enough to subract any mass that is
184 ! leaving, so the mvt just adds mass coming in from elsewhere. Because the
185 ! movers change change by stress period, their solute effects must be
186 ! added to the right-hand side of the gwt matrix equations.
187 ! ******************************************************************************
188 !
189 ! SPECIFICATIONS:
190 ! ------------------------------------------------------------------------------
191  ! -- modules
192  ! -- dummy
193  class(gwtmvttype) :: this
194  integer, intent(in) :: nodes
195  real(DP), intent(in), dimension(nodes) :: cold
196  integer(I4B), intent(in) :: nja
197  integer(I4B), intent(in) :: njasln
198  real(DP), dimension(njasln), intent(inout) :: amatsln
199  integer(I4B), intent(in), dimension(nja) :: idxglo
200  real(DP), intent(in), dimension(nodes) :: cnew
201  real(DP), intent(inout), dimension(nodes) :: rhs
202  ! -- local
203  integer(I4B) :: i, n
204  integer(I4B) :: id1, id2, nlist
205  integer(I4B) :: ipr, irc
206  integer(I4B) :: igwtnode
207  integer(I4B) :: nbudterm
208  real(DP) :: q, cp
209  real(DP), dimension(:), pointer :: concpak
210 ! ------------------------------------------------------------------------------
211  !
212  ! -- initialize the mass flow into advanced package from the mover
213  do i = 1, this%fmi%nflowpack
214  if (this%fmi%iatp(i) == 0) cycle
215  do n = 1, size(this%fmi%datp(i)%qmfrommvr)
216  this%fmi%datp(i)%qmfrommvr(n) = dzero
217  end do
218  end do
219  !
220  ! -- Add mover QC terms to the receiver packages
221  nbudterm = this%fmi%mvrbudobj%nbudterm
222  do i = 1, nbudterm
223  nlist = this%fmi%mvrbudobj%budterm(i)%nlist
224  if (nlist > 0) then
225  call this%fmi%get_package_index(this%fmi%mvrbudobj%budterm(i)%text2id1, ipr)
226  call this%fmi%get_package_index(this%fmi%mvrbudobj%budterm(i)%text2id2, irc)
227  if (this%fmi%iatp(ipr) /= 0) concpak => this%fmi%datp(ipr)%concpack
228  do n = 1, nlist
229  !
230  ! -- lak/sfr/maw/uzf id1 (provider) and id2 (receiver)
231  id1 = this%fmi%mvrbudobj%budterm(i)%id1(n)
232  id2 = this%fmi%mvrbudobj%budterm(i)%id2(n)
233  !
234  ! -- Obtain mover flow rate from the mover flow budget object
235  q = this%fmi%mvrbudobj%budterm(i)%flow(n)
236  !
237  ! -- Assign concentration of the provider
238  cp = dzero
239  if (this%fmi%iatp(ipr) /= 0) then
240  !
241  ! -- Provider package is being represented by an APT (SFT, LKT, MWT, UZT)
242  ! so set the concentration to the simulated concentation of APT
243  cp = concpak(id1)
244  else
245  !
246  ! -- Provider is a regular stress package (WEL, DRN, RIV, etc.) or the
247  ! provider is an advanced stress package but is not represented with
248  ! SFT, LKT, MWT, or UZT, so use the GWT cell concentration
249  igwtnode = this%fmi%gwfpackages(ipr)%nodelist(id1)
250  cp = cnew(igwtnode)
251  end if
252  !
253  ! -- add the mover rate times the provider concentration into the receiver
254  ! make sure these are accumulated since multiple providers can move
255  ! water into the same receiver
256  if (this%fmi%iatp(irc) /= 0) then
257  this%fmi%datp(irc)%qmfrommvr(id2) = this%fmi%datp(irc)%qmfrommvr(id2) - q * cp
258  end if
259  end do
260  end if
261  end do
262  !
263  ! -- Return
264  return
265  end subroutine mvt_fc

378  subroutine mvt_ot_bdsummary(this)
379 ! ******************************************************************************
380 ! mvt_ot_bdsummary -- Write mover budget to listing file
381 ! ******************************************************************************
382 !
383 ! SPECIFICATIONS:
384 ! ------------------------------------------------------------------------------
385  ! -- modules
386  use tdismodule, only: kstp, kper, delt
387  use arrayhandlersmodule, only: ifind, expandarray
388  ! -- dummy
389  class(gwtmvttype) :: this
390  ! -- locals
391  integer(I4B) :: i, j, n
392  real(DP), allocatable, dimension(:) :: ratin, ratout
393 ! ------------------------------------------------------------------------------
394  !
395  ! -- Allocate and initialize ratin/ratout
396  allocate(ratin(this%maxpackages), ratout(this%maxpackages))
397  do j = 1, this%maxpackages
398  ratin(j) = dzero
399  ratout(j) = dzero
400  enddo
401  !
402  ! -- Accumulate the rates
403  do i = 1, this%maxpackages
404 
405  do j = 1, this%budobj%nbudterm
406 
407  do n = 1, this%budobj%budterm(j)%nlist
408 
409  !
410  ! -- provider is inflow to mover
411  if(this%paknames(i) == this%budobj%budterm(j)%text2id1) then
412  ratin(i) = ratin(i) + this%budobj%budterm(j)%flow(n)
413  endif
414  !
415  ! -- receiver is outflow from mover
416  if(this%paknames(i) == this%budobj%budterm(j)%text2id2) then
417  ratout(i) = ratout(i) + this%budobj%budterm(j)%flow(n)
418  endif
419 
420  end do
421 
422  end do
423 
424  end do
425 
426  !
427  ! -- Send rates to budget object
428  call this%budget%reset()
429  do j = 1, this%maxpackages
430  call this%budget%addentry(ratin(j), ratout(j), delt, this%paknames(j))
431  enddo
432  !
433  ! -- Write the budget
434  call this%budget%budget_ot(kstp, kper, this%iout)
435  !
436  ! -- Deallocate
437  deallocate(ratin, ratout)
438  !
439  ! -- Output mvr budget
440  ! Not using budobj write_table here because it would result
441  ! in a table that has one entry. A custom table looks
442  ! better here with a row for each package.
