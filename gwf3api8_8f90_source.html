127  !> @ brief Fill A and r for the package
128  !!
129  !! Fill the coefficient matrix and right-hand side with the USR
130  !! package terms.
131  !!
132  !<
133  subroutine api_fc(this, rhs, ia, idxglo, amatsln)
134  ! -- dummy variables
135  class(apitype) :: this
136  real(DP), dimension(:), intent(inout) :: rhs !< right-hand side vector
137  integer(I4B), dimension(:), intent(in) :: ia !< pointer to the rows in A matrix
138  integer(I4B), dimension(:), intent(in) :: idxglo !< position of entries in A matrix
139  real(DP), dimension(:), intent(inout) :: amatsln !< A matrix for solution
140  ! -- local variables
141  integer(I4B) :: i
142  integer(I4B) :: n
143  integer(I4B) :: ipos
144  real(DP) :: qusr
145  !
146  ! -- pakmvrobj fc
147  if (this%imover == 1) then
148  call this%pakmvrobj%fc()
149  end if
150  !
151  ! -- Copy package rhs and hcof into solution rhs and amat
152  do i = 1, this%nbound
153  n = this%nodelist(i)
154  rhs(n) = rhs(n) + this%rhs(i)
155  ipos = ia(n)
156  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + this%hcof(i)
157  !
158  ! -- If mover is active and this boundary is discharging,
159  ! store available water (as positive value).
160  qusr = this%rhs(i) - this%hcof(i)*this%xnew(n)
161  if (this%imover == 1 .and. qusr > dzero) then
162  call this%pakmvrobj%accumulate_qformvr(i, qusr)
163  end if
164  end do
165  !
166  ! -- return
167  return
168  end subroutine api_fc
