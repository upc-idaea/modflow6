201  !> @ brief Formulate the package hcof and rhs terms.
202  !!
203  !! Formulate the hcof and rhs terms for the WEL package that will be
204  !! added to the coefficient matrix and right-hand side vector.
205  !!
206  !<
207  subroutine wel_cf(this, reset_mover)
208  ! -- dummy variables
209  class(weltype) :: this !< WelType object
210  logical, intent(in), optional :: reset_mover !< boolean for resetting mover
211  ! -- local variables
212  integer(I4B) :: i, node, ict
213  real(DP) :: qmult
214  real(DP) :: q
215  real(DP) :: tp
216  real(DP) :: bt
217  real(DP) :: thick
218  logical :: lrm
219  !
220  ! -- Return if no wells
221  if(this%nbound == 0) return
222  !
223  ! -- pakmvrobj cf
224  lrm = .true.
225  if (present(reset_mover)) lrm = reset_mover
226  if(this%imover == 1 .and. lrm) then
227  call this%pakmvrobj%cf()
228  endif
229  !
230  ! -- Calculate hcof and rhs for each well entry
231  do i = 1, this%nbound
232  node = this%nodelist(i)
233  this%hcof(i) = dzero
234  if(this%ibound(node) <= 0) then
235  this%rhs(i) = dzero
236  cycle
237  end if
238  q = this%bound(1,i)
239  if (this%iflowred /= 0 .and. q < dzero) then
240  ict = this%icelltype(node)
241  if (ict /= 0) then
242  tp = this%dis%top(node)
243  bt = this%dis%bot(node)
244  thick = tp - bt
245  tp = bt + this%flowred * thick
246  qmult = sqsaturation(tp, bt, this%xnew(node))
247  q = q * qmult
248  endif
249  end if
250  this%rhs(i) = -q
251  enddo
252  !
253  return
254  end subroutine wel_cf

256  !> @ brief Copy hcof and rhs terms into solution.
257  !!
258  !! Add the hcof and rhs terms for the WEL package to the
259  !! coefficient matrix and right-hand side vector.
260  !!
261  !<
262  subroutine wel_fc(this, rhs, ia, idxglo, amatsln)
263  ! -- dummy variables
264  class(weltype) :: this !< WelType object
265  real(DP), dimension(:), intent(inout) :: rhs !< right-hand side vector for model
266  integer(I4B), dimension(:), intent(in) :: ia !< solution CRS row pointers
267  integer(I4B), dimension(:), intent(in) :: idxglo !< mapping vector for model (local) to solution (global)
268  real(DP), dimension(:), intent(inout) :: amatsln !< solution coefficient matrix
269  ! -- local variables
270  integer(I4B) :: i
271  integer(I4B) :: n
272  integer(I4B) :: ipos
273  !
274  ! -- pakmvrobj fc
275  if(this%imover == 1) then
276  call this%pakmvrobj%fc()
277  endif
278  !
279  ! -- Copy package rhs and hcof into solution rhs and amat
280  do i = 1, this%nbound
281  n = this%nodelist(i)
282  rhs(n) = rhs(n) + this%rhs(i)
283  ipos = ia(n)
284  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + this%hcof(i)
285  !
286  ! -- If mover is active and this well is discharging,
287  ! store available water (as positive value).
288  if(this%imover == 1 .and. this%rhs(i) > dzero) then
289  call this%pakmvrobj%accumulate_qformvr(i, this%rhs(i))
290  endif
291  enddo
292  !
293  ! -- return
294  return
295  end subroutine wel_fc

297  !> @ brief Add Newton-Raphson terms for package into solution.
298  !!
299  !! Calculate and add the Newton-Raphson terms for the WEL package to the
300  !! coefficient matrix and right-hand side vector.
301  !!
302  !<
303  subroutine wel_fn(this, rhs, ia, idxglo, amatsln)
304  ! -- dummy variables
305  class(weltype) :: this !< WelType object
306  real(DP), dimension(:), intent(inout) :: rhs !< right-hand side vector for model
307  integer(I4B), dimension(:), intent(in) :: ia !< solution CRS row pointers
308  integer(I4B), dimension(:), intent(in) :: idxglo !< mapping vector for model (local) to solution (global)
309  real(DP), dimension(:), intent(inout) :: amatsln !< solution coefficient matrix
310  ! -- local variables
311  integer(I4B) :: i
312  integer(I4B) :: node
313  integer(I4B) :: ipos
314  integer(I4B) :: ict
315  real(DP) :: drterm
316  real(DP) :: q
317  real(DP) :: tp
318  real(DP) :: bt
319  real(DP) :: thick
320  !
321  ! -- Copy package rhs and hcof into solution rhs and amat
322  do i = 1, this%nbound
323  node = this%nodelist(i)
324  !
325  ! -- test if node is constant or inactive
326  if(this%ibound(node) <= 0) then
327  cycle
328  end if
329  !
330  ! -- well rate is possibly head dependent
331  ict = this%icelltype(node)
332  if (this%iflowred /= 0 .and. ict /= 0) then
333  ipos = ia(node)
334  q = -this%rhs(i)
335  if (q < dzero) then
336  ! -- calculate derivative for well
337  tp = this%dis%top(node)
338  bt = this%dis%bot(node)
339  thick = tp - bt
340  tp = bt + this%flowred * thick
341  drterm = sqsaturationderivative(tp, bt, this%xnew(node))
342  drterm = drterm * this%bound(1,i)
343  !--fill amat and rhs with newton-raphson terms
344  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + drterm
345  rhs(node) = rhs(node) + drterm * this%xnew(node)
346  end if
347  end if
348  end do
349  !
350  ! -- return
351  return
352  end subroutine wel_fn
