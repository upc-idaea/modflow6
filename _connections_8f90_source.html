219  data aname(6) /' ANGLDEGX'/

241  enddo
242  enddo

244  call this%parser%StoreErrorUnit()

246  endif
247  !

250  !
251  ! -- check for symmetry in ja (isym value of zero indicates there is no
252  ! symmetric connection
253  do n = 1, this%nodes
254  do ii = this%ia(n), this%ia(n + 1) - 1
255  m = this%ja(ii)
256  if(this%isym(ii) == 0) then
257  write(errmsg, fmtsymerrja) trim(adjustl(aname(2))), ii, n, m
258  call store_error(errmsg)
259  endif
260  enddo
261  enddo
262  if(count_errors() > 0) then
263  call this%parser%StoreErrorUnit()
264  call ustop()
265  endif
266  !
267  ! -- Fill the jas array, which maps any connection to upper triangle
268  call filljas(this%nodes, this%nja, this%ia, this%ja, this%isym, this%jas)
269  !
270  ! -- Put into symmetric array
271  do n = 1, this%nodes
272  do ii = this%ia(n) + 1, this%ia(n + 1) - 1
273  m = this%ja(ii)
274  if(ihctemp(ii) /= ihctemp(this%isym(ii))) then
275  write(errmsg, fmtsymerr) trim(adjustl(aname(3))), ii, this%isym(ii), &
276  ihctemp(ii), ihctemp(this%isym(ii)), n, m
277  call store_error(errmsg)
278  else

280  endif
281  enddo
282  enddo

284  call this%parser%StoreErrorUnit()
285  call ustop()
286  endif
287  !
288  ! -- Put cl12 into symmetric arrays cl1 and cl2
289  do n = 1, this%nodes
290  do ii = this%ia(n) + 1, this%ia(n + 1) - 1
291  m = this%ja(ii)
292  if(m > n) then
293  this%cl1(this%jas(ii)) = cl12temp(ii)
294  elseif(n > m) then
295  this%cl2(this%jas(ii)) = cl12temp(ii)
296  endif
297  enddo
298  enddo
299  !
300  ! -- Put HWVA into symmetric array based on the value of IHC
301  ! IHC = 0, vertical connection, HWVA is vertical flow area
302  ! IHC = 1, horizontal connection, HWVA is the width perpendicular to
303  ! flow
304  ! IHC = 2, horizontal connection for a vertically staggered grid.
305  ! HWVA is the width perpendicular to flow.
306  do n = 1, this%nodes
307  do ii = this%ia(n) + 1, this%ia(n + 1) - 1
308  m = this%ja(ii)
309  if(hwvatemp(ii) /= hwvatemp(this%isym(ii))) then
310  write(errmsg, fmtsymerr) trim(adjustl(aname(5))), ii, this%isym(ii), &
311  hwvatemp(ii), hwvatemp(this%isym(ii)), n, m
312  call store_error(errmsg)
313  endif
314  if(ihctemp(ii) < 0 .or. ihctemp(ii) > 2) then
315  write(errmsg, fmtihcerr) ihctemp(ii)
316  call store_error(errmsg)
317  endif
318  this%hwva(this%jas(ii)) = hwvatemp(ii)
319  enddo
320  enddo
321  if(count_errors() > 0) then
322  call this%parser%StoreErrorUnit()
323  call ustop()
324  endif
325  !
326  ! -- Put anglextemp into this%anglex; store only upper triangle
327  if(this%ianglex /= 0) then
328  do n = 1, this%nodes
329  do ii = this%ia(n) + 1, this%ia(n + 1) - 1
330  m = this%ja(ii)
331  if(n > m) cycle
332  this%anglex(this%jas(ii)) = angldegx(ii) * dpio180
333  enddo
334  enddo
335  else
336  do n = 1, size(this%anglex)
337  this%anglex(n) = dnodata
338  enddo
339  endif
340  !
341  ! -- Return
342  return
343  end subroutine con_finalize

362  ! -- local

367  logical :: isfound, endOfBlock

369  logical,dimension(nname) :: lname
370  character(len=24),dimension(nname) :: aname(nname)
371  character(len=300) :: ermsg
372  ! -- formats
373  character(len=*),parameter :: fmtsymerr = &
374  &"(/,'Error in array: ',(a),/, &
375  &'Array is not symmetric in positions: ',2i9,/, &
376  &'Values in these positions are: ', 2(1pg15.6))"

378  &"(/,'IHC must be 0, 1, or 2. Found: ',i0)"

389  !
390  ! -- Allocate space for connection arrays
391  call this%allocate_arrays()
392  !
393  ! -- get connectiondata block
394  call this%parser%GetBlock('CONNECTIONDATA', isfound, ierr)
395  lname(:) = .false.
396  if(isfound) then
397  write(iout,'(1x,a)')'PROCESSING CONNECTIONDATA'
398  do
399  call this%parser%GetNextLine(endofblock)
400  if (endofblock) exit
401  call this%parser%GetStringCaps(keyword)
402  select case (keyword)
403  case ('IAC')
404  call readarray(this%parser%iuactive, this%ia, aname(1), 1, &
405  this%nodes, iout, 0)
406  lname(1) = .true.
407  case ('JA')
408  call readarray(this%parser%iuactive, this%ja, aname(2), 1, &
409  this%nja, iout, 0)
410  lname(2) = .true.
411  case default
412  write(ermsg,'(4x,a,a)')'ERROR. UNKNOWN CONNECTIONDATA TAG: ', &
413  trim(keyword)
414  call store_error(ermsg)
415  call this%parser%StoreErrorUnit()
416  call ustop()
417  end select
418  end do
419  write(iout,'(1x,a)')'END PROCESSING CONNECTIONDATA'
420  else
421  call store_error('ERROR. REQUIRED CONNECTIONDATA BLOCK NOT FOUND.')
422  call this%parser%StoreErrorUnit()
423  call ustop()
424  end if
425  !
426  ! -- verify all items were read
427  do n = 1, nname
428  if(.not. lname(n)) then
429  write(ermsg,'(1x,a,a)') &
430  'ERROR. REQUIRED INPUT WAS NOT SPECIFIED: ',aname(n)

433  enddo

436  call ustop()
437  endif

441  this%ia(n) = this%ia(n) + this%ia(n-1)
442  enddo
443  do n = this%nodes + 1, 2, -1
444  this%ia(n) = this%ia(n - 1) + 1
445  enddo
446  this%ia(1) = 1
447  !
448  ! -- Convert any negative ja numbers to positive
449  do ii = 1, this%nja
450  if(this%ja(ii) < 0) this%ja(ii) = -this%ja(ii)
451  enddo
452  !

456  this%jas)
457  !
458  ! -- check for symmetry in ja
459  do n = 1, this%nodes
460  do ii = this%ia(n), this%ia(n + 1) - 1
461  m = this%ja(ii)
462  if(n /= this%ja(this%isym(ii))) then
463  write(line, fmtsymerr) aname(2), ii, this%isym(ii)

466  call ustop()
467  endif
468  enddo
469  enddo
470  !

473  call this%parser%StoreErrorUnit()
474  call ustop()
475  endif
476  !
477  ! -- Return
478  return
479  end subroutine read_connectivity_from_block

511  subroutine disconnections(this, name_model, nodes, ncol, nrow, nlay, &
512  nrsize, delr, delc, top, bot, nodereduced, &
513  nodeuser)
514 ! ******************************************************************************
515 ! disconnections -- Construct the connectivity arrays for a structured
516 ! three-dimensional grid.
517 ! ******************************************************************************
518 !
519 ! SPECIFICATIONS:
520 ! ------------------------------------------------------------------------------
521  ! -- modules
522  use constantsmodule, only: dhalf, dzero, dthree, dtwo, dpi
523  use sparsemodule, only: sparsematrix
524  use inputoutputmodule, only: get_node
525  ! -- dummy
526  class(connectionstype) :: this
527  character(len=*), intent(in) :: name_model
528  integer(I4B), intent(in) :: nodes
529  integer(I4B), intent(in) :: ncol
530  integer(I4B), intent(in) :: nrow
531  integer(I4B), intent(in) :: nlay
532  integer(I4B), intent(in) :: nrsize
533  real(DP), dimension(ncol), intent(in) :: delr
534  real(DP), dimension(nrow), intent(in) :: delc
535  real(DP), dimension(nodes), intent(in) :: top
536  real(DP), dimension(nodes), intent(in) :: bot
537  integer(I4B), dimension(:), target, intent(in) :: nodereduced
538  integer(I4B), dimension(:), intent(in) :: nodeuser
539  ! -- local
540  integer(I4B), dimension(:, :, :), pointer :: nrdcd_ptr => null() !non-contiguous because is a slice
541  integer(I4B), dimension(:), allocatable :: rowmaxnnz
542  type(sparsematrix) :: sparse
543  integer(I4B) :: i, j, k, kk, ierror, isympos, nodesuser
544  integer(I4B) :: nr, mr
545 ! ------------------------------------------------------------------------------
546  !
547  ! -- Allocate scalars
548  call this%allocate_scalars(name_model)
549  !
550  ! -- Set scalars
551  this%nodes = nodes
552  this%ianglex = 1
553  !
554  ! -- Setup the sparse matrix object
555  allocate(rowmaxnnz(this%nodes))
556  do i = 1, this%nodes
557  rowmaxnnz(i) = 6
558  enddo
559  call sparse%init(this%nodes, this%nodes, rowmaxnnz)
560  !
561  ! -- Create a 3d pointer to nodereduced for easier processing
562  if(nrsize /= 0) then
563  nrdcd_ptr(1:ncol, 1:nrow, 1:nlay) => nodereduced
564  endif
565  !
566  ! -- Add connections to sparse
567  do k = 1, nlay
568  do i = 1, nrow
569  do j = 1, ncol
570  !
571  ! -- Find the reduced node number and then cycle if the
572  ! node is always inactive
573  if(nrsize == 0) then
574  nr = get_node(k, i, j, nlay, nrow, ncol)
575  else
576  nr = nrdcd_ptr(j, i, k)
577  endif
578  if(nr <= 0) cycle
579  !
580  ! -- Process diagonal
581  call sparse%addconnection(nr, nr, 1)
582  !
583  ! -- Up direction
584  if(k > 1) then
585  do kk = k - 1, 1, -1
586  if(nrsize == 0) then
587  mr = get_node(kk, i, j, nlay, nrow, ncol)
588  else
589  mr = nrdcd_ptr(j, i, kk)
590  endif
591  if(mr >= 0) exit
592  enddo
593  if(mr > 0) then
594  call sparse%addconnection(nr, mr, 1)
595  endif
596  endif
597  !
598  ! -- Back direction
599  if(i > 1) then
600  if(nrsize == 0) then
601  mr = get_node(k, i-1, j, nlay, nrow, ncol)
602  else
603  mr = nrdcd_ptr(j, i-1, k)
604  endif
605  if(mr > 0) then
606  call sparse%addconnection(nr, mr, 1)
607  endif
608  endif
609  !
610  ! -- Left direction
611  if(j > 1) then
612  if(nrsize == 0) then
613  mr = get_node(k, i, j-1, nlay, nrow, ncol)
614  else
615  mr = nrdcd_ptr(j-1, i, k)
616  endif
617  if(mr > 0) then
618  call sparse%addconnection(nr, mr, 1)
619  endif
620  endif
621  !
622  ! -- Right direction
623  if(j < ncol) then
624  if(nrsize == 0) then
625  mr = get_node(k, i, j+1, nlay, nrow, ncol)
626  else
627  mr = nrdcd_ptr(j+1, i, k)
628  endif
629  if(mr > 0) then
630  call sparse%addconnection(nr, mr, 1)
631  endif
632  endif
633  !
634  ! -- Front direction
635  if(i < nrow) then !front
636  if(nrsize == 0) then
637  mr = get_node(k, i+1, j, nlay, nrow, ncol)
638  else
639  mr = nrdcd_ptr(j, i+1, k)
640  endif
641  if(mr > 0) then
642  call sparse%addconnection(nr, mr, 1)
643  endif
644  endif
645  !
646  ! -- Down direction
647  if(k < nlay) then
648  do kk = k + 1, nlay
649  if(nrsize == 0) then
650  mr = get_node(kk, i, j, nlay, nrow, ncol)
651  else
652  mr = nrdcd_ptr(j, i, kk)
653  endif
654  if(mr >= 0) exit
655  enddo
656  if(mr > 0) then
657  call sparse%addconnection(nr, mr, 1)
658  endif
659  endif
660  enddo
661  enddo
662  enddo
663  this%nja = sparse%nnz
664  this%njas = (this%nja - this%nodes) / 2
665  !
666  ! -- Allocate index arrays of size nja and symmetric arrays
667  call this%allocate_arrays()
668  !
669  ! -- Fill the IA and JA arrays from sparse, then destroy sparse
670  call sparse%filliaja(this%ia, this%ja, ierror)
671  call sparse%destroy()
672  !
673  ! -- fill the isym and jas arrays
674  call fillisym(this%nodes, this%nja, this%ia, this%ja, this%isym)
675  call filljas(this%nodes, this%nja, this%ia, this%ja, this%isym, this%jas)
676  !
677  ! -- Fill symmetric discretization arrays (ihc,cl1,cl2,hwva,anglex)
678  isympos = 1
679  do k = 1, nlay
680  do i = 1, nrow
681  do j = 1, ncol
682  !
683  ! -- cycle if node is always inactive
684  if(nrsize == 0) then
685  nr = get_node(k, i, j, nlay, nrow, ncol)
686  else
687  nr = nrdcd_ptr(j, i, k)
688  endif
689  if(nr <= 0) cycle
690  !
691  ! -- right connection
692  if(j < ncol) then
693  if(nrsize == 0) then
694  mr = get_node(k, i, j+1, nlay, nrow, ncol)
695  else
696  mr = nrdcd_ptr(j+1, i, k)
697  endif
698  if(mr > 0) then
699  this%ihc(isympos) = 1
700  this%cl1(isympos) = dhalf * delr(j)
701  this%cl2(isympos) = dhalf * delr(j + 1)
702  this%hwva(isympos) = delc(i)
703  this%anglex(isympos) = dzero
704  isympos = isympos + 1
705  endif
706  endif
707  !
708  ! -- front connection
709  if(i < nrow) then
710  if(nrsize == 0) then
711  mr = get_node(k, i+1, j, nlay, nrow, ncol)
712  else
713  mr = nrdcd_ptr(j, i+1, k)
714  endif
715  if(mr > 0) then
716  this%ihc(isympos) = 1
717  this%cl1(isympos) = dhalf * delc(i)
718  this%cl2(isympos) = dhalf * delc(i + 1)
719  this%hwva(isympos) = delr(j)
720  this%anglex(isympos) = dthree / dtwo * dpi
721  isympos = isympos + 1
722  endif
723  endif
724  !
725  ! -- down connection
726  if(k < nlay) then
727  do kk = k + 1, nlay
728  if(nrsize == 0) then
729  mr = get_node(kk, i, j, nlay, nrow, ncol)
730  else
731  mr = nrdcd_ptr(j, i, kk)
732  endif
733  if(mr >= 0) exit
734  enddo
735  if(mr > 0) then
736  this%ihc(isympos) = 0
737  this%cl1(isympos) = dhalf * (top(nr) - bot(nr))
738  this%cl2(isympos) = dhalf * (top(mr) - bot(mr))
739  this%hwva(isympos) = delr(j) * delc(i)
740  this%anglex(isympos) = dzero
741  isympos = isympos + 1
742  endif
743  endif
744  enddo
745  enddo
746  enddo
747  !
748  ! -- Deallocate temporary arrays
749  deallocate(rowmaxnnz)
750  !
751  ! -- If reduced system, then need to build iausr and jausr, otherwise point
752  ! them to ia and ja.
753  nodesuser = nlay * nrow * ncol
754  call this%iajausr(nrsize, nodesuser, nodereduced, nodeuser)
755  !
756  ! -- Return
757  return
758  end subroutine disconnections

788  real(DP), dimension(nodes), intent(in) :: top
789  real(DP), dimension(nodes), intent(in) :: bot
790  integer(I4B), dimension(:), intent(in) :: nodereduced
791  integer(I4B), dimension(:), intent(in) :: nodeuser

868  subroutine disuconnections(this, name_model, nodes, nodesuser, nrsize, &
869  nodereduced, nodeuser, iainp, jainp, &
870  ihcinp, cl12inp, hwvainp, angldegxinp)
871 ! ******************************************************************************
872 ! disuconnections -- Construct the connectivity arrays using disu
873 ! information. Grid may be reduced
874 ! ******************************************************************************
875 !
876 ! SPECIFICATIONS:
877 ! ------------------------------------------------------------------------------
878  ! -- modules
879  use constantsmodule, only: dhalf, dzero, dthree, dtwo, dpi
880  use sparsemodule, only: sparsematrix
881  use memorymanagermodule, only: mem_reallocate
882  ! -- dummy
883  class(connectionstype) :: this
884  character(len=*), intent(in) :: name_model
885  integer(I4B), intent(in) :: nodes
886  integer(I4B), intent(in) :: nodesuser
887  integer(I4B), intent(in) :: nrsize
888  integer(I4B), dimension(:), contiguous, intent(in) :: nodereduced
889  integer(I4B), dimension(:), contiguous, intent(in) :: nodeuser
890  integer(I4B), dimension(:), contiguous, intent(in) :: iainp
891  integer(I4B), dimension(:), contiguous, intent(in) :: jainp
892  integer(I4B), dimension(:), contiguous, intent(in) :: ihcinp
893  real(DP), dimension(:), contiguous, intent(in) :: cl12inp
894  real(DP), dimension(:), contiguous, intent(in) :: hwvainp
895  real(DP), dimension(:), contiguous, intent(in) :: angldegxinp
896  ! -- local
897  integer(I4B),dimension(:),allocatable :: ihctemp
898  real(DP),dimension(:),allocatable :: cl12temp
899  real(DP),dimension(:),allocatable :: hwvatemp
900  real(DP),dimension(:),allocatable :: angldegxtemp
901  integer(I4B) :: nr, nu, mr, mu, ipos, iposr, ierror
902  integer(I4B), dimension(:), allocatable :: rowmaxnnz
903  type(sparsematrix) :: sparse
904 ! ------------------------------------------------------------------------------
905  !
906  ! -- Allocate scalars
907  call this%allocate_scalars(name_model)
908  !
909  ! -- Set scalars
910  this%nodes = nodes
911  this%ianglex = 1
912  !
913  ! -- If not a reduced grid, then copy and finalize, otherwise more
914  ! processing is required
915  if (nrsize == 0) then
916  this%nodes = nodes
917  this%nja = size(jainp)
918  this%njas = (this%nja - this%nodes) / 2
919  call this%allocate_arrays()
920  do nu = 1, nodes + 1
921  this%ia(nu) = iainp(nu)
922  end do
923  do ipos = 1, this%nja
924  this%ja(ipos) = jainp(ipos)
925  end do
926  !
927  ! -- Call con_finalize to check inp arrays and push larger arrays
928  ! into compressed symmetric arrays
929  call this%con_finalize(ihcinp, cl12inp, hwvainp, angldegxinp)
930  !
931  else
932  ! -- reduced system requires more work
933  !
934  ! -- Setup the sparse matrix object
935  allocate(rowmaxnnz(this%nodes))
936  do nr = 1, this%nodes
937  nu = nodeuser(nr)
938  rowmaxnnz(nr) = iainp(nu + 1) - iainp(nu)
939  enddo
940  call sparse%init(this%nodes, this%nodes, rowmaxnnz)
941  !
942  ! -- go through user connectivity and create sparse
943  do nu = 1, nodesuser
944  nr = nodereduced(nu)
945  if (nr > 0) call sparse%addconnection(nr, nr, 1)
946  do ipos = iainp(nu) + 1, iainp(nu + 1) - 1
947  mu = jainp(ipos)
948  mr = nodereduced(mu)
949  if (nr < 1) cycle
950  if (mr < 1) cycle
951  call sparse%addconnection(nr, mr, 1)
952  enddo
953  enddo
954  this%nja = sparse%nnz
955  this%njas = (this%nja - this%nodes) / 2
956  !
957  ! -- Allocate index arrays of size nja and symmetric arrays
958  call this%allocate_arrays()
959  !
960  ! -- Fill the IA and JA arrays from sparse, then destroy sparse
961  call sparse%sort()
962  call sparse%filliaja(this%ia, this%ja, ierror)
963  call sparse%destroy()
964  deallocate(rowmaxnnz)
965  !
966  ! -- At this point, need to reduce ihc, cl12, hwva, and angldegx
967  allocate(ihctemp(this%nja))
968  allocate(cl12temp(this%nja))
969  allocate(hwvatemp(this%nja))
970  allocate(angldegxtemp(this%nja))
971  !
972  ! -- Compress user arrays into reduced arrays
973  iposr = 1
974  do nu = 1, nodesuser
975  nr = nodereduced(nu)
976  do ipos = iainp(nu), iainp(nu + 1) - 1
977  mu = jainp(ipos)
978  mr = nodereduced(mu)
979  if (nr < 1 .or. mr < 1) cycle
980  ihctemp(iposr) = ihcinp(ipos)
981  cl12temp(iposr) = cl12inp(ipos)
982  hwvatemp(iposr) = hwvainp(ipos)
983  angldegxtemp(iposr) = angldegxinp(ipos)
984  iposr = iposr + 1
985  end do
986  end do
987  !
988  ! -- call finalize
989  call this%con_finalize(ihctemp, cl12temp, hwvatemp, angldegxtemp)
990  !
991  ! -- deallocate temporary arrays
992  deallocate(ihctemp)
993  deallocate(cl12temp)
994  deallocate(hwvatemp)
995  deallocate(angldegxtemp)
996  end if
997  !
998  ! -- If reduced system, then need to build iausr and jausr, otherwise point
999  ! them to ia and ja.
1000  call this%iajausr(nrsize, nodesuser, nodereduced, nodeuser)
1001  !
1002  ! -- Return
1003  return
1004  end subroutine disuconnections

1006  subroutine iajausr(this, nrsize, nodesuser, nodereduced, nodeuser)
1007 ! ******************************************************************************
1008 ! iajausr -- Fill iausr and jausr if reduced grid, otherwise point them
1009 ! to ia and ja.
1010 ! ******************************************************************************
1011 !
1012 ! SPECIFICATIONS:
1013 ! ------------------------------------------------------------------------------
1014  ! -- modules
1015  use memorymanagermodule, only: mem_reallocate, mem_deallocate, mem_setptr
1016  ! -- dummy
1017  class(connectionstype) :: this
1018  integer(I4B), intent(in) :: nrsize
1019  integer(I4B), intent(in) :: nodesuser
1020  integer(I4B), dimension(:), intent(in) :: nodereduced
1021  integer(I4B), dimension(:), intent(in) :: nodeuser
1022  ! -- local
1023  integer(I4B) :: n, nr, ipos
1024 ! ------------------------------------------------------------------------------
1025  !
1026  ! -- If reduced system, then need to build iausr and jausr, otherwise point
1027  ! them to ia and ja.
1028  if(nrsize > 0) then
1029  !
1030  ! -- Create the iausr array of size nodesuser + 1. For excluded cells,
1031  ! iausr(n) and iausr(n + 1) should be equal to indicate no connections.
1032  call mem_reallocate(this%iausr, nodesuser+1, 'IAUSR', this%memoryPath)
1033  this%iausr(nodesuser + 1) = this%ia(this%nodes + 1)
1034  do n = nodesuser, 1, -1
1035  nr = nodereduced(n)
1036  if(nr < 1) then
1037  this%iausr(n) = this%iausr(n + 1)
1038  else
1039  this%iausr(n) = this%ia(nr)
1040  endif
1041  enddo
1042  !
1043  ! -- Create the jausr array, which is the same size as ja, but it
1044  ! contains user node numbers instead of reduced node numbers
1045  call mem_reallocate(this%jausr, this%nja, 'JAUSR', this%memoryPath)
1046  do ipos = 1, this%nja
1047  nr = this%ja(ipos)
1048  n = nodeuser(nr)
1049  this%jausr(ipos) = n
1050  enddo
1051  else
1052  ! -- iausr and jausr will be pointers
1053  call mem_deallocate(this%iausr)
1054  call mem_deallocate(this%jausr)
1055  call mem_setptr(this%iausr, 'IA', this%memoryPath)
1056  call mem_setptr(this%jausr, 'JA', this%memoryPath)
1057  endif
1058  !
1059  ! -- Return
1060  return
1061  end subroutine iajausr

1063  function getjaindex(this,node1,node2)
1064 ! ******************************************************************************
1065 ! Get the index in the JA array corresponding to the connection between
1066 ! two nodes of interest. Node1 is used as the index in the IA array, and
1067 ! IA(Node1) is the row index in the (nodes x nodes) matrix represented by
1068 ! the compressed sparse row format.
1069 !
1070 ! -1 is returned if either node number is invalid.
1071 ! 0 is returned if the two nodes are not connected.
1072 ! ******************************************************************************
1073 !
1074 ! SPECIFICATIONS:
1075 ! ------------------------------------------------------------------------------
1076  ! -- return
1077  integer(I4B) :: getjaindex
1078  ! -- dummy
1079  class(connectionstype) :: this
1080  integer(I4B), intent(in) :: node1, node2 ! nodes of interest
1081  ! -- local
1082  integer(I4B) :: i
1083 ! ------------------------------------------------------------------------------
1084  !
1085  ! -- error checking
1086  if (node1<1 .or. node1>this%nodes .or. node2<1 .or. node2>this%nodes) then
1087  getjaindex = -1 ! indicates error (an invalid node number)
1088  return
1089  endif
1090  !
1091  ! -- If node1==node2, just return the position for the diagonal.
1092  if (node1==node2) then
1093  getjaindex = this%ia(node1)
1094  return
1095  endif
1096  !
1097  ! -- Look for connection among nonzero elements defined for row node1.
1098  do i=this%ia(node1)+1,this%ia(node1+1)-1
1099  if (this%ja(i)==node2) then
1100  getjaindex = i
1101  return
1102  endif
1103  enddo
1104  !
1105  ! -- If execution reaches here, no connection exists
1106  ! between nodes of interest.
1107  getjaindex = 0 ! indicates no connection exists
1108  return
1109  end function getjaindex

1111  subroutine fillisym(neq, nja, ia, ja, isym)
1112 ! ******************************************************************************
1113 ! fillisym -- Private function to fill the isym array
1114 ! ******************************************************************************
1115 !
1116 ! SPECIFICATIONS:
1117 ! ------------------------------------------------------------------------------
1118  ! -- dummy
1119  integer(I4B),intent(in) :: neq
1120  integer(I4B),intent(in) :: nja
1121  integer(I4B),intent(inout),dimension(nja) :: isym
1122  ! -- local
1123  integer(I4B),intent(in),dimension(neq+1) :: ia
1124  integer(I4B),intent(in),dimension(nja) :: ja
1125  integer(I4B) :: n, m, ii, jj
1126 ! ------------------------------------------------------------------------------
1127  !
1128  do n=1, neq
1129  do ii = ia(n), ia(n + 1) - 1
1130  m = ja(ii)
1131  if(m /= n) then
1132  isym(ii) = 0
1133  search: do jj = ia(m), ia(m + 1) - 1
1134  if(ja(jj) == n) then
1135  isym(ii) = jj
1136  exit search
1137  endif
1138  enddo search
1139  else
1140  isym(ii) = ii
1141  endif
1142  enddo
1143  enddo
1144  !
1145  ! -- Return
1146  return
1147  end subroutine fillisym

1149  subroutine filljas(neq, nja, ia, ja, isym, jas)
1150 ! ******************************************************************************
1151 ! fillisym -- Private function to fill the jas array
1152 ! ******************************************************************************
1153 !
1154 ! SPECIFICATIONS:
1155 ! ------------------------------------------------------------------------------
1156  ! -- dummy
1157  integer(I4B),intent(in) :: neq
1158  integer(I4B),intent(in) :: nja
1159  integer(I4B),intent(in),dimension(neq+1) :: ia
1160  integer(I4B),intent(in),dimension(nja) :: ja
1161  integer(I4B),intent(in),dimension(nja) :: isym
1162  integer(I4B),intent(inout),dimension(nja) :: jas
1163  ! -- local
1164  integer(I4B) :: n, m, ii, ipos
1165 ! ------------------------------------------------------------------------------
1166  !
1167  ! -- set diagonal to zero and fill upper
1168  ipos = 1
1169  do n = 1, neq
1170  jas(ia(n)) = 0
1171  do ii = ia(n) + 1, ia(n + 1) - 1
1172  m = ja(ii)
1173  if(m > n) then
1174  jas(ii) = ipos
1175  ipos = ipos + 1
1176  endif
1177  enddo
1178  enddo
1179  !
1180  ! -- fill lower
1181  do n = 1, neq
1182  do ii = ia(n), ia(n + 1) - 1
1183  m = ja(ii)
1184  if(m < n) then
1185  jas(ii) = jas(isym(ii))
1186  endif
1187  enddo
1188  enddo
1189  !
1190  ! -- Return
1191  return
1192  end subroutine filljas

1197 ! ******************************************************************************
1198 ! vertexconnect -- routine to make cell connections from vertices

1200 !
1201 ! SPECIFICATIONS:
1202 ! ------------------------------------------------------------------------------
1203  ! -- modules
