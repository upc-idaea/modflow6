139  subroutine find_sft_package(this)
140 ! ******************************************************************************
141 ! find corresponding lkt package
142 ! ******************************************************************************
143 !
144 ! SPECIFICATIONS:
145 ! ------------------------------------------------------------------------------
146  ! -- modules
147  use memorymanagermodule, only: mem_allocate
148  ! -- dummy
149  class(gwtsfttype) :: this
150  ! -- local
151  character(len=LINELENGTH) :: errmsg
152  class(bndtype), pointer :: packobj
153  integer(I4B) :: ip, icount
154  integer(I4B) :: nbudterm
155  logical :: found
156 ! ------------------------------------------------------------------------------
157  !
158  ! -- Initialize found to false, and error later if flow package cannot
159  ! be found
160  found = .false.
161  !
162  ! -- If user is specifying flows in a binary budget file, then set up
163  ! the budget file reader, otherwise set a pointer to the flow package
164  ! budobj
165  if (this%fmi%flows_from_file) then
166  call this%fmi%set_aptbudobj_pointer(this%flowpackagename, this%flowbudptr)
167  if (associated(this%flowbudptr)) found = .true.
168  !
169  else
170  if (associated(this%fmi%gwfbndlist)) then
171  ! -- Look through gwfbndlist for a flow package with the same name as
172  ! this transport package name
173  do ip = 1, this%fmi%gwfbndlist%Count()
174  packobj => getbndfromlist(this%fmi%gwfbndlist, ip)
175  if (packobj%packName == this%flowpackagename) then
176  found = .true.
177  !
178  ! -- store BndType pointer to packobj, and then
179  ! use the select type to point to the budobj in flow package
180  this%flowpackagebnd => packobj
181  select type (packobj)
182  type is (sfrtype)
183  this%flowbudptr => packobj%budobj
184  end select
185  end if
186  if (found) exit
187  end do
188  end if
189  end if
190  !
191  ! -- error if flow package not found
192  if (.not. found) then
193  write(errmsg, '(a)') 'COULD NOT FIND FLOW PACKAGE WITH NAME '&
194  &// trim(adjustl(this%flowpackagename)) // '.'
195  call store_error(errmsg)
196  call this%parser%StoreErrorUnit()
197  call ustop()
198  endif
199  !
200  ! -- allocate space for idxbudssm, which indicates whether this is a
201  ! special budget term or one that is a general source and sink
202  nbudterm = this%flowbudptr%nbudterm
203  call mem_allocate(this%idxbudssm, nbudterm, 'IDXBUDSSM', this%memoryPath)
204  !
205  ! -- Process budget terms and identify special budget terms
206  write(this%iout, '(/, a, a)') &
207  'PROCESSING ' // ftype // ' INFORMATION FOR ', this%packName
208  write(this%iout, '(a)') ' IDENTIFYING FLOW TERMS IN ' // flowtype // ' PACKAGE'
209  write(this%iout, '(a, i0)') &
210  ' NUMBER OF ' // flowtype // ' = ', this%flowbudptr%ncv
211  icount = 1
212  do ip = 1, this%flowbudptr%nbudterm
213  select case(trim(adjustl(this%flowbudptr%budterm(ip)%flowtype)))
214  case('FLOW-JA-FACE')
215  this%idxbudfjf = ip
216  this%idxbudssm(ip) = 0
217  case('GWF')
218  this%idxbudgwf = ip
219  this%idxbudssm(ip) = 0
220  case('STORAGE')
221  this%idxbudsto = ip
222  this%idxbudssm(ip) = 0
223  case('RAINFALL')
224  this%idxbudrain = ip
225  this%idxbudssm(ip) = 0
226  case('EVAPORATION')
227  this%idxbudevap = ip
228  this%idxbudssm(ip) = 0
229  case('RUNOFF')
230  this%idxbudroff = ip
231  this%idxbudssm(ip) = 0
232  case('EXT-INFLOW')
233  this%idxbudiflw = ip
234  this%idxbudssm(ip) = 0
235  case('EXT-OUTFLOW')
236  this%idxbudoutf = ip
237  this%idxbudssm(ip) = 0
238  case('TO-MVR')
239  this%idxbudtmvr = ip
240  this%idxbudssm(ip) = 0
241  case('FROM-MVR')
242  this%idxbudfmvr = ip
243  this%idxbudssm(ip) = 0
244  case('AUXILIARY')
245  this%idxbudaux = ip
246  this%idxbudssm(ip) = 0
247  case default
248  !
249  ! -- set idxbudssm equal to a column index for where the concentrations
250  ! are stored in the concbud(nbudssm, ncv) array
251  this%idxbudssm(ip) = icount
252  icount = icount + 1
253  end select
254  write(this%iout, '(a, i0, " = ", a,/, a, i0)') &
255  ' TERM ', ip, trim(adjustl(this%flowbudptr%budterm(ip)%flowtype)), &
256  ' MAX NO. OF ENTRIES = ', this%flowbudptr%budterm(ip)%maxlist
257  end do
258  write(this%iout, '(a, //)') 'DONE PROCESSING ' // ftype // ' INFORMATION'
259  !
260  ! -- Return
261  return
262 end subroutine find_sft_package

515  subroutine sft_fill_budobj(this, idx, x, ccratin, ccratout)
516 ! ******************************************************************************
517 ! sft_fill_budobj -- copy flow terms into this%budobj
518 ! ******************************************************************************
519 !
520 ! SPECIFICATIONS:
521 ! ------------------------------------------------------------------------------
522  ! -- modules
523  ! -- dummy
524  class(gwtsfttype) :: this
525  integer(I4B), intent(inout) :: idx
526  real(DP), dimension(:), intent(in) :: x
527  real(DP), intent(inout) :: ccratin
528  real(DP), intent(inout) :: ccratout
529  ! -- local
530  integer(I4B) :: j, n1, n2
531  integer(I4B) :: nlist
532  real(DP) :: q
533  ! -- formats
534 ! -----------------------------------------------------------------------------

542  call this%budobj%budterm(idx)%update_term(n1, n2, q)
543  call this%apt_accumulate_ccterm(n1, q, ccratin, ccratout)
544  end do
545 
546 
547  ! -- EVAPORATION
548  idx = idx + 1
549  nlist = this%flowbudptr%budterm(this%idxbudevap)%nlist
550  call this%budobj%budterm(idx)%reset(nlist)
551  do j = 1, nlist
552  call this%sft_evap_term(j, n1, n2, q)
553  call this%budobj%budterm(idx)%update_term(n1, n2, q)
554  call this%apt_accumulate_ccterm(n1, q, ccratin, ccratout)
555  end do
556 
557 
558  ! -- RUNOFF
559  idx = idx + 1
560  nlist = this%flowbudptr%budterm(this%idxbudroff)%nlist
561  call this%budobj%budterm(idx)%reset(nlist)
562  do j = 1, nlist
563  call this%sft_roff_term(j, n1, n2, q)
564  call this%budobj%budterm(idx)%update_term(n1, n2, q)
565  call this%apt_accumulate_ccterm(n1, q, ccratin, ccratout)
566  end do
567 
568 
569  ! -- EXT-INFLOW
570  idx = idx + 1
571  nlist = this%flowbudptr%budterm(this%idxbudiflw)%nlist
572  call this%budobj%budterm(idx)%reset(nlist)
573  do j = 1, nlist
574  call this%sft_iflw_term(j, n1, n2, q)
575  call this%budobj%budterm(idx)%update_term(n1, n2, q)
576  call this%apt_accumulate_ccterm(n1, q, ccratin, ccratout)
577  end do
578 
579 
580  ! -- EXT-OUTFLOW

583  call this%budobj%budterm(idx)%reset(nlist)
584  do j = 1, nlist
585  call this%sft_outf_term(j, n1, n2, q)
586  call this%budobj%budterm(idx)%update_term(n1, n2, q)
587  call this%apt_accumulate_ccterm(n1, q, ccratin, ccratout)

590 
591  !
