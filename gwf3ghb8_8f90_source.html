155  subroutine ghb_cf(this, reset_mover)
156 ! ******************************************************************************
157 ! ghb_cf -- Formulate the HCOF and RHS terms
158 ! Subroutine: (1) skip if no ghbs
159 ! (2) calculate hcof and rhs
160 ! ******************************************************************************
161 !
162 ! SPECIFICATIONS:
163 ! ------------------------------------------------------------------------------
164  ! -- dummy
165  class(ghbtype) :: this
166  logical, intent(in), optional :: reset_mover
167  ! -- local
168  integer(I4B) :: i, node
169  logical :: lrm
170 ! ------------------------------------------------------------------------------
171  !
172  ! -- Return if no ghbs
173  if(this%nbound.eq.0) return
174  !
175  ! -- packmvrobj cf
176  lrm = .true.
177  if (present(reset_mover)) lrm = reset_mover
178  if(this%imover == 1 .and. lrm) then
179  call this%pakmvrobj%cf()
180  endif
181  !
182  ! -- Calculate hcof and rhs for each ghb entry
183  do i=1,this%nbound
184  node=this%nodelist(i)
185  if(this%ibound(node).le.0) then
186  this%hcof(i)=dzero
187  this%rhs(i)=dzero
188  cycle
189  endif
190  this%hcof(i) = -this%bound(2,i)
191  this%rhs(i) = -this%bound(2,i) * this%bound(1,i)
192  end do
193  !
194  ! -- return
195  return
196  end subroutine ghb_cf

198  subroutine ghb_fc(this, rhs, ia, idxglo, amatsln)
199 ! **************************************************************************
200 ! ghb_fc -- Copy rhs and hcof into solution rhs and amat
201 ! **************************************************************************
202 !
203 ! SPECIFICATIONS:
204 ! --------------------------------------------------------------------------
205  ! -- dummy
206  class(ghbtype) :: this
207  real(DP), dimension(:), intent(inout) :: rhs
208  integer(I4B), dimension(:), intent(in) :: ia
209  integer(I4B), dimension(:), intent(in) :: idxglo
210  real(DP), dimension(:), intent(inout) :: amatsln
211  ! -- local
212  integer(I4B) :: i, n, ipos
213  real(DP) :: cond, bhead, qghb
214 ! --------------------------------------------------------------------------
215  !
216  ! -- pakmvrobj fc
217  if(this%imover == 1) then
218  call this%pakmvrobj%fc()
219  endif
220  !
221  ! -- Copy package rhs and hcof into solution rhs and amat
222  do i = 1, this%nbound
223  n = this%nodelist(i)
224  rhs(n) = rhs(n) + this%rhs(i)
225  ipos = ia(n)
226  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + this%hcof(i)
227  !
228  ! -- If mover is active and this boundary is discharging,
229  ! store available water (as positive value).
230  bhead = this%bound(1,i)
231  if(this%imover == 1 .and. this%xnew(n) > bhead) then
232  cond = this%bound(2,i)
233  qghb = cond * (this%xnew(n) - bhead)
234  call this%pakmvrobj%accumulate_qformvr(i, qghb)
235  endif
236  enddo
237  !
238  ! -- return
239  return
240  end subroutine ghb_fc
