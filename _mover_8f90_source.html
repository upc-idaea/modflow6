260  subroutine fc(this)
261 ! ******************************************************************************
262 ! fc -- formulate coefficients
263 ! ******************************************************************************
264 !
265 ! SPECIFICATIONS:
266 ! ------------------------------------------------------------------------------
267  ! -- modules
268  ! -- dummy
269  class(mvrtype) :: this
270  ! -- local
271  real(DP) :: qavailable, qtanew, qpactual
272 ! ------------------------------------------------------------------------------
273  !
274  ! -- Set qa and this%qavailable equal to available water in package (qtomvr)
275  qavailable = this%qformvr_ptr
276  qtanew = this%qtformvr_ptr
277  this%qavailable = qavailable
278  !
279  ! -- Using the mover rules, calculate how much of the available water will
280  ! be provided from the mover to the receiver.
281  qpactual = this%qrcalc(qavailable, qtanew)
282  !
283  ! -- Store qpactual
284  this%qpactual = qpactual
285  !
286  ! -- Add the calculated qpactual term directly into the receiver package
287  ! qfrommvr array.
288  this%qfrommvr_ptr = this%qfrommvr_ptr + qpactual
289  !
290  ! -- Add the calculated qpactual term directly into the provider package
291  ! qtomvr array.
292  this%qtomvr_ptr = this%qtomvr_ptr + qpactual
293  !
294  ! -- Reduce the amount of water that is available in the provider package
295  ! qformvr array.
296  this%qformvr_ptr = this%qformvr_ptr - qpactual
297  !
298  ! -- return
299  return
300  end subroutine fc

302  function qrcalc(this, qa, qta) result(qr)
303 ! ******************************************************************************
304 ! qrcalc -- Calculate the rate of water provided to the receiver
305 ! ******************************************************************************
306 !
307 ! SPECIFICATIONS:
308 ! ------------------------------------------------------------------------------
309  ! -- modules
310  ! -- return
311  real(DP) :: qr
312  ! -- dummy
313  class(mvrtype) :: this
314  real(DP), intent(in) :: qa
315  real(DP), intent(in) :: qta
316  ! -- local
317 ! ------------------------------------------------------------------------------
318  ! -- Using the mover rules, calculate how much of the available water will
319  ! go to the receiver.
320  qr = dzero
321  ! -- Calculate qr
322  select case (this%imvrtype)
323  case(1)
324  ! -- FACTOR uses total available to make calculation, and then
325  ! limits qr by consumed available
326  if(qta > dzero) qr = qta * this%value
327  qr = min(qr, qa)
328  case(2)
329  ! -- EXCESS
330  if(qa > this%value) then
331  qr = qa - this%value
332  else
333  qr = dzero
334  endif
335  case(3)
336  ! -- THRESHOLD
337  if(this%value > qa) then
338  qr = dzero
339  else
340  qr = this%value
341  endif
342  case(4)
343  ! -- UPTO
344  if(qa > this%value) then
345  qr = this%value
346  else
347  qr = qa
348  endif
349  end select
350  !
351  ! -- return
352  return
353  end function qrcalc
