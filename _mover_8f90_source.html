255  subroutine fc(this)
256 ! ******************************************************************************
257 ! fc -- formulate coefficients
258 ! ******************************************************************************
259 !
260 ! SPECIFICATIONS:
261 ! ------------------------------------------------------------------------------
262  ! -- modules
263  ! -- dummy
264  class(mvrtype) :: this
265  ! -- local
266  real(DP) :: qavailable, qtanew, qpactual
267 ! ------------------------------------------------------------------------------
268  !
269  ! -- Set qa and this%qavailable equal to available water in package (qtomvr)
270  qavailable = this%qformvr_ptr
271  qtanew = this%qtformvr_ptr
272  this%qavailable = qavailable
273  !
274  ! -- Using the mover rules, calculate how much of the available water will
275  ! be provided from the mover to the receiver.
276  qpactual = this%qrcalc(qavailable, qtanew)
277  !
278  ! -- Store qpactual
279  this%qpactual = qpactual
280  !
281  ! -- Add the calculated qpactual term directly into the receiver package
282  ! qfrommvr array.
283  this%qfrommvr_ptr = this%qfrommvr_ptr + qpactual
284  !
285  ! -- Add the calculated qpactual term directly into the provider package
286  ! qtomvr array.
287  this%qtomvr_ptr = this%qtomvr_ptr + qpactual
288  !
289  ! -- Reduce the amount of water that is available in the provider package
290  ! qformvr array.
291  this%qformvr_ptr = this%qformvr_ptr - qpactual
292  !
293  ! -- return
294  return
295  end subroutine fc

297  function qrcalc(this, qa, qta) result(qr)
298 ! ******************************************************************************
299 ! qrcalc -- Calculate the rate of water provided to the receiver
300 ! ******************************************************************************
301 !
302 ! SPECIFICATIONS:
303 ! ------------------------------------------------------------------------------
304  ! -- modules
305  ! -- return
306  real(dp) :: qr
307  ! -- dummy
308  class(mvrtype) :: this
309  real(dp), intent(in) :: qa
310  real(dp), intent(in) :: qta
311  ! -- local
312 ! ------------------------------------------------------------------------------
313  ! -- Using the mover rules, calculate how much of the available water will
314  ! go to the receiver.
315  qr = dzero
316  ! -- Calculate qr
317  select case (this%imvrtype)
318  case(1)
319  ! -- FACTOR uses total available to make calculation, and then
320  ! limits qr by consumed available
321  if(qta > dzero) qr = qta * this%value
322  qr = min(qr, qa)
323  case(2)
324  ! -- EXCESS
325  if(qa > this%value) then
326  qr = qa - this%value
327  else
328  qr = dzero
329  endif
330  case(3)
331  ! -- THRESHOLD
332  if(this%value > qa) then
333  qr = dzero
334  else
335  qr = this%value
336  endif
337  case(4)
338  ! -- UPTO
339  if(qa > this%value) then
340  qr = this%value
341  else
342  qr = qa
343  endif
344  end select
345  !
346  ! -- return
347  return
348  end function qrcalc
