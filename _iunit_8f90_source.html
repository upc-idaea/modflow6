66 ! ******************************************************************************
67 ! addfile -- add an ftyp and unit number. Find the row for the ftyp and
68 ! store another iunit value.
69 ! ******************************************************************************
70 !
71 ! SPECIFICATIONS:
72 ! ------------------------------------------------------------------------------
73  ! -- modules

79  character(len=*), intent(in) :: namefilename
80  ! -- local

82  integer(I4B), allocatable, dimension(:) :: itemp
83  integer(I4B) :: i, irow
84 ! ------------------------------------------------------------------------------
85  !
86  ! -- Find the row containing ftyp
87  irow = 0
88  do i = 1, this%niunit
89  if(this%cunit(i) == ftyp) then
90  irow = i
91  exit
92  endif
93  enddo
94  if(irow == 0) then
95  write(errmsg, '(a,a)') 'Package type not supported: ', ftyp
96  call store_error(errmsg)
97  call store_error_filename(namefilename, terminate=.true.)
98  endif
99  !
100  ! -- Store the iunit number for this ftyp
101  if(this%iunit(irow)%nval == 0) then
102  allocate(this%iunit(irow)%iunit(1))
103  allocate(this%iunit(irow)%ipos(1))
104  this%iunit(irow)%nval=1
105  else
106  !
107  ! -- increase size of iunit
108  allocate(itemp(this%iunit(irow)%nval))
109  itemp(:) = this%iunit(irow)%iunit(:)
110  deallocate(this%iunit(irow)%iunit)
111  this%iunit(irow)%nval = this%iunit(irow)%nval + 1
112  allocate(this%iunit(irow)%iunit(this%iunit(irow)%nval))
113  this%iunit(irow)%iunit(1:this%iunit(irow)%nval - 1) = itemp(:)
114  !
115  ! -- increase size of ipos
116  itemp(:) = this%iunit(irow)%ipos(:)
117  deallocate(this%iunit(irow)%ipos)
118  allocate(this%iunit(irow)%ipos(this%iunit(irow)%nval))
119  this%iunit(irow)%ipos(1:this%iunit(irow)%nval - 1) = itemp(:)
120  !
121  ! -- cleanup temp
122  deallocate(itemp)
123  endif
124  this%iunit(irow)%iunit(this%iunit(irow)%nval) = iunit
125  this%iunit(irow)%ipos(this%iunit(irow)%nval) = ipos
126  !
