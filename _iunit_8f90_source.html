66 ! ******************************************************************************
67 ! addfile -- add an ftyp and unit number. Find the row for the ftyp and
68 ! store another iunit value.
69 ! ******************************************************************************
70 !
71 ! SPECIFICATIONS:
72 ! ------------------------------------------------------------------------------

80  ! -- local

83  integer(I4B) :: i, irow
84 ! ------------------------------------------------------------------------------
85  !
86  ! -- Find the row containing ftyp
87  irow = 0
88  do i = 1, this%niunit
89  if(this%cunit(i) == ftyp) then

91  exit
92  endif
93  enddo

95  write(errmsg, '(a,a)') 'Package type not supported: ', ftyp
96  call store_error(errmsg)
97  call store_error_filename(namefilename)
98  call ustop()
99  endif
100  !
101  ! -- Store the iunit number for this ftyp
102  if(this%iunit(irow)%nval == 0) then
103  allocate(this%iunit(irow)%iunit(1))
104  allocate(this%iunit(irow)%ipos(1))
105  this%iunit(irow)%nval=1
106  else
107  !
108  ! -- increase size of iunit
109  allocate(itemp(this%iunit(irow)%nval))
110  itemp(:) = this%iunit(irow)%iunit(:)
111  deallocate(this%iunit(irow)%iunit)
112  this%iunit(irow)%nval = this%iunit(irow)%nval + 1
113  allocate(this%iunit(irow)%iunit(this%iunit(irow)%nval))
114  this%iunit(irow)%iunit(1:this%iunit(irow)%nval - 1) = itemp(:)
115  !
116  ! -- increase size of ipos
117  itemp(:) = this%iunit(irow)%ipos(:)
118  deallocate(this%iunit(irow)%ipos)
119  allocate(this%iunit(irow)%ipos(this%iunit(irow)%nval))
120  this%iunit(irow)%ipos(1:this%iunit(irow)%nval - 1) = itemp(:)
121  !

126  this%iunit(irow)%ipos(this%iunit(irow)%nval) = ipos
