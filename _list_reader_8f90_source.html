157  subroutine set_openclose(this)
158 ! ******************************************************************************
159 ! set_openclose -- set up for open/close file

161 ! OPEN/CLOSE fname [(BINARY)]
162 ! ******************************************************************************
163 !
164 ! SPECIFICATIONS:
165 ! ------------------------------------------------------------------------------
166  ! -- modules
167  use inputoutputmodule, only: u8rdcom, urword, openfile

174  integer(I4B) :: idum, itmp
175  real(DP) :: r
176  logical :: exists
177  integer(I4B) :: nunopn = 99
178  character(len=LINELENGTH) :: fname
179  character(len=LINELENGTH) :: errmsg
180  ! -- formats
181  character(len=*), parameter :: fmtocne = &
182  "('Specified OPEN/CLOSE file ',(A),' does not exist')"
183  character(len=*), parameter :: fmtobf = &
184  "(1X,/1X,'OPENING BINARY FILE ON UNIT ',I0,':',/1X,A)"
185  character(len=*), parameter :: fmtobfnlist = &
186  "(1X, 'TO READ ', I0, ' RECORDS.')"
187  character(len=*), parameter :: fmtofnlist = &
188  "(1x,'TO READ ', I0, ' RECORDS.')"
189  character(len=*), parameter :: fmtof = &
190  "(1X,/1X,'OPENING FILE ON UNIT ',I0,':',/1X,A)"
191 ! ------------------------------------------------------------------------------
192  !
193  ! -- get filename
194  call urword(this%line, this%lloc, this%istart, this%istop, 0, idum, r, &
195  this%iout, this%in)

199  inquire(file=fname, exist=exists)

201  write(errmsg, fmtocne) this%line(this%istart:this%istop)
202  call store_error(errmsg)
203  call store_error('Specified OPEN/CLOSE file does not exist')
204  call store_error_unit(this%in)
205  call ustop()
206  endif
207  !

209  call urword(this%line, this%lloc, this%istart, this%istop, 1, idum, r, &

216  itmp = this%iout
217  if(this%iout > 0) then
218  itmp = 0
219  write(this%iout, fmtobf) this%inlist, trim(adjustl(fname))
220  if(this%nlist > 0) write(this%iout, fmtobfnlist) this%nlist
221  endif
222  call openfile(this%inlist, itmp, fname, 'OPEN/CLOSE', fmtarg_opt=form, &
223  accarg_opt=access)
224  else
225  itmp = this%iout
226  if(this%iout > 0) then
227  itmp = 0
228  write(this%iout, fmtof) this%inlist, trim(adjustl(fname))
229  if(this%nlist > 0) write(this%iout, fmtofnlist) this%nlist

231  call openfile(this%inlist, itmp, fname, 'OPEN/CLOSE')
232  end if

234  ! -- Set iclose to 1 because it is open/close, to indicate that the

236  this%iclose = 1
237  !
238  ! -- Read the first line from inlist to be consistent with how the list is
239  ! read when it is included in the package input file
240  if(this%ibinary /= 1) call u8rdcom(this%inlist, this%iout, this%line, &
241  this%ierr)
242  !
243  ! -- return
244  return
245  end subroutine set_openclose

276  subroutine read_binary(this)
277 ! ******************************************************************************
278 ! read_binary -- read the data from a binary file
279 ! ******************************************************************************
280 !
281 ! SPECIFICATIONS:
282 ! ------------------------------------------------------------------------------
283  ! -- modules
284  use constantsmodule, only: linelength, lenbigline
285  use inputoutputmodule, only: get_node
286  use simmodule, only: ustop, store_error
287  ! -- dummy
288  class(listreadertype) :: this
289  ! -- local
290  integer(I4B) :: mxlist, ldim, naux, nod, ii, jj
291  character(len=LINELENGTH) :: fname
292  character(len=LENBIGLINE) :: errmsg
293  integer(I4B), dimension(:), allocatable :: cellid
294  ! -- formats
295  character(len=*), parameter :: fmtmxlsterronly = &
296  "('ERROR READING LIST FROM FILE: '," // &
297  "a,' ON UNIT: ',I0," // &
298  "' THE NUMBER OF RECORDS ENCOUNTERED EXCEEDS THE MAXIMUM NUMBER " // &
299  "OF RECORDS. TRY INCREASING MAXBOUND FOR THIS LIST." // &
300  " NUMBER OF RECORDS: ',I0,' MAXBOUND: ',I0)"
301  character(len=*), parameter :: fmtlsterronly = &
302  "('ERROR READING LIST FROM FILE: '," // &
303  "1x,a,1x,' ON UNIT: ',I0)"
304 ! ------------------------------------------------------------------------------
305  !
306  ! -- determine array sizes
307  mxlist = size(this%rlist, 2)
308  ldim = size(this%rlist, 1)
309  naux = size(this%auxvar, 1)
310  !
311  ! -- Allocate arrays
312  allocate(cellid(this%ndim))
313  !
314  ii = 1
315  readloop: do
316  !
317  ! -- read layer, row, col, or cell number
318  read(this%inlist, iostat=this%ierr) cellid
319 
320  ! -- If not end of record, then store nodenumber, else
321  ! calculate lstend and nlist, and exit readloop
322  select case(this%ierr)
323  case(0)
324  !
325  ! -- Check range
326  if(ii > mxlist) then
327  inquire(unit=this%inlist, name=fname)
328  write(errmsg, fmtmxlsterronly) fname, this%inlist, ii, mxlist
329  call store_error(errmsg)
330  call ustop()
331  endif
332  !
333  ! -- Store node number and read the remainder of the record
334  if(this%ndim == 1) then
335  nod = cellid(1)
336  elseif(this%ndim == 2) then
337  nod = get_node(cellid(1), 1, cellid(2), &
338  this%mshape(1), 1, this%mshape(2))
339  else
340  nod = get_node(cellid(1), cellid(2), cellid(3), &
341  this%mshape(1), this%mshape(2), this%mshape(3))
342  endif
343  this%nodelist(ii) = nod
344  read(this%inlist, iostat=this%ierr) (this%rlist(jj,ii),jj=1,ldim), &
345  (this%auxvar(ii,jj),jj=1,naux)
346  if(this%ierr /= 0) then
347  inquire(unit=this%inlist, name=fname)
348  write(errmsg, fmtlsterronly) trim(adjustl(fname)), this%inlist
349  call store_error(errmsg)
350  call ustop()
351  endif
352  !
353  case(:-1)
354  !
355  ! -- End of record was encountered
356  this%nlist = ii - 1
357  exit readloop
358  !
359  case(1:)
360  !
361  ! -- Error
362  inquire(unit=this%inlist, name=fname)
363  write(errmsg, fmtlsterronly) trim(adjustl(fname)), this%inlist
364  call store_error(errmsg)
365  call ustop()
366  !
367  end select
368  !
369  ! -- If nlist is known, then exit when nlist values have been read
370  if(this%nlist > 0) then
371  if(ii == this%nlist) exit readloop
372  endif
373  !
374  ! -- increment ii
375  ii = ii + 1
376  !
377  enddo readloop
378  !
379  ! -- return
380  return
381  end subroutine read_binary

383  subroutine read_ascii(this)
384 ! ******************************************************************************
385 ! read_ascii -- read the data from an ascii file
386 ! ******************************************************************************
387 !
388 ! SPECIFICATIONS:
389 ! ------------------------------------------------------------------------------
390  ! -- modules
391  use constantsmodule, only: lenboundname, linelength, dzero
392  use inputoutputmodule, only: u8rdcom, urword, get_node
393  use simmodule, only: ustop, store_error, count_errors
394  use arrayhandlersmodule, only: expandarray
395  ! -- dummy
396  class(listreadertype) :: this
397  ! -- local
398  integer(I4B) :: mxlist, ldim, naux
399  integer(I4B) :: ii, jj, idum, nod, istat, increment
400  real(DP) :: r
401  integer(I4B), dimension(:), allocatable :: cellid
402  character(len=LINELENGTH) :: fname
403  character(len=LINELENGTH) :: errmsg
404  ! -- formats
405  character(len=*), parameter :: fmtmxlsterronly = &
406  "('***ERROR READING LIST. &
407  &THE NUMBER OF RECORDS ENCOUNTERED EXCEEDS THE MAXIMUM NUMBER " // &
408  "OF RECORDS. TRY INCREASING MAXBOUND FOR THIS LIST." // &
409  " NUMBER OF RECORDS: ',I0,' MAXBOUND: ',I0)"
410 ! ------------------------------------------------------------------------------
411  !
412  ! -- determine array sizes
413  mxlist = size(this%rlist, 2)
414  ldim = size(this%rlist, 1)
415  naux = size(this%auxvar, 1)
416  this%ntxtrlist = 0
417  this%ntxtauxvar = 0
418  !
419  ! -- Allocate arrays
420  allocate(cellid(this%ndim))
421  !
422  ii = 1
423  readloop: do
424  !
425  ! -- First line was already read, so don't read again
426  if(ii /= 1) call u8rdcom(this%inlist, 0, this%line, this%ierr)
427  !
428  ! -- If this is an unknown-length list, then check for END.
429  ! If found, then backspace, set nlist, and exit readloop.
430  if(this%nlist < 0) then
431  this%lloc = 1
432  call urword(this%line, this%lloc, this%istart, this%istop, 1, idum, r, &
433  this%iout, this%inlist)
434  if(this%line(this%istart:this%istop) == 'END' .or. this%ierr < 0) then
435  ! If ierr < 0, backspace was already performed in u8rdcom, so only
436  ! need to backspace if END was found.
437  if (this%ierr == 0) then
438  backspace(this%inlist)
439  endif
440  this%nlist = ii - 1
441  exit readloop
442  endif
443  endif
444  !
445  ! -- Check range
446  if(ii > mxlist) then
447  inquire(unit=this%inlist, name=fname)
448  write(errmsg, fmtmxlsterronly) ii, mxlist
449  call store_error(errmsg)
450  errmsg = 'Error occurred reading line: ' // trim(this%line)
451  call store_error(errmsg)
452  call store_error_unit(this%inlist)
453  call ustop()
454  endif
455  !
456  ! -- Read layer, row, column or cell number and assign to nodelist
457  this%lloc = 1
458  if(this%ndim == 3) then
459  !
460  ! -- Grid is structured; read layer, row, column
461  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
462  cellid(1), r, this%iout, this%inlist)
463  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
464  cellid(2), r, this%iout, this%inlist)
465  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
466  cellid(3), r, this%iout, this%inlist)
467  !
468  ! -- Check for illegal grid location
469  if(cellid(1) < 1 .or. cellid(1) > this%mshape(1)) then
470  write(errmsg, *) ' Layer number in list is outside of the grid', &
471  cellid(1)
472  call store_error(errmsg)
473  end if
474  if(cellid(2) < 1 .or. cellid(2) > this%mshape(2)) then
475  write(errmsg, *) ' Row number in list is outside of the grid', &
476  cellid(2)
477  call store_error(errmsg)
478  end if
479  if(cellid(3) < 1 .or. cellid(3) > this%mshape(3)) then
480  write(errmsg, *) ' Column number in list is outside of the grid', &
481  cellid(3)
482  call store_error(errmsg)
483  end if
484  !
485  ! -- Calculate nodenumber and put in nodelist
486  nod = get_node(cellid(1), cellid(2), cellid(3), &
487  this%mshape(1), this%mshape(2), this%mshape(3))
488  elseif(this%ndim == 2) then
489  !
490  ! -- Grid is disv
491  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
492  cellid(1), r, this%iout, this%inlist)
493  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
494  cellid(2), r, this%iout, this%inlist)
495  !
496  ! -- Check for illegal grid location
497  if(cellid(1) < 1 .or. cellid(1) > this%mshape(1)) then
498  write(errmsg, *) ' Layer number in list is outside of the grid', &
499  cellid(1)
500  call store_error(errmsg)
501  end if
502  if(cellid(2) < 1 .or. cellid(2) > this%mshape(2)) then
503  write(errmsg, *) ' Cell2d number in list is outside of the grid', &
504  cellid(2)
505  call store_error(errmsg)
506  end if
507  !
508  ! -- Calculate nodenumber and put in nodelist
509  nod = get_node(cellid(1), 1, cellid(2), &
510  this%mshape(1), 1, this%mshape(2))
511  else
512  !
513  ! -- Grid is unstructured; read layer and celld2d number
514  call urword(this%line, this%lloc, this%istart, this%istop, 2, nod, r, &
515  this%iout, this%inlist)
516  if(nod < 1 .or. nod > this%mshape(1)) then
517  write(errmsg, *) ' Node number in list is outside of the grid', nod
518  call store_error(errmsg)
519  end if
520  !
521  endif
522  !
523  ! -- Assign nod to nodelist
524  this%nodelist(ii) = nod
525  !
526  ! -- Read rlist
527  do jj = 1, ldim
528  call urword(this%line, this%lloc, this%istart, this%istop, 0, idum, &
529  r, this%iout, this%inlist)
530  read(this%line(this%istart:this%istop), *, iostat=istat) r
531  !
532  ! -- If a double precision value, then store in rlist, otherwise store
533  ! the text name and location
534  if (istat == 0) then
535  this%rlist(jj, ii) = r
536  else
537  this%rlist(jj, ii) = dzero
538  this%ntxtrlist = this%ntxtrlist + 1
539  if(this%ntxtrlist > size(this%txtrlist)) then
540  increment = size(this%txtrlist) * 0.2
541  increment = max(100, increment)
542  call expandarray(this%txtrlist, increment)
543  call expandarray(this%idxtxtrow, increment)
544  call expandarray(this%idxtxtcol, increment)
545  endif
546  this%txtrlist(this%ntxtrlist) = this%line(this%istart:this%istop)
547  this%idxtxtrow(this%ntxtrlist) = ii
548  this%idxtxtcol(this%ntxtrlist) = jj
549  endif
550  !
551  enddo
552  !
553  ! -- Read auxvar
554  do jj = 1, naux
555  call urword(this%line, this%lloc, this%istart, this%istop, 0, idum, &
556  r, this%iout, this%inlist)
557  read(this%line(this%istart:this%istop), *, iostat=istat) r
558  !
559  ! -- If a double precision value, then store in auxvar, otherwise store
560  ! the text name and location
561  if (istat == 0) then
562  this%auxvar(jj, ii) = r
563  else
564  this%auxvar(jj, ii) = dzero
565  this%ntxtauxvar = this%ntxtauxvar + 1
566  if(this%ntxtauxvar > size(this%txtauxvar)) then
567  increment = size(this%txtauxvar) * 0.2
568  increment = max(100, increment)
569  call expandarray(this%txtauxvar, increment)
570  call expandarray(this%idxtxtauxrow, increment)
571  call expandarray(this%idxtxtauxcol, increment)
572  endif
573  this%txtauxvar(this%ntxtauxvar) = this%line(this%istart:this%istop)
574  this%idxtxtauxrow(this%ntxtauxvar) = ii
575  this%idxtxtauxcol(this%ntxtauxvar) = jj
576  endif
577  !
578  enddo
579  !
580  ! -- Read the boundary names (only supported for ascii input)
581  if (this%inamedbound > 0) then
582  call urword(this%line, this%lloc, this%istart, this%istop, 1, idum, r, &
583  this%iout, this%inlist)
