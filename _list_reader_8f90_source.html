157  subroutine set_openclose(this)
158 ! ******************************************************************************
159 ! set_openclose -- set up for open/close file
160 !
161 ! OPEN/CLOSE fname [(BINARY)]
162 ! ******************************************************************************
163 !
164 ! SPECIFICATIONS:
165 ! ------------------------------------------------------------------------------
166  ! -- modules
167  use inputoutputmodule, only: u9rdcom, urword, openfile
168  use openspecmodule, only: form, access
169  use constantsmodule, only: linelength
170  use simmodule, only: store_error
171  ! -- dummy
172  class(listreadertype) :: this
173  ! -- local
174  integer(I4B) :: idum, itmp
175  real(DP) :: r
176  logical :: exists
177  integer(I4B) :: nunopn = 99
178  character(len=LINELENGTH) :: fname
179  character(len=LINELENGTH) :: errmsg
180  ! -- formats
181  character(len=*), parameter :: fmtocne = &
182  "('Specified OPEN/CLOSE file ',(A),' does not exist')"
183  character(len=*), parameter :: fmtobf = &
184  "(1X,/1X,'OPENING BINARY FILE ON UNIT ',I0,':',/1X,A)"
185  character(len=*), parameter :: fmtobfnlist = &
186  "(1X, 'TO READ ', I0, ' RECORDS.')"
187  character(len=*), parameter :: fmtofnlist = &
188  "(1x,'TO READ ', I0, ' RECORDS.')"
189  character(len=*), parameter :: fmtof = &
190  "(1X,/1X,'OPENING FILE ON UNIT ',I0,':',/1X,A)"
191 ! ------------------------------------------------------------------------------
192  !
193  ! -- get filename
194  call urword(this%line, this%lloc, this%istart, this%istop, 0, idum, r, &
195  this%iout, this%in)
196  fname = this%line(this%istart:this%istop)
197  !
198  ! -- check to see if file OPEN/CLOSE file exists
199  inquire(file=fname, exist=exists)
200  if (.not. exists) then
201  write(errmsg, fmtocne) this%line(this%istart:this%istop)
202  call store_error(errmsg)
203  call store_error('Specified OPEN/CLOSE file does not exist')
204  call store_error_unit(this%in)
205  endif
206  !
207  ! -- Check for (BINARY) keyword
208  call urword(this%line, this%lloc, this%istart, this%istop, 1, idum, r, &
209  this%iout, this%in)
210  if(this%line(this%istart:this%istop) == '(BINARY)') this%ibinary = 1
211  !
212  ! -- Open the file depending on ibinary flag
213  this%inlist = nunopn
214  if(this%ibinary == 1) then
215  itmp = this%iout
216  if(this%iout > 0) then
217  itmp = 0
218  write(this%iout, fmtobf) this%inlist, trim(adjustl(fname))
219  if(this%nlist > 0) write(this%iout, fmtobfnlist) this%nlist
220  endif
221  call openfile(this%inlist, itmp, fname, 'OPEN/CLOSE', fmtarg_opt=form, &
222  accarg_opt=access)
223  else
224  itmp = this%iout
225  if(this%iout > 0) then
226  itmp = 0
227  write(this%iout, fmtof) this%inlist, trim(adjustl(fname))
228  if(this%nlist > 0) write(this%iout, fmtofnlist) this%nlist
229  endif
230  call openfile(this%inlist, itmp, fname, 'OPEN/CLOSE')
231  end if
232  !
233  ! -- Set iclose to 1 because it is open/close, to indicate that the
234  ! file needs to be closed after the list is read
235  this%iclose = 1
236  !
237  ! -- Read the first line from inlist to be consistent with how the list is
238  ! read when it is included in the package input file
239  if(this%ibinary /= 1) call u9rdcom(this%inlist, this%iout, this%line, &
240  this%ierr)
241  !
242  ! -- return
243  return
244  end subroutine set_openclose

275  subroutine read_binary(this)
276 ! ******************************************************************************
277 ! read_binary -- read the data from a binary file
278 ! ******************************************************************************
279 !
280 ! SPECIFICATIONS:
281 ! ------------------------------------------------------------------------------
282  ! -- modules
283  use constantsmodule, only: linelength, lenbigline
284  use inputoutputmodule, only: get_node
285  use simmodule, only: store_error
286  ! -- dummy
287  class(listreadertype) :: this
288  ! -- local
289  integer(I4B) :: mxlist, ldim, naux, nod, ii, jj
290  character(len=LINELENGTH) :: fname
291  character(len=LENBIGLINE) :: errmsg
292  integer(I4B), dimension(:), allocatable :: cellid
293  ! -- formats
294  character(len=*), parameter :: fmtmxlsterronly = &
295  "('ERROR READING LIST FROM FILE: '," // &
296  "a,' ON UNIT: ',I0," // &
297  "' THE NUMBER OF RECORDS ENCOUNTERED EXCEEDS THE MAXIMUM NUMBER " // &
298  "OF RECORDS. TRY INCREASING MAXBOUND FOR THIS LIST." // &
299  " NUMBER OF RECORDS: ',I0,' MAXBOUND: ',I0)"
300  character(len=*), parameter :: fmtlsterronly = &
301  "('ERROR READING LIST FROM FILE: '," // &
302  "1x,a,1x,' ON UNIT: ',I0)"
303 ! ------------------------------------------------------------------------------
304  !
305  ! -- determine array sizes
306  mxlist = size(this%rlist, 2)
307  ldim = size(this%rlist, 1)
308  naux = size(this%auxvar, 1)
309  !
310  ! -- Allocate arrays
311  allocate(cellid(this%ndim))
312  !
313  ii = 1
314  readloop: do
315  !
316  ! -- read layer, row, col, or cell number
317  read(this%inlist, iostat=this%ierr) cellid
318 
319  ! -- If not end of record, then store nodenumber, else
320  ! calculate lstend and nlist, and exit readloop
321  select case(this%ierr)
322  case(0)
323  !
324  ! -- Check range
325  if(ii > mxlist) then
326  inquire(unit=this%inlist, name=fname)
327  write(errmsg, fmtmxlsterronly) fname, this%inlist, ii, mxlist
328  call store_error(errmsg, terminate=.true.)
329  endif
330  !
331  ! -- Store node number and read the remainder of the record
332  if(this%ndim == 1) then
333  nod = cellid(1)
334  elseif(this%ndim == 2) then
335  nod = get_node(cellid(1), 1, cellid(2), &
336  this%mshape(1), 1, this%mshape(2))
337  else
338  nod = get_node(cellid(1), cellid(2), cellid(3), &
339  this%mshape(1), this%mshape(2), this%mshape(3))
340  endif
341  this%nodelist(ii) = nod
342  read(this%inlist, iostat=this%ierr) (this%rlist(jj,ii),jj=1,ldim), &
343  (this%auxvar(ii,jj),jj=1,naux)
344  if(this%ierr /= 0) then
345  inquire(unit=this%inlist, name=fname)
346  write(errmsg, fmtlsterronly) trim(adjustl(fname)), this%inlist
347  call store_error(errmsg, terminate=.true.)
348  endif
349  !
350  case(:-1)
351  !
352  ! -- End of record was encountered
353  this%nlist = ii - 1
354  exit readloop
355  !
356  case(1:)
357  !
358  ! -- Error
359  inquire(unit=this%inlist, name=fname)
360  write(errmsg, fmtlsterronly) trim(adjustl(fname)), this%inlist
361  call store_error(errmsg, terminate=.true.)
362  !
363  end select
364  !
365  ! -- If nlist is known, then exit when nlist values have been read
366  if(this%nlist > 0) then
367  if(ii == this%nlist) exit readloop
368  endif
369  !
370  ! -- increment ii
371  ii = ii + 1
372  !
373  enddo readloop
374  !
375  ! -- return
376  return
377  end subroutine read_binary

379  subroutine read_ascii(this)
380 ! ******************************************************************************
381 ! read_ascii -- read the data from an ascii file
382 ! ******************************************************************************
383 !
384 ! SPECIFICATIONS:
385 ! ------------------------------------------------------------------------------
386  ! -- modules
387  use constantsmodule, only: lenboundname, linelength, dzero
388  use inputoutputmodule, only: u9rdcom, urword, get_node
389  use simmodule, only: store_error, count_errors
390  use arrayhandlersmodule, only: expandarray
391  ! -- dummy
392  class(listreadertype) :: this
393  ! -- local
394  integer(I4B) :: mxlist, ldim, naux
395  integer(I4B) :: ii, jj, idum, nod, istat, increment
396  real(DP) :: r
397  integer(I4B), dimension(:), allocatable :: cellid
398  character(len=LINELENGTH) :: fname
399  character(len=LINELENGTH) :: errmsg
400  ! -- formats
401  character(len=*), parameter :: fmtmxlsterronly = &
402  "('***ERROR READING LIST. &
403  &THE NUMBER OF RECORDS ENCOUNTERED EXCEEDS THE MAXIMUM NUMBER " // &
404  "OF RECORDS. TRY INCREASING MAXBOUND FOR THIS LIST." // &
405  " NUMBER OF RECORDS: ',I0,' MAXBOUND: ',I0)"
406 ! ------------------------------------------------------------------------------
407  !
408  ! -- determine array sizes
409  mxlist = size(this%rlist, 2)
410  ldim = size(this%rlist, 1)
411  naux = size(this%auxvar, 1)
412  this%ntxtrlist = 0
413  this%ntxtauxvar = 0
414  !
415  ! -- Allocate arrays
416  allocate(cellid(this%ndim))
417  !
418  ii = 1
419  readloop: do
420  !
421  ! -- First line was already read, so don't read again
422  if(ii /= 1) call u9rdcom(this%inlist, 0, this%line, this%ierr)
423  !
424  ! -- If this is an unknown-length list, then check for END.
425  ! If found, then backspace, set nlist, and exit readloop.
426  if(this%nlist < 0) then
427  this%lloc = 1
428  call urword(this%line, this%lloc, this%istart, this%istop, 1, idum, r, &
429  this%iout, this%inlist)
430  if(this%line(this%istart:this%istop) == 'END' .or. this%ierr < 0) then
431  ! If ierr < 0, backspace was already performed in u9rdcom, so only
432  ! need to backspace if END was found.
433  if (this%ierr == 0) then
434  backspace(this%inlist)
435  endif
436  this%nlist = ii - 1
437  exit readloop
438  endif
439  endif
440  !
441  ! -- Check range
442  if(ii > mxlist) then
443  inquire(unit=this%inlist, name=fname)
444  write(errmsg, fmtmxlsterronly) ii, mxlist
445  call store_error(errmsg)
446  errmsg = 'Error occurred reading line: ' // trim(this%line)
447  call store_error(errmsg)
448  call store_error_unit(this%inlist)
449  endif
450  !
451  ! -- Read layer, row, column or cell number and assign to nodelist
452  this%lloc = 1
453  if(this%ndim == 3) then
454  !
455  ! -- Grid is structured; read layer, row, column
456  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
457  cellid(1), r, this%iout, this%inlist)
458  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
459  cellid(2), r, this%iout, this%inlist)
460  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
461  cellid(3), r, this%iout, this%inlist)
462  !
463  ! -- Check for illegal grid location
464  if(cellid(1) < 1 .or. cellid(1) > this%mshape(1)) then
465  write(errmsg, *) ' Layer number in list is outside of the grid', &
466  cellid(1)
467  call store_error(errmsg)
468  end if
469  if(cellid(2) < 1 .or. cellid(2) > this%mshape(2)) then
470  write(errmsg, *) ' Row number in list is outside of the grid', &
471  cellid(2)
472  call store_error(errmsg)
473  end if
474  if(cellid(3) < 1 .or. cellid(3) > this%mshape(3)) then
475  write(errmsg, *) ' Column number in list is outside of the grid', &
476  cellid(3)
477  call store_error(errmsg)
478  end if
479  !
480  ! -- Calculate nodenumber and put in nodelist
481  nod = get_node(cellid(1), cellid(2), cellid(3), &
482  this%mshape(1), this%mshape(2), this%mshape(3))
483  elseif(this%ndim == 2) then
484  !
485  ! -- Grid is disv
486  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
487  cellid(1), r, this%iout, this%inlist)
488  call urword(this%line, this%lloc, this%istart, this%istop, 2, &
489  cellid(2), r, this%iout, this%inlist)
490  !
491  ! -- Check for illegal grid location
492  if(cellid(1) < 1 .or. cellid(1) > this%mshape(1)) then
493  write(errmsg, *) ' Layer number in list is outside of the grid', &
494  cellid(1)
495  call store_error(errmsg)
496  end if
497  if(cellid(2) < 1 .or. cellid(2) > this%mshape(2)) then
498  write(errmsg, *) ' Cell2d number in list is outside of the grid', &
499  cellid(2)
500  call store_error(errmsg)
501  end if
502  !
503  ! -- Calculate nodenumber and put in nodelist
504  nod = get_node(cellid(1), 1, cellid(2), &
505  this%mshape(1), 1, this%mshape(2))
506  else
507  !
508  ! -- Grid is unstructured; read layer and celld2d number
509  call urword(this%line, this%lloc, this%istart, this%istop, 2, nod, r, &
510  this%iout, this%inlist)
511  if(nod < 1 .or. nod > this%mshape(1)) then
512  write(errmsg, *) ' Node number in list is outside of the grid', nod
513  call store_error(errmsg)
514  end if
515  !
516  endif
517  !
518  ! -- Assign nod to nodelist
519  this%nodelist(ii) = nod
520  !
521  ! -- Read rlist
522  do jj = 1, ldim
523  call urword(this%line, this%lloc, this%istart, this%istop, 0, idum, &
524  r, this%iout, this%inlist)
525  read(this%line(this%istart:this%istop), *, iostat=istat) r
526  !
527  ! -- If a double precision value, then store in rlist, otherwise store
528  ! the text name and location
529  if (istat == 0) then
530  this%rlist(jj, ii) = r
531  else
532  this%rlist(jj, ii) = dzero
533  this%ntxtrlist = this%ntxtrlist + 1
534  if(this%ntxtrlist > size(this%txtrlist)) then
535  increment = size(this%txtrlist) * 0.2
536  increment = max(100, increment)
537  call expandarray(this%txtrlist, increment)
538  call expandarray(this%idxtxtrow, increment)
539  call expandarray(this%idxtxtcol, increment)
540  endif
541  this%txtrlist(this%ntxtrlist) = this%line(this%istart:this%istop)
542  this%idxtxtrow(this%ntxtrlist) = ii
543  this%idxtxtcol(this%ntxtrlist) = jj
544  endif
545  !
546  enddo
547  !
548  ! -- Read auxvar
549  do jj = 1, naux
550  call urword(this%line, this%lloc, this%istart, this%istop, 0, idum, &
551  r, this%iout, this%inlist)
552  read(this%line(this%istart:this%istop), *, iostat=istat) r
553  !
554  ! -- If a double precision value, then store in auxvar, otherwise store
555  ! the text name and location
556  if (istat == 0) then
557  this%auxvar(jj, ii) = r
558  else
559  this%auxvar(jj, ii) = dzero
560  this%ntxtauxvar = this%ntxtauxvar + 1
561  if(this%ntxtauxvar > size(this%txtauxvar)) then
562  increment = size(this%txtauxvar) * 0.2
563  increment = max(100, increment)
564  call expandarray(this%txtauxvar, increment)
565  call expandarray(this%idxtxtauxrow, increment)
566  call expandarray(this%idxtxtauxcol, increment)
567  endif
568  this%txtauxvar(this%ntxtauxvar) = this%line(this%istart:this%istop)
569  this%idxtxtauxrow(this%ntxtauxvar) = ii
570  this%idxtxtauxcol(this%ntxtauxvar) = jj
571  endif
572  !
573  enddo
574  !
575  ! -- Read the boundary names (only supported for ascii input)
576  if (this%inamedbound > 0) then
577  call urword(this%line, this%lloc, this%istart, this%istop, 1, idum, r, &
578  this%iout, this%inlist)
579  this%boundname(ii) = this%line(this%istart:this%istop)
580  endif
