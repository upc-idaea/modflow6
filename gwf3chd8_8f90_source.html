161  endif
162  enddo

274  subroutine chd_cq(this, x, flowja, iadv)
275 ! ******************************************************************************
276 ! chd_cq -- Calculate constant head flow. This method overrides bnd_cq().
277 ! ******************************************************************************
278 !
279 ! SPECIFICATIONS:
280 ! ------------------------------------------------------------------------------
281  ! -- modules
282  ! -- dummy
283  class(chdtype), intent(inout) :: this
284  real(DP), dimension(:), intent(in) :: x
285  real(DP), dimension(:), contiguous, intent(inout) :: flowja
286  integer(I4B), optional, intent(in) :: iadv
287  ! -- local
288  integer(I4B) :: i
289  integer(I4B) :: ipos
290  integer(I4B) :: node
291  integer(I4B) :: n2
292  integer(I4B) :: idiag
293  real(DP) :: rate
294  real(DP) :: ratein, rateout
295  real(DP) :: q
296 ! ------------------------------------------------------------------------------
297  !
298  ! -- If no boundaries, skip flow calculations.
299  if(this%nbound > 0) then
300  !
301  ! -- Loop through each boundary calculating flow.
302  do i = 1, this%nbound
303  node = this%nodelist(i)
304  idiag = this%dis%con%ia(node)
305  rate = dzero
306  ratein = dzero
307  rateout = dzero
308  !
309  ! -- Calculate the flow rate into the cell.
310  do ipos = this%dis%con%ia(node) + 1, &
311  this%dis%con%ia(node + 1) - 1
312  q = flowja(ipos)
313  rate = rate - q
314  ! -- only accumulate chin and chout for active
315  ! connected cells
316  n2 = this%dis%con%ja(ipos)
317  if (this%ibound(n2) > 0) then
318  if (q < dzero) then
319  ratein = ratein - q
320  else
321  rateout = rateout + q
322  end if
323  end if
324  end do
325  !
326  ! -- For chd, store total flow in rhs so it is available for other
327  ! calculations
328  this%rhs(i) = -rate
329  this%hcof(i) = dzero
330  !
331  ! -- Save simulated value to simvals array.
332  this%simvals(i) = rate
333  this%ratechdin(i) = ratein
334  this%ratechdout(i) = rateout
335  flowja(idiag) = flowja(idiag) + rate
336  !
337  end do
338  !
339  end if
340  !
341  ! -- return
342  return
343  end subroutine chd_cq
