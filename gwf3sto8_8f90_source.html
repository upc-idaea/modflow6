274  !> @ brief Fill A and right-hand side for the package
275  !!
276  !! Fill the coefficient matrix and right-hand side with the STO package terms.
277  !!
278  !<
279  subroutine sto_fc(this, kiter, hold, hnew, njasln, amat, idxglo, rhs)
280  ! -- modules
281  use tdismodule, only: delt
282  ! -- dummy variables
283  class(gwfstotype) :: this !< GwfStoType object
284  integer(I4B), intent(in) :: kiter !< outer iteration number
285  real(DP), intent(in), dimension(:) :: hold !< previous heads
286  real(DP), intent(in), dimension(:) :: hnew !< current heads
287  integer(I4B), intent(in) :: njasln !< size of the A matrix for the solution
288  real(DP), dimension(njasln), intent(inout) :: amat !< A matrix
289  integer(I4B), intent(in), dimension(:) :: idxglo !< global index model to solution
290  real(DP), intent(inout), dimension(:) :: rhs !< right-hand side
291  ! -- local variables
292  integer(I4B) :: n
293  integer(I4B) :: idiag
294  real(DP) :: tled
295  real(DP) :: sc1
296  real(DP) :: sc2
297  real(DP) :: rho1
298  real(DP) :: rho2
299  real(DP) :: sc1old
300  real(DP) :: sc2old
301  real(DP) :: rho1old
302  real(DP) :: rho2old
303  real(DP) :: tp
304  real(DP) :: bt
305  real(DP) :: snold
306  real(DP) :: snnew
307  real(DP) :: aterm
308  real(DP) :: rhsterm
309  ! -- formats
310  character(len=*), parameter :: fmtsperror = &
311  &"('DETECTED TIME STEP LENGTH OF ZERO. GWF STORAGE PACKAGE CANNOT BE ', &
312  &'USED UNLESS DELT IS NON-ZERO.')"
313  !
314  ! -- test if steady-state stress period
315  if (this%iss /= 0) return
316  !
317  ! -- Ensure time step length is not zero
318  if (delt == dzero) then
319  write (errmsg, fmtsperror)
320  call store_error(errmsg, terminate=.true.)
321  end if
322  !
323  ! -- set variables
324  tled = done/delt
325  !
326  ! -- loop through and calculate storage contribution to hcof and rhs
327  do n = 1, this%dis%nodes
328  idiag = this%dis%con%ia(n)
329  if (this%ibound(n) < 1) cycle
330  !
331  ! -- aquifer elevations and thickness
332  tp = this%dis%top(n)
333  bt = this%dis%bot(n)
334  !
335  ! -- aquifer saturation
336  if (this%iconvert(n) == 0) then
337  snold = done
338  snnew = done
339  else
340  snold = squadraticsaturation(tp, bt, hold(n), this%satomega)
341  snnew = squadraticsaturation(tp, bt, hnew(n), this%satomega)
342  end if
343  !
344  ! -- storage coefficients
345  sc1 = sscapacity(this%istor_coef, tp, bt, this%dis%area(n), this%ss(n))
346  rho1 = sc1 * tled
347  !
348  if(this%integratechanges /= 0) then
349  ! -- Integration of storage changes (e.g. when using TVS):
350  ! separate the old (start of time step) and new (end of time step)
351  ! primary storage capacities
352  sc1old = sscapacity(this%istor_coef, tp, bt, this%dis%area(n), &
353  this%oldss(n))
354  rho1old = sc1old * tled
355  else
356  ! -- No integration of storage changes: old and new values are
357  ! identical => normal MF6 storage formulation
358  rho1old = rho1
359  end if
360  !
361  ! -- calculate specific storage terms
362  call ssterms(this%iconvert(n), this%iorig_ss, this%iconf_ss, tp, bt, &
363  rho1, rho1old, snnew, snold, hnew(n), hold(n), &
364  aterm, rhsterm)
365  !
366  ! -- add specific storage terms to amat and rhs
367  amat(idxglo(idiag)) = amat(idxglo(idiag)) + aterm
368  rhs(n) = rhs(n) + rhsterm
369  !
370  ! -- specific yield
371  if (this%iconvert(n) /= 0) then
372  rhsterm = dzero
373  !
374  ! -- secondary storage coefficient
375  sc2 = sycapacity(this%dis%area(n), this%sy(n))
376  rho2 = sc2 * tled
377  !
378  if(this%integratechanges /= 0) then
379  ! -- Integration of storage changes (e.g. when using TVS):
380  ! separate the old (start of time step) and new (end of time step)
381  ! secondary storage capacities
382  sc2old = sycapacity(this%dis%area(n), this%oldsy(n))
383  rho2old = sc2old * tled
384  else
385  ! -- No integration of storage changes: old and new values are
386  ! identical => normal MF6 storage formulation
387  rho2old = rho2
388  end if
389  !
390  ! -- calculate specific storage terms
391  call syterms(tp, bt, rho2, rho2old, snnew, snold, &
392  aterm, rhsterm)
393 !
394  ! -- add specific yield terms to amat and rhs
395  amat(idxglo(idiag)) = amat(idxglo(idiag)) + aterm
396  rhs(n) = rhs(n) + rhsterm
397  end if
398  end do
399  !
400  ! -- return
401  return
402  end subroutine sto_fc

404  !> @ brief Fill Newton-Raphson terms in A and right-hand side for the package
405  !!
406  !! Fill the coefficient matrix and right-hand side with STO package
407  !! with Newton-Raphson terms.
408  !!
409  !<
410  subroutine sto_fn(this, kiter, hold, hnew, njasln, amat, idxglo, rhs)
411  ! -- modules
412  use tdismodule, only: delt
413  ! -- dummy variables
414  class(gwfstotype) :: this !< GwfStoType object
415  integer(I4B), intent(in) :: kiter !< outer iteration number
416  real(DP), intent(in), dimension(:) :: hold !< previous heads
417  real(DP), intent(in), dimension(:) :: hnew !< current heads
418  integer(I4B), intent(in) :: njasln !< size of the A matrix for the solution
419  real(DP), dimension(njasln), intent(inout) :: amat !< A matrix
420  integer(I4B), intent(in), dimension(:) :: idxglo !< global index model to solution
421  real(DP), intent(inout), dimension(:) :: rhs !< right-hand side
422  ! -- local variables
423  integer(I4B) :: n
424  integer(I4B) :: idiag
425  real(DP) :: tled
426  real(DP) :: sc1
427  real(DP) :: sc2
428  real(DP) :: rho1
429  real(DP) :: rho2
430  real(DP) :: tp
431  real(DP) :: bt
432  real(DP) :: tthk
433  real(DP) :: h
434  real(DP) :: snnew
435  real(DP) :: derv
436  real(DP) :: rterm
437  real(DP) :: drterm
438  !
439  ! -- test if steady-state stress period
440  if (this%iss /= 0) return
441  !
442  ! -- set variables
443  tled = done/delt
444  !
445  ! -- loop through and calculate storage contribution to hcof and rhs
446  do n = 1, this%dis%nodes
447  idiag = this%dis%con%ia(n)
448  if (this%ibound(n) <= 0) cycle
449  !
450  ! -- aquifer elevations and thickness
451  tp = this%dis%top(n)
452  bt = this%dis%bot(n)
453  tthk = tp - bt
454  h = hnew(n)
455  !
456  ! -- aquifer saturation
457  snnew = squadraticsaturation(tp, bt, h)
458  !
459  ! -- storage coefficients
460  sc1 = sscapacity(this%istor_coef, tp, bt, this%dis%area(n), this%ss(n))
461  sc2 = sycapacity(this%dis%area(n), this%sy(n))
462  rho1 = sc1*tled
463  rho2 = sc2*tled
464  !
465  ! -- calculate newton terms for specific storage
466  ! and specific yield
467  if (this%iconvert(n) /= 0) then
468  !
469  ! -- calculate saturation derivative
470  derv = squadraticsaturationderivative(tp, bt, h)
471  !
472  ! -- newton terms for specific storage
473  if (this%iconf_ss == 0) then
474  if (this%iorig_ss == 0) then
475  drterm = -rho1 * derv * (h - bt) + rho1 * tthk * snnew * derv
476  else
477  drterm = -(rho1 * derv * h)
478  end if
479  amat(idxglo(idiag)) = amat(idxglo(idiag)) + drterm
480  rhs(n) = rhs(n) + drterm*h
481  end if
482  !
483  ! -- newton terms for specific yield
484  ! only calculated if the current saturation
485  ! is less than one
486  if (snnew < done) then
487  ! -- calculate newton terms for specific yield
488  if (snnew > dzero) then
489  rterm = -rho2 * tthk * snnew
490  drterm = -rho2 * tthk * derv
491  amat(idxglo(idiag)) = amat(idxglo(idiag)) + drterm + rho2
492  rhs(n) = rhs(n) - rterm + drterm * h + rho2 * bt
493  end if
494  end if
495  end if
496  end do
497  !
498  ! -- return
499  return
500  end subroutine sto_fn

611  end if
612  !

616 
617  end if
618  this%strgsy(n) = rate
619  !
