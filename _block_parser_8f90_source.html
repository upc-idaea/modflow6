179  !!
180  !! Method to get the next line from a file.
181  !!
182  !<
183  subroutine getnextline(this, endOfBlock)
184  ! -- dummy variables
185  class(blockparsertype), intent(inout) :: this !< BlockParserType object
186  logical, intent(out) :: endOfBlock !< boolean indicating if the end of the block was read
187  ! -- local variables
188  integer(I4B) :: ierr
189  integer(I4B) :: ival
190  integer(I4B) :: istart
191  integer(I4B) :: istop
192  real(DP) :: rval
193  character(len=10) :: key
194  logical :: lineread
195  !
196  ! -- initialize local variables
197  endofblock = .false.
198  ierr = 0
199  lineread = .false.
200  !
201  ! -- read next line
202  loop1: do
203  if (lineread) exit loop1
204  call u9rdcom(this%iuext, this%iout, this%line, ierr)
205  this%lloc = 1
206  call urword(this%line, this%lloc, istart, istop, 0, ival, rval, &
207  this%iout, this%iuext)
208  key = this%line(istart:istop)
209  call upcase(key)
210  if (key == 'END' .or. key == 'BEGIN') then
211  call uterminate_block(this%inunit, this%iout, key, &
212  this%blockNameFound, this%lloc, this%line, &
213  ierr, this%iuext)
214  this%iuactive = this%iuext
215  endofblock = .true.
216  lineread = .true.
217  elseif (key == '') then
218  ! End of file reached.
219  ! If this is an OPEN/CLOSE file, close the file and read the next
220  ! line from this%inunit.
221  if (this%iuext /= this%inunit) then
222  close(this%iuext)
223  this%iuext = this%inunit
224  this%iuactive = this%inunit
225  else
226  errmsg = 'Unexpected end of file reached.'
227  call store_error(errmsg)
228  call this%StoreErrorUnit()
229  endif
230  else
231  this%lloc = 1
232  this%linesRead = this%linesRead + 1
233  lineread = .true.
234  endif
235  enddo loop1
236  !
237  ! -- return
238  return
239  end subroutine getnextline
