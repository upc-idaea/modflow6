127 subroutine get_filename(iunit, fname)
128 ! ******************************************************************************
129 ! Get filename from unit number. If the INQUIRE function returns the full
130 ! path (for example, the INTEL compiler) then the returned file name (fname)
131 ! is limited to the filename without the path.
132 ! ******************************************************************************
133 !
134 ! SPECIFICATIONS:
135 ! ------------------------------------------------------------------------------
136  ! -- modules
137  ! -- dummy
138  integer(I4B), intent(in) :: iunit
139  character(len=*), intent(inout) :: fname
140  ! -- local
141  integer(I4B) :: ipos
142  integer(I4B) :: ios
143  integer(I4B) :: ilen
144 ! ------------------------------------------------------------------------------
145  !
146  ! -- get file name from unit number
147  inquire(unit=iunit, name=fname)
148  !
149  ! -- determine the operating system
150  ios = get_os()
151  !
152  ! -- extract filename from full path, if present
153  ! forward slash on linux, unix, and osx
154  if (ios /= oswin) then
155  ipos = index(fname, '/', back=.true.)
156  end if
157  !
158  ! -- check for backslash on windows or undefined os and
159  ! forward slashes were not found
160  if (ios == oswin .or. ios == osundef) then
161  if (ipos < 1) then
162  ipos = index(fname, '\', back=.true.)
163  end if
164  end if
165  !
166  ! -- exclude the path from the file name
167  if (ipos > 0) then
168  ilen = len_trim(fname)
169  write(fname, '(a)') fname(ipos+1:ilen) // ' '
170  end if
171  !
172  ! -- return
173  return
174 end subroutine get_filename
