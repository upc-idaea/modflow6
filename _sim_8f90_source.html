154  !> @brief Get the file name
155  !!
156  !! Subroutine to get the file name from the unit number for a open file.
157  !! If the INQUIRE function returns the full path (for example, the INTEL
158  !! compiler) then the returned file name (fname) is limited to the filename
159  !! without the path.
160  !!
161  !<
162  subroutine get_filename(iunit, fname)
163  ! -- dummy variables
164  integer(I4B), intent(in) :: iunit !< open file unit number
165  character(len=*), intent(inout) :: fname !< file name attached to the open file unit number
166  ! -- local variables
167  integer(I4B) :: ipos
168  integer(I4B) :: ios
169  integer(I4B) :: ilen
170  !
171  ! -- get file name from unit number
172  inquire(unit=iunit, name=fname)
173  !
174  ! -- determine the operating system
175  ios = get_os()
176  !
177  ! -- extract filename from full path, if present
178  ! forward slash on linux, unix, and osx
179  if (ios /= oswin) then
180  ipos = index(fname, '/', back=.true.)
181  end if
182  !
183  ! -- check for backslash on windows or undefined os and
184  ! forward slashes were not found
185  if (ios == oswin .or. ios == osundef) then
186  if (ipos < 1) then
187  ipos = index(fname, '\', back=.true.)
188  end if
189  end if
190  !
191  ! -- exclude the path from the file name
192  if (ipos > 0) then
193  ilen = len_trim(fname)
194  write(fname, '(a)') fname(ipos+1:ilen) // ' '
195  end if
196  !
197  ! -- return
198  return
199  end subroutine get_filename

519  !> @brief Create final message
520  !!
521  !! Subroutine that creates the appropriate final message and
522  !! terminates the program with an error message, if necessary.
523  !!
524  !<
525  subroutine final_message()
526  ! -- modules
527  use simvariablesmodule, only: isimcnvg, numnoconverge, ireturnerr, &
528  isimcontinue
529  ! -- formats
530  character(len=*), parameter :: fmtnocnvg = &
531  "(1x, 'Simulation convergence failure occurred ', i0, ' time(s).')"
532  !
533  ! -- Write message if nonconvergence occured in at least one timestep
534  if(numnoconverge > 0) then
535  write(warnmsg, fmtnocnvg) numnoconverge
536  if (isimcontinue == 0) then
537  call sim_errors%store_message(warnmsg)
538  else
539  call sim_warnings%store_message(warnmsg)
540  end if
541  endif
542  !
543  ! -- write final message
544  if(isimcnvg == 0) then
545  call print_final_message('Premature termination of simulation.', iout)
546  else
547  call print_final_message('Normal termination of simulation.', iout)
548  endif
549  !
550  ! -- If the simulation did not converge and the continue
551  ! option was not set, then set the return code to 1. The
552  ! purpose of setting the returncode this way is that the
553  ! program will terminate without a stop code if the simulation
554  ! reached the end and the continue flag was set, even if the
555  ! the simulation did not converge.
556  if (isimcnvg == 0 .and. isimcontinue == 0) then
557  ireturnerr = 1
558  end if
559  !
560  ! -- destroy messages
561  call sim_errors%deallocate_message()
562  call sim_uniterrors%deallocate_message()
563  call sim_warnings%deallocate_message()
564  call sim_notes%deallocate_message()
565  !
566  ! -- return or halt
567  if (iforcestop == 1) then
568  call stop_with_error(ireturnerr)
569  end if
570
