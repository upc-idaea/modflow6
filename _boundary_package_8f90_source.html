592 ! ******************************************************************************
593 ! bnd_cq_simrate -- Calculate flow between package and aquifer and store in
594 ! simval
595 ! ******************************************************************************
596 !
597 ! SPECIFICATIONS:

611  ! -- formats
612 ! ------------------------------------------------------------------------------
613  !
614  ! -- If no boundaries, skip flow calculations.

629  end if
630  flowja(idiag) = flowja(idiag) + rrate
631  end if

660  real(DP) :: fact
661  real(DP) :: rrate
662  ! -- formats
663 ! ------------------------------------------------------------------------------
664  !
665  ! -- If no boundaries, skip flow calculations.
666  if (this%nbound > 0) then
667  !
668  ! -- Loop through each boundary calculating flow.
669  do i = 1, this%nbound
670  node = this%nodelist(i)
671  !
672  ! -- If cell is no-flow or constant-head, then ignore it.
673  rrate = dzero
674  if (node > 0) then
675  if(this%ibound(node) > 0) then
676  !
677  ! -- Calculate the flow rate into the cell.
678  q = this%simvals(i)
679 
680 
681  if (q < dzero) then
682  rrate = this%pakmvrobj%get_qtomvr(i)
683  !
684  ! -- Evaluate if qtomvr exceeds the calculated rrate.
685  ! When fact is greater than 1, qtomvr is numerically
686  ! larger than rrate (which should never happen) and
687  ! represents a water budget error. When this happens,
688  ! rrate is set to 0. so that the water budget error is
689  ! correctly accounted for in the listing water budget.
690  fact = -rrate / q
691  if (fact > done) then
692  ! -- all flow goes to mover
693  q = dzero
694  else
695  ! -- magnitude of rrate (which is negative) is reduced by
696  ! qtomvr (which is positive)
697  q = q + rrate
698  end if
699  this%simvals(i) = q
700 
701  if (rrate > dzero) then
702  rrate = -rrate
703  end if
704  end if
705  end if
706  end if
707  !
708  ! -- Save simulated value to simtomvr array.
709  this%simtomvr(i) = rrate
710  !
711  end do
712  endif
713  !
714  ! -- return
715  return
716  end subroutine bnd_cq_simtomvr
