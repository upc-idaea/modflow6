644  !> @ brief Calculate flow to the mover.
645  !!
646  !! Calculate the flow between package and the model that is sent to the
647  !! mover package and store in the simtomvr variable. This method only
648  !! needs to be overridden if a different calculation needs to be made.
649  !!
650  !<
651  subroutine bnd_cq_simtomvr(this, flowja)
652  ! -- dummy variables
653  class(bndtype) :: this !< BndType object
654  real(DP), dimension(:), intent(inout) :: flowja !< flow between package and model
655  ! -- local variables
656  integer(I4B) :: i
657  integer(I4B) :: node
658  real(DP) :: q
659  real(DP) :: fact
660  real(DP) :: rrate
661  !
662  ! -- If no boundaries, skip flow calculations.
663  if (this%nbound > 0) then
664  !
665  ! -- Loop through each boundary calculating flow.
666  do i = 1, this%nbound
667  node = this%nodelist(i)
668  !
669  ! -- If cell is no-flow or constant-head, then ignore it.
670  rrate = dzero
671  if (node > 0) then
672  if(this%ibound(node) > 0) then
673  !
674  ! -- Calculate the flow rate into the cell.
675  q = this%simvals(i)
676 
677 
678  if (q < dzero) then
679  rrate = this%pakmvrobj%get_qtomvr(i)
680  !
681  ! -- Evaluate if qtomvr exceeds the calculated rrate.
682  ! When fact is greater than 1, qtomvr is numerically
683  ! larger than rrate (which should never happen) and
684  ! represents a water budget error. When this happens,
685  ! rrate is set to 0. so that the water budget error is
686  ! correctly accounted for in the listing water budget.
687  fact = -rrate / q
688  if (fact > done) then
689  ! -- all flow goes to mover
690  q = dzero
691  else
692  ! -- magnitude of rrate (which is negative) is reduced by
693  ! qtomvr (which is positive)
694  q = q + rrate
695  end if
696  this%simvals(i) = q
697 
698  if (rrate > dzero) then
699  rrate = -rrate
700  end if
701  end if
702  end if
703  end if
704  !
705  ! -- Save simulated value to simtomvr array.
706  this%simtomvr(i) = rrate
707  !
708  end do
709  endif

712  return
