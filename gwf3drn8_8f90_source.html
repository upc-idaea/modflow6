291  subroutine drn_cf(this, reset_mover)
292 ! ******************************************************************************
293 ! drn_cf -- Formulate the HCOF and RHS terms
294 ! Subroutine: (1) skip if no drains
295 ! (2) calculate hcof and rhs
296 ! ******************************************************************************
297 !
298 ! SPECIFICATIONS:
299 ! ------------------------------------------------------------------------------
300  ! -- dummy
301  class(drntype) :: this
302  logical, intent(in), optional :: reset_mover
303  ! -- local
304  integer(I4B) :: i
305  integer(I4B) :: node
306  real(DP) :: cdrn
307  real(DP) :: drnbot
308  real(DP) :: fact
309  logical :: lrm
310 ! ------------------------------------------------------------------------------
311  !
312  ! -- Return if no drains
313  if(this%nbound == 0) return
314  !
315  ! -- pakmvrobj cf
316  lrm = .true.
317  if (present(reset_mover)) lrm = reset_mover
318  if (this%imover == 1 .and. lrm) then
319  call this%pakmvrobj%cf()
320  end if
321  !
322  ! -- Calculate hcof and rhs for each drn entry
323  do i = 1, this%nbound
324  node = this%nodelist(i)
325  if(this%ibound(node) <= 0) then
326  this%hcof(i) = dzero
327  this%rhs(i) = dzero
328  cycle
329  end if
330  !
331  ! -- set local variables for this drain
332  cdrn = this%bound(2,i)
333  !
334  ! -- calculate the drainage scaling factor
335  call this%get_drain_factor(i, fact, drnbot)
336  !
337  ! -- calculate rhs and hcof
338  this%rhs(i) = -fact * cdrn * drnbot
339  this%hcof(i) = -fact * cdrn
340  end do
341  !
342  ! -- return
343  return
344  end subroutine drn_cf

346  subroutine drn_fc(this, rhs, ia, idxglo, amatsln)
347 ! **************************************************************************
348 ! drn_fc -- Copy rhs and hcof into solution rhs and amat
349 ! **************************************************************************
350 !
351 ! SPECIFICATIONS:
352 ! --------------------------------------------------------------------------
353  ! -- dummy
354  class(drntype) :: this
355  real(DP), dimension(:), intent(inout) :: rhs
356  integer(I4B), dimension(:), intent(in) :: ia
357  integer(I4B), dimension(:), intent(in) :: idxglo
358  real(DP), dimension(:), intent(inout) :: amatsln
359  ! -- local
360  integer(I4B) :: i
361  integer(I4B) :: n
362  integer(I4B) :: ipos
363  real(DP) :: fact
364  real(DP) :: drnbot
365  real(DP) :: drncond
366  real(DP) :: qdrn
367 ! --------------------------------------------------------------------------
368  !
369  ! -- packmvrobj fc
370  if(this%imover == 1) then
371  call this%pakmvrobj%fc()
372  endif
373  !
374  ! -- Copy package rhs and hcof into solution rhs and amat
375  do i = 1, this%nbound
376  n = this%nodelist(i)
377  rhs(n) = rhs(n) + this%rhs(i)
378  ipos = ia(n)
379  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + this%hcof(i)
380  !
381  ! -- calculate the drainage scaling factor
382  call this%get_drain_factor(i, fact, drnbot)
383  !
384  ! -- If mover is active and this drain is discharging,
385  ! store available water (as positive value).
386  if(this%imover == 1 .and. fact > dzero) then
387  drncond = this%bound(2,i)
388  qdrn = fact * drncond * (this%xnew(n) - drnbot)
389  call this%pakmvrobj%accumulate_qformvr(i, qdrn)
390  endif
391  enddo
392  !
393  ! -- return
394  return
395  end subroutine drn_fc

397  subroutine drn_fn(this, rhs, ia, idxglo, amatsln)
398 ! **************************************************************************
399 ! drn_fn -- Fill newton terms
400 ! **************************************************************************
401 !
402 ! SPECIFICATIONS:
403 ! --------------------------------------------------------------------------
404  implicit none
405  ! -- dummy
406  class(drntype) :: this
407  real(DP), dimension(:), intent(inout) :: rhs
408  integer(I4B), dimension(:), intent(in) :: ia
409  integer(I4B), dimension(:), intent(in) :: idxglo
410  real(DP), dimension(:), intent(inout) :: amatsln
411  ! -- local
412  integer(I4B) :: i
413  integer(I4B) :: node
414  integer(I4B) :: ipos
415  real(DP) :: cdrn
416  real(DP) :: xnew
417  real(DP) :: drndepth
418  real(DP) :: drntop
419  real(DP) :: drnbot
420  real(DP) :: drterm
421 ! --------------------------------------------------------------------------
422 
423  !
424  ! -- Copy package rhs and hcof into solution rhs and amat
425  if (this%iauxddrncol /= 0) then
426  do i = 1, this%nbound
427  node = this%nodelist(i)
428  !
429  ! -- test if node is constant or inactive
430  if(this%ibound(node) <= 0) then
431  cycle
432  end if
433  !
434  ! -- set local variables for this drain
435  cdrn = this%bound(2,i)
436  xnew = this%xnew(node)
437  !
438  ! -- calculate the drainage depth and the top and bottom of
439  ! the conductance scaling elevations
440  call this%get_drain_elevations(i, drndepth, drntop, drnbot)
441  !
442  ! -- calculate scaling factor
443  if (drndepth /= dzero) then
444  drterm = sqsaturationderivative(drntop, drnbot, xnew, &
445  c1=-done, c2=dtwo)
446  drterm = drterm * cdrn * (drnbot - xnew)
447  !
448  ! -- fill amat and rhs with newton-raphson terms
449  ipos = ia(node)
450  amatsln(idxglo(ipos)) = amatsln(idxglo(ipos)) + drterm
451  rhs(node) = rhs(node) + drterm * xnew
452  end if
453  end do
454  end if
455  !
456  ! -- return
457  return
458  end subroutine drn_fn

538 ! ******************************************************************************

540 ! ******************************************************************************
541 !
542 ! SPECIFICATIONS:
543 ! ------------------------------------------------------------------------------
544  ! -- dummy

549  ! -- local

553  real(DP) :: drntop
554  real(DP) :: drnbot
555 ! ------------------------------------------------------------------------------
556  !
557  ! -- set local variables for this drain
558  node = this%nodelist(i)
559  xnew = this%xnew(node)
560  !
561  ! -- calculate the drainage depth and the top and bottom of

563  call this%get_drain_elevations(i, drndepth, drntop, drnbot)
564  !

566  if (present(opt_drnbot)) then
567  opt_drnbot = drnbot
568  end if
569  !
570  ! -- calculate scaling factor
571  if (drndepth /= dzero) then
572  if (this%icubic_scaling /= 0) then
573  factor = sqsaturation(drntop, drnbot, xnew, c1=-done, c2=dtwo)
574  else
575  factor = squadraticsaturation(drntop, drnbot, xnew, eps=dzero)
576  end if
577  else
578  if (xnew <= drnbot) then
579  factor = dzero
580  else
581  factor = done
582  end if
583  end if
584  !
585  ! -- return
586  return
