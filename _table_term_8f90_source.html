44 ! ******************************************************************************
45 ! initialize -- initialize the table term
46 ! ******************************************************************************
47 !
48 ! SPECIFICATIONS:
49 ! ------------------------------------------------------------------------------

54  integer(I4B), intent(in) :: width
55  integer(I4B), intent(in), optional :: alignment
56  ! -- local
57  character(len=LINELENGTH) :: string
58  character(len=LINELENGTH) :: tstring
59  character(len=LINELENGTH), allocatable, dimension(:) :: words
60  integer(I4B) :: nwords
61  integer(I4B) :: ilen
62  integer(I4B) :: i
63  integer(I4B) :: j
64 
65 ! ------------------------------------------------------------------------------
66  !
67  ! -- allocate scalars
68  call this%allocate_scalars()
69 
70  ! -- process dummy variables
71  this%tag = tag
72 
73  if (present(alignment)) then
74  this%alignment = alignment
75  else
76  this%alignment = tabcenter
77  end if
78 
79  this%width = width
80  !
81  ! -- parse tag into words
82  call parseline(tag, nwords, words, 0)
83  !
84  ! -- abbreviate any words that exceed the specified width
85  ! and trim trailing characters
86  do i = 1, nwords
87  ilen = len(trim(words(i)))
88  if (ilen > width) then
89  words(i)(width:width) = '.'
90  do j = width + 1, ilen
91  words(i)(j:j) = ' '
92  end do
93  end if
94  end do
95  !
96  ! -- combine words that fit into width
97  i = 0
98  do
99  i = i + 1
100  if (i > nwords) then
101  exit
102  end if
103  string = trim(adjustl(words(i)))
104  tstring = string
105  do j = i + 1, nwords
106  if (len(trim(adjustl(string))) > 0) then
107  tstring = trim(adjustl(tstring)) // ' ' // trim(adjustl(words(j)))
108  else
109  tstring = trim(adjustl(words(j)))
110  end if
111  ilen = len(trim(adjustl(tstring)))
112  if (ilen == 0) then
113  continue
114  else if (ilen <= width) then
115  words(j) = ' '
116  string = tstring
117  else
118  exit
119  end if
120  end do
121  words(i) = string
122  end do
123  !
124  ! -- calculate the number of header lines
125  do i = 1, nwords
126  ilen = len(trim(adjustl(words(i))))
127  if (ilen > 0) then
128  this%nheader_lines = this%nheader_lines + 1
129  end if
130  end do
131  !
132  ! allocate initial_lines and fill with words
133  allocate(this%initial_lines(this%nheader_lines))
134  do i = 1, this%nheader_lines
135  this%initial_lines(i) = words(i)(1:width)
136  end do
137  !
138  ! -- deallocate words
139  deallocate(words)
140  !
141  ! -- return
142  return
143 
144  end subroutine initialize
