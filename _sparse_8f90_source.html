140  subroutine insert(j, thisrow, inodup, iadded)
141  !insert j into thisrow (for row i)
142  !inodup=1 means do not include duplicate connections
143  !iadded is 1 if a new entry was added (meaning that nnz for the row was increased)
144  !iadded is 0 if duplicate and not added. Used to track total number of connections
145  ! -- dummy
146  integer(I4B), intent(in) :: j, inodup
147  type(rowtype), intent(inout) :: thisrow
148  integer(I4B), allocatable,dimension(:) :: iwk
149  integer(I4B), intent(inout) :: iadded
150  ! -- local
151  integer(I4B) :: jj, maxnnz
152  ! -- code
153  iadded = 0
154  maxnnz = ubound(thisrow%icolarray,dim=1)
155  if (thisrow%icolarray(1) == 0) then
156  thisrow%icolarray(1) = j
157  thisrow%nnz = thisrow%nnz + 1
158  iadded = 1
159  return
160  end if
161  if (thisrow%nnz == maxnnz) then
162  ! -- increase size of the row
163  allocate(iwk(thisrow%nnz))
164  iwk = thisrow%icolarray
165  deallocate(thisrow%icolarray)
166  ! -- Specify how to increase the size of the icolarray. Adding 1
167  ! will be most memory conservative, but may be a little slower
168  ! due to frequent allocate/deallocate. Another option would be
169  ! to double the size: maxnnz=maxnnz*2
170  maxnnz = maxnnz + 1
171  allocate(thisrow%icolarray(maxnnz))
172  thisrow%icolarray(1:thisrow%nnz) = iwk(1:thisrow%nnz)
173  thisrow%icolarray(thisrow%nnz+1:maxnnz) = 0
174  end if
175  if (inodup == 1) then
176  do jj = 1, thisrow%nnz
177  if (thisrow%icolarray(jj)==j) then
178  return
179  end if
180  end do
181  end if
182  !
183  ! -- add the connection to end
184  thisrow%nnz = thisrow%nnz + 1
185  thisrow%icolarray(thisrow%nnz) = j
186  iadded = 1
187  !
188  ! -- return
189  return
190  end subroutine insert
