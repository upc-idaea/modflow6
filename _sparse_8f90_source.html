161  subroutine insert(j, thisrow, inodup, iadded)
162  !insert j into thisrow (for row i)
163  !inodup=1 means do not include duplicate connections
164  !iadded is 1 if a new entry was added (meaning that nnz for the row was increased)
165  !iadded is 0 if duplicate and not added. Used to track total number of connections
166  ! -- dummy
167  integer(I4B), intent(in) :: j, inodup
168  type(rowtype), intent(inout) :: thisrow
169  integer(I4B), allocatable,dimension(:) :: iwk
170  integer(I4B), intent(inout) :: iadded
171  ! -- local
172  integer(I4B) :: jj, maxnnz
173  ! -- code
174  iadded = 0
175  maxnnz = ubound(thisrow%icolarray,dim=1)
176  if (thisrow%icolarray(1) == 0) then
177  thisrow%icolarray(1) = j
178  thisrow%nnz = thisrow%nnz + 1
179  iadded = 1
180  return
181  end if
182  if (thisrow%nnz == maxnnz) then
183  ! -- increase size of the row
184  allocate(iwk(thisrow%nnz))
185  iwk = thisrow%icolarray
186  deallocate(thisrow%icolarray)
187  ! -- Specify how to increase the size of the icolarray. Adding 1
188  ! will be most memory conservative, but may be a little slower
189  ! due to frequent allocate/deallocate. Another option would be
190  ! to double the size: maxnnz=maxnnz*2
191  maxnnz = maxnnz + 1
192  allocate(thisrow%icolarray(maxnnz))
193  thisrow%icolarray(1:thisrow%nnz) = iwk(1:thisrow%nnz)
194  thisrow%icolarray(thisrow%nnz+1:maxnnz) = 0
195  end if
196  if (inodup == 1) then
197  do jj = 1, thisrow%nnz
198  if (thisrow%icolarray(jj)==j) then
199  return
200  end if
201  end do
202  end if
203  !
204  ! -- add the connection to end
205  thisrow%nnz = thisrow%nnz + 1
206  thisrow%icolarray(thisrow%nnz) = j
207  iadded = 1
208  !
209  ! -- return
210  return
211  end subroutine insert
