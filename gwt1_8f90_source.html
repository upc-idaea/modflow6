641  subroutine gwt_cq(this, icnvg, isuppress_output)
642 ! ******************************************************************************
643 ! gwt_cq --Groundwater transport model calculate flow
644 ! Subroutine: (1) Calculate intercell flows (flowja)
645 ! ******************************************************************************
646 !
647 ! SPECIFICATIONS:
648 ! ------------------------------------------------------------------------------
649  ! -- modules
650  use sparsemodule, only: csr_diagsum
651  ! -- dummy
652  class(gwtmodeltype) :: this
653  integer(I4B), intent(in) :: icnvg
654  integer(I4B), intent(in) :: isuppress_output
655  ! -- local
656  integer(I4B) :: i
657  integer(I4B) :: ip
658  class(bndtype), pointer :: packobj
659 ! ------------------------------------------------------------------------------
660  !
661  ! -- Construct the flowja array. Flowja is calculated each time, even if
662  ! output is suppressed. (flowja is positive into a cell.) The diagonal
663  ! position of the flowja array will contain the flow residual after
664  ! these routines are called, so each package is responsible for adding
665  ! its flow to this diagonal position.
666  do i = 1, this%nja
667  this%flowja(i) = dzero
668  enddo
669  if(this%inadv > 0) call this%adv%adv_cq(this%x, this%flowja)
670  if(this%indsp > 0) call this%dsp%dsp_cq(this%x, this%flowja)
671  if(this%inmst > 0) call this%mst%mst_cq(this%dis%nodes, this%x, this%xold, &
672  this%flowja)
673  if(this%inssm > 0) call this%ssm%ssm_cq(this%flowja)
674  if(this%infmi > 0) call this%fmi%fmi_cq(this%x, this%flowja)
675  !
676  ! -- Go through packages and call cq routines. cf() routines are called
677  ! first to regenerate non-linear terms to be consistent with the final
678  ! conc solution.
679  do ip = 1, this%bndlist%Count()
680  packobj => getbndfromlist(this%bndlist, ip)
681  call packobj%bnd_cf(reset_mover=.false.)
682  call packobj%bnd_cq(this%x, this%flowja)
683  enddo
684  !
685  ! -- Finalize calculation of flowja by adding face flows to the diagonal.
686  ! This results in the flow residual being stored in the diagonal
687  ! position for each cell.
688  call csr_diagsum(this%dis%con%ia, this%flowja)
689  !
690  ! -- Return
691  return
692  end subroutine gwt_cq
