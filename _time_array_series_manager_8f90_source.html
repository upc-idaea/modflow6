104  subroutine tasmgr_ad(this)
105 ! ******************************************************************************
106 ! tasmgr_ad -- time step (or subtime step) advance.
107 ! Call this each time step or subtime step.
108 ! ******************************************************************************
109 !
110 ! SPECIFICATIONS:
111 ! ------------------------------------------------------------------------------
112  ! -- dummy
113  class(timearrayseriesmanagertype) :: this
114  ! -- local
115  type(timearrayserieslinktype), pointer :: tasLink => null()
116  type(timearrayseriestype), pointer :: timearrayseries => null()
117  integer(I4B) :: i, j, nlinks, nvals, isize1, isize2, inunit
118  real(DP) :: begintime, endtime
119  ! formats
120  character(len=*),parameter :: fmt5 = &
121  "(/,'Time-array-series controlled arrays in stress period ', &
122  &i0, ', time step ', i0, ':')"
123 10 format('"',a, '" package: ',a,' array obtained from time-array series "',a,'"')
124 ! ------------------------------------------------------------------------------
125  !
126  ! -- Initialize time variables
127  begintime = totimc
128  endtime = begintime + delt
129  !
130  ! -- Iterate through boundtaslinks and update specified
131  ! array with array of average values obtained from
132  ! appropriate time series.
133  if (associated(this%boundTasLinks)) then
134  nlinks = this%boundTasLinks%Count()
135  do i = 1, nlinks
136  taslink => gettimearrayserieslinkfromlist(this%boundTasLinks, i)
137  if (taslink%Iprpak == 1 .and. i==1) then
138  write(this%iout, fmt5) kper, kstp
139  endif
140  if (taslink%UseDefaultProc) then
141  timearrayseries => taslink%timeArraySeries
142  nvals = size(taslink%BndArray)
143  !
144  ! -- Fill the package array with integrated values
145  call timearrayseries%GetAverageValues(nvals, taslink%BndArray, &
146  begintime, endtime)
147  !
148  ! -- If conversion from flux to flow is required, multiply by cell area
149  if (taslink%ConvertFlux) then
150  call this%tasmgr_convert_flux(taslink)
151  endif
152  !
153  ! -- If PRINT_INPUT is specified, write information
154  ! regarding source of time-array series data
155  if (taslink%Iprpak == 1) then
156  write(this%iout,10) trim(taslink%PackageName), &
157  trim(taslink%Text), &
158  trim(taslink%timeArraySeries%Name)
159  endif
160  endif
161  if (i == nlinks) then
162  write(this%iout, '()')
163  endif
164  enddo
165  !
166  ! -- Now that all array values have been substituted, can now multiply
167  ! an array by a multiplier array
168  do i = 1, nlinks
169  taslink => gettimearrayserieslinkfromlist(this%boundTasLinks, i)
170  if (taslink%UseDefaultProc) then
171  if (associated(taslink%RMultArray)) then
172  isize1 = size(taslink%BndArray)
173  isize2 = size(taslink%RMultArray)
174  if (isize1 == isize2 .and. isize1 == nvals) then
175  do j = 1, nvals
176  taslink%BndArray(j) = taslink%BndArray(j) * taslink%RMultArray(j)
177  enddo
178  else
179  errmsg = 'Size mismatch between boundary and multiplier arrays' // &
180  ' using time-array series: ' // &
181  trim(taslink%TimeArraySeries%Name)
182  call store_error(errmsg)
183  inunit = taslink%TimeArraySeries%GetInunit()
184  call store_error_unit(inunit)
185  endif
186  endif
187  endif
188  enddo
189  endif
190  !
191  return
192  end subroutine tasmgr_ad

253  subroutine reset(this, pkgName)
254 ! ******************************************************************************
255 ! Reset -- zero out arrays that are represented with time series.
256 ! Delete all existing links from time array series to package arrays as they
257 ! will need to be created with a new BEGIN PERIOD block.
258 ! ******************************************************************************
259 !
260 ! SPECIFICATIONS:
261 ! ------------------------------------------------------------------------------
262  ! -- modules
263  ! -- dummy
264  class(timearrayseriesmanagertype) :: this
265  character(len=*), intent(in) :: pkgName
266  ! -- local
267  integer(I4B) :: i, j, nlinks
268  type(timearrayserieslinktype), pointer :: taslink
269 ! ------------------------------------------------------------------------------
270  !
271  ! -- Reassign all linked elements to zero
272  nlinks = this%boundTasLinks%Count()
273  do i = 1, nlinks
274  taslink => gettimearrayserieslinkfromlist(this%boundTasLinks, i)
275  if (associated(taslink)) then
276  do j = 1, size(taslink%BndArray)
277  taslink%BndArray(j) = dzero
278  enddo
279  endif
280  enddo
281  !
282  ! -- Delete all existing time array links
283  if (associated(this%boundTasLinks)) then
284  ! Deallocate and remove all links belonging to package
285  nlinks = this%boundTasLinks%Count()
286  do i = nlinks, 1, -1
287  taslink => gettimearrayserieslinkfromlist(this%boundTasLinks, i)
288  if (associated(taslink)) then
289  call taslink%da()

291  endif
292  enddo
293  endif
294  !
