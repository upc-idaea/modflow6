402  !> @ brief Define the solution
403  !!
404  !! Define a new solution. Must be called after the models and exchanges have
405  !! been added to solution. The order of the steps is (1) Allocate neq and nja,
406  !! (2) Assign model offsets and solution ids, (3) Allocate and initialize
407  !! the solution arrays, (4) Point each model's x and rhs arrays, and
408  !! (5) Initialize the sparsematrix instance
409  !!
410  !<
411  subroutine sln_df(this)
412  ! modules
413  use memorymanagermodule, only: mem_allocate
414  ! -- dummy variables
415  class(numericalsolutiontype) :: this !< NumericalSolutionType instance
416  ! -- local variables
417  class(numericalmodeltype), pointer :: mp => null()
418  integer(I4B) :: i
419  integer(I4B), allocatable, dimension(:) :: rowmaxnnz
420  !
421  ! -- calculate and set offsets
422  do i = 1, this%modellist%Count()
423  mp => getnumericalmodelfromlist(this%modellist, i)
424  call mp%set_idsoln(this%id)
425  call mp%set_moffset(this%neq)
426  this%neq = this%neq + mp%neq
427  enddo
428  !
429  ! -- Allocate and initialize solution arrays
430  call this%allocate_arrays()
431  !
432  ! -- Go through each model and point x, ibound, and rhs to solution
433  do i = 1, this%modellist%Count()
434  mp => getnumericalmodelfromlist(this%modellist, i)
435  call mp%set_xptr(this%x, 'X', this%name)
436  call mp%set_rhsptr(this%rhs, 'RHS', this%name)
437  call mp%set_iboundptr(this%active, 'IBOUND', this%name)
438  enddo
439  !
440  ! -- Create the sparsematrix instance
441  allocate(rowmaxnnz(this%neq))
442  do i=1,this%neq
443  rowmaxnnz(i)=4
444  enddo
445  call this%sparse%init(this%neq, this%neq, rowmaxnnz)
446  deallocate(rowmaxnnz)
447  !
448  ! -- Assign connections, fill ia/ja, map connections
449  call this%sln_connect()
450  !
451  ! -- return
452  return
453  end subroutine sln_df

1004  !! Calculate time step length.
1005  !!
1006  !<
1007  subroutine sln_calculate_delt(this)
1008  ! -- modules
1009  use tdismodule, only: kstp, kper, delt
1010  use adaptivetimestepmodule, only: ats_submit_delt
1011  use constantsmodule, only: dtwo, dthree
1012  ! -- dummy variables
1013  class(numericalsolutiontype) :: this !< NumericalSolutionType instance
1014  ! -- local variables
1015  integer(I4B) :: idir
1016  real(DP) :: delt_temp
1017  real(DP) :: fact_lower
1018  real(DP) :: fact_upper
1019  !
1020  ! -- increase or decrease delt based on kiter fraction. atsfrac should be
1021  ! a value of about 1/3. If the number of outer iterations is less than
1022  ! 1/3 of mxiter, then increase step size. If the number of outer
1023  ! iterations is greater than 2/3 of mxiter, then decrease step size.
1024  if (this%atsfrac > dzero) then
1025  delt_temp = delt
1026  fact_lower = this%mxiter * this%atsfrac
1027  fact_upper = this%mxiter - fact_lower
1028  if (this%iouttot_timestep < int(fact_lower)) then
1029  ! -- increase delt according to tsfactats
1030  idir = 1
1031  else if (this%iouttot_timestep > int(fact_upper)) then
1032  ! -- decrease delt according to tsfactats
1033  idir = -1
1034  else
1035  ! -- do not change delt
1036  idir = 0
1037  end if
1038  !
1039  ! -- submit stable dt for upcoming step
1040  call ats_submit_delt(kstp, kper, delt_temp, this%memoryPath, idir=idir)
1041  end if
1042  !
1043  return
1044  end subroutine sln_calculate_delt
