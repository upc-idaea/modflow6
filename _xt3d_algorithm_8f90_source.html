15 ! ******************************************************************************
16 !
17 !.....Compute the "conductances" in the normal-flux expression for an
18 ! interface (modflow-usg version). The cell on one side of
19 ! the interface is "cell 0", and the one on the other side is
20 ! "cell 1".
21 !
22 ! nnbrmx = maximum number of neighbors allowed for a cell.
23 ! nnbr0 = number of neighbors (local connections) for cell 0.
24 ! inbr0 = array with the list of neighbors for cell 0.
25 ! il01 = local node number of cell 1 with respect to cell 0.
26 ! vc0 = array of connection unit-vectors for cell 0 with its
27 ! neighbors.
28 ! vn0 = array of unit normal vectors for cell 0's interfaces.
29 ! dl0 = array of lengths contributed by cell 0 to its
30 ! connections with its neighbors.
31 ! dl0n = array of lengths contributed by cell 0's neighbors to
32 ! their connections with cell 0.
33 ! ck0 = conductivity tensor for cell 0.
34 ! nnbr1 = number of neighbors (local connections) for cell 1.
35 ! inbr1 = array with the list of neighbors for cell 1.
36 ! il10 = local node number of cell 0 with respect to cell 1.
37 ! vc1 = array of connection unit-vectors for cell 1 with its
38 ! neighbors.
39 ! vn1 = array of unit normal vectors for cell 1's interfaces.
40 ! dl1 = array of lengths contributed by cell 1 to its
41 ! connections with its neighbors.
42 ! dl1n = array of lengths contributed by cell 1's neighbors to
43 ! their connections with cell 1.
44 ! ck1 = conductivity tensor for cell1.
45 ! ar01 = area of interface (0,1).
46 ! ar10 = area of interface (1,0).
47 ! chat01 = "conductance" for connection (0,1).
48 ! chati0 = array of "conductances" for connections of cell 0.
49 ! (zero for connection with cell 1, as this connection is
50 ! already covered by chat01.)
51 ! chat1j = array of "conductances" for connections of cell 1.
52 ! (zero for connection with cell 0., as this connection is
53 ! already covered by chat01.)
54 ! ******************************************************************************
55 !
56 ! SPECIFICATIONS:
57 ! ------------------------------------------------------------------------------
58  ! -- dummy
59  integer(I4B) :: nnbrmx
60  integer(I4B) :: nnbr0
61  integer(I4B), dimension(nnbrmx) :: inbr0
62  integer(I4B) :: il01
63  real(DP), dimension(nnbrmx, 3) :: vc0
64  real(DP), dimension(nnbrmx, 3) :: vn0
65  real(DP), dimension(nnbrmx) :: dl0
66  real(DP), dimension(nnbrmx) :: dl0n
67  real(DP), dimension(3, 3) :: ck0
68  integer(I4B) :: nnbr1
69  integer(I4B), dimension(nnbrmx) :: inbr1
70  integer(I4B) :: il10
71  real(DP), dimension(nnbrmx) :: vc1
72  real(DP), dimension(nnbrmx) :: vn1
73  real(DP), dimension(nnbrmx) :: dl1
74  real(DP), dimension(nnbrmx) :: dl1n
75  real(DP), dimension(3, 3) :: ck1
76  real(DP) :: ar01
77  real(DP) :: ar10
78  real(DP) :: vcthresh
79  logical :: allhc0
80  logical :: allhc1
81  real(DP) :: chat01
82  real(DP), dimension(nnbrmx) :: chati0
83  real(DP), dimension(nnbrmx) :: chat1j
84  ! -- local
85  integer(I4B) :: i1
86  integer(I4B) :: i
87  real(DP) :: ahat0
88  real(DP) :: ahat1
89  real(DP) :: wght1
90  real(DP) :: wght0
91  real(DP), dimension(nnbrmx) :: bhat0
92  real(DP), dimension(nnbrmx) :: bhat1
93  real(DP) :: denom
94 ! ------------------------------------------------------------------------------
95 !
96 !.....Set the global cell number for cell 1, as found in the neighbor
97 ! list for cell 0. It is assumed that cells 0 and 1 are both
98 ! active, or else this subroutine would not have been called.
99  i1 = inbr0(il01)
100 !
101 !.....If area ar01 is zero (in which case ar10 is also zero, since
102 ! this can only happen here in the case of Newton), then the
103 ! "conductances" are all zero.
104  if (ar01.eq.0d0) then
105  chat01 = 0d0
106  do i=1,nnbrmx
107  chati0(i) = 0d0
108  chat1j(i) = 0d0
109  enddo
110 !.....Else compute "conductances."
111  else
112 !........Compute contributions from cell 0.
113  call abhats(nnbrmx,nnbr0,inbr0,il01,vc0,vn0,dl0,dl0n,ck0, &
114  vcthresh,allhc0,ar01,ahat0,bhat0)
115 !........Compute contributions from cell 1.
116  call abhats(nnbrmx,nnbr1,inbr1,il10,vc1,vn1,dl1,dl1n,ck1, &
117  vcthresh,allhc1,ar10,ahat1,bhat1)
118 !........Compute "conductances" based on the two flux estimates.
119  denom = (ahat0 + ahat1)
120  if (abs(denom) > dprec) then
121  wght1 = ahat0/(ahat0 + ahat1)
122  else
123  wght1 = done
124  end if
125  wght0 = 1d0 - wght1
126  chat01 = wght1*ahat1
127  do i=1,nnbrmx
128  chati0(i) = wght0*bhat0(i)
129  chat1j(i) = wght1*bhat1(i)
130  enddo
131  end if
132 !
133  return
134  end subroutine qconds

136  subroutine abhats(nnbrmx,nnbr,inbr,il01,vc,vn,dl0,dln,ck, &
137  vcthresh,allhc,ar01,ahat,bhat)
138 ! ******************************************************************************
139 !.....Compute "ahat" and "bhat" coefficients for one side of an
140 ! interface.
141 ! ******************************************************************************
142 !
143 ! SPECIFICATIONS:
144 ! ------------------------------------------------------------------------------
145  ! -- dummy
146  integer(I4B) :: nnbrmx
147  integer(I4B) :: nnbr
148  integer(I4B), dimension(nnbrmx) :: inbr
149  integer(I4B) :: il01
150  real(DP), dimension(nnbrmx, 3) :: vc
151  real(DP), dimension(nnbrmx, 3) :: vn
152  real(DP), dimension(nnbrmx) :: dl0
153  real(DP), dimension(nnbrmx) :: dln
154  real(DP), dimension(3, 3) :: ck
155  real(DP) :: vcthresh
156  logical :: allhc
157  real(DP) :: ar01
158  real(DP) :: ahat
159  real(DP), dimension(nnbrmx) :: bhat
160  ! -- local
161  logical :: iscomp
162  real(DP), dimension(nnbrmx, 3) :: vccde
163  real(DP), dimension(3, 3) :: rmat
164  real(DP), dimension(3) :: sigma
165  real(DP), dimension(nnbrmx) :: bd
166  real(DP), dimension(nnbrmx) :: be
167  real(DP), dimension(nnbrmx) :: betad
168  real(DP), dimension(nnbrmx) :: betae
169  integer(I4B) :: iml0
170  integer(I4B) :: il
171  real(DP) :: acd
172  real(DP) :: add
173  real(DP) :: aed
174  real(DP) :: ace
175  real(DP) :: aee
176  real(DP) :: ade
177  real(DP) :: determ
178  real(DP) :: oodet
179  real(DP) :: alphad
180  real(DP) :: alphae
181  real(DP) :: dl0il
182 ! ------------------------------------------------------------------------------
183 !
184 !.....Determine the basis vectors for local "(c, d, e)" coordinates
185 ! associated with the connection between cells 0 and 1, and
186 ! set the rotation matrix that transforms vectors from model
187 ! coordinates to (c, d, e) coordinates. (If no active
188 ! connection is found that has a non-negligible component
189 ! perpendicular to the primary connection, ilmo=0 is returned.)
190  call getrot(nnbrmx,nnbr,inbr,vc,il01,rmat,iml0)
191 !
192 !.....If no active connection with a non-negligible perpendicular
193 ! component, assume no perpendicular gradient and base gradient
194 ! solely on the primary connection. Otherwise, proceed with
195 ! basing weights on information from neighboring connections.
196  if (iml0.eq.0) then
197 !
198 !........Compute ahat and bhat coefficients assuming perpendicular
199 ! components of gradient are zero.
200  sigma(1) = dot_product(vn(il01, :), matmul(ck, rmat(:, 1)))
201  ahat = sigma(1) / dl0(il01)
202  bhat = 0d0
203 !
204  else
205 !
206 !........Transform local connection unit-vectors from model coordinates
207 ! to "(c, d, e)" coordinates associated with the connection
208 ! between cells 0 and 1.
209  call tranvc(nnbrmx,nnbr,rmat,vc,vccde)
210 !
211 !........Get "a" and "b" weights for first perpendicular direction.
212  call abwts(nnbrmx,nnbr,inbr,il01,2,vccde, &
213  vcthresh,dl0,dln,acd,add,aed,bd)
214 !
215 !........If all neighboring connections are user-designated as
216 ! horizontal, or if none have a non-negligible component in
217 ! the second perpendicular direction, assume zero gradient in
218 ! the second perpendicular direction. Otherwise, get "a" and
219 ! "b" weights for second perpendicular direction based on
220 ! neighboring connections.
221  if (allhc) then
222  ace = 0d0
223  aee = 1d0
224  ade = 0d0
225  be = 0d0
226  else
227  iscomp = .false.
228  do il = 1, nnbr
229  if ((il == il01) .or. (inbr(il) == 0)) then
230  cycle
231  else if (dabs(vccde(il, 3)) > 1d-10) then
232  iscomp = .true.
233  exit
234  end if
235  end do
236  if (iscomp) then
237  call abwts(nnbrmx,nnbr,inbr,il01,3,vccde, &
238  vcthresh,dl0,dln,ace,aee,ade,be)
239  else
240  ace = 0d0
241  aee = 1d0
242  ade = 0d0
243  be = 0d0
244  end if
245  end if
246 !
247 !........Compute alpha and beta coefficients.
248  determ = add * aee - ade * aed
249  oodet = 1d0 / determ
250  alphad = (acd * aee - ace * aed) * oodet
251  alphae = (ace * add - acd * ade) * oodet
252  betad = 0d0
253  betae = 0d0
254  do il = 1, nnbr
255 !...........If this is connection (0,1) or inactive, skip.
256  if ((il == il01) .or. (inbr(il) == 0)) cycle
257  betad(il) = (bd(il) * aee - be(il) * aed) * oodet
258  betae(il) = (be(il) * add - bd(il) * ade) * oodet
259  end do
260 !
261 !........Compute sigma coefficients.
262  sigma = matmul(vn(il01, :), matmul(ck, rmat))
263 !
264 !........Compute ahat and bhat coefficients.
265  ahat = (sigma(1) - sigma(2) * alphad - sigma(3) * alphae) / dl0(il01)
266  bhat = 0d0
267  do il = 1, nnbr
268 !...........If this is connection (0,1) or inactive, skip.
269  if ((il == il01) .or. (inbr(il) == 0)) cycle
270  dl0il = dl0(il) + dln(il)
271  bhat(il) = (sigma(2) * betad(il) + sigma(3) * betae(il)) / dl0il
272  end do
273 !........Set the bhat for connection (0,1) to zero here, since we have
274 ! been skipping it in our do loops to avoiding explicitly
275 ! computing it. This will carry through to the corresponding
276 ! chati0 and chat1j value, so that they too are zero.
277  bhat(il01) = 0d0
278 !
279  end if
280 !
281 !.....Multiply by area.
282  ahat = ahat * ar01
283  bhat = bhat * ar01
284 !
285  return
286  end subroutine abhats

288  subroutine getrot(nnbrmx,nnbr,inbr,vc,il01,rmat,iml0)
289 ! ******************************************************************************
290 !.....Compute the matrix that rotates the model-coordinate axes to
291 ! the "(c, d, e)-coordinate" axes associated with a connection.
292 ! This is also the matrix that transforms the components of a vector
293 ! from (c, d, e) coordinates to model coordinates. [Its transpose
294 ! transforms from model to (c, d, e) coordinates.]
295 !
296 ! vcc = unit vector for the primary connection, in model
297 ! coordinates.
298 ! vcd = unit vector for the first perpendicular direction,
299 ! in model coordinates.
300 ! vce = unit vector for the second perpendicular direction,
301 ! in model coordinates.
302 ! vcmax = unit vector for the connection with the maximum
303 ! component perpendicular to the primary connection,
304 ! in model coordinates.
305 ! rmat = rotation matrix from model to (c, d, e) coordinates.
306 !
307 ! ******************************************************************************
308 !
309 ! SPECIFICATIONS:
310 ! ------------------------------------------------------------------------------
311  ! -- dummy
312  integer(I4B) :: nnbrmx
313  integer(I4B) :: nnbr
314  integer(I4B), dimension(nnbrmx) :: inbr
315  real(DP), dimension(nnbrmx, 3) :: vc
316  integer(I4B) :: il01
317  real(DP), dimension(3, 3) :: rmat
318  integer(I4B) :: iml0
319  ! -- local
320  real(DP), dimension(3) :: vcc
321  real(DP), dimension(3) :: vcd
322  real(DP), dimension(3) :: vce
323  real(DP), dimension(3) :: vcmax
324  integer(I4B) :: il
325  real(DP) :: acmpmn
326  real(DP) :: cmp
327  real(DP) :: acmp
328  real(DP) :: cmpmn
329 ! ------------------------------------------------------------------------------
330 !
331 !.....set vcc.
332  vcc(:) = vc(il01,:)
333 !
334 !.....Set vcmax. (If no connection has a perpendicular component
335 ! greater than some tiny threshold, return with iml0=0 and
336 ! the first column of rmat set to vcc -- the other columns
337 ! are not needed.)
338  acmpmn = 1d0 - 1d-10
339  iml0 = 0
340  do il = 1, nnbr
341  if ((il.eq.il01).or.(inbr(il).eq.0)) then
342  cycle
343  else
344  cmp = dot_product(vc(il,:), vcc)
345  acmp = dabs(cmp)
346  if (acmp.lt.acmpmn) then
347  cmpmn = cmp
348  acmpmn = acmp
349  iml0 = il
350  end if
351  end if
352  enddo
353  if (iml0 == 0) then
354  rmat(:,1) = vcc(:)
355  goto 999
356  else
357  vcmax(:) = vc(iml0,:)
358  end if
359 !
360 !.....Set the first perpendicular direction as the direction that is
361 ! coplanar with vcc and vcmax and perpendicular to vcc.
362  vcd = vcmax - cmpmn * vcc
363  vcd = vcd / dsqrt(1d0 - cmpmn * cmpmn)
364 !
365 !.....Set the second perpendicular direction as the cross product of
366 ! the primary and first-perpendicular directions.
367  vce(1) = vcc(2)*vcd(3) - vcc(3)*vcd(2)
368  vce(2) = vcc(3)*vcd(1) - vcc(1)*vcd(3)
369  vce(3) = vcc(1)*vcd(2) - vcc(2)*vcd(1)
370 !
371 !.....Set the rotation matrix as the matrix with vcc, vcd, and vce
372 ! as its columns.
373  rmat(:,1) = vcc(:)
374  rmat(:,2) = vcd(:)
375  rmat(:,3) = vce(:)
376 !
377 999 return
378  end subroutine getrot

380  subroutine tranvc(nnbrmx,nnbrs,rmat,vc,vccde)
381 ! ******************************************************************************
382 !.....Transform local connection unit-vectors from model coordinates
383 ! to "(c, d, e)" coordinates associated with a connection.
384 !
385 ! nnbrs = number of neighbors (local connections)
386 ! rmat = rotation matrix from (c, d, e) to model coordinates.
387 ! vc = array of connection unit-vectors with respect to model
388 ! coordinates.
389 ! vccde = array of connection unit-vectors with respect to
390 ! (c, d, e) coordinates.
391 !
392 ! ******************************************************************************
393 !
394 ! SPECIFICATIONS:
395 ! ------------------------------------------------------------------------------
396  ! -- dummy
397  integer(I4B) :: nnbrmx
398  integer(I4B) :: nnbrs
399  real(DP), dimension(3, 3) :: rmat
400  real(DP), dimension(nnbrmx, 3) :: vc
401  real(DP), dimension(nnbrmx, 3) :: vccde
402  ! -- local
403  integer(I4B) :: il
404 ! ------------------------------------------------------------------------------
405 !
406 !.....Loop over the local connections, transforming the unit vectors.
407 ! Note that we are multiplying by the transpose of the
408 ! rotation matrix so that the transformation is from model
409 ! to (c, d, e) coordinates.
410  do il = 1, nnbrs
411  vccde(il,:) = matmul(transpose(rmat), vc(il,:))
412  enddo
413 !
414  return
415  end subroutine tranvc

417  subroutine abwts(nnbrmx,nnbr,inbr,il01,nde1,vccde, &
418  vcthresh,dl0,dln,acd,add,aed,bd)
419 ! ******************************************************************************
420 !.....Compute "a" and "b" weights for the local connections with respect
421 ! to the perpendicular direction of primary interest.
422 !
423 ! nde1 = number that indicates the perpendicular direction of
424 ! primary interest on this call: "d" (2) or "e" (3).
425 ! vccde = array of connection unit-vectors with respect to
426 ! (c, d, e) coordinates.
427 ! bd = array of "b" weights.
428 ! aed = "a" weight that goes on the matrix side of the 2x2
429 ! problem.
430 ! acd = "a" weight that goes on the right-hand side of the
431 ! 2x2 problem.
432 !
433 ! ******************************************************************************
434 !
435 ! SPECIFICATIONS:
436 ! ------------------------------------------------------------------------------
437  ! -- dummy
438  integer(I4B) :: nnbrmx
439  integer(I4B) :: nnbr
440  integer(I4B), dimension(nnbrmx) :: inbr
441  integer(I4B) :: il01
442  integer(I4B) :: nde1
443  real(DP), dimension(nnbrmx, 3) :: vccde
444  real(DP) :: vcthresh
445  real(DP), dimension(nnbrmx) :: dl0
446  real(DP), dimension(nnbrmx) :: dln
447  real(DP) :: acd
448  real(DP) :: add
449  real(DP) :: aed
450  real(DP), dimension(nnbrmx) :: bd
451  ! -- local
452  integer(I4B) :: nde2
453  integer(I4B) :: il
454  real(DP) :: vcmx
455  real(DP) :: dlm
456  real(DP) :: cosang
457  real(DP) :: dl4wt
458  real(DP) :: fact
459  real(DP) :: dsum
460  real(DP) :: oodsum
461  real(DP) :: fatten
462  real(DP), dimension(nnbrmx) :: omwt
463 ! ------------------------------------------------------------------------------
464 !
465 !.....Set the perpendicular direction of secondary interest.
466  nde2 = 5 - nde1
467 !
468 !.....Begin computing "omega" weights.
469  omwt = 0d0
470  dsum = 0d0
471  vcmx = 0d0
472  do il = 1, nnbr
473 !........if this is connection (0,1) or inactive, skip.
474  if ((il.eq.il01).or.(inbr(il).eq.0)) cycle
475  vcmx = max(dabs(vccde(il,nde1)), vcmx)
476  dlm = 5d-1*(dl0(il) + dln(il))
477 !...........Distance-based weighting. dl4wt is the distance between
478 ! the point supplying the gradient information and the
479 ! point at which the flux is being estimated. Could be
480 ! coded as a special case of resistance-based weighting
481 ! (by setting the conductivity matrix to be the identity
482 ! matrix), but this is more efficient.
483  cosang = vccde(il,1)
484  dl4wt = dsqrt(dlm*dlm + dl0(il01)*dl0(il01) &
485  - 2d0*dlm*dl0(il01)*cosang)
486  omwt(il) = dabs(vccde(il,nde1))*dl4wt
487  dsum = dsum + omwt(il)
488  end do
489 !
490 !.....Finish computing non-normalized "omega" weights. [Add a
491 ! tiny bit to dsum so that the normalized omega weight later
492 ! evaluates to (essentially) 1 in the case of a single relevant
493 ! connection, avoiding 0/0.]
494  dsum = dsum + 1d-10*dsum
495  do il = 1, nnbr
496 !........If this is connection (0,1) or inactive, skip.
497  if ((il.eq.il01).or.(inbr(il).eq.0)) cycle
498  fact = dsum - omwt(il)
499  omwt(il) = fact*dabs(vccde(il,nde1))
500  end do
501 !
502 !.....Compute "b" weights.
503  bd = 0d0
504  dsum = 0d0
505  do il = 1, nnbr
506 !........If this is connection (0,1) or inactive, skip.
507  if ((il.eq.il01).or.(inbr(il).eq.0)) cycle
508  bd(il) = omwt(il)*sign(1d0,vccde(il,nde1))
509  dsum = dsum + omwt(il)*dabs(vccde(il,nde1))
510  end do
511  oodsum = 1d0/dsum
512  do il = 1, nnbr
513 !........If this is connection (0,1) or inactive, skip.
514  if ((il.eq.il01).or.(inbr(il).eq.0)) cycle
515  bd(il) = bd(il)*oodsum
516  end do
517 !
518 !.....Compute "a" weights.
519  add = 1d0
520  acd = 0d0
521  aed = 0d0
522  do il = 1, nnbr
523 !........If this is connection (0,1) or inactive, skip.
524  if ((il.eq.il01).or.(inbr(il).eq.0)) cycle
525  acd = acd + bd(il)*vccde(il,1)
526  aed = aed + bd(il)*vccde(il,nde2)
527  end do
528 !
529 !.....Apply attenuation function to acd, aed, and bd.
530  if (vcmx.lt.vcthresh) then
531  fatten = vcmx/vcthresh
532  acd = acd*fatten
533  aed = aed*fatten
534  bd = bd*fatten
535  end if
536 !
537  end subroutine abwts
538 !
539 end module xt3dalgorithmmodule
