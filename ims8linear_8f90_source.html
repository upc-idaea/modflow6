859  !> @ brief Base linear accelerator subroutine
860  !!
861  !! Base linear accelerator subroutine that scales and reorders
862  !! the system of equations, if necessary, updates the preconditioner,
863  !! and calls the appropriate linear accelerator.
864  !!
865  !<
866  SUBROUTINE imslinear_ap(this,ICNVG,KSTP,KITER,IN_ITER, &
867  NCONV, CONVNMOD, CONVMODSTART, LOCDV, LOCDR, &
868  CACCEL, ITINNER, CONVLOCDV, CONVLOCDR, &
869  DVMAX, DRMAX, CONVDVMAX, CONVDRMAX)
870  ! -- modules
871  USE simmodule
872  ! -- dummy variables
873  CLASS(imslineardatatype), INTENT(INOUT) :: this !< ImsLinearDataType instance
874  integer(I4B), INTENT(INOUT) :: ICNVG !< convergence flag (1) non-convergence (0)
875  integer(I4B), INTENT(IN) :: KSTP !< time step number
876  integer(I4B), INTENT(IN) :: KITER !< outer iteration number
877  integer(I4B), INTENT(INOUT) :: IN_ITER !< inner iteration number
878  ! -- convergence information dummy variables
879  integer(I4B), INTENT(IN) :: NCONV !<
880  integer(I4B), INTENT(IN) :: CONVNMOD !<
881  integer(I4B), DIMENSION(CONVNMOD+1), INTENT(INOUT) ::CONVMODSTART !<
882  integer(I4B), DIMENSION(CONVNMOD), INTENT(INOUT) :: LOCDV !<
883  integer(I4B), DIMENSION(CONVNMOD), INTENT(INOUT) :: LOCDR !<
884  character(len=31), DIMENSION(NCONV), INTENT(INOUT) :: CACCEL !<
885  integer(I4B), DIMENSION(NCONV), INTENT(INOUT) :: ITINNER !<
886  integer(I4B), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVLOCDV !<
887  integer(I4B), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVLOCDR !<
888  real(DP), DIMENSION(CONVNMOD), INTENT(INOUT) :: DVMAX !<
889  real(DP), DIMENSION(CONVNMOD), INTENT(INOUT) :: DRMAX !<
890  real(DP), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVDVMAX !<
891  real(DP), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVDRMAX !<
892  ! -- local variables
893  integer(I4B) :: n
894  integer(I4B) :: innerit
895  integer(I4B) :: irc
896  integer(I4B) :: itmax
897  real(DP) :: tv
898  real(DP) :: rmax
899  !
900  ! -- set epfact based on timestep
901  IF (this%ICNVGOPT == 2) THEN
902  IF (kstp == 1) THEN
903  this%EPFACT = 0.01
904  ELSE
905  this%EPFACT = 0.10
906  END IF
907  ELSE IF (this%ICNVGOPT == 4) THEN
908  this%EPFACT = dem4
909  ELSE
910  this%EPFACT = done
911  END IF
912  !
913  ! -- SCALE PROBLEM
914  IF (this%ISCL.NE.0) THEN
915  CALL ims_base_scale(0,this%ISCL, &
916  this%NEQ,this%NJA,this%IA,this%JA, &
917  this%AMAT,this%X,this%RHS, &
918  this%DSCALE,this%DSCALE2)
919  END IF
920  !
921  ! -- PERMUTE ROWS, COLUMNS, AND RHS
922  IF (this%IORD.NE.0) THEN
923  CALL ims_dperm(this%NEQ, this%NJA, this%AMAT,this%JA,this%IA, &
924  & this%ARO,this%JARO,this%IARO,this%LORDER,this%ID,1)
925  CALL ims_vperm(this%NEQ, this%X, this%LORDER)
926  CALL ims_vperm(this%NEQ, this%RHS, this%LORDER)
927  this%IA0 => this%IARO
928  this%JA0 => this%JARO
929  this%A0 => this%ARO
930  ELSE
931  this%IA0 => this%IA
932  this%JA0 => this%JA
933  this%A0 => this%AMAT
934  END IF
935  !
936  ! -- UPDATE PRECONDITIONER
937  CALL ims_base_pcu(this%iout,this%NJA,this%NEQ,this%NIAPC,this%NJAPC, &
938  this%IPC, this%RELAX, this%A0, this%IA0, this%JA0, &
939  this%APC,this%IAPC,this%JAPC,this%IW,this%W, &
940  this%LEVEL, this%DROPTOL, this%NJLU, this%NJW, &
941  this%NWLU, this%JLU, this%JW, this%WLU)
942  !
943  ! -- INITIALIZE SOLUTION VARIABLE AND ARRAYS
944  IF (kiter == 1 ) this%NITERC = 0
945  irc = 1
946  icnvg = 0
947  DO n = 1, this%NEQ
948  this%D(n) = dzero
949  this%P(n) = dzero
950  this%Q(n) = dzero
951  this%Z(n) = dzero
952  END DO
953  !
954  ! -- CALCULATE INITIAL RESIDUAL
955  CALL ims_base_mv(this%NJA,this%NEQ,this%A0,this%X,this%D, &
956  this%IA0,this%JA0)
957  rmax = dzero
958  this%L2NORM0 = dzero
959  DO n = 1, this%NEQ
960  tv = this%D(n)
961  this%D(n) = this%RHS(n) - tv
962  IF (abs( this%D(n) ) > rmax ) rmax = abs( this%D(n) )
963  this%L2NORM0 = this%L2NORM0 + this%D(n) * this%D(n)
964  END DO
965  this%L2NORM0 = sqrt(this%L2NORM0)
966  !
967  ! -- CHECK FOR EXACT SOLUTION
968  itmax = this%ITER1
969  IF (rmax == dzero) THEN
970  itmax = 0
971  icnvg = 1
972  END IF
973  !
974  ! -- SOLUTION BY THE CONJUGATE GRADIENT METHOD
975  IF (this%ILINMETH == 1) THEN
976  CALL ims_base_cg(icnvg, itmax, innerit, &
977  this%NEQ, this%NJA, this%NIAPC, this%NJAPC, &
978  this%IPC, this%NITERC, this%ICNVGOPT, this%NORTH, &
979  this%DVCLOSE, this%RCLOSE, this%L2NORM0, &
980  this%EPFACT, this%IA0, this%JA0, this%A0, &
981  this%IAPC, this%JAPC, this%APC, &
982  this%X, this%RHS, this%D, this%P, this%Q, this%Z, &
983  this%NJLU, this%IW, this%JLU, &
984  nconv, convnmod, convmodstart, locdv, locdr, &
985  caccel, itinner, convlocdv, convlocdr, &
986  dvmax, drmax, convdvmax, convdrmax)
987  !
988  ! -- SOLUTION BY THE BICONJUGATE GRADIENT STABILIZED METHOD
989  ELSE IF (this%ILINMETH == 2) THEN
990  CALL ims_base_bcgs(icnvg, itmax, innerit, &
991  this%NEQ, this%NJA, this%NIAPC, this%NJAPC, &
992  this%IPC, this%NITERC, this%ICNVGOPT, this%NORTH,&
993  this%ISCL, this%DSCALE, &
994  this%DVCLOSE, this%RCLOSE, this%L2NORM0, &
995  this%EPFACT, this%IA0, this%JA0, this%A0, &
996  this%IAPC, this%JAPC, this%APC, &
997  this%X, this%RHS, this%D, this%P, this%Q, &
998  this%T, this%V, this%DHAT, this%PHAT, this%QHAT, &
999  this%NJLU, this%IW, this%JLU, &
1000  nconv, convnmod, convmodstart, locdv, locdr, &
1001  caccel, itinner, convlocdv, convlocdr, &
1002  dvmax, drmax, convdvmax, convdrmax)
1003  END IF
1004  !
1005  ! -- BACK PERMUTE AMAT, SOLUTION, AND RHS
1006  IF (this%IORD.NE.0) THEN
1007  CALL ims_dperm(this%NEQ, this%NJA, this%A0, this%JA0, this%IA0, &
1008  & this%AMAT, this%JA,this%IA,this%IORDER,this%ID,1)
1009  CALL ims_vperm(this%NEQ, this%X, this%IORDER)
1010  CALL ims_vperm(this%NEQ, this%RHS, this%IORDER)
1011  END IF
1012  !
1013  ! -- UNSCALE PROBLEM
1014  IF (this%ISCL.NE.0) THEN
1015  CALL ims_base_scale(1, this%ISCL, &
1016  this%NEQ, this%NJA, this%IA, this%JA, &
1017  this%AMAT, this%X, this%RHS, &
1018  this%DSCALE, this%DSCALE2)
1019  END IF
1020  !
1021  ! -- SET IMS INNER ITERATION NUMBER (IN_ITER) TO NUMBER OF
1022  ! IMSLINEAR INNER ITERATIONS (innerit)
1023  in_iter = innerit
1024  !
1025  ! -- RETURN
1026  RETURN
1027  END SUBROUTINE imslinear_ap
