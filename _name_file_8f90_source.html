157  subroutine namefile_openlistfile(this, iout)
158 ! ******************************************************************************
159 ! namefile_openlistfile -- Open the list file and set iout.
160 ! ******************************************************************************
161 !
162 ! SPECIFICATIONS:
163 ! ------------------------------------------------------------------------------
164  ! -- modules
165  use inputoutputmodule, only: getunit, upcase
166  ! -- dummy
167  class(namefiletype) :: this
168  integer(I4B), intent(inout) :: iout
169  ! -- local
170  logical :: found
171  character(len=LINELENGTH) :: fname
172  integer(I4B) :: i, istart, istop
173  integer(I4B) :: nwords
174  integer(I4B) :: ipos
175  character(len=LINELENGTH), allocatable, dimension(:) :: words
176  ! -- formats
177 ! ------------------------------------------------------------------------------
178  !
179  ! -- Go through the options and see if LIST was specified
180  found = .false.
181  ipos = 0
182  findloop: do i = 1, size(this%opts)
183  call parseline(this%opts(i), nwords, words)
184  call upcase(words(1))
185  if(words(1) == 'LIST') then
186  fname = words(2)
187  ipos = i
188  found = .true.
189  exit findloop
190  endif
191  enddo findloop
192  !
193  ! -- remove list file from options list
194  if (ipos > 0) then
195  call remove_character(this%opts, ipos)
196  end if
197  !
198  ! -- If LIST was not found, then set name of list file by replacing the
199  ! namefile extension with '.lst' If no extension then add to end
200  ! of namefile name.
201  if (.not. found) then
202  fname = ' '
203  istart = 0
204  istop = len_trim(this%filename)
205  do i = istop, 1, -1
206  if (this%filename(i:i) == '.') then
207  istart = i
208  exit
209  endif
210  enddo
211  if (istart == 0) istart = istop + 1
212  fname = this%filename(1:istart)
213  istop = istart + 3
214  fname(istart:istop) = '.lst'
215  endif
216  !
217  ! -- Open the list file
218  iout = getunit()
219  call openfile(iout, 0, trim(fname), 'LIST', filstat_opt='REPLACE')
220  this%opened_listfile = .true.
221  !
222  ! -- return
223  return
224  end subroutine namefile_openlistfile

370  endif
