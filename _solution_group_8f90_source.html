47  subroutine sgp_ca(this)
48 ! ******************************************************************************
49 ! sgp_ca -- Calculate the solution group
50 ! Solve each solution group and each solution. Start with converge
51 ! flag equal true and reset to zero if any non-convergence triggers
52 ! are encountered.
53 ! ******************************************************************************
54 !
55 ! SPECIFICATIONS:
56 ! ------------------------------------------------------------------------------
57  ! -- modules
58  use constantsmodule, only: linelength
59  use simvariablesmodule, only: iout, isimcnvg, laststepfailed
60  use simmodule, only: store_error, ustop
61  use tdismodule, only: kstp, kper
62  ! -- dummy
63  class(solutiongrouptype) :: this
64  ! -- local
65  class(basesolutiontype), pointer :: sp
66  integer(I4B) :: kpicard, isgcnvg, isuppress_output
67  integer(I4B) :: is, isoln
68  ! -- formats
69  character(len=*), parameter :: fmtnocnvg = &
70  "(1X,'Solution Group ', i0, ' did not converge for stress period ', i0, &
71  &' and time step ', i0)"
72 ! ------------------------------------------------------------------------------
73  !
74  ! -- Suppress output during picard iterations
75  if(this%mxiter > 1) then
76  isuppress_output = 1
77  else
78  isuppress_output = 0
79  endif
80  !
81  ! -- set failed flag
82  laststepfailed = 0
83  !
84  ! -- Picard loop
85  picardloop: do kpicard = 1, this%mxiter
86  if (this%mxiter > 1) then
87  write(iout,'(/a,i6/)') 'SOLUTION GROUP PICARD ITERATION: ', kpicard
88  end if
89  isgcnvg = 1
90  do is = 1, this%nsolutions
91  isoln = this%idsolutions(is)
92  sp => getbasesolutionfromlist(basesolutionlist, isoln)
93  call sp%sln_ca(isgcnvg, isuppress_output)
94  enddo
95  if(isgcnvg == 1) exit picardloop
96  enddo picardloop
97  !
98  ! -- if a picard loop was used and the solution group converged
99  ! then rerun the timestep and save the output. Or if there
100  ! is only one picard iteration, then do nothing as models
101  ! are assumed to be explicitly coupled.
102  if(isgcnvg == 1) then
103  if(this%mxiter > 1) then
104  isuppress_output = 0
105  do is = 1, this%nsolutions
106  isoln = this%idsolutions(is)
107  sp => getbasesolutionfromlist(basesolutionlist, isoln)
108  call sp%sln_ca(isgcnvg, isuppress_output)
109  enddo
110  endif
111  else
112  isimcnvg = 0
113  laststepfailed = 1
114  write(iout, fmtnocnvg) this%id, kper, kstp
115  endif
116  !
117  ! -- return
118  return
119  end subroutine sgp_ca
