362  subroutine npf_ad(this, nodes, hold, hnew, irestore)
363 ! ******************************************************************************
364 ! npf_ad -- Advance
365 ! Subroutine (1) Sets hold to bot whenever a wettable cell is dry
366 ! ******************************************************************************
367 !
368 ! SPECIFICATIONS:
369 ! ------------------------------------------------------------------------------
370  use tdismodule, only: kper, kstp
371  !
372  implicit none
373  class(gwfnpftype) :: this

377  integer(I4B), intent(in) :: irestore
378  integer(I4B) :: n
379 ! ------------------------------------------------------------------------------
380  !
381  ! -- loop through all cells and set hold=bot if wettable cell is dry
382  if(this%irewet > 0) then
383  do n = 1, this%dis%nodes
384  if(this%wetdry(n) == dzero) cycle
385  if(this%ibound(n) /= 0) cycle
386  hold(n) = this%dis%bot(n)
387  enddo
388  !
389  ! -- if restore state, then set hnew to DRY if it is a dry wettable cell
390  do n = 1, this%dis%nodes
391  if(this%wetdry(n) == dzero) cycle
392  if(this%ibound(n) /= 0) cycle
393  hnew(n) = dhdry
394  enddo
395  endif
396  !
397  ! -- TVK
398  if(this%intvk /= 0) then
399  call this%tvk%ad()
400  endif
401  !
402  ! -- If any K values have changed, we need to update CONDSAT or XT3D arrays
403  if(this%kchangeper == kper .and. this%kchangestp == kstp) then
404  if(this%ixt3d == 0) then
405  !
406  ! -- Update the saturated conductance for all connections
407  ! -- of the affected nodes
408  do n = 1, this%dis%nodes
409  if(this%nodekchange(n) == 1) then
410  call this%calc_condsat(n, .false.)
411  endif
412  enddo
413  else

580  subroutine npf_fn(this, kiter, njasln, amat, idxglo, rhs, hnew)
581 ! ******************************************************************************
582 ! npf_fn -- Fill newton terms
583 ! ******************************************************************************
584 !
585 ! SPECIFICATIONS:
586 ! ------------------------------------------------------------------------------
587  ! -- dummy
588  class(gwfnpftype) :: this
589  integer(I4B) :: kiter
590  integer(I4B),intent(in) :: njasln
591  real(DP),dimension(njasln),intent(inout) :: amat
592  integer(I4B),intent(in),dimension(:) :: idxglo
593  real(DP),intent(inout),dimension(:) :: rhs
594  real(DP),intent(inout),dimension(:) :: hnew
595  ! -- local
596  integer(I4B) :: nodes, nja
597  integer(I4B) :: n,m,ii,idiag
598  integer(I4B) :: isymcon, idiagm
599  integer(I4B) :: iups
600  integer(I4B) :: idn
601  real(DP) :: cond
602  real(DP) :: consterm
603  real(DP) :: filledterm
604  real(DP) :: derv
605  real(DP) :: hds
606  real(DP) :: term
607  real(DP) :: afac
608  real(DP) :: topup
609  real(DP) :: botup
610  real(DP) :: topdn
611  real(DP) :: botdn
612 ! ------------------------------------------------------------------------------
613  !
614  ! -- add newton terms to solution matrix
615  !
616  nodes = this%dis%nodes
617  nja = this%dis%con%nja
618  if(this%ixt3d /= 0) then
619  call this%xt3d%xt3d_fn(kiter, nodes, nja, njasln, amat, idxglo, rhs, hnew)
620  else
621  !
622  do n=1, nodes
623  idiag=this%dis%con%ia(n)
624  do ii=this%dis%con%ia(n)+1,this%dis%con%ia(n+1)-1
625  if (this%dis%con%mask(ii) == 0) cycle
626 
627  m=this%dis%con%ja(ii)
628  isymcon = this%dis%con%isym(ii)
629  ! work on upper triangle
630  if(m < n) cycle
631  if(this%dis%con%ihc(this%dis%con%jas(ii))==0 .and. &
632  this%ivarcv == 0) then
633  !call this%vcond(n,m,hnew(n),hnew(m),ii,cond)
634  ! do nothing
635  else
636  ! determine upstream node
637  iups = m
638  if (hnew(m) < hnew(n)) iups = n
639  idn = n
640  if (iups == n) idn = m
641  !
642  ! -- no newton terms if upstream cell is confined
643  if (this%icelltype(iups) == 0) cycle
644  !
645  ! -- Set the upstream top and bot, and then recalculate for a
646  ! vertically staggered horizontal connection
647  topup = this%dis%top(iups)
648  botup = this%dis%bot(iups)
649  if(this%dis%con%ihc(this%dis%con%jas(ii)) == 2) then
650  topup = min(this%dis%top(n), this%dis%top(m))
651  botup = max(this%dis%bot(n), this%dis%bot(m))
652  endif
653  !
654  ! get saturated conductivity for derivative
655  cond = this%condsat(this%dis%con%jas(ii))
656  !
657  ! -- if using MODFLOW-NWT upstream weighting option apply
658  ! factor to remove average thickness
659  if (this%inwtupw /= 0) then
660  topdn = this%dis%top(idn)
661  botdn = this%dis%bot(idn)
662  afac = dtwo / (done + (topdn - botdn) / (topup - botup))
663  cond = cond * afac
664  end if
665  !
666  ! compute additional term
667  consterm = -cond * (hnew(iups) - hnew(idn)) !needs to use hwadi instead of hnew(idn)
668  !filledterm = cond
669  filledterm = amat(idxglo(ii))
670  derv = squadraticsaturationderivative(topup, botup, hnew(iups), &
671  this%satomega, this%satmin)
672  idiagm = this%dis%con%ia(m)
673  ! fill jacobian for n being the upstream node
674  if (iups == n) then
675  hds = hnew(m)
676  !isymcon = this%dis%con%isym(ii)
677  term = consterm * derv
678  rhs(n) = rhs(n) + term * hnew(n) !+ amat(idxglo(isymcon)) * (dwadi * hds - hds) !need to add dwadi
679  rhs(m) = rhs(m) - term * hnew(n) !- amat(idxglo(isymcon)) * (dwadi * hds - hds) !need to add dwadi
680  ! fill in row of n
681  amat(idxglo(idiag)) = amat(idxglo(idiag)) + term
682  ! fill newton term in off diagonal if active cell
683  if (this%ibound(n) > 0) then
684  amat(idxglo(ii)) = amat(idxglo(ii)) !* dwadi !need to add dwadi
685  end if
686  !fill row of m
687  amat(idxglo(idiagm)) = amat(idxglo(idiagm)) !- filledterm * (dwadi - DONE) !need to add dwadi
688  ! fill newton term in off diagonal if active cell
689  if (this%ibound(m) > 0) then
690  amat(idxglo(isymcon)) = amat(idxglo(isymcon)) - term
691  end if
692  ! fill jacobian for m being the upstream node
693  else
694  hds = hnew(n)
695  term = -consterm * derv
696  rhs(n) = rhs(n) + term * hnew(m) !+ amat(idxglo(ii)) * (dwadi * hds - hds) !need to add dwadi
697  rhs(m) = rhs(m) - term * hnew(m) !- amat(idxglo(ii)) * (dwadi * hds - hds) !need to add dwadi
698  ! fill in row of n
699  amat(idxglo(idiag)) = amat(idxglo(idiag)) !- filledterm * (dwadi - DONE) !need to add dwadi
700  ! fill newton term in off diagonal if active cell
701  if (this%ibound(n) > 0) then
702  amat(idxglo(ii)) = amat(idxglo(ii)) + term
703  end if
704  !fill row of m
705  amat(idxglo(idiagm)) = amat(idxglo(idiagm)) - term
706  ! fill newton term in off diagonal if active cell
707  if (this%ibound(m) > 0) then
708  amat(idxglo(isymcon)) = amat(idxglo(isymcon)) !* dwadi !need to add dwadi
709  end if
710  end if
711  endif
712 
713  enddo
714  end do
715  !
716  end if
717  !
718  ! -- Return
719  return
720  end subroutine npf_fn

724 ! bnd_nur -- under-relaxation
725 ! Subroutine: (1) Under-relaxation of Groundwater Flow Model Heads for current
726 ! outer iteration using the cell bottoms at the bottom of the
727 ! model
728 ! ******************************************************************************
729 !
730 ! SPECIFICATIONS:
731 ! ------------------------------------------------------------------------------

2047  !> @brief preprocess the NPF input data
2048  !!
2049  !! This routine consists of the following steps:
2050  !!
2051  !! 1. convert cells to noflow when all transmissive parameters equal zero
2052  !! 2. perform initial wetting and drying
2053  !! 3. initialize cell saturation
2054  !! 4. calculate saturated conductance (when not xt3d)
2055  !! 5. If NEWTON under-relaxation, determine lower most node
2056  !<
2057  subroutine preprocess_input(this)
2058  use constantsmodule, only: linelength
2059  use memorymanagermodule, only: mem_allocate, mem_reallocate, mem_deallocate
2060  use simmodule, only: store_error, count_errors
2061  class(gwfnpftype) :: this !< the instance of the NPF package
2062  ! local
2063  integer(I4B) :: n, m, ii, nn
2064  real(DP) :: hyn, hym
2065  real(DP) :: satn, topn, botn
2066  integer(I4B) :: nextn
2067  real(DP) :: minbot, botm

2069  character(len=LINELENGTH) :: cellstr, errmsg

2071  character(len=*),parameter :: fmtcnv = &
2072  "(1X,'CELL ', A, &
2073  &' ELIMINATED BECAUSE ALL HYDRAULIC CONDUCTIVITIES TO NODE ARE 0.')"

2075  "(1X,'Negative cell thickness at cell ', A)"

2077  "(1X,'Initial head, bottom elevation:',1P,2G13.5)"
2078  character(len=*),parameter :: fmttebe = &
2079  "(1X,'Top elevation, bottom elevation:',1P,2G13.5)"
2080  !
2081  do n = 1, this%dis%nodes
2082  this%ithickstartflag(n) = 0
2083  end do
2084  !
2085  ! -- Insure that each cell has at least one non-zero transmissive parameter
2086  ! Note that a cell can be deactivated even if it has a valid connection
2087  ! to another model.
2088  nodeloop: do n = 1, this%dis%nodes
2089  !
2090  ! -- Skip if already inactive
2091  if(this%ibound(n) == 0) then
2092  if(this%irewet /= 0) then
2093  if(this%wetdry(n) == dzero) cycle nodeloop
2094  else
2095  cycle nodeloop
2096  endif
2097  endif
2098  !
2099  ! -- Cycle if k11 is not zero
2100  if(this%k11(n) /= dzero) cycle nodeloop

2102  ! -- Cycle if at least one vertical connection has non-zero k33
2103  ! for n and m
2104  do ii = this%dis%con%ia(n) + 1, this%dis%con%ia(n + 1) - 1
2105  m = this%dis%con%ja(ii)
2106  if(this%dis%con%ihc(this%dis%con%jas(ii)) == 0) then
2107  hyn = this%k11(n)
2108  if(this%ik33 /= 0) hyn = this%k33(n)
2109  if(hyn /= dzero) then
2110  hym = this%k11(m)
2111  if(this%ik33 /= 0) hym = this%k33(m)
2112  if(hym /= dzero) cycle
2113  endif
2114  endif
2115  enddo
2116  !
2117  ! -- If this part of the loop is reached, then all connections have
2118  ! zero transmissivity, so convert to noflow.
2119  this%ibound(n) = 0
2120  this%hnew(n) = this%hnoflo
2121  if(this%irewet /= 0) this%wetdry(n) = dzero
2122  call this%dis%noder_to_string(n, cellstr)
2123  write(this%iout, fmtcnv) trim(adjustl(cellstr))
2124  !
2125  enddo nodeloop
2126  !
2127  ! -- Preprocess cell status and heads based on initial conditions
2128  if (this%inewton == 0) then
2129  !
2130  ! -- For standard formulation (non-Newton) call wetdry routine
2131  call this%wd(0, this%hnew)
2132  else
2133  !
2134  ! -- Newton formulation, so adjust heads to be above bottom
2135  ! (Not used in present formulation because variable cv
2136  ! cannot be used with Newton)
2137  if (this%ivarcv == 1) then
2138  do n = 1, this%dis%nodes
2139  if (this%hnew(n) < this%dis%bot(n)) then
2140  this%hnew(n) = this%dis%bot(n) + dem6
2141  end if
2142  end do
2143  end if
2144  end if
2145  !
2146  ! -- Initialize sat to zero for ibound=0 cells, unless the cell can
2147  ! rewet. Initialize sat to the saturated fraction based on strt
2148  ! if icelltype is negative and the THCKSTRT option is in effect.
2149  ! Initialize sat to 1.0 for all other cells in order to calculate
2150  ! condsat in next section.
2151  do n = 1, this%dis%nodes
2152  if(this%ibound(n) == 0) then
2153  this%sat(n) = done
2154  if(this%icelltype(n) < 0 .and. this%ithickstrt /= 0) then
2155  this%ithickstartflag(n) = 1
2156  this%icelltype(n) = 0
2157  endif
2158  else
2159  topn = this%dis%top(n)
2160  botn = this%dis%bot(n)
2161  if(this%icelltype(n) < 0 .and. this%ithickstrt /= 0) then
2162  call this%thksat(n, this%ic%strt(n), satn)
2163  if(botn > this%ic%strt(n)) then
2164  call this%dis%noder_to_string(n, cellstr)
2165  write(errmsg, fmtnct) trim(adjustl(cellstr))
2166  call store_error(errmsg)
2167  write(errmsg, fmtihbe) this%ic%strt(n), botn
2168  call store_error(errmsg)
2169  endif
2170  this%ithickstartflag(n) = 1
2171  this%icelltype(n) = 0
2172  else
2173  satn = done
2174  if(botn > topn) then
2175  call this%dis%noder_to_string(n, cellstr)
2176  write(errmsg, fmtnct) trim(adjustl(cellstr))
2177  call store_error(errmsg)
2178  write(errmsg, fmttebe) topn, botn
2179  call store_error(errmsg)
2180  endif
2181  endif
2182  this%sat(n) = satn
2183  endif
2184  enddo
2185  if(count_errors() > 0) then
2186  call this%parser%StoreErrorUnit()
2187  endif
2188  !
2189  ! -- Calculate condsat, but only if xt3d is not active. If xt3d is
2190  ! active, then condsat is allocated to size of zero.
2191  if (this%ixt3d == 0) then
2192  !
2193  ! -- Calculate the saturated conductance for all connections assuming
2194  ! that saturation is 1 (except for case where icelltype was entered
2195  ! as a negative value and THCKSTRT option in effect)
2196  do n = 1, this%dis%nodes
2197  call this%calc_condsat(n, .true.)
2198  enddo
2199  !
2200  endif
2201  !
2202  ! -- Determine the lower most node
2203  if (this%igwfnewtonur /= 0) then
2204  call mem_reallocate(this%ibotnode, this%dis%nodes, 'IBOTNODE', &
2205  trim(this%memoryPath))
2206  do n = 1, this%dis%nodes
2207  !
2208  minbot = this%dis%bot(n)
2209  nn = n
2210  finished = .false.
2211  do while(.not. finished)
2212  nextn = 0
2213  !
2214  ! -- Go through the connecting cells
2215  do ii = this%dis%con%ia(nn) + 1, this%dis%con%ia(nn + 1) - 1
2216  !
2217  ! -- Set the m cell number
2218  m = this%dis%con%ja(ii)

2949  function condmean(k1, k2, thick1, thick2, cl1, cl2, width, iavgmeth)
2950 ! ******************************************************************************
2951 ! condmean -- Calculate the conductance between two cells
2952 !
2953 ! k1 is hydraulic conductivity for cell 1 (in the direction of cell2)
2954 ! k2 is hydraulic conductivity for cell 2 (in the direction of cell1)
2955 ! thick1 is the saturated thickness for cell 1
2956 ! thick2 is the saturated thickness for cell 2
2957 ! cl1 is the distance from the center of cell1 to the shared face with cell2
2958 ! cl2 is the distance from the center of cell2 to the shared face with cell1
2959 ! h1 is the head for cell1
2960 ! h2 is the head for cell2
2961 ! width is the width perpendicular to flow
2962 ! iavgmeth is the averaging method:
2963 ! 0 is harmonic averaging
2964 ! 1 is logarithmic averaging
2965 ! 2 is arithmetic averaging of sat thickness and logarithmic averaging of
2966 ! hydraulic conductivity
2967 ! 3 is arithmetic averaging of sat thickness and harmonic averaging of
2968 ! hydraulic conductivity
2969 ! ******************************************************************************
2970 !
2971 ! SPECIFICATIONS:
2972 ! ------------------------------------------------------------------------------
2973  ! -- return
2974  real(dp) :: condmean
2975  ! -- dummy
2976  real(dp), intent(in) :: k1
2977  real(dp), intent(in) :: k2
2978  real(dp), intent(in) :: thick1
2979  real(dp), intent(in) :: thick2
2980  real(dp), intent(in) :: cl1
2981  real(dp), intent(in) :: cl2
2982  real(dp), intent(in) :: width
2983  integer(I4B), intent(in) :: iavgmeth
2984  ! -- local
2985  real(dp) :: t1
2986  real(dp) :: t2
2987  real(dp) :: tmean, kmean, denom
2988 ! ------------------------------------------------------------------------------
2989  !
2990  ! -- Initialize
2991  t1 = k1 * thick1
2992  t2 = k2 * thick2
2993  !
2994  ! -- Averaging
2995  select case (iavgmeth)
2996  !
2997  ! -- Harmonic-mean method
2998  case(0)
2999  !
3000  if (t1*t2 > dzero) then
3001  condmean = width * t1 * t2 / (t1 * cl2 + t2 * cl1)
3002  else
3003  condmean = dzero
3004  end if
3005  !
3006  ! -- Logarithmic-mean method
3007  case(1)
3008  if (t1*t2 > dzero) then
3009  tmean = logmean(t1, t2)
3010  else
3011  tmean = dzero
3012  endif
3013  condmean = tmean * width / (cl1 + cl2)
3014  !
3015  ! -- Arithmetic-mean thickness and logarithmic-mean hydraulic conductivity
3016  case(2)
3017  if (k1*k2 > dzero) then
3018  kmean = logmean(k1, k2)
3019  else
3020  kmean = dzero
3021  endif
3022  condmean = kmean * dhalf * (thick1 + thick2) * width / (cl1 + cl2)
3023  !
3024  ! -- Arithmetic-mean thickness and harmonic-mean hydraulic conductivity
3025  case(3)
3026  denom = (k1 * cl2 + k2 * cl1)
3027  if (denom > dzero) then
3028  kmean = k1 * k2 / denom
3029  else
3030  kmean = dzero
3031  end if
3032  condmean = kmean * dhalf * (thick1 + thick2) * width
3033  end select
3034  !
3035  ! -- Return
3036  return
3037  end function condmean

3067  function hyeff_calc(k11, k22, k33, ang1, ang2, ang3, vg1, vg2, vg3, &
3068  iavgmeth) result(hyeff)
3069 ! ******************************************************************************
3070 ! hyeff_calc -- Calculate the effective horizontal hydraulic conductivity from
3071 ! an ellipse using a specified direction (unit vector vg1, vg2, vg3).
3072 ! k11 is the hydraulic conductivity of the major ellipse axis
3073 ! k22 is the hydraulic conductivity of first minor axis
3074 ! k33 is the hydraulic conductivity of the second minor axis
3075 ! ang1 is the counter-clockwise rotation (radians) of the ellipse in
3076 ! the (x, y) plane
3077 ! ang2 is the rotation of the conductivity ellipsoid upward or
3078 ! downward from the (x, y) plane
3079 ! ang3 is the rotation of the conductivity ellipsoid about the major
3080 ! axis
3081 ! vg1, vg2, and vg3 are the components of a unit vector in model coordinates
3082 ! in the direction of the connection between cell n and m

3084 ! if one, then use arithmetic averaging.
3085 ! ******************************************************************************
3086 !
3087 ! SPECIFICATIONS:
3088 ! ------------------------------------------------------------------------------
3089  ! -- modules
3090  use constantsmodule, only: done
3091  ! -- result
3092  real(dp) :: hyeff
3093  ! -- dummy
3094  real(dp), intent(in) :: k11
3095  real(dp), intent(in) :: k22
3096  real(dp), intent(in) :: k33
3097  real(dp), intent(in) :: ang1
3098  real(dp), intent(in) :: ang2
3099  real(dp), intent(in) :: ang3
3100  real(dp), intent(in) :: vg1
3101  real(dp), intent(in) :: vg2
3102  real(dp), intent(in) :: vg3
3103  integer(I4B), intent(in) :: iavgmeth
3104  ! -- local
3105  real(dp) :: s1, s2, s3, c1, c2, c3
3106  real(dp), dimension(3,3) :: r

3109 ! ------------------------------------------------------------------------------
3110  !
3111  ! -- Sin and cos of angles
3112  s1 = sin(ang1)
3113  c1 = cos(ang1)

3115  c2 = cos(ang2)
3116  s3 = sin(ang3)
3117  c3 = cos(ang3)
3118  !
3119  ! -- Rotation matrix
3120  r(1,1) = c1*c2
3121  r(1,2) = c1*s2*s3 - s1*c3
3122  r(1,3) = -c1*s2*c3 - s1*s3
3123  r(2,1) = s1*c2
3124  r(2,2) = s1*s2*s3 + c1*c3
3125  r(2,3) = -s1*s2*c3 + c1*s3
3126  r(3,1) = s2
3127  r(3,2) = -c2*s3
3128  r(3,3) = c2*c3
3129  !
3130  ! -- Unit vector in direction of n-m connection in a local coordinate
3131  ! system aligned with the ellipse axes
3132  ve1 = r(1, 1) * vg1 + r(2, 1) * vg2 + r(3, 1) * vg3
3133  ve2 = r(1, 2) * vg1 + r(2, 2) * vg2 + r(3, 2) * vg3
3134  ve3 = r(1, 3) * vg1 + r(2, 3) * vg2 + r(3, 3) * vg3
3135  !

3137  ! or arithmetic (2) weighting
3138  hyeff = dzero
3139  if (iavgmeth == 0) then
3140  !
3141  ! -- Arithmetic weighting. If principal direction corresponds exactly with
3142  ! unit vector then set to principal direction. Otherwise weight it.
3143  dnum = done
3144  d1 = ve1 ** 2
3145  d2 = ve2 ** 2
3146  d3 = ve3 ** 2
3147  if (ve1 /= dzero) then
3148  dnum = dnum * k11
3149  d2 = d2 * k11
3150  d3 = d3 * k11
3151  end if
3152  if (ve2 /= dzero) then
3153  dnum = dnum * k22
3154  d1 = d1 * k22
3155  d3 = d3 * k22
3156  end if
3157  if (ve3 /= dzero) then
3158  dnum = dnum * k33
3159  d1 = d1 * k33
3160  d2 = d2 * k33
3161  end if
3162  denom = d1 + d2 + d3
3163  if (denom > dzero) hyeff = dnum / denom
3164  else if (iavgmeth == 1) then
3165  ! -- arithmetic
3166  hyeff = ve1 ** 2 * k11 + ve2 ** 2 * k22 + ve3 ** 2 * k33
3167  end if

3169  ! -- Return
3170  return
3171  end function hyeff_calc
