597  topdn = this%dis%top(idn)
598  botdn = this%dis%bot(idn)
599  afac = dtwo / (done + (topdn - botdn) / (topup - botup))

607  derv = squadraticsaturationderivative(topup, botup, hnew(iups), &
608  this%satomega, this%satmin)

613  !isymcon = this%dis%con%isym(ii)
614  term = consterm * derv

619  ! fill newton term in off diagonal if active cell
620  if (this%ibound(n) > 0) then
621  amat(idxglo(ii)) = amat(idxglo(ii)) !* dwadi !need to add dwadi
622  end if
623  !fill row of m
624  amat(idxglo(idiagm)) = amat(idxglo(idiagm)) !- filledterm * (dwadi - DONE) !need to add dwadi
625  ! fill newton term in off diagonal if active cell

627  amat(idxglo(isymcon)) = amat(idxglo(isymcon)) - term
628  end if
629  ! fill jacobian for m being the upstream node
630  else
631  hds = hnew(n)
632  term = -consterm * derv
633  rhs(n) = rhs(n) + term * hnew(m) !+ amat(idxglo(ii)) * (dwadi * hds - hds) !need to add dwadi
634  rhs(m) = rhs(m) - term * hnew(m) !- amat(idxglo(ii)) * (dwadi * hds - hds) !need to add dwadi
635  ! fill in row of n
636  amat(idxglo(idiag)) = amat(idxglo(idiag)) !- filledterm * (dwadi - DONE) !need to add dwadi
637  ! fill newton term in off diagonal if active cell
638  if (this%ibound(n) > 0) then
639  amat(idxglo(ii)) = amat(idxglo(ii)) + term
640  end if
641  !fill row of m
642  amat(idxglo(idiagm)) = amat(idxglo(idiagm)) - term
643  ! fill newton term in off diagonal if active cell
644  if (this%ibound(m) > 0) then
645  amat(idxglo(isymcon)) = amat(idxglo(isymcon)) !* dwadi !need to add dwadi
646  end if
647  end if
648  endif
649 
650  enddo
651  end do
652  !
653  end if
654  !
655  ! -- Return
656  return
657  end subroutine npf_fn

663 ! outer iteration using the cell bottoms at the bottom of the

1960  !> @brief preprocess the NPF input data
1961  !!
1962  !! This routine consists of the following steps:
1963  !!
1964  !! 1. convert cells to noflow when all transmissive parameters equal zero
1965  !! 2. perform initial wetting and drying
1966  !! 3. initialize cell saturation
1967  !! 4. calculate saturated conductance (when not xt3d)
1968  !! 5. If NEWTON under-relaxation, determine lower most node
1969  !<
1970  subroutine preprocess_input(this)
1971  use constantsmodule, only: linelength
1972  use memorymanagermodule, only: mem_allocate, mem_reallocate, mem_deallocate
1973  use simmodule, only: store_error, ustop, count_errors
1974  class(gwfnpftype) :: this !< the instance of the NPF package
1975  ! local
1976  integer(I4B), dimension(:), pointer, contiguous :: ithickstartflag
1977  integer(I4B) :: n, m, ii, nn, ihc
1978  real(DP) :: hyn, hym
1979  real(DP) :: csat
1980  real(DP) :: fawidth
1981  real(DP) :: satn, topn, topm, botn
1982  real(DP) :: hn, hm
1983  integer(I4B) :: nextn
1984  real(DP) :: minbot, botm
1985  logical :: finished
1986  character(len=LINELENGTH) :: cellstr, errmsg
1987  ! format strings
1988  character(len=*),parameter :: fmtcnv = &
1989  "(1X,'CELL ', A, &
1990  &' ELIMINATED BECAUSE ALL HYDRAULIC CONDUCTIVITIES TO NODE ARE 0.')"
1991  character(len=*),parameter :: fmtnct = &
1992  "(1X,'Negative cell thickness at cell ', A)"
1993  character(len=*),parameter :: fmtihbe = &
1994  "(1X,'Initial head, bottom elevation:',1P,2G13.5)"
1995  character(len=*),parameter :: fmttebe = &
1996  "(1X,'Top elevation, bottom elevation:',1P,2G13.5)"
1997  !
1998  ! -- allocate temporary storage to handle thickstart option
1999  allocate(ithickstartflag(this%dis%nodes))
2000  do n = 1, this%dis%nodes
2001  ithickstartflag(n) = 0
2002  end do
2003  !
2004  ! -- Insure that each cell has at least one non-zero transmissive parameter
2005  ! Note that a cell can be deactivated even if it has a valid connection
2006  ! to another model.
2007  nodeloop: do n = 1, this%dis%nodes
2008  !
2009  ! -- Skip if already inactive
2010  if(this%ibound(n) == 0) then
2011  if(this%irewet /= 0) then
2012  if(this%wetdry(n) == dzero) cycle nodeloop
2013  else
2014  cycle nodeloop
2015  endif
2016  endif
2017  !
2018  ! -- Cycle if k11 is not zero
2019  if(this%k11(n) /= dzero) cycle nodeloop
2020  !
2021  ! -- Cycle if at least one vertical connection has non-zero k33
2022  ! for n and m
2023  do ii = this%dis%con%ia(n) + 1, this%dis%con%ia(n + 1) - 1
2024  m = this%dis%con%ja(ii)
2025  if(this%dis%con%ihc(this%dis%con%jas(ii)) == 0) then
2026  hyn = this%k11(n)
2027  if(this%ik33 /= 0) hyn = this%k33(n)
2028  if(hyn /= dzero) then
2029  hym = this%k11(m)
2030  if(this%ik33 /= 0) hym = this%k33(m)
2031  if(hym /= dzero) cycle
2032  endif
2033  endif
2034  enddo
2035  !
2036  ! -- If this part of the loop is reached, then all connections have
2037  ! zero transmissivity, so convert to noflow.
2038  this%ibound(n) = 0
2039  this%hnew(n) = this%hnoflo
2040  if(this%irewet /= 0) this%wetdry(n) = dzero
2041  call this%dis%noder_to_string(n, cellstr)
2042  write(this%iout, fmtcnv) trim(adjustl(cellstr))
2043  !
2044  enddo nodeloop
2045  !
2046  ! -- Preprocess cell status and heads based on initial conditions
2047  if (this%inewton == 0) then
2048  !
2049  ! -- For standard formulation (non-Newton) call wetdry routine
2050  call this%wd(0, this%hnew)
2051  else
2052  !
2053  ! -- Newton formulation, so adjust heads to be above bottom
2054  ! (Not used in present formulation because variable cv
2055  ! cannot be used with Newton)
2056  if (this%ivarcv == 1) then
2057  do n = 1, this%dis%nodes
2058  if (this%hnew(n) < this%dis%bot(n)) then
2059  this%hnew(n) = this%dis%bot(n) + dem6
2060  end if
2061  end do
2062  end if
2063  end if
2064  !
2065  ! -- Initialize sat to zero for ibound=0 cells, unless the cell can
2066  ! rewet. Initialize sat to the saturated fraction based on strt
2067  ! if icelltype is negative and the THCKSTRT option is in effect.
2068  ! Initialize sat to 1.0 for all other cells in order to calculate
2069  ! condsat in next section.
2070  do n = 1, this%dis%nodes
2071  if(this%ibound(n) == 0) then
2072  this%sat(n) = done
2073  if(this%icelltype(n) < 0 .and. this%ithickstrt /= 0) then
2074  ithickstartflag(n) = 1

2081  call this%thksat(n, this%ic%strt(n), satn)
2082  if(botn > this%ic%strt(n)) then
2083  call this%dis%noder_to_string(n, cellstr)
2084  write(errmsg, fmtnct) trim(adjustl(cellstr))

2086  write(errmsg, fmtihbe) this%ic%strt(n), botn
2087  call store_error(errmsg)
2088  endif
2089  ithickstartflag(n) = 1
2090  this%icelltype(n) = 0
2091  else
2092  satn = done
2093  if(botn > topn) then
2094  call this%dis%noder_to_string(n, cellstr)
2095  write(errmsg, fmtnct) trim(adjustl(cellstr))
2096  call store_error(errmsg)
2097  write(errmsg, fmttebe) topn, botn
2098  call store_error(errmsg)
2099  endif
2100  endif
2101  this%sat(n) = satn
2102  endif
2103  enddo
2104  if(count_errors() > 0) then
2105  call this%parser%StoreErrorUnit()
2106  call ustop()
2107  endif
2108  !
2109  ! -- Calculate condsat, but only if xt3d is not active. If xt3d is
2110  ! active, then condsat is allocated to size of zero.
2111  if (this%ixt3d == 0) then
2112  !
2113  ! -- Calculate the saturated conductance for all connections assuming
2114  ! that saturation is 1 (except for case where icelltype was entered
2115  ! as a negative value and THCKSTRT option in effect)
2116  do n = 1, this%dis%nodes
2117  !
2118  topn = this%dis%top(n)
2119  !
2120  ! -- Go through the connecting cells
2121  do ii = this%dis%con%ia(n) + 1, this%dis%con%ia(n + 1) - 1
2122  !
2123  ! -- Set the m cell number and cycle if lower triangle connection
2124  m = this%dis%con%ja(ii)
2125  if (m < n) cycle
2126  ihc = this%dis%con%ihc(this%dis%con%jas(ii))
2127  topm = this%dis%top(m)
2128  hyn = this%hy_eff(n, m, ihc, ipos=ii)
2129  hym = this%hy_eff(m, n, ihc, ipos=ii)
2130  if (ithickstartflag(n) == 0) then
2131  hn = topn
2132  else
2133  hn = this%ic%strt(n)
2134  end if
2135  if (ithickstartflag(m) == 0) then
2136  hm = topm
2137  else
2138  hm = this%ic%strt(m)
2139  end if
2140  !
2141  ! -- Calculate conductance depending on whether connection is
2142  ! vertical (0), horizontal (1), or staggered horizontal (2)
2143  if(ihc == 0) then
2144  !
2145  ! -- Vertical conductance for fully saturated conditions
2146  csat = vcond(1, 1, 1, 1, 0, 1, 1, done, &
2147  this%dis%bot(n), this%dis%bot(m), &
2148  hyn, hym, &
2149  this%sat(n), this%sat(m), &
2150  topn, topm, &
2151  this%dis%bot(n), this%dis%bot(m), &
2152  this%dis%con%hwva(this%dis%con%jas(ii)))
2153  else

2156  fawidth = this%dis%con%hwva(this%dis%con%jas(ii))
2157  csat = hcond(1, 1, 1, 1, this%inewton, 0, &
2158  this%dis%con%ihc(this%dis%con%jas(ii)), &
2159  this%icellavg, this%iusgnrhc, this%inwtupw, &
2160  done, &
2161  hn, hm, this%sat(n), this%sat(m), hyn, hym, &
2162  topn, topm, &
2163  this%dis%bot(n), this%dis%bot(m), &
2164  this%dis%con%cl1(this%dis%con%jas(ii)), &
2165  this%dis%con%cl2(this%dis%con%jas(ii)), &
2166  fawidth, this%satomega, this%satmin)
2167  end if
2168  this%condsat(this%dis%con%jas(ii)) = csat
2169  enddo
2170  enddo
2171  !
2172  endif
2173  !
2174  ! -- Determine the lower most node
2175  if (this%igwfnewtonur /= 0) then
2176  call mem_reallocate(this%ibotnode, this%dis%nodes, 'IBOTNODE', &
2177  trim(this%memoryPath))
2178  do n = 1, this%dis%nodes
2179  !
2180  minbot = this%dis%bot(n)
2181  nn = n
2182  finished = .false.
2183  do while(.not. finished)
2184  nextn = 0
2185  !
2186  ! -- Go through the connecting cells
2187  do ii = this%dis%con%ia(nn) + 1, this%dis%con%ia(nn + 1) - 1
2188  !
2189  ! -- Set the m cell number
2190  m = this%dis%con%ja(ii)
2191  botm = this%dis%bot(m)
2192  !
2193  ! -- select vertical connections: ihc == 0
2194  if(this%dis%con%ihc(this%dis%con%jas(ii)) == 0) then
2195  if (m > nn .and. botm < minbot) then
2196  nextn = m
2197  minbot = botm
2198  end if
2199  end if
2200  end do
2201  if (nextn > 0) then
2202  nn = nextn
2203  else
2204  finished = .true.
2205  end if
2206  end do
2207  this%ibotnode(n) = nn
2208  end do
2209  end if
2210  !
2211  ! -- nullify unneeded gwf pointers
2212  this%igwfnewtonur => null()
2213  !
2214  ! - clean up local storage

2218  return

2561 !

2600  integer(I4B) :: indk
2601  real(dp) :: satmin
2602  real(dp) :: sn

2700

2952  real(dp), intent(in) :: k11
2953  real(dp), intent(in) :: k22
2954  real(dp), intent(in) :: k33

2973  c2 = cos(ang2)

2975  c3 = cos(ang3)

2978  r(1,1) = c1*c2
2979  r(1,2) = c1*s2*s3 - s1*c3
2980  r(1,3) = -c1*s2*c3 - s1*s3

2982  r(2,2) = s1*s2*s3 + c1*c3

2985  r(3,2) = -c2*s3
2986  r(3,3) = c2*c3

2989  ! system aligned with the ellipse axes

2991  ve2 = r(1, 2) * vg1 + r(2, 2) * vg2 + r(3, 2) * vg3
2992  ve3 = r(1, 3) * vg1 + r(2, 3) * vg2 + r(3, 3) * vg3

3003  d2 = ve2 ** 2
3004  d3 = ve3 ** 2

3012  d1 = d1 * k22

3017  d1 = d1 * k33
3018  d2 = d2 * k33
3019  end if

3021  if (denom > dzero) hyeff = dnum / denom
3022  else if (iavgmeth == 1) then
3023  ! -- arithmetic
3024  hyeff = ve1 ** 2 * k11 + ve2 ** 2 * k22 + ve3 ** 2 * k33
