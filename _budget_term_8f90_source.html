174 ! ******************************************************************************
175 ! accumulate_flow -- calculate ratin and ratout for all the flow terms
176 ! ******************************************************************************

189  ratin = dzero

201  subroutine save_flows(this, dis, ibinun, kstp, kper, delt, pertim, totim, &
202  iout)
203 ! ******************************************************************************
204 ! save_flows -- write flows to a binary file
205 ! ******************************************************************************
206 !
207 ! SPECIFICATIONS:
208 ! ------------------------------------------------------------------------------
209  ! -- modules
210  ! -- dummy
211  class(budgettermtype) :: this
212  class(disbasetype), intent(in) :: dis
213  integer(I4B), intent(in) :: ibinun
214  integer(I4B), intent(in) :: kstp
215  integer(I4B), intent(in) :: kper
216  real(DP), intent(in) :: delt
217  real(DP), intent(in) :: pertim
218  real(DP), intent(in) :: totim
219  integer(I4B), intent(in) :: iout
220  ! -- local
221  integer(I4B) :: nlist
222  integer(I4B) :: i
223  integer(I4B) :: n1
224  integer(I4B) :: n2
225  real(DP) :: q
226 ! ------------------------------------------------------------------------------
227  !
228  ! -- Count the size of the list and exclude ids less than or equal to zero
229  nlist = 0
230  do i = 1, this%nlist
231  n1 = this%id1(i)
232  n2 = this%id2(i)
233  if (n1 <= 0 .or. n2 <= 0) cycle
234  nlist = nlist + 1
235  end do
236  !
237  ! -- Write the header
238  call ubdsv06(kstp, kper, this%flowtype, &
239  this%text1id1, this%text2id1, &
240  this%text1id2, this%text2id2, &
241  ibinun, this%naux, this%auxtxt, &
242  nlist, 1, 1, nlist, &
243  iout, delt, pertim, totim)
244  !
245  ! -- Write each entry
246  do i = 1, this%nlist
247  q = this%flow(i)
248  n1 = this%id1(i)
249  n2 = this%id2(i)
250  if (n1 <= 0 .or. n2 <= 0) cycle
251  call dis%record_mf6_list_entry(ibinun, n1, n2, q, &
252  this%naux, this%auxvar(:, i), &
253  olconv=this%olconv1, &
254  olconv2=this%olconv2)
255  end do
256  end subroutine save_flows

360 !

362 ! ------------------------------------------------------------------------------
363  ! -- modules

372  real(DP), intent(inout) :: totim
373  ! -- local

376  integer(I4B) :: n1
377  integer(I4B) :: n2

392  deallocate(this%auxtxt)
393  allocate(this%auxtxt(this%naux))
394  end if
395  end if
396  if (this%naux > 0) read(ibinun) (this%auxtxt(j), j = 1, this%naux)
397  read(ibinun) this%nlist
398  if (.not. associated(this%id1)) then
399  this%maxlist = this%nlist
400  allocate(this%id1(this%maxlist))
401  allocate(this%id2(this%maxlist))
402  allocate(this%flow(this%maxlist))
403  allocate(this%auxvar(this%naux, this%maxlist))
404  else
405  if (this%nlist > this%maxlist) then
406  this%maxlist = this%nlist

410  deallocate(this%auxvar)
411  allocate(this%id1(this%maxlist))

413  allocate(this%flow(this%maxlist))
414  allocate(this%auxvar(this%naux, this%maxlist))

416  end if

419  read(ibinun) n2
420  read(ibinun) q
