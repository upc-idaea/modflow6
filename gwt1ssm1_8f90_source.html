193  subroutine ssm_term(this, ipackage, ientry, rrate, rhsval, hcofval)
194 ! ******************************************************************************
195 ! ssm_term
196 ! ******************************************************************************
197 !
198 ! SPECIFICATIONS:
199 ! ------------------------------------------------------------------------------
200  ! -- dummy
201  class(gwtssmtype) :: this
202  integer(I4B), intent(in) :: ipackage
203  integer(I4B), intent(in) :: ientry
204  real(DP), intent(out), optional :: rrate
205  real(DP), intent(out), optional :: rhsval
206  real(DP), intent(out), optional :: hcofval
207  ! -- local
208  integer(I4B) :: n
209  integer(I4B) :: iauxpos
210  real(DP) :: qbnd
211  real(DP) :: ctmp
212  real(DP) :: omega
213  real(DP) :: hcoftmp
214  real(DP) :: rhstmp
215 ! ------------------------------------------------------------------------------
216  !
217  ! -- retrieve node number, qbnd and iauxpos
218  hcoftmp = dzero
219  rhstmp = dzero
220  ctmp = dzero
221  n = this%fmi%gwfpackages(ipackage)%nodelist(ientry)
222  !
223  ! -- If cell is active (ibound > 0) then calculate values
224  if (this%ibound(n) > 0) then
225  !
226  ! -- retrieve qbnd and iauxpos
227  qbnd = this%fmi%gwfpackages(ipackage)%get_flow(ientry)
228  iauxpos = this%iauxpak(ipackage)
229  !
230  ! -- assign values for hcoftmp, rhstmp, and ctmp for subsequent assigment
231  ! of hcof, rhs, and rate
232  if(iauxpos >= 0) then
233  !
234  ! -- concentration is zero or stored in auxiliary variable; if
235  ! qbnd is positive, then concentration represents the inflow
236  ! concentration. If qbnd is negative, then the outflow concentration
237  ! is set equal to the simulated cell concentration
238  if (qbnd >= dzero) then
239  if (iauxpos > 0) then
240  ctmp = this%fmi%gwfpackages(ipackage)%auxvar(iauxpos, ientry)
241  else
242  ctmp = dzero
243  end if
244  omega = dzero ! rhs
245  else
246  ctmp = this%cnew(n)
247  omega = done ! lhs
248  end if
249  else if (iauxpos < 0) then
250  !
251  ! -- concentration stored in auxiliary variable; negative iauxpos value
252  ! indicates that this is a mixed sink type where the concentration
253  ! value represents the injected concentration if qbnd is positive.
254  ! If qbnd is negative, then the withdrawn water is equal to the
255  ! minimum of the aux concentration and the cell concentration.
256  ctmp = this%fmi%gwfpackages(ipackage)%auxvar(-iauxpos, ientry)
257  if (qbnd >= dzero) then
258  omega = dzero ! rhs (ctmp is aux value)
259  else
260  if (ctmp < this%cnew(n)) then
261  omega = dzero ! rhs (ctmp is aux value)
262  else
263  omega = done ! lhs (ctmp is cell concentration)
264  ctmp = this%cnew(n)
265  end if
266  end if
267  endif
268  !
269  ! -- Add terms based on qbnd sign
270  if(qbnd <= dzero) then
271  hcoftmp = qbnd * omega
272  else
273  rhstmp = -qbnd * ctmp * (done - omega)
274  endif
275  !
276  ! -- end of active ibound
277  end if
278  !
279  ! -- set requested values
280  if (present(hcofval)) hcofval = hcoftmp
281  if (present(rhsval)) rhsval = rhstmp
282  if (present(rrate)) rrate = hcoftmp * ctmp - rhstmp
283  !
284  ! -- return
285  return
286  end subroutine ssm_term

374  subroutine ssm_bd(this, isuppress_output, model_budget)
375 ! ******************************************************************************
376 ! ssm_bd -- Calculate budget terms
377 ! ******************************************************************************
378 !
379 ! SPECIFICATIONS:
380 ! ------------------------------------------------------------------------------
381  ! -- modules
382  use tdismodule, only: delt
383  use budgetmodule, only: budgettype
384  ! -- dummy
385  class(gwtssmtype) :: this
386  integer(I4B), intent(in) :: isuppress_output
387  type(budgettype), intent(inout) :: model_budget
388  ! -- local
389  character(len=LENPACKAGENAME) :: rowlabel = 'SSM'
390  integer(I4B) :: ip
391  integer(I4B) :: i
392  real(DP) :: rate
393  real(DP) :: rin
394  real(DP) :: rout
395 ! ------------------------------------------------------------------------------
396  !
397  ! -- do for each flow package, unless it is being handled by an advanced
398  ! transport package
399  do ip = 1, this%fmi%nflowpack
400  !
401  ! -- cycle if package is being managed as an advanced package
402  if (this%fmi%iatp(ip) /= 0) cycle
403  !
404  ! -- Initialize the rate accumulators
405  rin = dzero
406  rout = dzero
407  !
408  ! -- do for each boundary
409  do i = 1, this%fmi%gwfpackages(ip)%nbound
410  call this%ssm_term(ip, i, rrate=rate)
411  if(rate < dzero) then
412  rout = rout - rate
413  else
414  rin = rin + rate
415  endif
416  !
417  enddo
418  !
419  call model_budget%addentry(rin, rout, delt, &
420  this%fmi%flowpacknamearray(ip), &
421  isuppress_output, rowlabel=rowlabel)
422  enddo
423  !
424  ! -- Return
425  return
426  end subroutine ssm_bd
