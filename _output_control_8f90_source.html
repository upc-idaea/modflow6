94  !!
95  !! Read a period data block.
96  !!
97  !<
98  subroutine oc_rp(this)
99  ! -- modules
100  use tdismodule, only: kper, nper
101  use constantsmodule, only: linelength
102  use simmodule, only: store_error, store_error_unit, count_errors
103  ! -- dummy
104  class(outputcontroltype) :: this !< OutputControlType object
105  ! -- local
106  integer(I4B) :: ierr, ival, ipos
107  logical :: isfound, found, endOfBlock
108  character(len=:), allocatable :: line

112  ! -- formats
113  character(len=*), parameter :: fmtboc = &
114  "(1X,/1X,'BEGIN READING OUTPUT CONTROL FOR STRESS PERIOD ',I0)"
115  character(len=*), parameter :: fmteoc = &
116  "(/,1X,'END READING OUTPUT CONTROL FOR STRESS PERIOD ',I0)"
117  character(len=*), parameter :: fmterr = &
118  "(' ERROR READING OUTPUT CONTROL PERIOD BLOCK: ')"
119  character(len=*), parameter :: fmtroc = &
120  "(1X,/1X,'OUTPUT CONTROL FOR STRESS PERIOD ',I0, &
121  &' IS REPEATED USING SETTINGS FROM A PREVIOUS STRESS PERIOD.')"
122  character(len=*), parameter :: fmtpererr = &
123  "(1x,'CURRENT STRESS PERIOD GREATER THAN PERIOD IN OUTPUT CONTROL.')"
124  character(len=*), parameter :: fmtpererr2 = &
125  "(1x,'CURRENT STRESS PERIOD: ',I0,' SPECIFIED STRESS PERIOD: ',I0)"
126  !
127  ! -- Read next block header if kper greater than last one read
128  if (this%iperoc < kper) then
129  !
130  ! -- Get period block
131  call this%parser%GetBlock('PERIOD', isfound, ierr, &
132  supportopenclose=.true.)
133  !
134  ! -- If end of file, set iperoc past kper, else parse line
135  if (ierr < 0) then
136  this%iperoc = nper + 1
137  write(this%iout, '(/,1x,a)') 'END OF FILE DETECTED IN OUTPUT CONTROL.'
138  write(this%iout, '(1x,a)') 'CURRENT OUTPUT CONTROL SETTINGS WILL BE '
139  write(this%iout, '(1x,a)') 'REPEATED UNTIL THE END OF THE SIMULATION.'
140  else
141  !
142  ! -- Read period number
143  ival = this%parser%GetInteger()
144  !
145  ! -- Check to see if this is a valid kper
146  if(ival <= 0 .or. ival > nper) then
147  write(ermsg, '(a,i0)') 'PERIOD NOT VALID IN OUTPUT CONTROL: ', ival
148  call store_error(ermsg)
149  write(ermsg, '(a, a)') 'LINE: ', trim(adjustl(line))
150  call store_error(ermsg)
151  endif
152  !
153  ! -- Check to see if specified is less than kper
154  if(ival < kper) then
155  write(ermsg, fmtpererr)
156  call store_error(ermsg)
157  write(ermsg, fmtpererr2) kper, ival
158  call store_error(ermsg)
159  write(ermsg, '(a, a)') 'LINE: ', trim(adjustl(line))
160  call store_error(ermsg)
161  endif
162  !
163  ! -- Stop or set iperoc and continue
164  if(count_errors() > 0) then
165  call this%parser%StoreErrorUnit()
166  endif
167  this%iperoc = ival
168  endif
169  end if
170  !
171  ! -- Read the stress period block
172  if (this%iperoc == kper) then
173  !
174  ! -- Clear io flags
175  do ipos = 1, size(this%ocdobj)
176  ocdobjptr => this%ocdobj(ipos)
177  call ocdobjptr%psmobj%init()
178  enddo
179  !
180  ! -- Output control time step matches simulation time step.
181  write(this%iout,fmtboc) this%iperoc
182  !
183  ! -- loop to read records
184  recordloop: do
185  !
186  ! -- Read the line
187  call this%parser%GetNextLine(endofblock)
188  if (endofblock) exit
189  call this%parser%GetStringCaps(keyword1)
190  !
191  ! -- Set printsave string and then read the record type (e.g.
192  ! BUDGET, HEAD)
193  printsave = keyword1
194  call this%parser%GetStringCaps(keyword2)
195  !
196  ! -- Look through the output control data objects that are
197  ! available and set ocdobjptr to the correct one based on
198  ! cname. Set found to .false. if not a valid record type.
199  found = .false.
200  do ipos = 1, size(this%ocdobj)
201  ocdobjptr => this%ocdobj(ipos)
202  if(keyword2 == trim(ocdobjptr%cname)) then
203  found = .true.
204  exit
205  endif
206  enddo
207  if (.not. found) then
208  call this%parser%GetCurrentLine(line)
209  write(ermsg, fmterr)
210  call store_error(ermsg)
211  call store_error('UNRECOGNIZED KEYWORD: '//keyword2)
212  call store_error(trim(line))
213  call this%parser%StoreErrorUnit()
214  endif
215  call this%parser%GetRemainingLine(line)
216  call ocdobjptr%psmobj%rp(trim(printsave)//' '//line, &
217  this%iout)
218  call ocdobjptr%ocd_rp_check(this%parser%iuactive)
219  !
220  ! -- End of recordloop
221  enddo recordloop
222  write(this%iout,fmteoc) this%iperoc
223  else
224  !
225  ! -- Write message that output control settings are from a previous
226  ! stress period.
227  write(this%iout, fmtroc) kper
228  endif
229  !
230  ! -- return
231  return
232  end subroutine oc_rp
